༄༅༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྱ་ག་ཡཱ་ཤིརྵ་ནཱ་མ་མ་ཧཱ་ཡཱ་ན་སཱུ་ཏྲ། བོད་སྐད་དུ། འཕགས་པ་ག་ཡཱ་མགོའི་རི་ཞེས་བྱ་བ་ཐེག་པ་ཆེན་པོའི་མདོ། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །འདི་སྐད་བདག་གིས་ཐོས་པའི་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་ནས་ཡུན་རིང་པོ་མ་ལོན་པར་བཅོམ་ལྡན་འདས་རི་ག་ཡཱ་ན་ག་ཡཱ་མགོའི་མཆོད་རྟེན་ན། དགེ་སློང་སྟོང་དུ་ཚང་བའི་དགེ་སློང་གི་དགེ་འདུན་ཆེན་པོ་ཐམས་ཅད་ཀྱང་སྔོན་རལ་པ་ཅན་དུ་གྱུར་པ། བྱ་བ་བྱས་པ། བྱེད་པ་བྱས་པ། ཁུར་བོར་བ། བདག་གི་དོན་རྗེས་སུ་ཐོབ་པ། སྲིད་པ་ཀུན་ཏུ་སྦྱོར་བ། ཡོངས་སུ་ཟད་པ། ཡང་དག་པའི་ཤེས་པས་སེམས་ཤིན་ཏུ་རྣམ་པར་གྲོལ་བ། སེམས་ཀྱི་དབང་ཐམས་ཅད་ཀྱི་དམ་པའི་ཕ་རོལ་དུ་སོན་པ། དགྲ་བཅོམ་པ་ཤ་སྟག་དང་། བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རབ་ཏུ་མང་པོ་དང་ཐབས་ཅིག་ཏུ་བཞུགས་ཏེ། དེ་ནས་བཅོམ་ལྡན་འདས་གཅིག་པུ་དབེན་པར་གཤེགས་ཏེ། ནང་དུ་ཡང་དག་བཞག་ནས་ཆོས་ཀྱི་དབྱིངས་ཉིད་ལ་བརྟགས་ཏེ། བདག་གིས་བྱང་ཆུབ་ནི་ཐོབ། བདག་གིས་ཡེ་ཤེས་ནི་ཁོང་དུ་ཆུད། བདག་གིས་བྱ་བ་ནི་བྱས། བདག་གིས་ཁུར་ནི་བོར། བདག་གིས་འཁོར་བའི་འབྲོག་དགོན་པ་ལས་ནི་རྒལ། བདག་གིས་མ་རིག་པ་ནི་བསལ། བདག་གིས་རིག་པ་ནི་སྒྲུབ། བདག་གིས་ཟུག་རྔུ་ནི་ཕྱུང་། བདག་གི་སྲེད་པ་ནི་ཟད། བདག་གིས་ཆོས་ཀྱི་གྲུ་ནི་སྦྱར། བདག་གིས་ཆོས་ཀྱི་རྔ་ནི་བརྡུངས། བདག་གིས་ཆོས་ཀྱི་དུང་ནི་བུས། བདག་གིས་ཆོས་ཀྱི་རྒྱལ་མཚན་ནི་བསྒྲེང་། བདག་གིས་འཁོར་བའི་ལུགས་ནི་བཟློག །བདག་གིས་ཆོས་ཀྱི་ལུགས་ནི་བསྙད། བདག་གིས་ལམ་ངན་པ་ནི་བཅད། བདག་གིས་ལམ་ནི་ཡང་དག་པར་བསྟན། བདག་གིས་ཞིང་ངན་པ་དག་ནི་རྣམ་པར་བཟློག །བདག་གིས་བསོད་ནམས་ཀྱི་ཞིང་དག་ནི་ཡང་དག་པར་བསྟན་ན། འདི་ལྟར་བདག་གིས་བྱང་ཆུབ་དེ་ཇི་ལྟར་ཁོང་དུ་ཆུད་པ་དང་། གང་གིས་ཁོང་དུ་ཆུད་པ་དང་། སུས་ཁོང་དུ་ཆུད་པར་འགྱུར་བའི་ཆོས་དེ་ལ་ངེས་པར་བརྟག་པར་བྱ་སྟེ། ཅི་ལུས་ཀྱིས་ཁོང་དུ་ཆུད་དམ། འོན་ཏེ་སེམས་ཀྱིས་ཁོང་དུ་ཆུད། གལ་ཏེ་ལུས་ཀྱིས་ཁོང་དུ་ཆུད་དུ་ནི་ལུས་ནི་བེམས་པོ། གཡོ་བ་མེད་པ་དང་། རྩྭ་དང་། ཤིང་དང་། རྩིག་པ་དང་། བོང་བ་དང་། མིག་ཡོར་ལྟ་བུ་སྟེ། སེམས་མེད་པ། འབྱུང་བ་ཆེན་པོ་བཞི་ལས་གྱུར་པ། ཕ་མའི་མི་གཙང་བ་ལས་བྱུང་བ་དང་། མི་རྟག་པ་དང་། རྟག་ཏུ་བསྐུ་བ་དང་། བཀྲུ་བ་དང་། མཉེ་དགོས་པ་དང་། འཇིག་པ་དང་། རྣམ་པར་འགྱེས་པ་དང་། རྣམ་པར་འཇིག་པའི་ཆོས་ཅན་ནོ། །སེམས་ནི་སྒྱུ་མ་དང་མཚུངས་པ། རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བ། གནས་མེད་པ། མཚན་ཉིད་མེད་པ། དངོས་པོ་མེད་པ། བླ་གབ་མེད་པའོ། །བྱང་ཆུབ་ནི་མིང་ཙམ་ཐ་སྙད་ཙམ་སྟེ། སྒྲ་སྐད་མེད་པ། དཔྱད་དུ་མེད་པ། གྲུབ་པ་མེད་པ། རྒྱུ་བ་མེད་པ། འཇུག་པ་མེད་པ། རྣམ་པར་རིག་པ་མེད་པ། གདགས་པར་བྱ་བ་མ་ཡིན་པ། འོང་བ་དང་འགྲོ་བ་ལས་ཡང་དག་པར་འདས་པ། གདགས་པ་ལས་ཡང་དག་པར་འདས་པ། ཁམས་གསུམ་པ་ལས་ཡང་དག་པར་འདས་པ། མ་མཐོང་བ། མ་ཐོས་པ། བྱེ་བྲག་མ་ཕྱེད་པ། གཞི་མེད་པ། དམིགས་པ་མེད་པ། རྒྱུ་བ་མེད་པ། སྤྲོས་པ་མེད་པ། འཇུག་པ་མེད་པ། བརྟག་ཏུ་མེད་པ། དཔག་ཏུ་མེད་པ། བསྟན་དུ་མེད་པ། སྒྲ་མེད་པ། ཡི་གེ་མེད་པ། ཚིག་གི་ལམ་མེད་པའོ། །དེ་ལྟར་ན་སུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ། གང་གིས་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ། མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་བྱ་བ་ནི་གང་ཡིན། ཆོས་འདི་དག་ཐམས་ཅད་ནི་མིང་ཙམ། བཏགས་པ་ཙམ། བརྡ་ཙམ། ཐ་སྙད་ཙམ་སྟེ། བརྟགས་པ། ཡོངས་སུ་བརྟགས་པ། མ་བྱུང་བ། བྱུང་བ་མེད་པ། རྫས་མ་ཡིན་པ། རྫས་མེད་པ། ལེན་པ་མེད་པ། བསྟན་དུ་མེད་པ། མངོན་པར་ཆགས་པ་མེད་པ་ཡིན་ཏེ། དེ་ལ་ནི་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བ་སུ་ཡང་མེད། ཅིས་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བ་ཡང་མེད། མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་བྱ་བ་ཅི་ཡང་མེད། རྣམ་པ་གང་གིས་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་བྱ་བ་ཡང་མེད་དེ། འདི་ལ་གང་དེ་ལྟར་མངོན་པར་རྫོགས་པར་བྱང་ཆུབ་པ་དེའི་ཕྱིར། བྱང་ཆུབ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བ་ཞེས་བྱ་སྟེ། དེ་ལ་བྱང་ཆུབ་ནི་འགྱུར་བ་མེད་པ། མི་འགྱུར་བ། མཚན་ཉིད་མེད་པའོ་སྙམ་མོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ལ་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པས་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་གལ་ཏེ་བྱང་ཆུབ་ཀྱི་མཚན་ཉིད་དེ་ལྟ་བུ་ལགས་ན། བཅོམ་ལྡན་འདས་རིགས་ཀྱི་བུའམ། རིགས་ཀྱི་བུ་མོས་ཇི་ལྟར་བྱང་ཆུབ་ཀྱི་སླད་དུ་འཇུག་པར་བགྱི། བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། འཇམ་དཔལ་བྱང་ཆུབ་དེ་ཇི་ལྟ་བ་དེ་བཞིན་དུ་བྱང་ཆུབ་སེམས་དཔས་བྱང་ཆུབ་ཀྱི་ཕྱིར་འཇུག་པར་བྱའོ། །འཇམ་དཔལ་གྱིས་གསོལ་པ། བཅོམ་ལྡན་འདས་བྱང་ཆུབ་ཇི་ལྟར་རབ་ཏུ་འཚལ་བར་འགྱུར། བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། འཇམ་དཔལ་བྱང་ཆུབ་ནི་ཁམས་གསུམ་པ་ལས་ཡང་དག་པར་འདས་པ། ཐ་སྙད་ལས་ཡང་དག་པར་འདས་པ། ཡི་གེའི་སྒྲ་སྐད་ལས་ཡང་དག་པར་འདས་པ། སྒྲ་ལས་ཡང་དག་པར་འདས་པ། འཇུག་པ་མེད་པ། གཟོད་མ་ཉིད་ནས་འཇུག་པ་མེད་པའོ། །འཇམ་དཔལ་དེ་ལྟ་བས་ན། བྱང་ཆུབ་སེམས་དཔའ་འཇུག་པ་ལ་ཡང་དག་པར་འདས་པ་རྣམས་ཀྱིས་བྱང་ཆུབ་ཀྱི་ཕྱིར་འཇུག་པར་བྱའོ། །འཇམ་དཔལ་འཇུག་པ་མེད་པའི་འཇུག་པ་ནི་བྱང་ཆུབ་ལ་འཇུག་པའོ། །འཇམ་དཔལ་དངོས་པོ་མེད་པ་ལ་འཇུག་པ་ནི་བྱང་ཆུབ་ལ་འཇུག་པའོ། །འཇམ་དཔལ་བླ་གབ་མེད་པ་ལ་འཇུག་པ་ནི་བྱང་ཆུབ་ལ་འཇུག་པའོ། །འཇམ་དཔལ་ཆོས་ཀྱི་དབྱིངས་ཀྱི་རང་བཞིན་ལ་འཇུག་པ་ནི་བྱང་ཆུབ་ལ་འཇུག་པའོ། །འཇམ་དཔལ་ཆོས་ཐམས་ཅད་ལ་མངོན་པར་ཞེན་པ་མེད་པ་ལ་འཇུག་པ་ནི་བྱང་ཆུབ་ལ་འཇུག་པའོ། །འཇམ་དཔལ་ཡང་དག་པའི་མཐའ་དབྱེར་མེད་པ་ལ་འཇུག་པ་ནི་བྱང་ཆུབ་ལ་འཇུག་པའོ། །འཇམ་དཔལ་དོར་བ་མེད་པ་དང་། གཞག་པ་མེད་པ་ལ་འཇུག་པ་ནི་བྱང་ཆུབ་ལ་འཇུག་པའོ། །འཇམ་དཔལ་ཐ་དད་པ་མ་ཡིན་པ་དང་། ཐ་དད་པ་མེད་པ་ལ་འཇུག་པ་ནི་བྱང་ཆུབ་ལ་འཇུག་པའོ། །འཇམ་དཔལ་སོ་སོ་ཡང་དག་པར་རིག་པའི་མིག་ཡོར་དང་། བྲག་ཅ་དང་། ནམ་མཁའ་དང་། ཆུ་ཟླ་དང་། སྨིག་རྒྱུ་ལ་འཇུག་པ་ནི་བྱང་ཆུབ་ལ་འཇུག་པའོ། །དེ་ནས་ལྷའི་བུ་ཟླ་བ་དྲི་མ་མེད་པའི་གཟི་བརྗིད་འོད་ཟེར་གྱི་འོད་ཀྱིས་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། །འཇམ་དཔལ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་སྤྱོད་པ་རྩོམ་པ་ནི་ཅི། གནས་ནི་གང་། འཇམ་དཔལ་གྱིས་སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་སྤྱོད་པའི་རྩོམ་པ་ནི་སྙིང་རྗེ་ཆེན་པོའོ། །གནས་ནི་སེམས་ཅན་ནོ། །སྨྲས་པ། འཇམ་དཔལ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་སྙིང་རྗེ་ཆེན་པོ་ཅིས་བསྡུས། འཇམ་དཔལ་གྱིས་སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་སྙིང་རྗེ་ཆེན་པོ་ནི་གཡོ་མེད་པས་བསྡུས་སོ། །སྨྲས་པ། འཇམ་དཔལ་གཡོ་མེད་པ་ཅིས་བསྡུས། འཇམ་དཔལ་གྱིས་སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་གཡོ་མེད་པ་ནི་སེམས་ཅན་ཐམས་ཅད་ལ་སྙོམས་པའི་སེམས་ཀྱིས་བསྡུས་པའོ། །སྨྲས་པ། འཇམ་དཔལ་སེམས་ཅན་ཐམས་ཅད་ལ་སྙོམས་པའི་སེམས་ཅིས་བསྡུས། འཇམ་དཔལ་གྱིས་སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་སེམས་ཅན་ཐམས་ཅད་ལ་སྙོམས་པའི་སེམས་ནི་ཐ་དད་པ་མ་ཡིན་པ་དང་། ཐ་དད་པ་མེད་པའི་སྤྱོད་པས་བསྡུས་སོ། །སྨྲས་པ། འཇམ་དཔལ་ཐ་དད་པ་མ་ཡིན་པ་དང་། ཐ་དད་པ་མེད་པའི་སྤྱོད་པ་ཅིས་བསྡུས། འཇམ་དཔལ་གྱིས་སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་ཐ་དད་པ་མ་ཡིན་པ་དང་། ཐ་དད་པ་མེད་པའི་སྤྱོད་པ་ནི་བསམ་པ་དང་། ལྷག་པའི་བསམ་པས་བསྡུས་པའོ། །སྨྲས་པ། འཇམ་དཔལ་བསམ་པ་དང་། ལྷག་པའི་བསམ་པ་ཅིའི་ཕྱིར་བྱུང་། སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་བསམ་པ་དང་། ལྷག་པའི་བསམ་པ་ནི་བྱང་ཆུབ་ཀྱི་སེམས་ཀྱི་ཕྱིར་བྱུང་ངོ་། །སྨྲས་པ། འཇམ་དཔལ་བྱང་ཆུབ་ཀྱི་སེམས་ཅིའི་ཕྱིར་བྱུང་། སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་བྱང་ཆུབ་ཀྱི་སེམས་ནི་ཕ་རོལ་ཏུ་ཕྱིན་པ་དྲུག་གི་ཕྱིར་བྱུང་ངོ་། །སྨྲས་པ། འཇམ་དཔལ་ཕ་རོལ་ཏུ་ཕྱིན་པ་དྲུག་ཅིའི་ཕྱིར་བྱུང་། སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་དྲུག་པོ་དག་ནི་ཐབས་དང་ཤེས་རབ་གཉིས་ཀྱི་ཕྱིར་བྱུང་ངོ་། །སྨྲས་པ། འཇམ་དཔལ་ཐབས་དང་ཤེས་རབ་ཅིའི་ཕྱིར་བྱུང་། སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་ཐབས་དང་ཤེས་རབ་ནི་བག་ཡོད་པའི་ཕྱིར་བྱུང་ངོ་། །སྨྲས་པ། འཇམ་དཔལ་བག་ཡོད་པ་ཅིའི་ཕྱིར་བྱུང་། སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་བག་ཡོད་པ་ནི་ལེགས་པར་སྤྱོད་པ་རྣམ་པ་གསུམ་གྱི་ཕྱིར་བྱུང་ངོ་། །སྨྲས་པ། འཇམ་དཔལ་ལེགས་པར་སྤྱོད་པ་རྣམ་པ་གསུམ་ཅིའི་ཕྱིར་བྱུང་། སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་ལེགས་པར་སྤྱོད་པ་རྣམ་པ་གསུམ་ནི་དགེ་བ་བཅུའི་ལས་ཀྱི་ལམ་གྱི་ཕྱིར་བྱུང་ངོ་། །སྨྲས་པ། འཇམ་དཔལ་དགེ་བ་བཅུའི་ལས་ཀྱི་ལམ་དག་ཅིའི་ཕྱིར་བྱུང་། སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་དགེ་བ་བཅུའི་ལས་ཀྱི་ལམ་དག་ནི་སྡོམ་པའི་ཕྱིར་བྱུང་ངོ་། །སྨྲས་པ། འཇམ་དཔལ་སྡོམ་པ་ཅིའི་ཕྱིར་བྱུང་། སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་སྡོམ་པ་ནི་ཚུལ་བཞིན་ཡིད་ལ་བྱ་བའི་ཕྱིར་བྱུང་ངོ་། །སྨྲས་པ། འཇམ་དཔལ་ཚུལ་བཞིན་ཡིད་ལ་བྱ་བ་ཅིའི་ཕྱིར་བྱུང་། སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་ཚུལ་བཞིན་ཡིད་ལ་བྱ་བ་ནི་སོ་སོར་རྟོག་པའི་ཕྱིར་བྱུང་ངོ་། །སྨྲས་པ། འཇམ་དཔལ་སོ་སོར་རྟོག་པ་ཅིའི་ཕྱིར་བྱུང་། སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་སོ་སོར་རྟོག་པ་ནི་དྲན་པའི་ཕྱིར་བྱུང་ངོ་། །སྨྲས་པ། འཇམ་དཔལ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་སེམས་བསྐྱེད་པ་དག་རྒྱུར་བསྡུ་བ་དང་། འབྲས་བུར་བསྡུ་བར་འགྱུར་བ་དུ། འཇམ་དཔལ་གྱིས་སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་སེམས་བསྐྱེད་པ་དག་རྒྱུར་བསྡུ་བ་དང་། འབྲས་བུར་བསྡུ་བར་འགྱུར་བ་ནི་བཞི་པོ་འདི་དག་སྟེ། བཞི་གང་ཞེ་ན། དང་པོ་སེམས་བསྐྱེད་པ་དང་། སྤྱོད་པ་དང་འབྲེལ་པ་དང་། ཕྱིར་མི་ལྡོག་པ་དང་། སྐྱེ་བ་གཅིག་གིས་ཐོགས་པའོ། །དེ་ལ་དང་པོ་སེམས་བསྐྱེད་པ་ནི་སྤྱོད་པ་དང་འབྲེལ་པའི་རྒྱུར་བལྟའོ། །སྤྱོད་པ་དང་འབྲེལ་པ་ནི་ཕྱིར་མི་ལྡོག་པའི་རྒྱུར་བལྟའོ། །ཕྱིར་མི་ལྡོག་པ་ནི་སྐྱེ་བ་གཅིག་གིས་ཐོགས་པའི་རྒྱུར་བལྟའོ། །སྐྱེ་བ་གཅིག་གིས་ཐོགས་པ་ནི་འབྲས་བུ་དང་འབྲེལ་པའི་རྒྱུར་བལྟའོ། །ལྷའི་བུ་གཞན་ཡང་དཔེར་ན་ས་བོན་བཏབ་པ་དེ་ལྟ་བུར་ནི་དང་པོ་སེམས་བསྐྱེད་པ་བལྟའོ། །དཔེར་ན་ས་བོན་ལས་མྱུ་གུ་མངོན་པར་འགྲུབ་པ་དེ་ལྟ་བུར་ནི་སྤྱོད་པ་དང་འབྲེལ་པའི་སེམས་བསྐྱེད་པ་བལྟའོ། །དཔེར་ན་སྡོང་བུ་དང་། འདབ་མ་དང་། མེ་ཏོག་དང་། འབྲས་བུ་འབྱུང་བ་དེ་ལྟ་བུར་ཕྱིར་མི་ལྡོག་པའི་སེམས་བསྐྱེད་པ་བལྟའོ། །དཔེར་ན་འབྲས་བུ་ཡོངས་སུ་སྨིན་པ་དེ་ལྟ་བུར་ནི་སྐྱེ་བ་གཅིག་གིས་ཐོགས་པའི་སེམས་བསྐྱེད་པ་བལྟའོ། །ལྷའི་བུ་གཞན་ཡང་དཔེར་ན་ཤིང་རྟ་མཁན་ཤིང་བསྡུ་བ་ཤེས་པ་དེ་ལྟ་བུར་ནི་དང་པོ་སེམས་བསྐྱེད་པ་བལྟའོ། །དཔེར་ན་ཤིང་རྟའི་ཡན་ལག་ཡོངས་སུ་སྦྱང་བ་ཤེས་པ་དེ་ལྟ་བུར་ནི་སྤྱོད་པ་དང་འབྲེལ་པའི་སེམས་བསྐྱེད་པ་བལྟའོ། །དཔེར་ན་ཤིང་རྟ་ཡན་ལག་སྦྱོར་བ་ཤེས་པ་དེ་ལྟ་བུར་ནི་ཕྱིར་མི་ལྡོག་པའི་སེམས་བསྐྱེད་པ་བལྟའོ། །དཔེར་ན་ཁལ་དགལ་བ་ཤེས་པ་དེ་ལྟ་བུར་ནི་སྐྱེ་བ་གཅིག་གིས་ཐོགས་པའི་སེམས་བསྐྱེད་པ་བལྟའོ། ལྷའི་བུ་གཞན་ཡང་ཟླ་བ་ཚེས་པའི་དཀྱིལ་འཁོར་མངོན་པར་འགྲུབ་པ་དེ་ལྟ་བུར་ནི་དང་པོ་སེམས་བསྐྱེད་པ་བལྟའོ། །དཔེར་ན་ཚེས་ལྔ་དང་། དྲུག་དང་། བདུན་གྱི་ནུབ་མོ་ཟླ་བའི་དཀྱིལ་འཁོར་ཤར་བ་དེ་ལྟ་བུར་ནི་སྤྱོད་པ་དང་འབྲེལ་པའི་སེམས་བསྐྱེད་པ་བལྟའོ། །དཔེར་ན་ཚེས་བཅུའི་ནུབ་མོ་ཟླ་བའི་དཀྱིལ་འཁོར་ཤར་བ་དེ་ལྟ་བུར་ནི་ཕྱིར་མི་ལྡོག་པའི་སེམས་བསྐྱེད་པ་བལྟའོ། །དཔེར་ན་ཚེས་བཅུ་བཞིའི་ནུབ་མོ་ཟླ་བའི་དཀྱིལ་འཁོར་ཤར་བ་དེ་ལྟ་བུར་ནི་སྐྱེ་བ་གཅིག་གིས་ཐོགས་པའི་སེམས་བསྐྱེད་པ་བལྟའོ། །དཔེར་ན་ཚེས་བཅྭ་ལྔའི་ནུབ་མོ་ཟླ་བའི་དཀྱིལ་འཁོར་ཤར་བ་དེ་ལྟ་བུར་ནི་དེ་བཞིན་གཤེགས་པའི་ཡེ་ཤེས་བལྟའོ། །ལྷའི་བུ་གཞན་ཡང་སེམས་བསྐྱེད་པ་དང་པོ་ནི་ཉན་ཐོས་ཀྱི་ས་ལས་ཡང་དག་པར་འདའོ། །སེམས་བསྐྱེད་པ་གཉིས་པ་ནི་རང་སངས་རྒྱས་ཀྱི་ས་ལས་ཡང་དག་པར་འདའོ། །སེམས་བསྐྱེད་པ་གསུམ་པ་ནི་མ་ངེས་པའི་ས་ལས་ཡང་དག་པར་འདའོ། །སེམས་བསྐྱེད་པ་བཞི་པ་ནི་ངེས་པའི་ཕུང་པོ་ལ་གནས་སོ། །ལྷའི་བུ་གཞན་ཡང་དང་པོ་སེམས་བསྐྱེད་པ་ནི་ཡི་གེའི་ཕྱི་མོ་འཛིན་པའི་ཚུལ་དུ་བལྟའོ། །དཔེར་ན་རྣམ་པར་དབྱེ་བ་རབ་ཏུ་རྣམ་པར་འབྱེད་པ་ཤེས་པ་དེ་ལྟ་བུར་ནི་སྤྱོད་པ་དང་འབྲེལ་པའི་སེམས་བསྐྱེད་པ་བལྟའོ། །དཔེར་ན་གྲངས་དང་། རྩིས་དང་། ཤོད་དགོད་པ་དེ་ལྟ་བུར་ནི་ཕྱིར་མི་ལྡོག་པའི་སེམས་བསྐྱེད་པ་བལྟའོ། །དཔེར་ན་བསྟན་བཅོས་རྣམས་ཁོང་དུ་ཆུད་པ་དེ་ལྟ་བུར་ནི་སྐྱེ་བ་གཅིག་གིས་ཐོགས་པའི་སེམས་བསྐྱེད་པ་བལྟའོ། །ལྷའི་བུ་གཞན་ཡང་སེམས་བསྐྱེད་པ་དང་པོ་ནི་རྒྱུར་འབྲེལ་པའོ། །སེམས་བསྐྱེད་པ་གཉིས་པ་ནི་ཡེ་ཤེས་དང་འབྲེལ་པའོ། །སེམས་བསྐྱེད་པ་གསུམ་པ་ནི་སྤོང་བ་དང་འབྲེལ་པའོ། །སེམས་བསྐྱེད་པ་བཞི་པ་ནི་འབྲས་བུ་དང་འབྲེལ་པའོ། །ལྷའི་བུ་གཞན་ཡང་སེམས་བསྐྱེད་པ་དང་པོ་ནི་རྒྱུས་བསྡུས་པའོ། །སེམས་བསྐྱེད་པ་གཉིས་པ་ནི་ཡེ་ཤེས་ཀྱིས་བསྡུས་པའོ། །སེམས་བསྐྱེད་པ་གསུམ་པ་ནི་སྤོང་བས་བསྡུས་པའོ། །སེམས་བསྐྱེད་པ་བཞི་པ་ནི་འབྲས་བུས་བསྡུས་པའོ། །ལྷའི་བུ་གཞན་ཡང་སེམས་བསྐྱེད་པ་དང་པོ་ནི་རྒྱུ་ལས་ཡང་དག་པར་བྱུང་བའོ། །སེམས་བསྐྱེད་པ་གཉིས་པ་ནི་ཡེ་ཤེས་ལས་ཡང་དག་པར་བྱུང་བའོ། །སེམས་བསྐྱེད་པ་གསུམ་པ་ནི་སྤོང་བ་ལས་ཡང་དག་པར་བྱུང་བའོ། །སེམས་བསྐྱེད་པ་བཞི་པ་ནི་འབྲས་བུ་ལས་ཡང་དག་པར་བྱུང་བའོ། །ལྷའི་བུ་གཞན་ཡང་སེམས་བསྐྱེད་པ་དང་པོ་ནི་རྒྱུ་རབ་ཏུ་དབྱེ་བའི་ཡན་ལག་གོ། །སེམས་བསྐྱེད་པ་གཉིས་པ་ནི་ཡེ་ཤེས་རབ་ཏུ་དབྱེ་བའི་ཡན་ལག་གོ། །སེམས་བསྐྱེད་པ་གསུམ་པ་ནི་སྤོང་བ་རབ་ཏུ་དབྱེ་བའི་ཡན་ལག་གོ། །སེམས་བསྐྱེད་པ་བཞི་པ་ནི་འབྲས་བུ་རབ་ཏུ་དབྱེ་བའི་ཡན་ལག་གོ། །ལྷའི་བུ་གཞན་ཡང་སེམས་བསྐྱེད་པ་དང་པོ་ནི་སྨན་བསྡུ་བའི་ཚུལ་དུ་བལྟའོ། །སེམས་བསྐྱེད་པ་གཉིས་པ་ནི་སྨན་རབ་ཏུ་དབྱེ་བ་ལ་མཁས་པའི་ཚུལ་དུ་བལྟའོ། །སེམས་བསྐྱེད་པ་གསུམ་པ་ནི་སྨན་སྦྱོར་བ་ལ་མཁས་པའི་ཚུལ་དུ་བལྟའོ། །སེམས་བསྐྱེད་པ་བཞི་པ་ནི་ནད་པ་གསོ་བ་ལ་མཁས་པའི་ཚུལ་དུ་བལྟའོ། །ལྷའི་བུ་གཞན་ཡང་སེམས་བསྐྱེད་པ་དང་པོ་ནི་ཆོས་ཀྱི་རྒྱལ་པོའི་རིགས་ཀྱི་རྒྱུད་དུ་འབྱུང་བའི་ཚུལ་དུ་བལྟའོ། །སེམས་བསྐྱེད་པ་གཉིས་པ་ནི་ཆོས་ཀྱི་རྒྱལ་པོའི་རིགས་ཀྱི་རྒྱུད་ཀྱི་བསླབ་པ་ཡང་དག་པར་འཛིན་པའི་ཚུལ་དུ་བལྟའོ། །སེམས་བསྐྱེད་པ་གསུམ་པ་ནི་ཆོས་ཀྱི་རྒྱལ་པོའི་རིགས་ཀྱི་རྒྱུད་ཀྱི་བསླབ་པ་ཚར་ཕྱིན་པའི་ཚུལ་དུ་བལྟའོ། །སེམས་བསྐྱེད་པ་བཞི་པ་ནི་ཆོས་ཀྱི་རྒྱལ་པོའི་རིགས་ཀྱི་རྒྱུད་ཀྱི་བསླབ་པ་ལ་དབང་སྒྱུར་བའི་ཚུལ་དུ་བལྟའོ། །དེ་ནས་ལྷའི་བུ་འོད་རྣམ་པར་ངེས་པའི་འོད་ཟེར་གྱི་འོད་ཀྱིས་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། །འཇམ་དཔལ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ལམ་གང་དང་ལྡན་ན་མྱུར་དུ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བའི་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་ལམ་མདོར་བསྡུས་པ་གང་། འཇམ་དཔལ་གྱིས་སྨྲས་པ། ལྷའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་ལམ་ནི་མདོར་བསྡུས་ན་འདི་གཉིས་ཏེ། བྱང་ཆུབ་སེམས་དཔའ་དག་ལམ་འདི་གཉིས་དང་ལྡན་ན་མྱུར་དུ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱའོ། །གཉིས་གང་ཞེ་ན་འདི་ལྟ་སྟེ། ཐབས་དང་ཤེས་རབ་བོ། །དེ་ལ་ཐབས་ནི་བསྡུ་བ་ཤེས་པའོ། །ཤེས་རབ་ནི་ཡོངས་སུ་གཅོད་པ་ཤེས་པའོ། །ཐབས་ནི་སེམས་ཅན་ལ་བལྟ་བའོ། །ཤེས་རབ་ནི་ཆོས་ཐམས་ཅད་ལ་མི་ལྟ་བའོ། །ཐབས་ནི་ཆོས་ལ་སྦྱོར་བ་ཤེས་པའོ། །ཤེས་རབ་ནི་བྲལ་བ་ཤེས་པའོ། །ཐབས་ནི་རྒྱུ་ལ་དམིགས་པ་ཤེས་པའོ། །ཤེས་རབ་ནི་རྒྱུ་ཟད་པར་འགྲོ་བའོ། །ཐབས་ནི་ཆོས་ཐ་དད་པ་ཤེས་པའོ། །ཤེས་རབ་ནི་ཆོས་ཀྱི་དབྱིངས་དབྱེར་མེད་པ་ཤེས་པའོ། །ཐབས་ནི་སངས་རྒྱས་ཀྱི་ཞིང་གི་རྒྱན་ཤེས་པའོ། །ཤེས་རབ་ནི་སངས་རྒྱས་ཀྱི་ཞིང་མཉམ་པ་ཉིད་དབྱེར་མེད་པར་ཤེས་པའོ། །ཐབས་ནི་སེམས་ཅན་གྱི་དབང་པོ་དང་སྤྱོད་པ་ལ་འཇུག་པ་ཤེས་པའོ། །ཤེས་རབ་ནི་སེམས་ཅན་དམིགས་སུ་མེད་པར་ཤེས་པའོ། །ཐབས་ནི་བྱང་ཆུབ་ཀྱི་རྒྱན་ཐོབ་པར་བྱེད་པ་ཤེས་པའོ། །ཤེས་རབ་ནི་སངས་རྒྱས་ཀྱི་ཆོས་ཐམས་ཅད་མངོན་པར་རྫོགས་པར་བྱང་ཆུབ་པ་ཤེས་པའོ། །ལྷའི་བུ་གཞན་ཡང་མདོར་བསྡུས་པའི་ལམ་གཉིས་ཏེ། གཉིས་གང་ཞེ་ན། ཚོགས་ཀྱི་ལམ་དང་། ཡོངས་སུ་གཅོད་པའི་ལམ་མོ། །དེ་ལ་ཚོགས་ཀྱི་ལམ་ནི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ལྔ་པོ་དག་གོ། །ཡོངས་སུ་གཅོད་པའི་ལམ་ནི་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའོ། །གཞན་ཡང་མདོར་བསྡུས་པའི་ལམ་གཉིས་ཏེ། གཉིས་གང་ཞེ་ན། ཆགས་པ་དང་བཅས་པའི་ལམ་དང་། ཆགས་པ་མེད་པའི་ལམ་མོ། །དེ་ལ་ཆགས་པ་དང་བཅས་པའི་ལམ་ནི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ལྔ་པོ་དག་གོ། །ཆགས་པ་མེད་པའི་ལམ་ནི་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའོ། །གཞན་ཡང་མདོར་བསྡུས་པའི་ལམ་གཉིས་ཏེ། གཉིས་གང་ཞེ་ན། ཟག་པ་དང་བཅས་པའི་ལམ་དང་ཟག་པ་མེད་པའི་ལམ་མོ། །དེ་ལ་ཟག་པ་དང་བཅས་པའི་ལམ་ནི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ལྔ་པོ་དག་གོ། །ཟག་པ་མེད་པའི་ལམ་ནི་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའོ། །གཞན་ཡང་མདོར་བསྡུས་པའི་ལམ་གཉིས་ཏེ། གཉིས་གང་ཞེ་ན། ཚད་ཡོད་པར་བྱས་པའི་ལམ་དང་། ཚད་མེད་པར་བྱས་པའི་ལམ་མོ། །དེ་ལ་ཚད་ཡོད་པར་བྱས་པའི་ལམ་ནི་མཚན་མའི་གནས་སོ། །ཚད་མེད་པར་བྱས་པའི་ལམ་ནི་མཚན་མ་མེད་པའི་གནས་སོ། །གཞན་ཡང་མདོར་བསྡུས་པའི་ལམ་གཉིས་ཏེ། གཉིས་གང་ཞེ་ན། ཤེས་པའི་ལམ་དང་སྤོང་བའི་ལམ་མོ། །དེ་ལ་ཤེས་པའི་ལམ་ནི་ས་དང་པོ་ནས་བཟུང་སྟེ་ས་བདུན་པའི་བར་དུའོ། །སྤོང་བའི་ལམ་ནི་ས་བརྒྱད་པ་ནས་བཟུང་སྟེ་ས་བཅུ་པའི་བར་དུའོ། །དེ་ནས་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་ཡེ་ཤེས་སྒྲུབ་པ་དཔའ་བོས་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། །འཇམ་དཔལ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་ཇི་ལྟར་ན་དོན་གྱིས་རབ་ཏུ་ཕྱེ་བ་ཡིན། ཇི་ལྟར་ན་ཡེ་ཤེས་ཀྱིས་རབ་ཏུ་ཕྱེ་བ་ཡིན། འཇམ་དཔལ་གྱིས་སྨྲས་པ། རིགས་ཀྱི་བུ་དོན་ནི་ལྡན་པ་མེད་པའོ། །ཡེ་ཤེས་ནི་ལྡན་པའོ། །སྨྲས་པ། འཇམ་དཔལ་ཅིའི་ཕྱིར་དོན་ནི་ལྡན་པ་མེད་ལ་ཡེ་ཤེས་ནི་ལྡན་པ་ཡིན། འཇམ་དཔལ་གྱིས་སྨྲས་པ། རིགས་ཀྱི་བུ་དོན་ནི་འདུས་མ་བྱས་པ་སྟེ། ཆོས་གང་དང་ཡང་ལྡན་པའམ་མི་ལྡན་པ་མ་ཡིན་པར་ཁོང་དུ་ཆུད་པར་བྱའོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། དོན་ནི་འགྱུར་བ་མེད་པའི་མཚན་ཉིད་ཡིན་ཏེ། དེ་ནི་ཆོས་གང་དང་ཡང་ལྡན་པའམ་མི་ལྡན་པ་མ་ཡིན་པར་ཁོང་དུ་ཆུད་པར་བྱའོ། །དོན་ནི་གཟོད་མ་ནས་ཡོངས་སུ་གྲུབ་པ་མེད་པའི་ཕྱིར་ཆོས་གང་དང་ཡང་ལྡན་པའམ་མི་ལྡན་པ་མ་ཡིན་པར་ཁོང་དུ་ཆུད་པར་བྱའོ། །དོན་ནི་དོར་བ་མེད་པ་དང་གཞག་པ་མེད་པའི་ཕྱིར་ཆོས་གང་དང་ཡང་ལྡན་པའམ་མི་ལྡན་པ་མ་ཡིན་པར་ཁོང་དུ་ཆུད་པར་བྱའོ། །ཡེ་ཤེས་ཞེས་བྱ་བ་ནི་ལམ་སྟེ། ལམ་ནི་སེམས་དང་ཡང་དག་པར་ལྡན་པ་ཡིན་གྱི་མི་ལྡན་པ་མ་ཡིན་ཏེ། རིགས་ཀྱི་བུ་དེའི་ཕྱིར་ཡེ་ཤེས་ནི་ལྡན་པ་ཞེས་བྱ་སྟེ། མི་ལྡན་པ་མ་ཡིན་ནོ། །ཡེ་ཤེས་ནི་སྤོང་བ་དང་ཡང་དག་པར་ལྡན་པ་ཡིན་གྱི་མི་ལྡན་པ་མ་ཡིན་ཏེ། རིགས་ཀྱི་བུ་དེའི་ཕྱིར་ཡེ་ཤེས་ནི་ལྡན་པ་ཞེས་བྱ་སྟེ། མི་ལྡན་པ་མ་ཡིན་ནོ། །ཡེ་ཤེས་ཞེས་བྱ་བ་ནི་ཕུང་པོ་ལ་མཁས་པ་དང་། ཁམས་ལ་མཁས་པ་དང་། སྐྱེ་མཆེད་ལ་མཁས་པ་དང་། རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བ་ལ་མཁས་པ་དང་། གནས་དང་གནས་མ་ཡིན་པ་རྣམ་པར་དཔྱད་པ་ལ་མཁས་པ་སྟེ། རིགས་ཀྱི་བུ་དེའི་ཕྱིར་ཡེ་ཤེས་ནི་ལྡན་པ་ཞེས་བྱ་སྟེ། །མི་ལྡན་པ་མ་ཡིན་ནོ། །རིགས་ཀྱི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་ཤེས་པ་རྣམ་པ་བཅུ་སྟེ། རྣམ་པ་བཅུ་གང་ཞེ་ན། འདི་ལྟ་སྟེ། རྒྱུ་ཤེས་པ་དང་། འབྲས་བུ་ཤེས་པ་དང་། དོན་ཤེས་པ་དང་། ཐབས་ཤེས་པ་དང་། ཤེས་རབ་ཤེས་པ་དང་། བསྡུ་བ་ཤེས་པ་དང་། ཕ་རོལ་ཏུ་ཕྱིན་པ་ཤེས་པ་དང་། སྙིང་རྗེ་ཆེན་པོ་ཤེས་པ་དང་། སེམས་ཅན་ཡོངས་སུ་སྨིན་པར་བྱ་བ་ཤེས་པ་དང་། ཆོས་ཐམས་ཅད་ལ་མངོན་པར་ཞེན་པ་མེད་པར་ཤེས་པ་སྟེ། རིགས་ཀྱི་བུ་རྣམ་པ་དེ་བཅུ་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་ཤེས་པའོ། །རིགས་ཀྱི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་རྩོམ་པ་རྣམ་པ་བཅུ་སྟེ། རྣམ་པ་བཅུ་གང་ཞེ་ན། འདི་ལྟ་སྟེ། སེམས་ཅན་ཐམས་ཅད་ཀྱི་ལུས་ཀྱི་ལས་ཡོངས་སུ་སྦྱང་བའི་ཕྱིར་ལུས་ཀྱི་རྩོམ་པ་དང་། སེམས་ཅན་ཐམས་ཅད་ཀྱི་ངག་གི་ལས་ཡོངས་སུ་སྦྱང་བའི་ཕྱིར་ངག་གི་རྩོམ་པ་དང་། སེམས་ཅན་ཐམས་ཅད་ཀྱི་ཡིད་ཀྱི་ལས་ཡོངས་སུ་སྦྱང་བའི་ཕྱིར་སེམས་ཀྱི་རྩོམ་པ་དང་། སེམས་ཅན་ཐམས་ཅད་ལ་མངོན་པར་ཞེན་པ་མེད་པར་བྱ་བའི་ཕྱིར་ནང་གི་རྩོམ་པ་དང་། སེམས་ཅན་ཐམས་ཅད་ལ་མཚུངས་པར་སྦྱར་བར་བྱ་བའི་ཕྱིར་ཕྱིའི་རྩོམ་པ་དང་། སངས་རྒྱས་ཀྱི་ཡེ་ཤེས་ཡོངས་སུ་རྫོགས་པར་བྱ་བའི་ཕྱིར་ཡེ་ཤེས་ཀྱི་རྩོམ་པ་དང་། སངས་རྒྱས་ཀྱི་ཞིང་ཐམས་ཅད་ཀྱི་ཡོན་ཏན་བཀོད་པ་ཀུན་ཏུ་བསྟན་པའི་ཕྱིར་ཞིང་གི་རྩོམ་པ་དང་། ཉོན་མོངས་པའི་ནད་ཀྱི་སྨན་ཡོངས་སུ་ཤེས་པར་བྱ་བའི་ཕྱིར་སེམས་ཅན་ཡོངས་སུ་སྨིན་པར་བྱ་བའི་རྩོམ་པ་དང་། ངེས་པའི་ཕུང་པོ་ཡོངས་སུ་རྫོགས་པར་བྱ་བའི་ཕྱིར་དེ་ཁོ་ནའི་རྩོམ་པ་དང་། ཁམས་གསུམ་པ་ཐམས་ཅད་ལ་སེམས་མངོན་པར་ཞེན་པ་མེད་པར་བྱ་བའི་ཕྱིར་འདུས་མ་བྱས་ཀྱི་ཡེ་ཤེས་ཡོངས་སུ་རྫོགས་པར་བྱ་བའི་རྩོམ་པ་སྟེ། །རིགས་ཀྱི་བུ་རྣམ་པ་འདི་བཅུ་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་རྩོམ་པའོ། །རིགས་ཀྱི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྦྱོར་བ་བཅུ་སྟེ། བཅུ་གང་ཞེ་ན། འདི་ལྟ་སྟེ། ཕ་རོལ་ཏུ་ཕྱིན་པའི་སྦྱོར་བ་དང་། བསྡུ་བའི་དངོས་པོའི་སྦྱོར་བ་དང་། ཤེས་རབ་ཀྱི་སྦྱོར་བ་དང་། ཐབས་ཀྱི་སྦྱོར་བ་དང་། སྙིང་རྗེ་ཆེན་པོའི་སྦྱོར་བ་དང་། བསོད་ནམས་ཀྱི་ཚོགས་ཡོངས་སུ་ཚོལ་བའི་སྦྱོར་བ་དང་། ཡེ་ཤེས་ཀྱི་ཚོགས་ཡོངས་སུ་ཚོལ་བའི་སྦྱོར་བ་དང་། བསམ་པ་རྣམ་པར་སྦྱོང་བའི་སྦྱོར་བ་དང་། བདེན་པ་རྣམ་པར་དཔྱད་པའི་སྦྱོར་བ་དང་། སྡུག་པ་དང་མི་སྡུག་པ་ཐམས་ཅད་ལ་མངོན་པར་ཞེན་པ་མེད་པའི་སྦྱོར་བ་སྟེ། །རིགས་ཀྱི་བུ་བཅུ་པོ་དེ་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྦྱོར་བའོ། །རིགས་ཀྱི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་མི་ཟད་པ་ལ་རྟོག་པ་བཅུ་སྟེ། བཅུ་གང་ཞེ་ན། འདི་ལྟ་སྟེ། ལུས་མི་ཟད་པ་ལ་རྟོག་པ་དང་། དངོས་པོ་མི་ཟད་པ་ལ་རྟོག་པ་དང་། ཆོས་མི་ཟད་པ་ལ་རྟོག་པ་དང་། སྲེད་པ་མི་ཟད་པ་ལ་རྟོག་པ་དང་། ལྟ་བ་མི་ཟད་པ་ལ་རྟོག་པ་དང་། ཚོགས་མི་ཟད་པ་ལ་རྟོག་པ་དང་། ཉེ་བར་ལེན་པ་མི་ཟད་པ་ལ་རྟོག་པ་དང་། མངོན་པར་ཞེན་པ་མེད་པ་མི་ཟད་པ་ལ་རྟོག་པ་དང་། འདུ་བ་མི་ཟད་པ་ལ་རྟོག་པ་དང་། བྱང་ཆུབ་ཀྱི་སྙིང་པོ་ཤེས་པ་མི་ཟད་པ་ལ་རྟོག་པ་སྟེ། རིགས་ཀྱི་བུ་བཅུ་པོ་དེ་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་མི་ཟད་པ་ལ་རྟོག་པའོ། །རིགས་ཀྱི་བུ་གཞན་ཡང་བཅུ་པོ་འདི་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་འདུལ་བའི་དངོས་པོ་བཅུ་སྟེ། བཅུ་གང་ཞེ་ན། འདི་ལྟ་སྟེ། གཏོང་བའི་ཆར་རབ་ཏུ་འབེབས་པས་སེར་སྣའི་སེམས་འདུལ་བ་ཉིད་དང་། ལས་རྣམ་གསུམ་ཡོངས་སུ་སྦྱོང་བས་འཆལ་པའི་ཚུལ་ཁྲིམས་ཀྱི་སེམས་འདུལ་བ་ཉིད་དང་། བྱམས་པ་སྒོམ་པས་ཁོང་ཁྲོ་བའི་སེམས་འདུལ་བ་ཉིད་དང་། སངས་རྒྱས་ཀྱི་ཆོས་ཡོངས་སུ་བཙལ་བ་ལ་མི་སྐྱོ་བས་ལེ་ལོའི་སེམས་འདུལ་བ་ཉིད་དང་། བསམ་གཏན་དང་རྣམ་པར་ཐར་པར་རྣམ་པར་འཕྲུལ་པས་མི་དགེ་བའི་སེམས་འདུལ་བ་ཉིད་དང་། ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པས་གཏན་ལ་འབེབས་པ་ལ་མཁས་པའི་ཚོགས་སོགས་པས་མི་ཤེས་པའི་སེམས་འདུལ་བ་ཉིད་དང་། ལམ་གྱི་ཚོགས་བསྐྱེད་པས་ཉོན་མོངས་པ་འདུལ་བ་ཉིད་དང་། ཕྱིན་ཅི་མ་ལོག་པའི་ལམ་བསྐྱེད་པས་བདེན་པའི་ཚོགས་སོགས་པས་འདུལ་བ་ཉིད་དང་། དུས་དང་དུས་མ་ཡིན་པར་དབང་སྒྱུར་བས་སེམས་འདུལ་བ་ཉིད་དང་། བདག་མེད་པའི་ཆོས་ལ་རྟོགས་པས་བདག་འདུལ་བ་ཉིད་དེ། རིགས་ཀྱི་བུ་བཅུ་པོ་དེ་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་འདུལ་བའི་དངོས་པོའོ། །རིགས་ཀྱི་བུ་གཞན་ཡང་བཅུ་པོ་འདི་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་དུལ་བའི་ས་སྟེ། བཅུ་གང་ཞེ་ན། འདི་ལྟ་སྟེ། ལུས་ཀྱི་ལས་མི་དགེ་བ་རྣམ་པ་གསུམ་ཡོངས་སུ་སྤོང་བས་ལུས་དུལ་བ་ཉིད་དང་། ངག་གི་ལས་རྣམ་པ་བཞི་ཡོངས་སུ་སྤོང་བས་ངག་དུལ་བ་ཉིད་དང་། ཡིད་ཀྱི་ཉེས་པར་སྤྱོད་པ་རྣམ་པ་གསུམ་ཡོངས་སུ་སྤོང་བས་སེམས་དུལ་བ་ཉིད་དང་། རང་གི་ལུས་ལ་མངོན་པར་ཞེན་པ་མེད་པའི་ཚུལ་གྱིས་ནང་དུལ་བ་ཉིད་དང་། ཆོས་ཐམས་ཅད་ལ་མངོན་པར་ཞེན་པ་མེད་པའི་ཚུལ་གྱིས་ཕྱི་རོལ་གྱི་ཡུལ་དུལ་བ་ཉིད་དང་། ལམ་ལ་མངོན་པར་ཞེན་པ་མེད་པའི་ཚུལ་གྱིས་ཡེ་ཤེས་ཀྱི་ཚོགས་ལ་མངོན་པར་ཞེན་པ་མེད་པ་དུལ་བ་ཉིད་དང་། འཕགས་པའི་སའི་ངོ་བོ་ཉིད་ལ་རྟོག་པའི་ཚུལ་གྱིས་བདག་ལ་སྟོད་པ་དུལ་བ་ཉིད་དང་། ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ལ་གནས་པའི་ཚུལ་གྱིས་ཕྱི་མའི་མཐའ་རྣམ་པར་དག་པ་ཡོངས་སུ་སྦྱོང་བ་དུལ་བ་ཉིད་དང་། སེམས་ཅན་ཐམས་ཅད་ལ་མི་སླུ་བའི་ཚུལ་གྱིས་ཐ་སྙད་ཤིན་ཏུ་ཡོངས་སུ་དུལ་བ་ཉིད་དང་། སེམས་ཅན་ཐམས་ཅད་ཡོངས་སུ་སྨིན་པར་བྱེད་པའི་སྙིང་རྗེ་ཆེན་པོའི་ཚུལ་གྱིས་ལུས་དང་སེམས་ལ་མི་ལྟ་བ་དུལ་བ་ཉིད་དེ། རིགས་ཀྱི་བུ་བཅུ་པོ་དེ་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་དུལ་བའི་སའོ། །རིགས་ཀྱི་བུ་གཞན་ཡང་བྱང་ཆུབ་ནི་སྒྲུབ་པ་སྙིང་པོར་བྱེད་པའི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་ཡིན་ཏེ། ལོག་པར་སྒྲུབ་པ་སྙིང་པོར་བྱེད་པ་རྣམས་ཀྱི་ནི་མ་ཡིན་ནོ། །དེ་ལ་སྒྲུབ་པ་སྙིང་པོར་བྱེད་པ་ནི་ཇི་སྐད་སྨྲས་པ་དེ་བཞིན་དུ་བྱེད་པ་ཡིན་ནོ། །ལོག་པར་སྒྲུབ་པ་སྙིང་པོར་བྱེད་པ་ནི་ཚིག་གི་དངོས་པོ་ཁོ་ནའོ། །དེ་ནི་སྒྲུབ་པ་སྙིང་པོར་བྱེད་པ་མང་བ་ཡིན་གྱི། ལོག་པར་སྒྲུབ་པ་སྙིང་པོར་བྱེད་པ་མང་བ་ནི་མ་ཡིན་ནོ། །རིགས་ཀྱི་བུ་གཞན་ཡང་གཉིས་པོ་འདི་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྒྲུབ་པ་སྟེ། གཉིས་གང་ཞེ་ན། འདི་ལྟ་སྟེ། ལམ་སྒྲུབ་པ་དང་། སྤོང་བ་སྒྲུབ་པ་སྟེ། །རིགས་ཀྱི་བུ་གཉིས་པོ་དེ་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྒྲུབ་པའོ། །རིགས་ཀྱི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྒྲུབ་པ་གཉིས་ཏེ། གཉིས་གང་ཞེ་ན། འདི་ལྟ་སྟེ། བདག་འདུལ་བའི་སྒྲུབ་པ་དང་། སེམས་ཅན་ཡོངས་སུ་སྨིན་པར་བྱ་བའི་སྒྲུབ་པ་སྟེ། །རིགས་ཀྱི་བུ་གཉིས་པོ་དེ་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྒྲུབ་པའོ། །རིགས་ཀྱི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྒྲུབ་པ་གཉིས་ཏེ། གཉིས་གང་ཞེ་ན། འདི་ལྟ་སྟེ། །རྩོལ་བ་དང་བཅས་པའི་ཤེས་པས་སྒྲུབ་པ་དང་། ལྷུན་གྱིས་གྲུབ་པའི་ཡེ་ཤེས་ཀྱིས་སྒྲུབ་པ་སྟེ། རིགས་ཀྱི་བུ་གཉིས་པོ་དེ་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྒྲུབ་པའོ། །རིགས་ཀྱི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྒྲུབ་པ་གཉིས་ཏེ། གཉིས་གང་ཞེ་ན། འདི་ལྟ་སྟེ། ས་རྣམ་པར་གཞག་པ་ལ་མཁས་པར་སྒྲུབ་པ་དང་། ས་དབྱེར་མེད་པ་ལ་མཁས་པར་སྒྲུབ་པ་སྟེ། རིགས་ཀྱི་བུ་གཉིས་པོ་དེ་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྒྲུབ་པའོ། །རིགས་ཀྱི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྒྲུབ་པ་གཉིས་ཏེ། གཉིས་གང་ཞེ་ན། འདི་ལྟ་སྟེ། སའི་ཉེས་པ་རྣམ་པར་སྤོང་བའི་སྒྲུབ་པ་དང་། ས་ནས་སར་འཕར་བ་ལ་མཁས་པ་ཡོངས་སུ་རྫོགས་པར་བྱ་བའི་སྒྲུབ་པ་སྟེ། རིགས་ཀྱི་བུ་གཉིས་པོ་དེ་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྒྲུབ་པའོ། །རིགས་ཀྱི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྒྲུབ་པ་གཉིས་ཏེ། གཉིས་གང་ཞེ་ན། འདི་ལྟ་སྟེ། ཉན་ཐོས་དང་། རང་སངས་རྒྱས་ཀྱི་ས་བསྟན་པའི་སྒྲུབ་པ་དང་། སངས་རྒྱས་ཀྱི་བྱང་ཆུབ་ལས་ཕྱིར་མི་ལྡོག་པ་ལ་མཁས་པའི་སྒྲུབ་པ་སྟེ། རིགས་ཀྱི་བུ་གཉིས་པོ་དེ་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྒྲུབ་པའོ། །རིགས་ཀྱི་བུ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་སྒྲུབ་པ་ནི་འདི་ལྟ་བུ་འདི་དག་སྟེ། དེ་ལ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱིས་འབད་པར་བྱའོ། །རིགས་ཀྱི་བུ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་དེ་ལྟར་སྒྲུབ་པ་རྣམས་ལ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་རྙེད་དཀའ་བ་མ་ཡིན་ནོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་བཀའ་སྩལ་པ། འཇམ་དཔལ་ཁྱོད་ཀྱིས་ཚིག་འདི་ལེགས་པར་བཤད་པ་ལེགས་སོ་ལེགས་སོ། །བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་སྩལ་ནས། འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་དང་། ལྷ་དང་། མི་དང་། ལྷ་མ་ཡིན་དང་། དྲི་ཟར་བཅས་པའི་འཇིག་རྟེན་ཡི་རངས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལ་མངོན་པར་བསྟོད་དོ། །འཕགས་པ་ག་ཡཱ་མགོའི་རི་ཞེས་བྱ་བ་ཐེག་པ་ཆེན་པོའི་མདོ་རྫོགས་སོ།། །།རྒྱ་གར་གྱི་མཁན་པོ་སུ་རེན་དྲ་བོ་དྷི་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙྪ་བ་བན་དེ་ཡེ་ཤེས་སྡེས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པ།། །
