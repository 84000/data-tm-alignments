༄༅༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྱ་མ་ཧཱ་མེ་གྷ། བོད་སྐད་དུ། འཕགས་པ་སྤྲིན་ཆེན་པོ། སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །འདི་སྐད་བདག་གིས་ཐོས་པ་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་ཀླུའི་རྒྱལ་པོ་དགའ་བོ་དང་། ཉེ་དགའ་བོའི་གནས་ན་ནོར་བུ་རིན་པོ་ཆེ་དཔལ་གྱི་སྙིང་པོའི་སྤྲིན་གྱི་དཀྱིལ་འཁོར་ཆེན་པོའི་ཁང་པ་བརྩེགས་པ་ན་དགེ་སློང་གི་དགེ་འདུན་ཆེན་པོ་དང་། བྱང་ཆུབ་སེམས་དཔའི་དགེ་འདུན་ཆེན་པོ་དང་། ཀླུའི་རྒྱལ་པོའི་ཚོགས་ཆེན་པོ་འདི་ལྟ་སྟེ། ཀླུའི་རྒྱལ་པོ་དགའ་བོ་དང་། ཉེ་དགའ་དང་། རྒྱ་མཚོ་དང་། མ་དྲོས་པ་དང་། གཟི་ཅན་དང་། ཆུ་ལྷ་དང་། འཇོག་པོ་དང་། ཡུལ་འཁོར་སྲུང་དང་། ནོར་རྒྱས་ཀྱི་བུ་དང་། ཀླུའི་རྒྱལ་པོ་བཏང་བཟུང་དང་། ས་སྲུང་གི་བུ་དང་། མདོངས་བུ་དང་། དཔལ་གྱི་གཟི་བརྗིད་དང་། དཔལ་བཟངས་དང་། གློག་གི་ཕྲེང་བ་དང་། གཙུག་ན་ནོར་བུ་རིན་པོ་ཆེ་དང་། གཙུག་ན་ནོར་བུ་ཐོགས་དང་། སྣང་ལྡན་གཙུག་ཕུད་ཅན་དང་། གཙུག་ཕུད་འཇའ་ཅན་དང་། ཀླུའི་རྒྱལ་པོ་དབང་པོའི་རྒྱལ་མཚན་དང་། དབང་པོའི་མཆོད་སྡོང་དང་། འཛམ་བུའི་རྒྱལ་མཚན་དང་། བཀྲ་ཤིས་ལྡན་དང་། དཀྱིལ་འཁོར་ཆེན་པོ་ཅན་དང་། ལག་གིས་འགྲོ་ཆེན་པོ་དང་། སྐར་མདོག་དང་། ཟླ་བའི་གཟི་བརྗིད་དང་། དཔལ་གྱི་བློ་གྲོས་དང་། ཟླ་མཐོང་དང་། ཀླུའི་རྒྱལ་པོ་ལེགས་མཐོང་དང་། ཤིན་ཏུ་བརྟན་དང་། མགུལ་ན་ནོར་བུ་དང་། སྤྲིན་བྱུང་དང་། ཆར་རྒྱུན་དང་། རབ་འཇོམས་དང་། རྣམ་པར་འཇོམས་དང་། མཐུ་བསྒྱིངས་དང་། གདེངས་ཀ་ཆེན་པོ་དང་། དབྱངས་ཟབ་དང་། ཀླུའི་རྒྱལ་པོ་ཆེར་སྒྲོགས་དང་། རྣམ་པར་སྒྲ་འབྱིན་དང་། ཨུད་པལ་དང་། རྩལ་ཆེན་དང་། གཤོལ་མདའ་འཛིན་དང་། སྣ་ཚོགས་སྡེ་དང་། ཞགས་པ་ཆེན་པོ་དང་། ཀླུའི་རྒྱལ་པོ་ཨེ་ལའི་འདབ་མ་དང་། ཀུན་ཏུ་སྣང་དང་། རྣ་རྩུབ་དང་། དུང་དང་། ད་ར་ད་ར་དང་། ཀླུའི་རྒྱལ་པོ་ཨུ་པ་ད་ར་ད་ར་དང་། བདེ་བྱེད་དང་། ལག་འགྲོ་དང་། ལག་འགྲོ་ཆེན་པོ་དང་། དྲི་མ་ཆེན་པོ་དང་། འཕུར་ལྡིང་དང་། སོག་མ་མེད་དང་། འཕྱང་བ་དང་། སྲིན་བུ་གནས་དང་། ནག་པོ་དང་། ཀླུའི་རྒྱལ་པོ་དབང་པོ་སྡེ་དང་། འདམ་བུ་དང་། ཉེ་བའི་འདམ་བུ་དང་། དུང་ཐོགས་དང་། འགྲོ་ལྡིང་དང་། རབ་མཛེས་དང་། གླང་ཆེན་སྣ་དང་། རྣོན་པོ་དང་། སེར་སྐྱ་དང་། གློག་འབར་དང་། གློག་འོད་ཆེན་པོ་དང་། སྟོབས་ལྷ་དང་། ཆུ་སྟོབས་དང་། ཀླུའི་རྒྱལ་པོ་དམ་པ་འགེབས་དང་། བདུད་རྩི་ཅན་དང་། མུ་སྟེགས་ཅན་དང་། བཻ་དཱུརྱའི་འོད་ཅན་དང་། གསེར་མདོག་སྒྲ་དང་། གསེར་འོད་དང་། ཟླ་བའི་རྒྱལ་མཚན་འོད་དང་། ཉི་འོད་དང་། འཆར་འོད་དང་། བལ་གླང་མགོ་དང་། ཀླུའི་རྒྱལ་པོ་དཀར་པོ་དང་། ནག་པོ་དང་། གཤིན་རྗེ་དང་། དགེ་ཚུལ་དང་། སྦལ་པ་དང་། ངང་པའི་གཟི་བརྗིད་དང་། མུ་ཁྱུད་འཛིན་དང་། ས་འཛིན་དང་། མགོ་སྟོང་ཅན་དང་། གཙུག་ན་ནོར་བུ་དང་། ཀླུའི་རྒྱལ་པོ་དོན་ཡོད་མཐོང་དང་། སྤྲིན་ཆེན་དང་། སྡེ་བཟངས་དང་། ས་སྐྱོང་དང་། མིས་བྱིན་དང་། རྣམ་པར་དུལ་དང་། རྒ་བ་འབྱུང་དང་། ཆུ་སྲིན་ཀུན་བྷི་ར་དང་། ཆུ་སྲིན་ཀུན་བྷ་རའི་གདོང་དང་། དུག་འཛིན་དང་། ཀླུའི་རྒྱལ་པོ་མྱུར་གདུག་དང་། པད་མ་ཀ་དང་། འཇུག་མ་དང་། རབ་འཕྱང་དང་། འཇིགས་བྱེད་དང་། གཟི་བརྗིད་མཛེས་དང་། མགོ་ལྔ་པ་དང་། བལ་ལི་ར་དང་། ཤིང་རྟས་རྒྱལ་དང་། ཀླུའི་རྒྱལ་པོ་བླ་མ་དང་། ཀླུའི་རྒྱལ་པོ་མཇུག་རིངས་དང་། མགོ་བོ་ཆེ་དང་། བིམ་པི་ཀ་དང་། བི་ཏ་ར་དང་། མགྱོགས་འགྲོ་དང་། ཀླུའི་རྒྱལ་པོ་མགོ་གསུམ་དང་། ཀླུ་སྡེ་དང་། གཟི་ཆེན་དང་། འདམ་བུ་བྱིན་དང་། ཀླུའི་རྒྱལ་པོ་སྐྲག་བྱེད་དང་། འོད་འབར་དང་། མགོ་བདུན་པ་དང་། སྡོང་ཆེན་དང་། མཐོང་དགའ་དང་། རབ་ཏུ་ཆེར་གཏུམས་དང་། དྲི་མེད་གཟི་བརྗིད་དང་། མིག་བཟངས་དང་། དྲག་པོ་ཆེན་པོ་དང་། འོད་འཕྲོ་ཕུང་པོ་དང་། ཀླུའི་རྒྱལ་པོ་གཏུམ་པོ་དང་། ཁྲོ་བོ་དྲག་ཤུལ་ཅན་དང་། རིན་ཆེན་ཆུ་སྦྱིན་དང་། ཆུ་སྦྱིན་སྤྲིན་ཆེན་དང་། དབང་པོའི་འོད་དང་། རྐང་པས་འཐུང་དང་། སྤྲིན་ཟླ་དང་། རྒྱ་མཚོའི་སྤྲིན་དང་། མེ་ཏོག་དྲི་ཆེན་དང་། ཀུ་མུ་དའི་འབྱུང་གནས་དང་། ཀླུའི་རྒྱལ་པོ་རིན་ཆེན་མིག་དང་། ཏོག་ཆེན་པོའི་རྒྱལ་མཚན་དང་། སྤྲིན་ཆེན་སྙིང་པོ་དང་། ཁ་བ་ཅན་དང་། དཔལ་གྱི་སྙིང་པོ་དང་། སྤྲིན་མཐུ་ཅན་དང་། མཚན་མོ་འཛིན་དང་། ཆུ་སྦྱིན་དང་། ཆུ་སྦྱིན་ཆེན་པོ་དང་། ཀླུའི་རྒྱལ་པོ་གཟི་བརྗིད་མེ་འདྲ་དང་། སྤྲིན་གྱི་དབང་ཕྱུག་རྒྱལ་པོ་དང་། ཁྲོ་མེད་དང་། སའི་བུམ་པ་དང་། ཀླུ་དཔའ་དང་། ཆུ་འཛིན་དང་། སྤྲིན་གདུགས་དང་། ཨིང་གི་རའི་གདོང་དང་། གཟི་བརྗིད་ཅན་དང་། སྤྲིན་ལས་མངོན་བྱུང་དང་། ཀླུའི་རྒྱལ་པོ་གོམ་སྟབས་མཐའ་ཡས་དང་། ཆ་བྱད་མཛེས་དང་། ལུས་ཆེན་དང་། སྤྱང་ལྟོ་དང་། ཞི་བ་དང་། དཔའ་བ་དང་། གསེར་ངད་ཅན་དང་། མཆོག་ཏུ་བརྩོན་དང་། དུག་དྲག་དང་། དབྱངས་སྙན་དང་། ཀླུའི་རྒྱལ་པོ་བདུད་རྩིའི་སྙིང་པོ་དང་། རབ་ཏུ་ཆར་འབེབས་ཆེན་པོ་དང་། རྣམ་པར་གཡོ་དང་། འཁྲུག་སྒྲ་དང་། རྔ་སྒྲ་དང་། བདུད་རྩི་རྒྱུན་དང་། དགའ་བར་བྱེད་པ་དང་། འབྲུག་སྒྲོགས་དང་། དཔའ་སྡེ་དང་། སྲེད་མེད་ཀྱི་བུ་དང་། རྒྱ་མཚོའི་གློང་དང་། ཀླུའི་རྒྱལ་པོ་ཡོན་པོ་དང་། ཀླུའི་རྒྱལ་པོ་ཆེན་པོ་དེ་དག་ལ་སོགས་པ་ཀླུ་བྱེ་བ་ཁྲག་ཁྲིག་བརྒྱ་སྟོང་ཕྲག་བརྒྱད་ཅུ་རྩ་བཞི་དག་འདུས་ཏེ་འཁོད་པ་དང་ཐབས་ཅིག་ཏུ་བཞུགས་སོ། །དེའི་ཚེ་ཀླུའི་རྒྱལ་པོ་དེ་དག་འཁོར་དང་བཅས་ཏེ་སྟན་ལས་ལངས་ནས་བླ་གོས་ཕྲག་པ་གཅིག་ཏུ་གཟར་ཏེ། པུས་མོ་གཡས་པའི་ལྷ་ང་ས་ལ་བཙུགས་ནས། བཅོམ་ལྡན་འདས་ག་ལ་བ་དེ་ལོགས་སུ་ཐལ་མོ་སྦྱར་བ་བཏུད་ནས་བླ་མ་ལ་གུས་པ་དང་། ཕུ་དུད་ཆེན་པོ་བྱེད་པས་མེ་ཏོག་དང་། བདུག་པ་དང་། སྤོས་དང་། ཕྲེང་བ་དང་། བྱུག་པ་དང་། ཕྱེ་མ་དང་། ན་བཟའ་དང་། གདུགས་དང་། རྒྱལ་མཚན་དང་། བ་དན་དང་། དར་གྱི་ལྡ་ལྡི་དང་། རོལ་མོའི་སྒྲ་དང་། སིལ་སྙན་དང་། ཕེག་རྡོབ་པ་དང་། གླུ་དབྱངས་དང་། རིན་པོ་ཆེའི་མེ་ཏོག་དང་། རིན་པོ་ཆེའི་ལྡ་ལྡི་དང་། མུ་ཏིག་གི་ཕྲེང་བ་དང་། ཀླུའི་མེ་ཏོག་དང་། མུ་ཏིག་གི་དྲ་བ་མཆོག་ཏུ་མཛེས་པ་རྣམ་པ་སྣ་ཚོགས་དཔག་ཏུ་མེད་གྲངས་མེད་པ་དག་བཅོམ་ལྡན་འདས་ལ་ཕུལ་ནས་བསྐོར་བ་བྱས་ཏེ་ཕྱོགས་གཅིག་ཏུ་འདུག་གོ། །ཕྱོགས་གཅིག་ཏུ་འདུག་ནས་སྨོན་ལམ་རྣམས་བཏབ་པ། ལུས་རྒྱ་མཚོ་འཇིག་རྟེན་གྱི་ཁམས་རྒྱ་མཚོ་ཐམས་ཅད་ཀྱི་རྡུལ་ཕྲ་རབ་ཀྱི་རྡུལ་སྙེད་ཀྱིས་སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའི་འཁོར་གྱི་དཀྱིལ་འཁོར་རྒྱ་མཚོ་ཐམས་ཅད་དང་། འཇིག་རྟེན་གྱི་ཁམས་རབ་འབྱམ་རྒྱ་མཚོ་ཐམས་ཅད་དང་། ས་དང་། ཆུ་དང་། མེ་དང་། རླུང་ཐམས་ཅད་ཀྱི་རྡུལ་ཕྲ་རབ་ཀྱི་རྡུལ་རྣམས་དང་། གཟུགས་སུ་སྣང་བ་ཐམས་ཅད་ཀྱི་རྡུལ་ཕྲ་རབ་ཀྱི་རྡུལ་རྣམས་ལས་རྡུལ་ཕྲ་རབ་ཀྱི་རྡུལ་རེ་རེ་ལ་ཡང་ལུས་ཀྱི་སྤྲིན། རྒྱ་མཚོ་བགྲང་བའི་རྒྱ་མཚོ་ཐམས་ཅད་ལས་ཡང་དག་པར་འདས་པ་གྲངས་མེད། དཔག་ཏུ་མེད། བསམ་གྱིས་མི་ཁྱབ་མཚུངས་པ་མེད། གཞལ་དུ་མེད། བརྗོད་དུ་མེད་པ་ལས་ཡང་དག་པར་འདས་པ་དག་དང་། ལུས་རེ་རེ་ལ་ཡང་ལག་པ་རྒྱ་མཚོའི་སྤྲིན་ཕྱོགས་ཀུན་གྱི་རྒྱུན་དུ་མངོན་པར་ཕྱོགས་པ་དཔག་ཏུ་མེད་གྲངས་མེད་པ་དག་བྱིན་གྱིས་བརླབས་ནས། རྡུལ་ཕྲ་རབ་ཀྱི་རྡུལ་གྱི་ཆ་རེ་རེ་ལས་ཀྱང་མཆོད་པ་ཐམས་ཅད་ཀྱི་སྤྲིན་རྒྱ་མཚོ་ཕྱོགས་ཀུན་གྱི་གཞི་རྒྱས་པར་འགེངས་པ། མ་འདྲེས་པ་རྣམས་ཀྱིས་སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་མཆོད་པ་བྱའོ་སྙམ་པ་འདི་ལྟ་སྟེ། ཀུན་དུ་སྤྱོད་པའི་ལུས་རྒྱ་མཚོ་སྤྲིན་མ་འདྲེས་པ་དཔག་ཏུ་མེད། གྲངས་མེད། བསམ་གྱིས་མི་ཁྱབ། གཞལ་དུ་མེད། ཚད་མེད། བརྗོད་དུ་མེད་པ་དག་གིས་ནམ་མཁའི་དཀྱིལ་ཀུན་དུ་ཁེབས་པར་བྱིན་གྱིས་བརླབས་ནས་བྱང་ཆུབ་སེམས་དཔའི་ལུས་རྒྱ་མཚོ་སྤྲིན་རྣམས་ཀྱིས་ཇི་ལྟ་བ་དེ་བཞིན་དུ་རིན་པོ་ཆེ་ཐམས་ཅད་ཀྱི་ཁ་དོག་དང་། འོད་ཟེར་སྟུག་པོ་དང་། ཉི་མ་དང་། ཟླ་བ་ཐམས་ཅད་ལྟ་བུའི་ལུས་ཀྱི་འཁོར་གྱི་དཀྱིལ་འཁོར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་ཐམས་ཅད་ཀྱི་ཕྲེང་བ་དང་། མེ་ཏོག་གི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་ཐམས་ཅད་ཀྱི་སྣང་བ་ཆེན་པོའི་སྙིང་པོའི་ཁང་པ་བརྩེགས་པ་རྒྱ་མཚོའི་སྤྲིན་རྣམས་དང་། ཕྱེ་མ་དང་། ཤིང་ལྗོན་པ་ཐམས་ཅད་ཀྱི་མཛོད་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སྤོས་དང་། བདུག་པ་ཐམས་ཅད་ཀྱི་གཟུགས་ཀུན་དུ་སྟོན་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སྒྲ་ཐམས་ཅད་སྒྲོགས་པའི་རོལ་མོའི་སྒྲ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སྤོས་ཐམས་ཅད་ཀྱི་ཤིང་ལྗོན་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་ཀྱིས་ནམ་མཁའི་དཀྱིལ་ཀུན་དུ་ཁེབས་པར་བྱིན་གྱིས་བརླབས་ནས་དེ་ལ་སོགས་པ་མཆོད་པ་ཐམས་ཅད་ཀྱི་སྤྲིན་རྒྱ་མཚོ་དཔག་ཏུ་མེད། གྲངས་མེད། བསམ་གྱིས་མི་ཁྱབ། གཞལ་དུ་མེད། ཚད་མེད། བརྗོད་དུ་མེད་པ། མ་འདྲེས་པ་རྣམས་ཀྱིས། སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་རྒྱ་མཚོ་ཐམས་ཅད་ལ་བསྟི་སྟང་དུ་བྱའོ། །བླ་མར་བྱའོ། །རི་མོར་བྱའོ། །མཆོད་པར་བྱའོ། །ཡུལ་ཐམས་ཅད་དུ་བཀོད་པ་རྣམ་པར་སྣང་བར་བྱས་པའི་སྙིང་པོ་ཅན་གྱི་ནོར་བུའི་རྒྱལ་པོ་རྒྱ་མཚོ་སྤྲིན་རྣམས་ཀྱིས་ནམ་མཁའི་དཀྱིལ་ཀུན་དུ་ཁེབས་པར་བྱིན་གྱིས་བརླབས་ཏེ། སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་རྒྱ་མཚོ་ཐམས་ཅད་ལ་བསྟི་སྟང་དུ་བྱའོ། །བླ་མར་བྱའོ། །རི་མོར་བྱའོ། །མཆོད་པར་བྱའོ། །ཐམས་ཅད་ཀུན་དུ་སྣང་བའི་རིན་པོ་ཆེའི་ཆར་འབབ་པ་བཀོད་པའི་ནོར་བུའི་རྒྱལ་པོ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་ཐམས་ཅད་ཀྱི་འོད་འཕྲོ་བ་འབར་བ་དང་ལྡན་པ་སངས་རྒྱས་ལ་ངེས་པར་གྱུར་པའི་དབྱངས་ཀྱི་ནོར་བུའི་རྒྱལ་པོ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཀུན་དུ་སྒྲས་འགེངས་པ་སངས་རྒྱས་ཀྱི་ཆོས་ཐམས་ཅད་ཀྱི་དབྱངས་ཅན་ནོར་བུའི་རྒྱལ་པོ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཀུན་ནས་སྒོ་རིན་པོ་ཆེ་འོད་ཟེར་གྱིས་སངས་རྒྱས་ཀྱི་སྤྲུལ་པ་སྣང་བར་བྱེད་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། འཁོར་གྱི་དཀྱིལ་འཁོར་མ་འདྲེས་པ་རྣམས་སུ་བཀོད་པ་ཐམས་ཅད་སྣང་བ་ཀུན་དུ་སྟོན་པའི་ནོར་བུའི་རྒྱལ་པོ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། འོད་འཕྲོ་བ་སྒྲོན་མ་སངས་རྒྱས་ཀྱི་ཡུལ་ཐམས་ཅད་ཀྱི་རྗེས་སུ་འཇུག་པའི་ནོར་བུའི་རྒྱལ་པོ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སངས་རྒྱས་ཀྱི་ཞིང་བསམ་གྱིས་མི་ཁྱབ་པ་རྣམས་སུ་དེ་བཞིན་གཤེགས་པའི་གཞལ་མེད་ཁང་སྣང་བ་ཀུན་དུ་སྟོན་པའི་ནོར་བུའི་རྒྱལ་པོ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་སྣ་ཚོགས་ཐམས་ཅད་ཀྱི་རྡུལ་ལ་དུས་གསུམ་གྱི་སངས་རྒྱས་ཀྱི་སྐུ་སྣང་བ་རྣམ་པར་སྣང་བར་བྱེད་པའི་ནོར་བུའི་རྒྱལ་པོ་རྒྱ་མཚོ་སྤྲིན་རྣམས་ཀྱིས་ནམ་མཁའི་དཀྱིལ་ཀུན་དུ་ཁེབས་པར་བྱིན་གྱིས་བརླབས་ཏེ། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་རྒྱ་མཚོ་སྤྲིན་ཐམས་ཅད་ལ་བསྟི་སྟང་དུ་བྱའོ། །བླ་མར་བྱའོ། །རི་མོར་བྱའོ། །མཆོད་པར་བྱའོ། །རིན་པོ་ཆེ་དང་། སྤོས་དང་། མེ་ཏོག་སྣ་ཚོགས་ཐམས་ཅད་ཀྱི་ཁང་པ་བརྩེགས་པ་མ་འདྲེས་པ་རྒྱ་མཚོའི་སྤྲིན་རྣམས་དང་། ནོར་བུ་རིན་པོ་ཆེའི་རྒྱལ་པོ་ཁ་དོག་མཐའ་ཡས་པ་ཐམས་ཅད་ཀྱིས་རབ་ཏུ་བརྒྱན་པའི་ཁང་པ་བརྩེགས་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེའི་སྒྲོན་མ་དང་། སྤོས་ཀྱི་འོད་འཕྲོ་བ་ཅན་གྱི་ཁང་པ་བརྩེགས་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། མུ་ཏིག་ཐམས་ཅད་ཀྱིས་རྣམ་པར་སྤྲས་པའི་ཁང་པ་བརྩེགས་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། མེ་ཏོག་ཐམས་ཅད་ཀྱི་ཟེ་བ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེའི་ཕྲེང་བ་མཐའ་ཡས་པས་ལེགས་པར་བརྒྱན་པའི་ཁང་པ་བརྩེགས་པའི་རྒྱ་མཚོ་སྤྲིན་རྣམས་ཀྱིས་ནམ་མཁའི་དཀྱིལ་ཀུན་དུ་ཁེབས་པར་བྱིན་གྱིས་བརླབས་ནས། ཕྱོགས་བཅུ་ཀུན་དུ་འོད་ཟེར་སྣང་བའི་སྙིང་པོ་ཅན། བཀོད་པ་ཐམས་ཅད་ཡང་དག་པར་འདུ་བའི་ཁང་པ་བརྩེགས་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་ཐམས་ཅད་ཀྱི་རྡུལ་ཕྲ་རབ་ཀྱིས་རབ་ཏུ་བརྒྱན་པ་བཀོད་པ་དུ་མ་ཀུན་དུ་སྟོན་པའི་ཁང་པ་བརྩེགས་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། བཀོད་པ་ཐམས་ཅད་ཀྱིས་རབ་ཏུ་བརྒྱན་པ་ཀུན་དུ་རྒྱས་པར་འགེངས་པའི་ཁང་པ་བརྩེགས་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སྒོ་ཀུན་དུ་མེ་ཏོག་གི་རྟ་བབས་དྲིལ་བུའི་དྲ་བ་དཔྱངས་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་ཀྱིས་ནམ་མཁའི་དཀྱིལ་ཀུན་དུ་ཁེབས་པར་བྱིན་གྱིས་བརླབས་ཏེ། སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་རྒྱ་མཚོ་ཐམས་ཅད་ལ་བསྟི་སྟང་དུ་བྱའོ། །བླ་མར་བྱའོ། །རི་མོར་བྱའོ། །མཆོད་པར་བྱའོ། །གསེར་དང་། རིན་པོ་ཆེ་སྣ་ཚོགས་ཀྱི་ཙོང་ཙོང་དང་། ཕྲེང་བ་དང་། རིན་པོ་ཆེའི་མགུལ་གདུབ་ཀྱི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། མེ་ཏོག་ལྟར་སྣང་བ་སྣ་ཚོགས་ཀྱི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཨན་ད་རྙིལ་དང་། འཛམ་བུའི་ཆུ་བོའི་གསེར་གྱི་པད་མ་སྣ་ཚོགས་ཀྱི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ནོར་བུ་རིན་པོ་ཆེའི་སྒྲོན་མའི་པད་མའི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་ནོར་བུ་རིན་པོ་ཆེ་སྣང་གསལ་རྒྱལ་མཚན་གྱི་པད་མ་སྣ་ཚོགས་ཀྱི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེས་བརྒྱན་པ་པད་མ་སྣ་ཚོགས་ཀྱི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཨན་ད་རྙིལ་མཛེས་པར་འབར་བའི་འོད་ཟེར་ལྡན་པའི་པད་མའི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། འོད་ཟེར་འབར་བ་ཟད་མི་ཤེས་པའི་གཟི་བརྗིད་དང་ལྡན་པའི་པད་མའི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་འབར་བ་ཐམས་ཅད་བྱུང་བའི་པད་མའི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་འབར་བ་མཆོག་དམ་པ་ཐམས་ཅད་བྱུང་བའི་པད་མའི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སངས་རྒྱས་ཀྱི་གསུང་སྒྲའི་འོད་ཟེར་དང་ལྡན་པའི་པད་མའི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་མ་འདྲེས་པ་རྣམས་ཀྱིས་ནམ་མཁའི་དཀྱིལ་ཀུན་དུ་ཁེབས་པར་བྱིན་གྱིས་བརླབས་ཏེ། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་རྒྱ་མཚོ་ཐམས་ཅད་ལ་བསྟི་སྟང་དུ་བྱའོ། །བླ་མར་བྱའོ། །རི་མོར་བྱའོ། །མཆོད་པར་བྱའོ། །སྤོས་དང་། ནོར་བུ་སྣ་ཚོགས་ཐམས་ཅད་ཀྱི་ཤིང་ལྗོན་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཀུན་ནས་སྒོ་ལོ་མ་ལས་ཐལ་མོ་སྦྱར་བའི་སྤྲིན་འབྱུང་བའི་སྤོས་ཀྱི་ཤིང་ལྗོན་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཤིང་ལྗོན་པ་ཐམས་ཅད་ཀྱི་གཟུགས་མཐའ་ཡས་པ་བཀོད་པ་ཀུན་དུ་སྟོན་པའི་ཤིང་ལྗོན་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། མེ་ཏོག་ཐམས་ཅད་ཀྱི་སྤྲིན་རབ་ཏུ་འཕྱང་བའི་སྤྲིན་ཤིང་ལྗོན་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། བཀོད་པ་ཐམས་ཅད་འབྱུང་བའི་མཛོད་མཐའ་ཡས་པ་དང་ལྡན་པའི་ཤིང་ལྗོན་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་འོད་འཕྲོ་བ་ཐམས་ཅད་ཀྱི་དཀྱིལ་འཁོར་རྣམ་པར་སྣང་བར་བྱེད་པའི་ཤིང་ལྗོན་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། བྱང་ཆུབ་སེམས་དཔའི་ལུས་ཕྱེད་ཙན་དན་ཐམས་ཅད་ཀྱི་ཕྱེ་མར་ཀུན་དུ་སྟོན་པའི་ཤིང་ལྗོན་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། བྱང་ཆུབ་ཀྱི་སྙིང་པོ་ཐམས་ཅད་ཀྱི་བཀོད་པ་བསམ་གྱིས་མི་ཁྱབ་པའི་འོད་ཅན་གྱི་ཤིང་ལྗོན་པའི་རིན་པོ་ཆེའི་གོས་ཐམས་ཅད་ཀྱི་མཛོད་ཉི་མ་ལྟར་སྣང་བར་བྱས་པ་དང་ལྡན་པའི་ཤིང་ལྗོན་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སྒྲ་དབྱངས་ཐམས་ཅད་སྒྲོགས་པ་ཡིད་དུ་འོང་བའི་དབྱངས་ཀུན་དུ་གཏོང་བའི་ཤིང་ལྗོན་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་ཁ་དོག་མཐའ་ཡས་པའི་པད་མའི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཀུན་དུ་མངོན་པར་ཕྱོགས་པའི་ནོར་བུའི་རྒྱལ་པོ་སྣང་བར་བྱེད་པའི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རྒྱན་བཀོད་པ་ཐམས་ཅད་ཀྱིས་རབ་ཏུ་བརྒྱན་པའི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་སྣ་ཚོགས་འོད་འཕྲོ་བའི་སྒྲོན་མའི་ཕྲེང་བའི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཀུན་ནས་སྒྲ་རིན་པོ་ཆེའི་ཤིང་ལྗོན་པ་འབྱུང་བའི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སྤོས་དང་མེ་ཏོག་དང་། པད་མ་ཐམས་ཅད་ཀྱི་ཕྲེང་བ་རིན་པོ་ཆེའི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སངས་རྒྱས་ཐམས་ཅད་ཀྱི་གདན་བཀོད་པ་ཀུན་དུ་སྟོན་པའི་ནོར་བུ་རིན་པོ་ཆེའི་རྒྱལ་པོའི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། བཀོད་པ་ཐམས་ཅད་ཀྱི་ཕྲེང་བ་རབ་ཏུ་དཔྱངས་པའི་རྟ་བབས་སྟེགས་བུས་རབ་ཏུ་བརྒྱན་པ་རིན་པོ་ཆེའི་སྤྲིན་གྱི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ནོར་བུ་རིན་པོ་ཆེའི་ཡལ་ག་དང་། ཕྱེ་མའི་མཛོད་ཡོད་པའི་སྡོང་པོ་ཐམས་ཅད་ཀྱི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སྤོས་སྣ་ཚོགས་དང་རིན་པོ་ཆེའི་དྲིལ་བུ་གཡེར་ཀའི་དྲ་བ་ཀུན་ནས་རྒྱན་ཉི་མ་སྣང་བར་བྱེད་པའི་སྙིང་པོ་ཅན་གྱི་སེང་གེའི་ཁྲི་རྒྱ་མཚོ་སྤྲིན་རྣམས་ཀྱིས་ནམ་མཁའི་དཀྱིལ་ཀུན་དུ་ཁེབས་པར་བྱིན་གྱིས་བརླབས་ཏེ། སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་རྒྱ་མཚོ་ཐམས་ཅད་ལ་བསྟི་སྟང་དུ་བྱའོ། །བླ་མར་བྱའོ། །རི་མོར་བྱའོ། །མཆོད་པར་བྱའོ། །ཡིད་བཞིན་གྱི་ནོར་བུའི་རྒྱལ་པོ་ཐམས་ཅད་ཀྱི་གཞལ་མེད་ཁང་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཨན་ད་རྙིལ་གྱི་ཟེ་བ་ཅན་གྱི་མེ་ཏོག་བཀོད་པ་ཐམས་ཅད་ཀྱི་བླ་རེ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སྤོས་དང་། ནོར་བུ་ཐམས་ཅད་ཀྱི་བླ་རེ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེའི་འོད་འཕྲོ་བའི་སྒྲོན་མའི་གཟུགས་ཀྱི་བླ་རེ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སངས་རྒྱས་ཀྱི་རྫུ་འཕྲུལ་རྣམ་པར་སྟོན་པར་སྣང་བའི་དབྱངས་ཅན་གྱི་ནོར་བུའི་རྒྱལ་པོའི་བླ་རེ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ནོར་བུ་སྣ་ཚོགས་ཀྱི་གོས་བཀོད་པ་ཐམས་ཅད་སྣང་བ་ཀུན་ཏུ་སྟོན་པའི་བླ་རེ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། མེ་ཏོག་ཐམས་ཅད་ཀྱི་ཆུ་བོ་འབར་བར་སྣང་བའི་རིན་པོ་ཆེའི་བླ་རེ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། དྲིལ་བུ་སྣ་ཚོགས་ཀྱི་སྒྲས་ཀུན་དུ་རྒྱས་པར་འགེངས་པའི་དྲ་བའི་བླ་རེ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། པད་མ་འབར་བ་ཁ་དོག་མཐའ་ཡས་པ་སྣ་ཚོགས་སུ་སྣང་བའི་ནོར་བུའི་སྙིང་པོ་ཅན་གྱི་པད་མའི་དྲ་བའི་བླ་རེ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། གསེར་གྱི་ཟེ་བ་ཅན་གྱི་འོད་ཟེར་སྣང་བའི་རྒྱལ་མཚན་གྱི་བླ་རེ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། བཀོད་པ་བསམ་གྱིས་མི་ཁྱབ་པར་སྣང་བའི་འོད་མཐའ་ཡས་པ་བཀོད་པ་ཐམས་ཅད་ཀྱི་རྒྱན་གྱི་བླ་རེ་རྒྱ་མཚོ་སྤྲིན་རྣམས་ཀྱིས་ནམ་མཁའི་དཀྱིལ་ཀུན་དུ་ཁེབས་པར་བྱིན་གྱིས་བརླབས་ཏེ། སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་གང་ཆེན་མཚོ་སྤྲིན་རྒྱ་མཚོ་ཐམས་ཅད་ལ་བསྟི་སྟང་དུ་བྱའོ། །བླ་མར་བྱའོ། །རི་མོར་བྱའོ། །མཆོད་པར་བྱའོ། །ནོར་བུ་རིན་པོ་ཆེ་སྣ་ཚོགས་ཐམས་ཅད་ཀྱི་གདུགས་རྒྱ་མཚོ་སྤྲིན་རྣམས་ཀྱིས་ནམ་མཁའི་དཀྱིལ་ཀུན་དུ་ཁེབས་པར་བྱིན་གྱིས་བརླབས་ལ་རིན་པོ་ཆེ་བཅུའི་འོད་ཟེར་སྣང་བ་བཀོད་པའི་མེ་ཏོག་གི་གདུགས་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། མུ་ཏིག་ཁ་དོག་མཐའ་ཡས་པའི་མཛོད་ཀྱི་གདུགས་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ཀྱི་སྙིང་རྗེའི་སྒོའི་སྒྲའི་ནོར་བུའི་རྒྱལ་པོའི་གདུགས་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་སྣ་ཚོགས་ཀྱི་འོད་འཕྲོ་བའི་ཕྲེང་བའི་གདུགས་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེའི་རྡུལ་ཀུན་དུ་སྣང་བས་རབ་ཏུ་བརྒྱན་པའི་དྲིལ་བུའི་དྲ་བ་དཔྱངས་པའི་གདུགས་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ནོར་བུ་ཐམས་ཅད་ཀྱི་སྡོང་པོའི་ཡལ་གའི་རྒྱན་ཅན་གྱི་གདུགས་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཉི་མ་རྣམ་པར་སྣང་བར་བྱེད་པའི་འོད་འཕྲོ་བའི་ནོར་བུའི་རྒྱལ་པོ་དང་། སྤོས་དང་། བདུག་པ་ཐམས་ཅད་ཀྱི་གདུགས་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཙན་དན་གྱི་ཕྱེ་མའི་མཛོད་ཀུན་དུ་རྒྱས་པར་འགེངས་པའི་གདུགས་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སངས་རྒྱས་ཀྱི་ཡུལ་རྒྱ་ཆེན་པོ་སྣང་བར་བྱེད་པ་ཀུན་དུ་བཀོད་པས་རྒྱས་པར་འགེངས་པའི་གདུགས་རྒྱ་མཚོ་སྤྲིན་མ་འདྲེས་པ་རྣམས་ཀྱིས་སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་རྒྱ་མཚོ་ཐམས་ཅད་ལ་བསྟི་སྟང་དུ་བྱའོ། །བླ་མར་བྱའོ། །རི་མོར་བྱའོ། །མཆོད་པར་བྱའོ། །རིན་པོ་ཆེ་ཐམས་ཅད་སྣང་བའི་དཀྱིལ་འཁོར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་འོད་འཕྲོ་བ་ཐམས་ཅད་ཀྱི་གཟུགས་ཀྱི་འོད་ཀྱི་དཀྱིལ་འཁོར་རྒྱ་མཚོ་སྤྲིན་མ་འདྲེས་པ་རྣམས་དང་། མེ་ཏོག་གི་སྤྲིན་སྣང་བར་བྱེད་པའི་འོད་ཀྱི་དཀྱིལ་འཁོར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་ཐམས་ཅད་ཀྱི་འོད་ཟེར་སངས་རྒྱས་ཀྱི་སྤྲུལ་པའི་འོད་ཀྱི་དཀྱིལ་འཁོར་དང་ལྡན་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སངས་རྒྱས་ཀྱི་ཞིང་ཐམས་ཅད་སྣང་བར་གཏོགས་པའི་འོད་ཀྱི་དཀྱིལ་འཁོར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཀུན་ནས་སྒོ་སངས་རྒྱས་ཀྱི་ཡུལ་སྒྲོགས་པ་རིན་པོ་ཆེའི་ཡལ་གའི་འོད་ཀྱི་དཀྱིལ་འཁོར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། བཻ་དཱུརྱ་དང་། རིན་པོ་ཆེའི་རིགས་དང་། ནོར་བུ་རིན་པོ་ཆེའི་རྒྱལ་པོ་ཐམས་ཅད་ཀྱི་འོད་ཟེར་གྱི་འོད་ཀྱི་དཀྱིལ་འཁོར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སེམས་ཅན་གྱི་གཟུགས་མཐའ་ཡས་པ་དག་སེམས་ཀྱི་སྐད་ཅིག་ལ་ཀུན་དུ་སྟོན་པའི་འོད་ཀྱི་དཀྱིལ་འཁོར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སངས་རྒྱས་ཐམས་ཅད་ཀྱི་སྨོན་ལམ་ལས་བྱུང་བ་ཡིད་དུ་འོང་བའི་སྒྲ་རབ་ཏུ་འབྱིན་པ་དང་ལྡན་པའི་འོད་ཀྱི་དཀྱིལ་འཁོར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། འཁོར་གྱི་དཀྱིལ་འཁོར་ཐམས་ཅད་དུ་སེམས་ཅན་འདུལ་བའི་དབྱངས་ཅན་ནོར་བུའི་རྒྱལ་པོའི་འོད་ཀྱི་དཀྱིལ་འཁོར་རྒྱ་མཚོ་སྤྲིན་རྣམས་ཀྱིས་སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་རྒྱ་མཚོ་ཐམས་ཅད་ལ་བསྟི་སྟང་དུ་བྱའོ། །བླ་མར་བྱའོ། །རི་མོར་བྱའོ། །མཆོད་པར་བྱའོ། །ནོར་བུ་ཐམས་ཅད་ཀྱི་མཛོད་ཀྱི་འོད་ཟེར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སངས་རྒྱས་ཐམས་ཅད་ཀྱི་གཟུགས་དང་། སྒྲ་དང་། དྲི་དང་། རོ་དང་། རེག་བྱའི་འོད་ཟེར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་ཐམས་ཅད་ཀྱི་འོད་འཕྲོ་བ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སངས་རྒྱས་ཐམས་ཅད་ཀྱི་ཆོས་སྒྲོགས་པ་རྒྱས་པར་འགེངས་པའི་འོད་ཟེར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སངས་རྒྱས་ཀྱི་ཞིང་བཀོད་པ་ཐམས་ཅད་སྣང་བར་བྱེད་པའི་འོད་ཟེར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། མེ་ཏོག་གི་ཁང་པ་བརྩེགས་པ་ཐམས་ཅད་ཀྱི་འོད་ཟེར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་ཐམས་ཅད་ཀྱི་རྡུལ་གྱི་འོད་ཟེར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། བསྐལ་པ་གཅིག་ནས་གཅིག་ཏུ་བརྒྱུད་པ་ཐམས་ཅད་དུ་སངས་རྒྱས་འབྱུང་བས་སེམས་ཅན་ཡོངས་སུ་སྨིན་པར་བྱེད་པ་སྒྲོགས་པ་ཅན་གྱི་འོད་ཟེར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་མི་ཟད་པ་ཐམས་ཅད་བསྟན་པ་དང་། མེ་ཏོག་ཐམས་ཅད་ཀྱི་ཟེ་བའི་འོད་ཟེར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཁྲི་སྟན་བཀོད་པ་ཐམས་ཅད་ཀུན་ཏུ་སྟོན་པའི་འོད་ཟེར་རྒྱ་མཚོ་སྤྲིན་མ་འདྲེས་པ་རྣམས་ཀྱིས་སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་རྒྱ་མཚོ་ཐམས་ཅད་ལ་བསྟི་སྟང་དུ་བྱའོ། །བླ་མར་བྱའོ། །རི་མོར་བྱའོ། །མཆོད་པར་བྱའོ། །རིན་པོ་ཆེའི་འོད་འཕྲོ་བ་ཁ་དོག་མཐའ་ཡས་པ་རྒྱ་མཚོ་སྤྲིན་མ་འདྲེས་པ་རྣམས་དང་། ཀུན་ཏུ་འོད་ཀྱི་ནོར་བུའི་རྒྱལ་པོ་འོད་འཕྲོ་བ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སངས་རྒྱས་ཐམས་ཅད་ཀྱི་ཞིང་རྒྱ་ཆེན་པོ་བཀོད་པ་སྣང་བར་བྱེད་པའི་འོད་འཕྲོ་བ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སྤོས་ཐམས་ཅད་ཀྱི་འོད་འཕྲོ་བ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། བཀོད་པ་ཐམས་ཅད་ཀྱི་འོད་འཕྲོ་བ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། སངས་རྒྱས་ཐམས་ཅད་ཀྱི་སྤྲུལ་པའི་འོད་འཕྲོ་བ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་སྣ་ཚོགས་ཀྱི་ཤིང་གི་ཟེ་བ་འོད་འཕྲོ་བ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། གོས་ཐམས་ཅད་ཀྱི་འོད་འཕྲོ་བ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། བྱང་ཆུབ་སེམས་དཔའ་མཐའ་ཡས་པའི་སྤྱོད་པ་དབྱངས་ཅན་ནོར་བུའི་རྒྱལ་པོའི་འོད་འཕྲོ་བ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། མུ་ཏིག་གི་སྒྲོན་མ་ཐམས་ཅད་ཀྱི་འོད་འཕྲོ་བ་རྒྱ་མཚོ་སྤྲིན་མ་འདྲེས་པ་རྣམས་ཀྱིས་སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་གང་ཆེན་མཚོའི་སྤྲིན་གྱི་རྒྱ་མཚོ་ཐམས་ཅད་ལ་བསྟི་སྟང་དུ་བྱའོ། །བླ་མར་བྱའོ། །རི་མོར་བྱའོ། །མཆོད་པར་བྱའོ། །སྤོས་དང་མེ་ཏོག་སྣ་ཚོགས་ཐམས་ཅད་ཀྱི་རྒྱན་རྒྱ་མཚོ་སྤྲིན་བསམ་གྱིས་མི་ཁྱབ་པ་རྣམས་དང་། རིན་པོ་ཆེ་འོད་འཕྲོ་བ་ཐམས་ཅད་ཀྱི་པད་མའི་དྲ་བ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ནོར་བུ་རིན་པོ་ཆེ་ཁ་དོག་མཐའ་ཡས་པའི་འོད་ཀྱི་དཀྱིལ་འཁོར་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་ཐམས་ཅད་ཀྱི་ཁ་དོག་དང་ལྡན་པའི་མུ་ཏིག་གི་མཛོད་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་དང་། སྤོས་དང་། ཙན་དན་ཐམས་ཅད་ཀྱི་ཕྱེ་མ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེ་ཐམས་ཅད་ཀྱི་གདུགས་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ནོར་བུའི་རྒྱལ་པོ་ཡིད་དུ་འོང་ཞིང་སྡུག་པ་མཛེས་པའི་དབྱངས་ཅན་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཉི་མའི་འོད་ཀྱི་ནོར་བུའི་ཚོགས་ཀྱི་ཕྲེང་བ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། རིན་པོ་ཆེའི་མཛོད་མཐའ་ཡས་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་དང་། ཐམས་ཅད་ཀུན་ཏུ་བཟང་པོའི་ལུས་མ་འདྲེས་པ་རྒྱ་མཚོ་སྤྲིན་རྣམས་ཀྱིས་སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་གང་ཆེན་མཚོའི་སྤྲིན་གྱི་རྒྱ་མཚོ་ཐམས་ཅད་ལ་བསྟི་སྟང་དུ་བྱའོ། །བླ་མར་བྱའོ། །རི་མོར་བྱའོ། །མཆོད་པར་བྱའོ། །ཞེས་དེ་ལྟ་བུའི་སྨོན་ལམ་བཏབ་ནས་ཀླུའི་རྒྱལ་པོ་དེ་དག་གིས་ཡང་བཅོམ་ལྡན་འདས་ལ་ལན་གསུམ་བསྐོར་བ་བྱས་ཏེ་ཞབས་ལ་ཕྱག་འཚལ་བ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་གནང་ནས་རང་རང་གི་སྟན་ལ་འཁོད་དོ། །དེའི་ཚེ་སྟོང་གསུམ་གྱི་སྟོང་ཆེན་པོ་ཀླུའི་བདག་པོ་ཆེན་པོ་ཕྱིར་མི་ལྡོག་པའི་སྨོན་ལམ་གྱི་སྟོབས་ཅན་གཡོག་མཐའ་ཡས་རྒྱ་མཚོའི་སྤྲིན་བཀོད་པའི་གཟི་བརྗིད་ཀྱི་དཀྱིལ་འཁོར་གདུགས་ལྟ་བུའི་རྒྱལ་པོ་ཞེས་བྱ་བ་དེ་བཞིན་གཤེགས་པ་ལ་མཆོད་པ་བྱ་བ་དང་། ཕྱག་བྱ་བ་དང་། བསྙེན་བཀུར་བྱ་བ་དང་། ཆོས་མཉན་པའི་ཕྱིར་འཛམ་བུའི་གླིང་འདིར་འོངས་སོ། །དེ་ནས་ཀླུའི་བདག་པོ་གཡོག་མཐའ་ཡས་རྒྱ་མཚོའི་སྤྲིན་བཀོད་པའི་གཟི་བརྗིད་ཀྱི་དཀྱིལ་འཁོར་གདུགས་ལྟ་བུའི་རྒྱལ་པོ་སྟན་ལས་ལངས་ཏེ་བླ་གོས་ཕྲག་པ་གཅིག་ཏུ་གཟར་ནས་པུས་མོ་གཡས་པའི་ལྷ་ང་ས་ལ་བཙུགས་ཏེ་བཅོམ་ལྡན་འདས་ག་ལ་བ་དེ་ལོགས་སུ་ཐལ་མོ་སྦྱར་བ་བཏུད་དེ། བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །གལ་ཏེ་ཞུས་ན་ཞུ་བ་ལུང་བསྟན་པའི་སླད་དུ་བཅོམ་ལྡན་འདས་ཀྱིས་བདག་ལ་སྐབས་ཕྱེ་ན། བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ལ་བདག་ཕྱོགས་འགའ་ཞིག་ཞུའོ། །དེ་སྐད་ཅེས་གསོལ་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་ཀླུའི་བདག་པོ་གཡོག་མཐའ་ཡས་རྒྱ་མཚོའི་སྤྲིན་བཀོད་པའི་གཟི་བརྗིད་ཀྱི་དཀྱིལ་འཁོར་གདུགས་ལྟ་བུའི་རྒྱལ་པོ་ལ་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། །ཀླུའི་བདག་པོ་ཅི་དང་། ཅི་འདོད་པ་དྲིས་ཤིག་དང་། ངས་དྲིས་པ་དེ་དང་དེའི་ལུང་བསྟན་པས་ཁྱོད་ཀྱི་སེམས་རངས་པར་བྱའོ། །དེ་སྐད་ཅེས་བཀའ་སྩལ་པ་དང་། བཅོམ་ལྡན་འདས་ལ་ཀླུའི་བདག་པོ་གཡོག་མཐའ་ཡས་རྒྱ་མཚོའི་སྤྲིན་བཀོད་པའི་གཟི་བརྗིད་ཀྱི་དཀྱིལ་འཁོར་གདུགས་ལྟ་བུའི་རྒྱལ་པོས་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་ཇི་ལྟར་ན་ཀླུ་ཐམས་ཅད་ཀྱི་ཀླུའི་སྡུག་བསྔལ་ཐམས་ཅད་རྒྱུན་ཆད་པར་འགྱུར། བདེ་བ་ཐམས་ཅད་དང་ལྡན་པ་དང་རབ་ཏུ་དགའ་བར་འགྱུར། འཛམ་བུའི་གླིང་འདིར་དུས་དུས་སུ་ཆར་གྱི་རྒྱུན་འབེབ་པར་འགྱུར། རྩྭ་དང་། ཤིང་གེལ་བ་དང་། སྨན་དང་། ནགས་ཚལ་ཐམས་ཅད་འཁྲུང་བར་འགྱུར། འབྲུ་ཐམས་ཅད་སྐྱེ་བར་འགྱུར། བཅུད་ཐམས་ཅད་ཡང་དག་པར་སྐྱེ་བར་འགྱུར། གང་གིས་འཛམ་བུའི་གླིང་གི་མི་རྣམས་བདེ་བ་དང་ལྡན་པར་འགྱུར། དེ་སྐད་ཅེས་གསོལ་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་ཀླུའི་བདག་པོ་གཡོག་མཐའ་ཡས་རྒྱ་མཚོའི་སྤྲིན་བཀོད་པའི་གཟི་བརྗིད་ཀྱི་དཀྱིལ་འཁོར་གདུགས་ལྟ་བུའི་རྒྱལ་པོ་ལ་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། །ཀླུའི་བདག་པོ་ཁྱོད་སེམས་ཅན་ཐམས་ཅད་ལ་ཕན་པ་དང་། བདེ་བའི་ཕྱིར་ཞུགས་ཏེ་དེ་བཞིན་གཤེགས་པ་ལ་དོན་འདི་ཡོངས་སུ་འདྲི་བར་སེམས་པ་ལེགས་སོ། །ལེགས་སོ། །ཀླུའི་བདག་པོ་དེའི་ཕྱིར་ལེགས་པར་རབ་ཏུ་ཉོན་ལ་ཡིད་ལ་ཟུང་ཤིག་དང་། ངས་ཁྱོད་ལ་བཤད་དོ། །ཀླུའི་བདག་པོ་ཆོས་གཅིག་དང་ལྡན་ན། ཀླུ་ཐམས་ཅད་ཀྱི་ཀླུའི་སྡུག་བསྔལ་ཐམས་ཅད་རྒྱུན་ཆད་པར་འགྱུར་བདེ་བ་དང་ལྡན་པར་འགྱུར་རོ། །ཆོས་གཅིག་པོ་གང་ཞེ་ན། འདི་ལྟ་སྟེ། བྱམས་པའོ། །ཀླུའི་བདག་པོ་བྱམས་པ་དེ་ལ་གནས་པའི་ལྷ་དང་། མི་རྣམས་ནི་མེས་མི་ཚིག །མཚོན་གྱིས་མི་ཚུགས། ཆུས་མི་ཁྱེར། དུག་གིས་མི་ཚུགས། ཕ་རོལ་གྱི་དམག་ཚོགས་ཀྱིས་ཟིལ་གྱིས་མི་ནོན། བདེ་བར་གཉིད་ཀྱིས་ལོག་པར་འགྱུར། བདེ་བར་སད་པར་འགྱུར། རང་གི་བསོད་ནམས་ཀྱིས་ཡོངས་སུ་བསྲུང་བར་འགྱུར། བསོད་ནམས་ཆེན་པོའི་གཟི་བརྗིད་ཀྱིས་གཟི་ཅན་དུ་འགྱུར། ལྷ་དང་བཅས་པའི་འཇིག་རྟེན་གྱིས་མི་བརྫི་བར་འགྱུར། མཛེས་པ་དང་མཐོང་ན་དགའ་བར་འགྱུར། འགྲོ་བ་ཐམས་ཅད་དུ་ཐགས་ཐོགས་མེད་པར་འགྱུར། སྡུག་བསྔལ་ཐམས་ཅད་རྒྱུན་ཆད་པ་དང་རབ་ཏུ་དགའ་བར་འགྱུར། བདེ་བ་ཐམས་ཅད་དང་ལྡན་པ་དང་། མིའི་ཆོས་བླ་མ་རྟོགས་ནས། ལུས་ཞིག་པའི་འོག་ཏུ་ཚངས་པའི་འཇིག་རྟེན་དུ་སྐྱེ་བར་འགྱུར་ཏེ། ཀླུའི་བདག་པོ་ལྷ་དང་མིའི་བྱམས་པ་ལ་གནས་པ་རྣམས་ཀྱི་ཕན་ཡོན་ནི་འདི་དག་གོ། །ཀླུའི་བདག་པོ་དེ་བས་ན་བྱམས་པའི་ལུས་ཀྱི་ལས་དང་། བྱམས་པའི་ངག་གི་ལས་དང་། བྱམས་པའི་ཡིད་ཀྱི་ལས་ཀྱིས་གནས་པར་བྱའོ། །ཀླུའི་བདག་པོ་གཞན་ཡང་བདེ་བ་ཐམས་ཅད་སྦྱིན་པ་ཞེས་བྱ་བའི་གཟུངས་བརྗོད་པར་བྱ་སྟེ། དེས་ཀླུ་ཐམས་ཅད་ཀྱི་ཀླུའི་སྡུག་བསྔལ་ཐམས་ཅད་རྒྱུན་གཅོད་པར་བྱེད། བདེ་བ་ཐམས་ཅད་སྦྱིན་པར་བྱེད་དེ། འཛམ་བུའི་གླིང་འདིར་དུས་དུས་སུ་ཆར་གྱི་རྒྱུན་འབེབས་པར་བྱེད། རྩྭ་དང་། ཤིང་གེལ་པ་དང་། སྨན་དང་། ནགས་ཚལ་དང་། ལོ་ཐོག་ཐམས་ཅད་སྐྱེ་བར་བྱེད་དོ། །ཀླུའི་བདག་པོ་དེ་ལ་བདེ་བ་ཐམས་ཅད་སྦྱིན་པ་ཞེས་བྱ་བའི་གཟུངས་དེ་གང་ཞེ་ན། ཏདྱ་ཐཱ། ངྷཱ་ར་ཎི་དྷཱ་ར་ཎི། ཨུདྡྷཱ་ར་ཎི། སམ་པྲ་ཏིཥྛི་ཏེ། བི་ཛ་ཡ་བརྞ། ས་ཏྱ་པྲ་ཏི་ཛྙཱ་ས་ར་དཱ། ཛྙཱ་ན་བ་ཏི། ཨུ་ཏ་པ་ད་ནི་བི་ནཱ་མ་ནི། ཨ་བྷི་ཥེ་ཙ་ནི། ཨ་བྷི་བྱཱ་ཧཱ་ར། ཤུ་བྷ་བ་ཏི། ཛཱི་མ་ཏུ། ཨེ་ཧི་ཀུ་མ་བཱ་ལན། བ་ཧ་ཧ་ར། ཀླེ་ཤཱ་ན་དྷུ་ན་པཱ་བ་མ། ཤོ་དྷ་ཡ་མཱརྒ་ན། ནི་རཱ་ཧི་ཀ་དྷརྨཱ་ཏཱ་ཤུད་དྷཱ། ལོ་ཀ་བི་ཏི་མི་ར་རཱ་ཛ་ས་དུ་ཁ། ཤ་མན་སརྦ་བུདྡྷ་ཨ་བ་ལོ་ཀ་ན་ཨ་དྷི་ཥྛི་ཏེ། པྲ་ཛྙཱ་ཛྙཱ་ན་བྷེ་སྭཱ་ཧཱ། ཀླུའི་བདག་པོ་གཟུངས་ཀྱི་ཚིག་དོན་ཐམས་ཅད་སྒྲུབ་པར་བྱེད་པ་ཆོས་ཀྱི་སྐུ་ལ་འཇུག་པ། སངས་རྒྱས་ཐམས་ཅད་ཀྱིས་བྱིན་གྱིས་བརླབས་པ། བདེ་བ་ཐམས་ཅད་སྦྱིན་པ་འདི་དག་གཟུང་བར་བྱ། བརྗོད་པར་བྱའོ། །ཀླུའི་བདག་པོ་གཞན་ཡང་སྤྲིན་གྱི་རིགས་ལས་འབྱུང་བའི་བྱིན་གྱི་རླབས་བཀོད་པ་གཟི་བརྗིད་ཀྱི་སྙིང་པོ་སྤྲུལ་པ་སྣང་བར་བྱེད་པ། ཡེ་ཤེས་ཀྱི་ཏོག་གི་ཆུའི་རྒྱུན་གྱི་སྤྲིན་གྱི་དཀྱིལ་འཁོར་དཔལ་དང་ལྡན་པ་གསེར་གྱི་འོད་ཟེར་རྣམ་པར་སྣང་བར་བྱེད་པ། སྐྲའི་རྩེ་མོའི་ཆ་ཤས་གཅིག་ལས་བྱུང་བའི་རུས་དང་རིགས་ཀྱི་དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱི་མཚན་བཟུང་བར་བྱ། ཡིད་ལ་བྱའོ། །དེ་དག་གིས་ཀླུ་ཐམས་ཅད་དང་། ཀླུའི་རིགས་ཐམས་ཅད་དང་། ཀླུའི་རུས་ཐམས་ཅད་དང་། ཀླུའི་ཆོ་རིགས་ཐམས་ཅད་དང་། ཀླུའི་འབྱུང་བ་ཐམས་ཅད་དང་། ཀླུའི་སྐྱེ་གནས་ཐམས་ཅད་དང་། ཀླུའི་རྒྱལ་པོ་ཐམས་ཅད་དང་། ཀླུའི་བུ་མོ་ཐམས་ཅད་དང་། ཀླུའི་གཡོག་ཐམས་ཅད་ཀྱི་ཀླུའི་སྡུག་བསྔལ་ཐམས་ཅད་རྒྱུན་གཅོད་པར་བྱེད། བདེ་བའི་ཡོ་བྱད་ཐམས་ཅད་སྒྲུབ་པར་བྱེད་དོ། །ཀླུའི་བདག་པོ་དེ་ལ་དེ་བཞིན་གཤེགས་པའི་མཚན་དེ་དག་གང་ཞེ་ན། འདི་ལྟ་སྟེ། བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་རྣམ་པར་སྣང་མཛད་སྙིང་པོ་ཅན་གྱི་སྤྲིན་ཆེན་པོ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་ཀླུའི་རིགས་ལས་བྱུང་བ་སྤྲིན་གྱིས་རྣམ་པར་མཛེས་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་གྱི་ཆུ་འཛིན་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དཔལ་ལྡན་སྤྲིན་གྱི་གཟི་བརྗིད་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་འབྱུང་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་རླུང་གི་དཀྱིལ་འཁོར་ཐམས་ཅད་རྣམ་པར་འཇོམས་པའི་སྤྲིན་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་ནམ་མཁའ་དཔག་མེད་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་པོའི་གློག་འབར་བའི་ཚོགས་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་མཐུ་རྩལ་དཔའ་བོ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་མཛེས་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་སྟུག་ཆེན་པོ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་གྱི་དཀྱིལ་འཁོར་ཆེན་པོ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་གྱི་འོད་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་སེང་གེ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་གདུགས་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་བལྟ་སྡུག་སྤྲིན་ཆེན་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་པོས་ཁེབས་པ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ལས་བྱུང་བ་སྣང་བར་བྱེད་པའི་དཀྱིལ་འཁོར་ཕྱོགས་བཅུ་མཐོ་འགེབས་རྔའི་སྒྲ་དབྱངས་ཤིན་ཏུ་རབ་སྒྲོགས་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་བསིལ་འབྱུང་མི་དགའ་རབ་འཇོམས་སྒྲ་ཟབ་སྒྲོགས་པ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་བདལ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་གང་གཱའི་ཆུ་སྤྲིན་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་རྩལ་གྱིས་སྤྲིན་ལྟར་དཔའ་བོ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་འབྲུག་སྒྲ་བསྒྲགས་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཀུན་དུ་སྟོན་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་རྒྱ་ཆེན་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་འཛག་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ལྟར་རྣམ་པར་སྒྲོགས་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ལྟར་འཕགས་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ལྟ་བུའི་ན་བཟའ་མནབས་པ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་གྱིས་ལོ་ཏོག་སྐྱེད་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་གྱི་གནས་ལ་གནས་པ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་ཆར་སྤྲིན་གྱི་སྤྲིན་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་འདུ་བར་མཛད་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་རྣམ་པར་རྒྱས་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་ཨུད་པ་ལ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་སྤོས་ཀྱི་སྐུ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་འཕགས་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་དབང་ཕྱུག་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་འོད་མཛད་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་བྱིན་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་ནོར་བུ་རིན་པོ་ཆེ་ཆེན་པོའི་སྤྲིན་ལྟ་བུ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ལྟར་རྣམ་པར་འཇོམས་པ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་གྱི་ཆུ་བོ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་སྡུད་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དུས་མ་ཡིན་པའི་སྤྲིན་དང་ལྕེ་འབབ་པ་རྣམ་པར་འཇོམས་པ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་པོ་ལྟར་ནམ་མཁའ་མཐོང་ནས་བསྒྲགས་པ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་སྒྲ་འབྱིན་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་ཆར་ཆུ་སྐྱེས་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྙོམས་པའི་གཟི་བརྗིད་སྒྲུབ་པའི་སྤྲིན་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་རོ་བྲོ་བ་དྲུག་དང་ལྡན་པའི་ཆུ་ཆེན་པོའི་སྤྲིན་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་ཆུ་འཐོར་སྤྲིན་ཆེན་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་ཆེན་རྒྱ་མཚོ་འགེངས་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་ཆར་མི་འབབ་པ་ན་སྤྲིན་གྱིས་ཀུན་དུ་འགེབས་པ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་སྤྲིན་གཟུགས་མཐའ་ཡས་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་སྤྲིན་ཆེན་པོ་ཐམས་ཅད་རྣམ་པར་སྒྲུབ་པ། འཛམ་བུའི་སྤྲིན་གྱི་དཔལ་གྱི་གཟི་བརྗིད་འོད་མཛད་པ་སྤྲིན་གྱི་འོད་ཟེར་ཅན་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །ཀླུའི་བདག་པོ་དེ་བཞིན་གཤེགས་པའི་མཚན་འདི་དག་བརྗོད་པས་ཀླུ་ཐམས་ཅད་ཀྱི་ཀླུའི་སྡུག་བསྔལ་ཐམས་ཅད་རྒྱུན་འཆད་པ་དང་། བདེ་བ་ཐམས་ཅད་དང་ལྡན་པ་དང་། འཛམ་བུའི་གླིང་འདིར་དུས་དུས་སུ་ཆར་གྱི་རྒྱུན་འབབ་པ་དང་། རྩྭ་དང་། ཤིང་གེལ་པ་དང་། སྨན་དང་། ནགས་ཚལ་དང་། ལོ་ཐོག་ཐམས་ཅད་སྐྱེད་པར་བྱེད་དོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ལ་སྟོང་གསུམ་གྱི་སྟོང་ཆེན་པོའི་ཀླུའི་བདག་པོ་གཡོག་མཐའ་ཡས་རྒྱ་མཚོའི་སྤྲིན་བཀོད་པའི་གཟི་བརྗིད་ཀྱི་དཀྱིལ་འཁོར་གདུགས་ལྟ་བུའི་རྒྱལ་པོས་གསོལ་བ་བཏབ་པ། བཅོམ་ལྡན་འདས་སླད་མའི་ཚེ་སླད་མའི་དུས་ན་འཛམ་བུའི་གླིང་འདིར་ཆར་མི་འབབ་པའི་ཚེ་གང་དག་བརྗོད་ན་ཆར་གྱི་རྒྱུན་ཆེན་པོ་འབབ་པར་བགྱིད་པ་དང་། མི་བཟད་པའི་དུས་ཀྱི་ཚེ་དང་། གནོད་པ་འབྱུང་བའི་དུས་ཀྱི་ཚེ་དང་། ཆོས་མ་ལགས་པས་སྐྱེ་བོ་འཐབ་པའི་དུས་ཀྱི་ཚེ་དང་། ཡམས་ཀྱི་ནད་དང་། གནོད་པ་འབྱུང་བའི་དུས་ཀྱི་ཚེ་དང་། ནད་དང་འཆི་བའི་དུས་ཀྱི་ཚེ་དང་། སྐར་མ་མ་རུངས་པ་གནས་པའི་དུས་ཀྱི་ཚེ། ཡམས་ཀྱི་ནད་དང་། འཚེ་བ་དང་། གནོད་པ་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་བགྱིད་པ་དེ་ལྟ་བུའི་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་རྣམས་བཤད་དུ་གསོལ། སངས་རྒྱས་ཀྱི་བྱིན་གྱིས་རླབས་ཀྱིས་བྱིན་གྱིས་བརླབས་པར་གསོལ། གང་དག་ཀླུ་ཐམས་ཅད་ལ་ཡང་དག་པར་བསྐུལ་མ་འདེབས་པར་འགྱུར་བ་དང་། ལྷ་ཐམས་ཅད་རབ་ཏུ་དགའ་བར་འགྱུར་བ་དང་། བདུད་ཐམས་ཅད་རྣམ་པར་འཇོམས་པར་འགྱུར་བ་དང་། སེམས་ཅན་ཐམས་ཅད་ཀྱི་འཇིགས་པ་ཐམས་ཅད་འཇིག་པར་འགྱུར་བ་དང་། གནོད་པ་ཐམས་ཅད་ཟློག་པར་འགྱུར་བ་དང་། ཞི་བ་དང་བདེ་ལེགས་སུ་འགྱུར་བ་བགྱིད་པ་དང་། རྒྱུ་སྐར་མ་རུངས་པས་བགྱིས་པ་རབ་ཏུ་ཞི་བར་འགྱུར་བ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་ཆར་པའི་བར་ཆད་ལྔ་གསུངས་པ་གང་ལགས་པ་དེ་དག་ཐམས་ཅད་སེལ་བར་འགྱུར་བ་དང་། འཛམ་བུའི་གླིང་འདིར་ཆར་པའི་རྒྱུན་ལེགས་པར་འབབ་པར་འགྱུར་བ་དེ་ལྟ་བུའི་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་རྣམས་བཅོམ་ལྡན་འདས་མཆོག་ཏུ་ཐུགས་རྗེ་དང་ལྡན་པ། སེམས་ཅན་ཐམས་ཅད་ལ་ཕན་པར་ཐུགས་བརྩེ་བ་དང་ལྡན་པས་བཤད་དུ་གསོལ། བཅོམ་ལྡན་འདས་བདག་དེ་བཞིན་གཤེགས་པ་ལ་གསོལ་པ་འཚལ་ལོ། །དེ་སྐད་ཅེས་གསོལ་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་ཀླུའི་རྒྱལ་པོ་གཡོག་མཐའ་ཡས་རྒྱ་མཚོའི་སྤྲིན་བཀོད་པའི་གཟི་བརྗིད་ཀྱི་དཀྱིལ་འཁོར་གདུགས་ལྟ་བུའི་རྒྱལ་པོ་ལ་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། །ཀླུའི་བདག་པོ་ཁྱོད་སེམས་ཅན་ཐམས་ཅད་ཀྱི་དོན་དང་། ཕན་པ་དང་། བདེ་བའི་ཕྱིར་དེ་བཞིན་གཤེགས་པ་ལ་གསོལ་བ་འདེབས་པ་ལེགས་སོ་ལེགས་སོ། །ཀླུའི་བདག་པོ་དེའི་ཕྱིར་ལེགས་པར་རབ་ཏུ་ཉོན་ལ་ཡིད་ལ་ཟུང་ཤིག་དང་། སྙིང་རྗེ་ཆེན་པོ་ལས་བྱུང་བ་འབྲུག་ཆེན་པོའི་སྒྲ་ཅན་བསྒྱིངས་པ་དཔའ་བའི་ཏོག་ཅེས་བྱ་བའི་གཟུངས་སེམས་ཅན་ཐམས་ཅད་ཀྱི་དོན་དང་། ཕན་པ་དང་། བདེ་བའི་ཕྱིར་སངས་རྒྱས་ཐམས་ཅད་ཀྱིས་གསུངས་པ། བྱིན་གྱིས་བརླབས་པ། རྗེས་སུ་ཡི་རང་བ་བཤད་པར་བྱ་སྟེ། དེ་ནི་ཆར་མེད་པའི་ཚེ་ཆར་འབབ་པར་བྱེད། ཆར་ཆེས་ན་གཅོད་འཆི་བའི་གནོད་པ་རབ་ཏུ་ཞི་བར་བྱེད། ཀླུ་ཐམས་ཅད་སྐུལ་བར་བྱེད། ལྷ་ཐམས་ཅད་སིམ་པར་བྱེད། བདུད་ཐམས་ཅད་རྣམ་པར་འཇོམས་པར་བྱེད། སེམས་ཅན་ཐམས་ཅད་བདེ་བ་ཐམས་ཅད་དང་ལྡན་པར་བྱེད་དོ། །ཏདྱ་ཐཱ། མ་ཧཱ་ཛྙཱ་ན་ཨ་བ་བྷཱ་སཱ་ནི། ཤྲཱི་ཏེ་ཛོ་ལཀྵྨི་བྲྀཥྚི་པིཀྲ་མ། བཛྲ་སིང་ཧ་ཏ་ནེ། པ་ར་མ་བི་རཱ་ཛ། ནིརྨ་ལ་གུ་ཎ་ཀེ་ཏུ་སཱུརྱ་པྲ་བྷེ། བི་མ་ལཱང་ག་ཡཥྚི། བྷ་ར་བྷ་ར། སམ་བྷ་ར་སམ་བྷ་ར། དུ་དུམ་བུ། ཧ་ན་ཧ་ན། མ་ཧ་པྲ་བྷེ། བི་བྷཱུ་ཏ་མ་ཧཱ་མ་ཧོ་ཨན་དྷ་ཀཱ་རེ། པྲཛྙཱ་གྲ་ཤུད་དྷེ། པ་རི་པཱུ་རྞྞ་མཻ་ཏྲེ། མཻ་ཏྲ་བི་ར་ཎ་སྐན་དྷ། མཻ་ཏྲཱ་མ་བུདྡྷ་རེ། ཛ་ལ་ཛ་ལ། ཛ་ལ་ཛ་ལ། ཛ་ལམ་བུདྡྷ་རེ། བོ་དྷྱད་ག་ཀུ་སུ་མེ། ད་ཤ་བ་ལེ། ཙ་ཏུར་བེ་ཤྭ་ར་དྷེ། ཨ་ཤྚཱ་ད་ཤ། ཨ་པ་ནི་ག་བུད་དྷ་དྷ་རྨེ། ཤུ་བྷ་མ་དི། པུ་ཎྱ་རཱ་ཤི། ཤུ་ཀླ་དྷར་མ་ས་མ་ནྭི་ཏེ། གམ་བྷཱི་རེ། བི་ར་ཛ་སེ། བི་མ་ལེ་བི་ཤི་ཥ་པྲཱ་པྟེ། ནི་རཱ་སྲ་པ་དྷརྨེ། སརྦ་ལོ་ཀཛྱཻཥྛ། ཛེཥྛ་པར་པྲ་བ་རེ། ཨ་ནུཏྟ་རེ། ཨ་སང་གེ །དྷ་ར་དྷ་ར། དྷི་རི་དྷི་རི། དྷུ་རུ་དྷུ་ར་ཤཱནྟ་མ་ཏེ། ཤཱན་ཏ་བེ་གེ །ཙ་ར་ཙ་ར། ཙ་ར་ཙ་ར། ཙི་རི་ཙི་རི། ཙུ་རུ་ཙུ་རུ། པ་ར་མ་བུད་དྷཱ་ནུ་མ་ཏེ། མ་ཧ་པྲ་ཛྙཱ་པཱ་ར་མི་ཏེ་སྭཱ་ཧཱ། དེ་བཞིན་གཤེགས་པ་རྣམ་པར་སྣང་མཛད་ཡེ་ཤེས་རྒྱ་མཚོ་དང་ལྡན་པ་ལ་ཕྱག་འཚལ་ལོ། །སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །དེ་ནས་འཛམ་བུའི་གླིང་འདིར་ཆར་དབབ་པའི་ཕྱིར་ཀླུ་ཐམས་ཅད་ཀྱི་སྙིང་པོ་བརྗོད་པར་བྱའོ། །སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ཀྱི་བདེན་པས་ཆར་པའི་བར་ཆད་བྱེད་པ་ལྔ་བསལ་བར་བྱའོ། །སདྱཱ་ཐེ་དན། ས་ར་ས་ར། སི་རི་སི་རི། སུ་རུ་སུ་རུ་ནཱ་གཱ་ནཱན། ཛ་བ་ཛ་བ། ཛི་བི་ཛི་བི། ཛུ་བུ་ཛུ་བུ། ཀླུ་ཆེན་པོ་དག་ཚུར་ཤེག །སངས་རྒྱས་ཀྱི་བདེན་པས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག །ཙ་ར་ཙ་ར། ཙི་རི་ཙི་རི། ཙུ་རུ་ཙུ་རུ། ཀླུའི་བདག་པོ་ཆེན་པོ་དག་ཚུར་ཤེག །ཀྱེ་ཀླུ་ཆེན་པོ་དག །སངས་རྒྱས་ཀྱི་བདེན་པས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག །བ་ར་བ་ར། བི་རི་བི་རི། བུ་རུ་བུ་རུ། སངས་རྒྱས་ཀྱི་བདེན་པས་ཀླུ་ཐམས་ཅད་དགུག་པར་བྱའོ། །བྱམས་པའི་སེམས་དང་། སྙིང་རྗེའི་སེམས་དང་། དགའ་བའི་སེམས་དང་། བཏང་སྙོམས་ཀྱི་སེམས་དང་། སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ཀྱི་བྱིན་གྱི་རླབས་དང་། བདེན་པ་དང་། བྱམས་པ་ཆེན་པོས་ཀླུའི་བདག་པོ་ཆེན་པོ་དག་ཚུར་ཤེག །སངས་རྒྱས་རྣམས་དང་། སངས་རྒྱས་ཀྱི་ཆོས་རྣམས་དང་། བྱང་ཆུབ་སེམས་དཔའི་ཡོན་ཏན་རྣམས་དྲན་པར་གྱིས་ཤིག །བྷ་ར་བྷ་ར། བྷི་རི་བྷི་རི། བྷུ་རུ་བྷུ་རུ། ཆུ་ཆེན་པོའི་ཆུ་དང་། སྤྲིན་གྱི་ཆུ་འཆང་བ་ཅན་ལག་གིས་འགྲོ་ཆེན་པོའི་གཡོག་རྣམས་བྱམས་པའི་སེམས་ཀྱིས་ཚུར་ཤེག །སྟོན་པའི་བསྟན་པ་དམ་པ་དྲན་པར་གྱིས་ཤིག །གྷ་ཌ་གྷ་ཌ། གྷི་ཌི་གྷི་ཌི། གྷུ་ཌུ་གྷུ་ཌུ། དྲག་ཤུལ་ཅན་ཁྲོ་བ་ཤུགས་ཆེན་པོ་དང་ལྡན་པ། དུག་ཆེན་པོའི་ལྕེ་འབྱིན་པ་རྣམས་བྱམས་པའི་སེམས་དང་ལྡན་པར་ཚུར་ཤེག །དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་བདེན་པས་འཛམ་བུའི་གླིང་འདིར་ཆར་ཕོབ་ཤིག །སྭཱ་ཧཱ། ཏ་ཊ་ཏ་ཊ། ཏི་ཊི་ཏི་ཊི། ཏུ་ཊུ་ཏུ་ཊུ། ནོར་བུ་ཆེན་པོའི་ཅོད་པན་ཐོགས་པ་དང་ལྡན་པ་སྦྲུལ་གདུག་པའི་གཟུགས་ཅན་རྣམས་དཀོན་མཆོག་གསུམ་གྱི་བྱིན་གྱི་རླབས་དྲན་པར་གྱིས་ལ། རྡོ་རྗེ་འཆང་གི་བདེན་པས་འཛམ་བུའི་གླིང་འདིར་ཆར་ཕོབ་ཤིག །སྭཱ་ཧཱ། ཀ་ལ་ཀ་ལ། ཀི་ལི་ཀི་ལི། ཀུ་ལུ་ཀུ་ལུ། ཆུ་ཆེན་པོ་ལ་གནས་པ་དང་ལྡན་པ་སྤྲིན་ཆེན་པོ་བརྩེགས་པའི་བཞོན་པ་ལ་ཞོན་ནས་མངོན་པར་འགྲོ་བ་རྣམས་ཚུར་ཤེག །བྱམས་པའི་སེམས་ཀྱིས་འཛམ་བུའི་གླིང་འདིར་ཆར་གྱི་རྒྱུན་ཐོང་ཤིག །དེ་བཞིན་གཤེགས་པའི་བདེན་པ་དང་། དེ་བཞིན་གཤེགས་པའི་བྱིན་གྱི་རླབས་ཀྱིས་ལག་ན་རྡོ་རྗེས་བསྒོ་བ། ར་ལ་ར་ལ། རི་ལི་རི་ལི། རུ་ལུ་རུ་ལུ། ལག་གིས་འགྲོ་བ་ཐམས་ཅད་ཁྱེད་རྣམས་གཉིད་མེད་པར་དུས་གསུམ་གྱི་དེ་བཞིན་གཤེགས་པའི་བདེན་པས། གྷ་མ་གྷ་མ། གྷི་མི་གྷི་མི། གྷུ་མུ་གྷུ་མུ་སྭཱ་ཧཱ། ཀླུ་ཐམས་ཅད་དགུག་གོ། །བྱམས་པའི་སེམས་ཀྱིས་བྱང་ཆུབ་ཀྱི་སེམས་སྔོན་དུ་འགྲོ་བས། ན་ར་ན་ར། ནི་རི་ནི་རི། ནུ་རུ་ནུ་རུ། སྭཱ་ཧཱ། ཡོན་པོ་སྣ་ཚོགས་སུ་གྱུར་མགོ་སྟོང་པ། མགོ་དང་མིག་དམར་བ། སྟོབས་ཆེ་བའི་ལྟོ་འཕྱེ་ཆེན་པོ་རྣམས་དགུག་པར་བྱའོ། །ཀྱེ་ཀྱེ་ལག་གིས་འགྲོ་ཆེན་པོ་དག་མཆོག་ཏུ་ཐུགས་རྗེ་ཅན། བསོད་ནམས་ཐམས་ཅད་ཀྱི་གཟི་བརྗིད་ཀྱིས་བརྗིད་པ་ཉོན་མོངས་པ་དང་བྲལ་བ། དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱི་བྱིན་གྱི་རླབས་དྲན་པར་གྱིས་ཤིག །ག་ཊ་ག་ཊ། གི་ཊི་གི་ཊི། གུ་ཊུ་གུ་ཊུ། སྭཱ་ཧཱ། སྟོབས་དང་། ཕ་རོལ་གནོན་པ་དང་། གཟི་བརྗིད་ཐོགས་པ་མེད་པ་དང་ལྡན་པ་དག་གིས་འཛམ་བུའི་གླིང་འདིར་ཆར་གྱི་རྒྱུན་རྣམས་ཕོབ་ཤིག །ཤ་ར་ཤ་ར། ཤི་རི་ཤི་རི། ཤུ་རུ་ཤུ་རུ། སྭཱ་ཧཱ། ཀྱེ་ཀྱེ་ཀླུ་ཆེན་པོ་རྣམས་རང་རང་གི་རིགས་དང་། ཆོ་རིགས་དྲན་པར་གྱིས་ལ་མ་ཤོལ་བར་ལྷ་ཐམས་ཅད་ཀྱི་བདེན་པའི་བྱིན་གྱི་རླབས་ཀྱིས་འཛམ་བུའི་གླིང་འདིར་ཆར་གྱི་རྒྱུན་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཚངས་པའི་བདེན་པའི་བྱིན་གྱི་རླབས་ཀྱིས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། བརྒྱ་བྱིན་གྱི་བདེན་པས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། རྒྱལ་པོ་ཆེན་པོ་བཞིའི་བདེན་པས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུ་ཆེན་པོ་དག་བརྒྱད་པོའི་བདེན་པས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུ་ཆེན་པོ་དག་རྒྱུན་དུ་ཞུགས་པའི་བདེན་པས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུ་ཆེན་པོ་དག་ལན་ཅིག་ཕྱིར་འོང་བའི་བདེན་པས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུ་ཆེན་པོ་དག་ཕྱིར་མི་འོང་བའི་བདེན་པས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུ་ཆེན་པོ་དག་དགྲ་བཅོམ་པའི་བདེན་པས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུ་ཆེན་པོ་དག་རང་སངས་རྒྱས་ཀྱི་བདེན་པས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུ་ཆེན་པོ་དག་སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ཀྱི་བདེན་པས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུ་ཆེན་པོ་དག་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་བདེན་པ་དང་བྱིན་གྱི་རླབས་ཀྱིས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ལྷ་ཐམས་ཅད་ཀྱི་བདེན་པས་གནོད་པ་ཐམས་ཅད་ཞི་བར་གྱིས་ཤིག་སྭཱ་ཧཱ། ཀླུ་ཐམས་ཅད་ཀྱི་བདེན་པས་ས་ཆེན་པོ་འདིར་རབ་ཏུ་ཆར་ཕོབ་ཤིག་སྭཱ་ཧཱ། གནོད་སྦྱིན་ཐམས་ཅད་ཀྱི་བདེན་པས་སེམས་ཅན་ཐམས་ཅད་སྲུངས་ཤིག་སྭཱ་ཧཱ། དྲི་ཟ་ཐམས་ཅད་ཀྱི་བདེན་པས་མི་རྣམས་ཀྱི་འཁྲུག་པ་དང་གནོད་པ་ཐམས་ཅད་སོལ་ཅིག་སྭཱ་ཧཱ། ལྷ་མ་ཡིན་ཐམས་ཅད་ཀྱི་བདེན་པས་རྒྱུ་སྐར་མ་རུངས་པ་ཐམས་ཅད་རྣམ་པར་ཟློག་ཅིག་སྭཱ་ཧཱ། ནམ་མཁའ་ལྡིང་ཐམས་ཅད་ཀྱི་བདེན་པས་ཀླུ་ཐམས་ཅད་བྱམས་པར་བྱོས་ལ་འཛམ་བུའི་གླིང་འདིར་ཆར་ཆེན་པོའི་རྒྱུན་ཕོབ་ཤིག་སྭཱ་ཧཱ། མི་འམ་ཅི་ཐམས་ཅད་ཀྱི་བདེན་པས་སྡིག་པ་ཐམས་ཅད་ཞི་བར་གྱིས་ཤིག་སྭཱ་ཧཱ། སེམས་ཅན་ཐམས་ཅད་ཚིམ་པར་གྱིས་ཤིག་སྭཱ་ཧཱ། ལྟོ་འཕྱེ་ཆེན་པོ་ཐམས་ཅད་ཀྱི་བདེན་པས་ཡངས་ཤིང་རྒྱ་ཆེ་བའི་ཆར་གྱི་རྒྱུན་ཕོབ་ཤིག་སྭཱ་ཧཱ། འདིར་ཆར་ལ་སྒྲིབ་པ་ལྔ་པོ་དག་ཡང་དག་པར་ཟུང་ཤིག་སྭཱ་ཧཱ། མི་ཐམས་ཅད་ཀྱི་བདེན་པས་མི་ཐམས་ཅད་ལ་ཡོངས་སུ་སྲུངས་ཤིག་སྭཱ་ཧཱ། ཀ་རཀ་ར། ཀི་རི་ཀི་རི། ཀུ་རུ་ཀུ་རུ། ད་ར་ད་ར། དི་རི་དི་རི། དུ་རུ་དུ་རུ། ན་ཊ་ན་ཊ། ནི་ཊི་ནི་ཊི། ནུ་ཊུ་ནུ་ཊུ། ཤིན་ཏུ་མགྱོགས་པར་བཞུད་མ། སྤྲིན་ཆེན་པོའི་ཆུ་འཛིན་མ། སྤྲིན་ལྡན་མ། སྤྲིན་ཆེན་ལྡན་མ། སྤྲིན་ཆེན་ཆུ་འཛིན་མ། སྤྲིན་གསལ་མ། སྤྲིན་འབྱུང་མ། དུས་སྤྲིན་མ། སྤྲིན་གྱི་འབྱུང་གནས་མ། སྤྲིན་སྒྲོགས་མ། སྤྲིན་བཞུགས་མ། སྤྲིན་གྱི་ཅོད་པན་ལྡན་མ། སྤྲིན་གྱི་ཕྲེང་བ་མངའ་མ། སྤྲིན་གྱིས་རྣམ་པར་བརྒྱན་མ། སྤྲིན་བཅིབས་མ། སྤྲིན་ལ་གནས་བཅས་མ། སྤྲིན་གྱི་སྙིང་པོ། སྤྲིན་གྱི་རལ་པ་ཅན། སྤྲིན་འོད་མ། སྤྲིན་གྱི་འཁོར་ལྡན་མ། རྒྱ་ཆེན་སྤྲིན་བཞུགས་མ། མཆོད་ཕྱིར་སྤྲིན་ཐོགས་མ། ལོ་ཐོག་འོངས་སྒྲུབ་མ། རི་སུལ་ན་བཞུགས་མ། ཀླུའི་ཡུམ་བཅོམ་ལྡན་འདས་མ། སྤྲིན་ཆེན་དཔལ་ལྡན་སྐར་མདོག་ཅན། བསིལ་རེག་མ། ཚུབ་མ་ཆེན་པོ་སྤྱོད་ཡུལ་མ། ཀླུ་ཆེན་རྣམ་པར་རོལ་མཛད་མ། བཅོམ་ལྡན་འདས་མ། པོན་ཏྲ་བཅུད་ཀྱིས་ལེན་དྲུག་འཛིན་མ། སངས་རྒྱས་ཀྱི་བདེན་པས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་དབབ་ཏུ་གསོལ་སྭཱ་ཧཱ། གྷ་ར་གྷ་ར། གྷི་རི་གྷི་རི། གྷུ་རུ་གྷུ་རུ། གྷི་རི་ནི་གྷརི་ནི། གྷུ་མ་གྷུ་མ་གྷུ་མ་གྷུ་མ་གྷུ་མ་གྷུ་མ་གྷུ་མ་གྷུ་མ་གྷུ་མ་རི། དབུ་དགུ་དང་ལྡན་མ། སྤྲིན་ཆེན་ཕྲེང་ལྡན་མ། ཀློག་ཚོགས་ཕྲེང་ལྡན་མ། ལག་གིས་འགྲོ་ཐམས་ཅད་འཛིན་མ། སྤྲིན་ལྟ་བུའི་ན་བཟའ་མན་བསམ། དུག་དྲག་པོ་ཐམས་ཅད་སྤྱོད་ཡུལ་མ། སྤྲིན་བཀོད་པ་ལ་བཅིབས་མ། འབྲུག་སྒྲོགས་མ། སྒྲ་སྒྲོགས་མ། ཀླུའི་ཚོགས་ལ་སྐུལ་མ། ལྷ་མོ་སྤྲིན་ཆེན་ཕྲེང་ལྡན་མ། ཀླུ་ཐམས་ཅད་ལ་མ་ཤོལ་བར་དེ་བཞིན་གཤེགས་པའི་བྱིན་གྱི་རླབས་ཀྱིས་འཛམ་བུའི་གླིང་འདིར་ཆར་ཕོབ་ཤིག་ཅེས་བསྐུལ་དུ་གསོལ་སྭཱ་ཧཱ། ཡ་ར་ཡ་ར། ཡི་རི་ཡི་རི། ཡུ་རུ་ཡུ་རུ། ཛ་ར་ཛ་ར། ཛི་རི་ཛི་རི། ཛུ་རུ་ཛུ་རུ། ཛྷ་ར་ཛྷ་ར། གུ་ཏུ་གུ་ཏུ། གུ་གུ་ཏུ། ག་ཊ་ག་ཊ། གི་ཊི་གི་ཊྭི། ཧ་ར་ཧ་ར། ཧི་ལི་ཧི་ལི། མུ་ལུ་མུ་ལུ། ཏ་ལ་ཏ་ལ། ཏི་ལི་ཏི་ལི། ཏུ་ལུ་ཏུ་ལུ། ཧ་ན་ཧ་ན། ད་ཧ་ད་ཧ། པ་ཙ་པ་ཙ། གྲྀཧྞ་གྲྀཧྞ། མར་དམར་ད། པྲ་མར་ད་པྲ་བརཤ་བིག་ནན། མ་ཏྲེ་ཡོད་ཉཱ་པ་ཡ་ཏི། སྭཱ་ཧཱ། བུད་དྷེ་བུད་དྷེ། བུ་བུད་དྷེ། སུ་བུད་དྷེ། ཧ་ར་པཱ་པ་ན། སརྦ་ས་ཏྭཱ་ནཱན། ཨ་དྷིཤ་ཡ་པུ་ཎྱ་མ། སརྦ་བུད་དྷ་ནཱན། དྷ་རི་ནི་དྷ་རེ། ཤུ་བྷ་མ་ཏེ། གུ་ཎཱ་གྲཔྲཱ་པ་ནེ། མ་ཧཱ་ཛྙཱ་ནོ་ལ་ཀེ །ཤུཀླ་དྷརྨེ། སངྱ་པྲ་ཏི་ཛྙཱ། མ་ཧཱ་ཡཱ་ནཱ་དྷྱུ་ཥི་ཏེ་ལོ་ཀ་ཛྱཻཥྛ་བྷ་ག་བ་ཏི། བུད་དྷ་མཻ་ཏྲེ་ཨ་ཡུ་བུ་ར་ཡ་སརྦ་ཀྵེ་ཏྲཱ་ནི། ཤུ་ཀླེ་ཤྭེ་ཏྭམ། པ་ར་དཱ་རེ། པ་ན་ཊ་ར་བ་སི་ནི། དྷུ་དྷུ་རུ། ཤ་མ་ཤ་མ། ཤཱནྟ་མཱ་ན་སེ། ཆར་གྱི་བགེགས་ཐམས་ཅད་རྣམ་པར་བསལ་བར་གསོལ་སྭཱ་ཧཱ། དུས་གསུམ་གྱི་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་བདེན་པས་བྱམས་པའི་ཐུགས་ཀྱིས་ཡང་དག་པ་བརྟུལ་ཞུགས་དང་། དཀའ་ཐུབ་ཀྱི་ཐུགས་སྐྱོན་མེད་པས་ཀླུ་ཆེན་པོ་རྣམས་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །ཀླུའི་རྒྱལ་པོ་ཆེན་པོ་གཡོག་མཐའ་ཡས་རྒྱ་མཚོའི་སྤྲིན་བཀོད་པའི་གཟི་བརྗིད་ཀྱི་དཀྱིལ་འཁོར་གདུགས་ལྟ་བུའི་རྒྱལ་པོ་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་རབ་ཏུ་ཆར་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་དགའ་བོ་དང་ཉེ་དགའ་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་རབ་ཏུ་ཆར་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་རྒྱ་མཚོ་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་རབ་ཏུ་ཆར་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་མ་དྲོས་པ་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་རབ་ཏུ་ཆར་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་གཟི་ཅན་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་ཆུ་ལྷ་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་འཇོག་པོ་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་ཡུལ་འཁོར་སྲུང་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་ནོར་རྒྱས་ཀྱི་བུ་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་བཏང་བཟུང་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་ས་སྲུང་གི་བུ་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་ནུ་ཏ་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་དཔལ་གྱི་གཟི་བརྗིད་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་དཔའ་བཟངས་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་གློག་གི་ཕྲེང་བ་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་གཙུག་ན་ནོར་བུ་ཆེན་པོ་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་གཙུག་ན་ནོར་བུ་ཐོགས་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ཀླུའི་རྒྱལ་པོ་སྣང་ལྡན་གཙུག་པུད་ཅན་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། དེ་དག་ལ་སོགས་པ་ཀླུའི་རྒྱལ་པོ་ཐམས་ཅད་ལ་ཡང་དག་པར་བསྐུལ་ལོ། །འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ནཱ་གེ་ན་གེ། །མ་ཧཱ་ནཱ་གེ །གྷོ་ར་མཱ་ན་སེ། ནཱ་ག་ཧྲྀད་ཡ། དྷུ་མཀུ་ལེ། ཨུ་གྲ་རོ་ཥེ། པྲ་ཙཎྜ་ཏེ་ཛོ། བི་ཥོ་གྲེ། ཨཱ་ཤཱ་བི་ཤེ། ཨ་ཧི་གྷོ་རེ། ཀྲིཥྼ་པིང་ག་ལེ། ཙ་ན་ཙ་ལེ། ལོ་ལ་ཛི་ཧྭེ། མ་ཧཱ་ཕ་ན་ཀ་རེ། ཀཱ་ལ་པཱ་ཤེ། རཽཾ་དྲ་བཱ་སི་ནི། ཊུ་ཊུ་མ་བེ། གྷ་ཎ་གྷ་ཎ། ཤི་ཁི་ནི། ཀ་ཎ་ཀ་ཎ། ག་ཎ་ག་ཎ། མ་ཧཱ་ག་ཎ་ག་ཎེ། པ་ར་པ་ར། པི་རི་པི་རི། པུ་རུ་པུ་རུ། བི་སྥུརྫ་ནེ། བྷུ་རུ་བྷུ་རུ། མ་ཧཱ་བྷོ་གེ །མ་ཎི་དྷ་རི། ཕི་རི་ཕི་རི། ཕུ་རུ་ཕུ་རུ། ཕ་ར་ཕ་ར། བར་ཥ་བར་ཥ། ཛ་ལཱམ་བུ་དྷ་རེ། ཤམ་བུ་ཤམ་བུ། བ་ཏ་ཌུ་ཏུམ་པེ། དྷུད་དྷུད་དྷུདདྷུམ་བེ། མེ་གྷ་པྲ་བྷེ། མེ་གྷ་བཱ་ཧིནེ། ཌྷ་ཀ་ཌྷ་ཀ་ཌྷ་ཀ་ཌྷུ་ཌྷུམ་བེ། གྷ་ཎཱ་གྷ་ཎཱ། ཤི་ཀི་ནི། ཀ་ན་ཀ་ན། ག་ཎ་ག་ཎ། མ་ཧཱ་ནཱ་ག་ག་ནེ། ནི་རཱ་ཏྲ་ཎྱ། པ་ཛ་ར་ད་ཀཱ་རི་མ་ཧཱ་ནཱ་གཱ་དྷི་བ་ཏ་ཡེ། གྷུ་མ་གྷུ་མ་གྷུ་མ་པ་ཡ་སྟི་ཀ། ཛཱང་གཱ་རི་བྷུ་ཛང་ཤ་མེ། པི་ཀ་ཏ་སང་ཀ་ཏེ། གྷོ་ར་བི་སྥུརྫ་ནི་བི་ཛྲྀམ་བྷནྟེ། ཀླུ་ཐམས་ཅད་དགུག་གོ། །སངས་རྒྱས་ཐམས་ཅད་ཀྱི་བྱིན་གྱི་རླབས་དང་། དུས་གསུམ་གྱི་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་བདེན་པ་དང་། བྱམས་པའི་ཐུགས་ཀྱིས་འཛམ་བུའི་གླིང་འདིར་ཆར་རབ་ཏུ་ཕོབ་ཤིག་སྭཱ་ཧཱ། ན་མཿསརྦ་བུ་ངྡྷེ་བྷྱཿསིད་དྷན་ཏུ་མནྟྲ་པ་དཱ་ནི་སྭཱ་ཧཱ། འཕགས་པ་སྤྲིན་ཆེན་པོ་ཐེག་པ་ཆེན་པོའི་མདོ། སྟོང་ཕྲག་བརྒྱ་པ་ལས་ཆར་དབབ་པ་རླུང་གི་དཀྱིལ་འཁོར་ཅེས་བྱ་བའི་ལེའུ་སྟེ་དྲུག་ཅུ་རྩ་བཞིའོ།། །།ཆོ་ག་ནི་ཆར་ཆེར་དབབ་པར་འདོད་ན་བླ་གབ་མེད་པའི་ས་ཕྱོགས་གཙང་མར་བླ་རེ་སྔོན་པོ་བྲེ། བ་དན་སྔོན་པོ་དཔྱང་ངོ་། །ཆོས་སྨྲ་བ་ལུས་བཀྲུས་ལ་ཡན་ལག་བརྒྱད་པ་ལ་ཉེ་བར་གནས་ཤིང་གོས་གཙང་མ་སྔོན་པོ་བགོས་ལ། སྤོས་དྲི་ཞིམ་པོས་བསྐུས་ནས་ཟས་དཀར་གསུམ་ཟོས་ལ་སྟན་སྔོན་པོའི་སྟེང་དུ་འདུག་ནས། ཁ་ཤར་ཕྱོགས་སུ་བལྟས་ཏེ། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ལ་གསོལ་བ་བཏབ་ནས་སྤྲིན་ཆེན་པོའི་དཀྱིལ་འཁོར་གྱི་ལེའུ་འདི་ཉིན་མཚན་རྒྱུན་མི་འཆད་པར་བཀླག་པར་བྱའོ། །ཆུ་སྔོན་པོས་བཀང་བའི་བུམ་པ་གང་བ་བཞི་གཞག་པར་བྱའོ། །གཏོར་མ་དང་། བདུག་པ་དང་། མེ་ཏོག་ཅི་ནུས་ཀྱིས་སྦྱར་རོ། །དེ་ལ་ཆོས་སྨྲ་བས་ཕྱོགས་བཞིར་བའི་ལྕི་བའི་ཁུ་བས་བྲི་བར་བྱ་སྟེ། ཤར་ཕྱོགས་སུ་ཁྲུ་གསུམ་ཙམ་དུ་ཀླུའི་རྒྱལ་པོ་མགོ་གསུམ་པ་ཀླུའི་གཡོག་འཁོར་སྣ་ཚོགས་དང་བཅས་པ་བའི་ལྕི་བས་བྲི་བར་བྱའོ། །ལྷོ་ཕྱོགས་སུ་ཁྲུ་ལྔ་ཙམ་དུ་ཀླུའི་རྒྱལ་པོ་མགོ་ལྔ་པ་ཀླུའི་གཡོག་འཁོར་སྣ་ཚོགས་དང་བཅས་པ་བྲི་བར་བྱའོ། །ནུབ་ཕྱོགས་སུ་ཁྲུ་བདུན་ཙམ་དུ་ཀླུའི་རྒྱལ་པོ་མགོ་བདུན་པ་ཀླུའི་གཡོག་འཁོར་སྣ་ཚོགས་དང་བཅས་པ་བྲི་བར་བྱའོ། །བྱང་ཕྱོགས་སུ་ཁྲུ་དགུ་ཙམ་དུ་ཀླུའི་རྒྱལ་པོ་མགོ་དགུ་པ་ཀླུའི་གཡོག་འཁོར་སྣ་ཚོགས་དང་བཅས་པ་བྲི་བར་བྱའོ། །ཆོས་སྨྲ་བས་རང་གི་ལུས་བསྲུངས་ལ་བྱམས་པ་ལ་གནས་ཏེ་སེམས་ཅན་ཐམས་ཅད་ལ་སྙིང་རྗེའི་སེམས་ཉེ་བར་གཞག་ནས་སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་གསོལ་བ་བཏབ་ལ་ཀླུ་རྣམས་དང་དགེ་བའི་རྩ་བ་བགོ་བར་བྱ་སྟེ། འདི་ནི་ཆོ་གའོ། །གང་ཞིག་ཆར་མི་འབབ་པའི་ཚེ། སྤྲིན་ཆེན་པོའི་རླུང་གི་དཀྱིལ་འཁོར་གྱི་ལེའུ་འདི་ཀློག་ན། ཉིན་ཞག་གཅིག་གམ། གཉིས་ནས་ཉིན་ཞག་བདུན་ཚུན་ཆད་གདོན་མི་ཟ་བར་ཆར་འབབ་པར་འགྱུར་རོ། །རྒྱ་མཚོ་ནི་དུས་ལས་ཡོལ་ཀྱང་སྲིད་ཀྱི། ཐུབ་པ་ཆེན་པོའི་གསུང་རབ་ཆར་གྱི་ཆོ་ག་འདི་ནི་གཞན་དུ་མི་འགྱུར་རོ། །དེ་ལྟ་མོད་ཀྱི་ཚུལ་ཁྲིམས་དང་ཡོན་ཏན་ལ་གནས་ཏེ། འཆལ་པའི་ཚུལ་ཁྲིམས་དང་། ཤ་དང་ཉ་ལ་སོགས་པ་ཡོངས་སུ་སྤང་བར་བྱའོ། །འོ་ཐུག་དང་ཞོ་དང་འོ་མ་དང་། འབྲས་ཆན་བུ་རམ་ཅན་ལ་སོགས་པ་དང་། མངར་གསུམ་ཟ་ཞིང་བཀླག་པར་བྱའོ། །དེ་ལྟར་ན་འགྲུབ་པར་འགྱུར་ཏེ་ཐུབ་པའི་རྒྱལ་པོས་གསུངས་པ་བཞིན་དུ་ཐེ་ཚོམ་མེད་པར་བྱའོ། །འཕགས་པ་སྤྲིན་ཆེན་པོ་ཐེག་པ་ཆེན་པོའི་མདོ་ལས་ཆར་དབབ་པ་རླུང་གི་དཀྱིལ་འཁོར་གྱི་ལེའུ་ཞེས་བྱ་བ་དྲུག་ཅུ་རྩ་བཞི་པ་ཆོ་ག་དང་བཅས་པ་རྫོགས་སོ།། །།རྒྱ་གར་གྱི་མཁན་པོ་ཛི་ན་མི་ཏྲ་དང་། ཤཱི་ལེན་དྲ་བོ་དྷི་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་བན་དེ་ཡེ་ཤེས་སྡེས་བསྒྱུར་ཅིང་ཞུས་ཏེ་སྐད་གསར་ཆད་ཀྱིས་ཀྱང་བཅོས་ནས་གཏན་ལ་ཕབ་པ།།
