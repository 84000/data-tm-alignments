རྒྱ་གར་སྐད་དུ། བོ་དྷི་ས་ཏྭ་པྲ་ཏི་མོཀྵ་ཙ་ཏུཥྐ་ནི་ར་ཧཱ་ར་ནཱ་མ་མ་ཧཱ་ཡཱ་ན་སཱུ་ཏྲ། བོད་སྐད་དུ། བྱང་ཆུབ་སེམས་དཔའི་སོ་སོར་ཐར་པ་ཆོས་བཞི་སྒྲུབ་པ་ཞེས་བྱ་བ་ཐེག་པ་ཆེན་པོའི་མདོ། བཅོམ་ལྡན་འདས་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་ཕྱག་འཚལ་ལོ། །འདི་སྐད་བདག་གིས་ཐོས་པ་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་རྒྱལ་པོའི་ཁབ་བྱ་རྒོད་ཀྱི་ཕུང་པོའི་རི་ལ། ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུ་དང་། མཽད་གལ་གྱི་བུ་ལ་སོགས་པ་དགེ་སློང་སྟོང་ཉིས་བརྒྱ་ལྔ་བཅུའི་དགེ་སློང་གི་དགེ་འདུན་ཆེན་པོ་ཐམས་ཅད་ཀྱང་དགྲ་བཅོམ་པ། ཟག་པ་ཟད་པ། ཉོན་མོངས་པ་མེད་པ། དབང་དུ་གྱུར་པ་དང་། བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རབ་ཏུ་མང་པོ་དང་ཐབས་ཅིག་ཏུ་བཞུགས་སོ། །དེ་ནས་ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུ་བཅོམ་ལྡན་འདས་ག་ལ་བ་དེ་ལོགས་སུ་སོང་སྟེ་ཕྱིན་ནས་བཅོམ་ལྡན་འདས་ཀྱི་ཞབས་ལ་མགོ་བོས་ཕྱག་འཚལ་ནས་ཕྱོགས་གཅིག་ཏུ་འདུག་གོ། །ཕྱོགས་གཅིག་ཏུ་འདུག་ནས་ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུས་བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་བདེན་པའི་རྣམ་པའི་བྱ་བ་ནི་གང་། བཅོམ་ལྡན་འདས་ཇི་ལྟར་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་བདེན་པར་སྨྲ་ཞིང་ཡང་དག་པར་སྨྲ་བས་མྱུར་དུ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་འགྱུར་ལགས། དེ་སྐད་ཅེས་གསོལ་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུ་ལ་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། །ཤཱ་རིའི་བུ་ཆོས་བཞི་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་བདེན་པར་སྨྲ་ཞིང་ཡང་དག་པར་སྨྲ་བ་ཡིན་ཏེ། མྱུར་དུ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་འགྱུར་རོ། །བཞི་གང་ཞེ་ན། ཤཱ་རིའི་བུ་འདི་ལ་རིགས་ཀྱི་བུའམ། རིགས་ཀྱི་བུ་མོ་དགེ་བའི་ཆོས་ཐམས་ཅད་ཡོངས་སུ་བཙལ་བའི་ཕྱིར་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་སེམས་བསྐྱེད་ནས་བརྩོན་འགྲུས་བརྩམས་པས་གནས་པ་ཡིན་ཏེ། ཤཱ་རིའི་བུ་དཔེར་ན་སྐྱེས་བུ་གྲོང་ཁྱེར་གྱི་ཚད་འཇལ་བར་འདོད་པ་དེ་གྲོང་ཁྱེར་གྱི་བྱ་བ་ལ་བརྩོན་པ་སྐྱེད་པར་བྱེད་དོ། །ཤཱ་རིའི་བུ་སྐྱེས་བུ་དེའི་གྲོང་ཁྱེར་དེ་ཡང་དག་པར་འགྱུར་བ་དང་། མི་རྣམས་གྲོང་ཁྱེར་དུ་གནས་པར་འགྱུར་རོ། །ཞེས་རེ་བར་བྱ་བ་ཡིན་ལ། གྲོང་ཁྱེར་དུ་སོ་སོར་འཇོག་པར་འདོད་པས་ཀྱང་ཡང་དག་པའི་ཤེས་རབ་ལ་བརྟེན་ནས་གནས་ཐམས་ཅད་ཡོངས་སུ་འཛིན་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་དེ་བཞིན་དུ་དུས་གང་གི་ཚེ་དགེ་བའི་ཆོས་ཐམས་ཅད་ཡོངས་སུ་བཙལ་བར་བྱ་བའི་ཕྱིར་བྱང་ཆུབ་སེམས་དཔའ་བརྩོན་འགྲུས་རྩོམ་པ་ལ་གནས་ཏེ་ཆོས་ཐམས་ཅད་ཡོངས་སུ་འཛིན་པར་བྱེད་དོ། །ཤཱ་རིའི་བུ་གང་གིས་མྱུར་དུ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བ་འདི་ནི་བྱང་ཆུབ་སེམས་དཔའི་བདེན་པའི་རྣམ་པ་དང་པོའོ། །ཤཱ་རིའི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་ཆོས་ཐོས་པ་ནན་ཏན་གྱིས་སྒྲུབ་པར་བྱེད་པ་གང་ཡིན་པ་དེ་ནི་དགེ་བའི་ཆོས་རྣམས་མྱུར་དུ་སྨིན་པར་འགྱུར་ཏེ། ཤཱ་རིའི་བུ་འདི་ལྟ་སྟེ། དཔེར་ན་སྐྱེས་བུ་ནད་ཀྱིས་ཐེབས་པར་གྱུར་པ་དེ་ལ་སྨན་པས་སྨན་ཉེ་བར་བསྟན་ལ། དེས་ཇི་སྐད་བསྟན་པ་བཞིན་དུ་སོ་སོར་བསྟེན་པས་ནད་དེ་ལས་མྱུར་དུ་ཐར་བར་འགྱུར་རོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། དེས་སྨན་ཚུལ་བཞིན་དུ་སོ་སོར་བསྟེན་པའི་ཕྱིར་རོ། །ཤཱ་རིའི་བུ་དེ་བཞིན་དུ་བྱང་ཆུབ་སེམས་དཔའ་ཆོས་ཐོས་པ་ནན་ཏན་གྱིས་སྒྲུབ་པར་བྱེད་པ་གང་ཡིན་པ་དེ་ནི་དགེ་བའི་ཆོས་རྣམས་མྱུར་དུ་ཡོངས་སུ་སྨིན་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་གང་གིས་མྱུར་དུ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བ་འདི་ནི་བྱང་ཆུབ་སེམས་དཔའི་བདེན་པའི་རྣམ་པ་གཉིས་པའོ། །ཤཱ་རིའི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་ཆོས་ཐམས་ཅད་ལ་གཞན་གྱི་འདུ་ཤེས་སྐྱེད་པར་བྱེད་དེ་ཆོས་ཅུང་ཟད་ཀྱང་མི་ལེན་ཏོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། ཉེ་བར་ལེན་པ་ལས་ནི་འཇིགས་པ་འབྱུང་བས་ཆོས་ཐམས་ཅད་ལ་འཇིགས་པའི་འདུ་ཤེས་སྐྱེད་པར་བྱེད་དེ། འཇིགས་པའི་འདུ་ཤེས་སྐྱེད་པ་ལས་ནི་ཡིད་རྟོན་པར་མི་བྱེད་དོ། །ཡིད་མི་རྟོན་པ་ཡོད་ན་ཇི་ལྟར་ཐར་པར་བྱ། ཇི་ལྟར་ན་བདེ་ལེགས་སུ་གནས་པར་འགྱུར་ཞེས་ལམ་ཚོལ་བར་བྱེད་དོ། །ཤཱ་རིའི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་སྐྱེ་བ་དང་། རྒ་བ་དང་། ན་བ་དང་། འཆི་བ་དང་། མྱ་ངན་དང་། སྨྲེ་སྔགས་འདོན་པ་དང་། སྡུག་བསྔལ་བ་དང་། ཡིད་མི་བདེ་བ་དང་། འཁྲུག་པ་རྣམས་ཀྱིས་ཡོངས་སུ་བསྐོར་བ་ཡོད་ན་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་ཇི་ལྟར་ཐར་བར་བྱ། ཇི་ལྟར་ན་བདེ་ལེགས་སུ་གནས་པར་འགྱུར་ཞེས་ལམ་ཡོངས་སུ་ཚོལ་ཞིང་ཕྱོགས་བཞིར་ལེགས་པར་གསུངས་པ་ཡོངས་སུ་ཚོལ་ལོ། །ཤཱ་རིའི་བུ་གང་གིས་མྱུར་དུ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བ་འདི་ནི་བྱང་ཆུབ་སེམས་དཔའི་བདེན་པའི་རྣམ་པ་གསུམ་པའོ། །ཤཱ་རིའི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་སྦྱིན་བདག་ཏུ་འགྱུར་བ་དང་། སྡུག་བསྔལ་གྱི་ཆོས་རྣམས་ཀྱིས་ཡོངས་སུ་མི་སྐྲག་ཅིང་མ་གོས་པས་རྟག་ཏུ་བརྩོན་པ་དང་ལྡན་པ་ཡིན་ཏེ། སེམས་ཅན་གང་སུ་དག་སྡུག་བསྔལ་དང་། ཡིད་མི་བདེ་བ་དང་། འཁྲུག་པ་རྣམས་བྱུང་བ་ཉེ་བར་ཞི་བར་བྱ་བའི་ཕྱིར་བདག་གིས་དེ་དག་བསལ་བར་བྱའོ་ཞེས་སེམས་སྐྱེད་པར་བྱེད་ཅིང་ཇི་ལྟར་བདག་ལ་ཟག་པ་མེད་པའི་བདེ་བ་མྱོང་བ་སྐྱེས་པ་དེ་བཞིན་དུ་བདག་གིས་སེམས་ཅན་ཐམས་ཅད་ལ་བསྒྲུབ་པར་བྱའོ་ཞེས་བརྩོན་འགྲུས་རྩོམ་པར་བྱེད་དོ། །དེ་དེ་ལྟར་བརྩོན་འགྲུས་རྩོམ་པ་ན་རྙེད་པའི་འདུ་ཤེས་སྐྱེད་པར་མི་བྱེད་དོ། །ཤཱ་རིའི་བུ་འདི་ལྟ་སྟེ། དཔེར་ན་སྐྱེས་བུ་དབྱར་ཟླ་འབྲིང་པོའི་ཟླ་བ་ལ་འབྲོག་དགོན་པར་སོང་བ་ན། འབྲོག་དགོན་པ་དེ་ན་ཡང་མི་རྒྱུ་བ་མེད་ལ་ཤིང་མེད་ཅིང་ཆུ་མེད་པ་དང་། དེའི་སྟེང་ནས་ཉི་མའི་དཀྱིལ་འཁོར་ཡང་ཤར་བར་གྱུར། དེར་ས་གཞི་ཡང་རབ་ཏུ་འབར་ཞིང་ཚ་བར་གྱུར་ཏེ། དེའི་མདུན་དུ་ཆུ་བོའི་རྒྱུན་ཆེན་པོ་འབབ་པར་གྱུར་པ་ན་དེ་དེ་ལ་ངོ་མཚར་གྱི་འདུ་ཤེས་སྐྱེད་པར་བྱེད་དོ། །ཤཱ་རིའི་བུ་དེ་བཞིན་དུ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་ཅན་ཐམས་ཅད་ཀྱི་ཕྱིར་སྡུག་བསྔལ་ཟད་པར་འགྱུར་བའི་ལམ་ཇི་ལྟ་བུ་བཟུང་ལ། གལ་ཏེ་བྱང་ཆུབ་སེམས་དཔའ་ལམ་དེ་ཡོངས་སུ་བཟུང་ཡང་བསྐལ་པ་བྱེ་བ་འདས་ནས་ཐ་ན་འདུག་པ་ཙམ་གྱི་བདེ་བ་འདོད་པའི་སེམས་གཅིག་སྐྱེས་ན་ཡང་དེ་ལ་བྱང་ཆུབ་སེམས་དཔས་འདི་ལྟར་སེམས་གཟེངས་བསྟོད་པར་བྱ་སྟེ། བདག་གིས་གང་གཟུང་ནས་འདུག་པ་དེ་ཡང་མང་དུ་ཡིན་ཏེ། སེམས་ཅན་ཐམས་ཅད་ལ་གནོང་བསྐུར་རོ། །ཤཱ་རིའི་བུ་གང་གིས་མྱུར་དུ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བ་འདི་ནི་བྱང་ཆུབ་སེམས་དཔའི་བདེན་པའི་རྣམ་པ་བཞི་པའོ། །ཤཱ་རིའི་བུ་ཅིའི་ཕྱིར་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་པ་ཞེས་བྱ་ཞེ་ན། མ་མཁྱེན་པའི་མཁྱེན་པ་མཁྱེན་པས་མངོན་པར་མཁྱེན་པ་དང་། མཁྱེན་པའི་ཆོས་དེ་ལ་རབ་ཏུ་གནས་པས་སྤྱོད་པ་དེས་ན་མངོན་པར་སངས་རྒྱས་པ་ཞེས་བྱའོ། །བླ་ན་མེད་པ་ཞེས་བྱ་བ་ཡང་དེ་བཞིན་གཤེགས་པས་ཆོས་གང་རྡུལ་ཙམ་ཡང་ལུས་ཤིང་མ་རྒལ་བ་མེད་པ་དེས་ན་དེ་བཞིན་གཤེགས་པ་སྟེ་ཕ་རོལ་ཏུ་བྱོན་པ་ཞེས་བྱའོ། །ཤཱ་རིའི་བུ་ཅིའི་ཕྱིར་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ཞེས་བྱ་ཞེ་ན། ཐུགས་མཉམ་པ་དང་། ཐུགས་རྙོག་པ་མེད་པ་དང་། ཐུགས་མི་འཁྲུགས་པས་ཆོས་རྣམས་མངོན་པར་མཁྱེན་པ་དང་། ཆོས་ཐམས་ཅད་མཉམ་པ་ཉིད་དང་། ཆོས་ཐམས་ཅད་མི་མཉམ་པ་མེད་པར་མངོན་པར་མཁྱེན་པ་དེས་ན་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ཞེས་བྱའོ། །ཤཱ་རིའི་བུ་ཆོས་འདི་བཞི་དང་ལྡན་པའི་བྱང་ཆུབ་སེམས་དཔའ་ནི་བརྩོན་འགྲུས་མི་འཕྲོགས་པར་གྱུར་ཏེ། ཉོན་མོངས་པ་ཐམས་ཅད་ཀྱིས་མི་གོས་པར་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱའོ། །ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའི་ཆོས་བཞི་སྒྲུབ་པ་འདི་ནི་དེ་བཞིན་གཤེགས་པས་མཁྱེན་པ་དང་། གཟིགས་པ་དང་། ཐུགས་སུ་ཆུད་པ་དང་། བརྙེས་པ་དང་། མངོན་སུམ་དུ་མཛད་པས་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བ་ཡིན་ཏེ། བྱང་ཆུབ་སེམས་དཔའ་གནས་དེ་དག་ལས་མ་ཉམས་ཤིང་བསྒྲུབ་པའི་རྣམ་པ་ཀུན་རྫོགས་པས་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་ཆོས་བཞི་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔས་བསྔགས་པའི་བྱ་བ་རྗེས་སུ་འཐོབ་པར་འགྱུར་ཏེ། བཞི་གང་ཞེ་ན། ཤཱ་རིའི་བུ་འདི་ལ་བྱང་ཆུབ་སེམས་དཔའ་ལྷག་པའི་ང་རྒྱལ་ཅན་དུ་མི་འགྱུར་བ་ཡིན། ང་རྒྱལ་ཚར་གཅོད་པའི་ཕྱིར་ཞུགས་པ་ཡིན། ཆགས་པ་མེད་པར་འགྱུར་བ་ཡིན། ཆགས་པ་སྤོང་བའི་ཕྱིར་ཞུགས་པ་ཡིན། ཞེ་སྡང་མེད་པར་འགྱུར་བ་ཡིན། བྱམས་པ་བསྒོམ་པ་ལ་མངོན་པར་དགའ་བ་ཡིན། རྨོངས་པ་མེད་པར་འགྱུར་བ་ཡིན། ཤེས་རབ་ཀྱི་སྟོབས་དང་ལྡན་པ་ཡིན། ཤེས་རབ་ཉེ་བར་རྟོན་པ་ཡིན། ཤེས་རབ་ཀྱིས་སྤྱོད་པ་ཡིན། ཤེས་རབ་ཀྱི་བདག་པོར་གྱུར་པ་ཡིན། ཤེས་རབ་ཀྱིས་ངེས་པར་འབྱུང་བ་ཡིན། རྨོངས་པའི་དྲ་བ་སྤོང་བའི་ཕྱིར་ཞུགས་པ་ཡིན། མུན་པ་མུན་ནག་འཇོམས་པའི་ཕྱིར་ཞུགས་པ་ཡིན། ཤེས་རབ་ཀྱི་རྒྱུ་ལྷུར་ལེན་ཅིང་མཐུ་དང་ལྡན་པ་ཡིན། ཤེས་རབ་ངེས་པར་རྙེད་པ་ཡིན་ཏེ། ཤཱ་རིའི་བུ་ཆོས་དང་པོ་འདི་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་བསྔགས་པའི་བྱ་བ་རྗེས་སུ་འཐོབ་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་ནི་སེམས་ཅན་རྣམས་ཀྱི་ཐར་པའི་བྱ་བ་དོན་དུ་གཉེར་བ་ཡིན་ཏེ། ཤཱ་རིའི་བུ་འདི་ལྟ་སྟེ། དཔེར་ན་སྐྱེས་བུ་ཞིག་བུ་གཅིག་པ་སྡུག་ཅིང་ཡིད་དུ་འོང་བ་མི་འཐུན་པ་མེད་པ་དེ་རྒྱལ་པོས་གསད་སར་བོར་བ་མཐོང་བར་གྱུར་ལ། དེའི་ཕ་དང་མ་ཀྱེ་མའོ་བུ་གཅིག་པ་ཀྱི་ཧུད་བུ་གཅིག་བདག་ཅག་འབོར་རམ་ཞེས་ཀུན་ཏུ་སྨྲ་ཞིང་སྨྲེ་སྔགས་འདོན་པས་རྣམ་པར་སྨྲ་བར་བྱེད་དོ། །ཤཱ་རིའི་བུ་དེ་བཞིན་དུ་དུས་གང་གི་ཚེ་ལྷག་པའི་བསམ་པ་ཕུན་སུམ་ཚོགས་པའི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་ཅན་གང་ཡང་རུང་བ་ཞིག་སྡིག་པ་བྱེད་པ་མཐོང་བ་ན། །ཀྱེ་མའོ་གང་དུ་སེམས་ཅན་ཤེས་རབ་འཆལ་བ་རྣམས་ནི་བདག་གི་ཡང་དག་པའི་འཚོ་བ་ལས་རྣམ་པར་ཉམས་པར་བྱེད་དོ། །ཞེས་དེ་ལ་བྱང་ཆུབ་སེམས་དཔའ་སྙིང་རྗེ་སྐྱེ་བར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་གཞན་བསྲུང་ངོ་སྙམ་ན་བདག་བསྲུང་བར་བྱའོ། །ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་ཕན་པ་འདོད་པ་འདི་ལྟ་བུ་དང་ལྡན་པས་སྲོག་གི་ཕྱིར་ཡང་སྡིག་པའི་ལས་མི་བྱེད་དོ། །ཤཱ་རིའི་བུ་ཆོས་གཉིས་པོ་འདི་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་བསྔགས་པའི་བྱ་བ་རྗེས་སུ་འཐོབ་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་བསྟི་སྟང་གིས་ཆོས་ཚོལ་བར་བྱེད་ཅིང་བསྟི་སྟང་གིས་ཆོས་ཡོངས་སུ་འཛིན་པར་བྱེད་ལ། བསྟི་སྟང་གིས་ཆོས་ཀྱི་སྦྱིན་པ་སྦྱིན་པར་བྱེད་དོ། །ཤཱ་རིའི་བུ་ཇི་ལྟར་བྱང་ཆུབ་སེམས་དཔས་བསྟི་སྟང་གིས་ཆོས་བཙལ་བར་བྱ་ཞེ་ན། ཤཱ་རིའི་བུ་འདི་ལ་བྱང་ཆུབ་སེམས་དཔའ་ཆོས་འདོད་པ་མཁན་པོ་དང་སློབ་དཔོན་ལ་གུས་པས་གང་ཡོངས་སུ་མི་གཏོང་བའི་རིན་པོ་ཆེ་དེ་འཇིག་རྟེན་ན་མེད་དོ། །གང་ལུས་ཀྱིས་རིམ་གྲོ་མི་བྱེད་པ་དེ་མེད་དོ། །གང་རྐང་པས་བཏང་བ་ལ་མི་སྤྲོ་བ་མེད་དོ། །གང་ངག་གི་ལས་ལ་མི་སྤྲོ་བ་དེ་མེད་དོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། དེ་བཞིན་དུ་སྦྱར་ཏེ་ཆོས་རིན་པོ་ཆེ་ནི་རིན་པོ་ཆེ་ཐམས་ཅད་ལས་ཁྱད་པར་ཅན་དུ་གྱུར་པ་ཡིན་པས་བདག་གིས་རིན་པོ་ཆེ་ཐམས་ཅད་བཏང་ནས་ཆོས་རིན་པོ་ཆེ་བཙལ་ཞིང་གཟུང་བར་བྱའོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། ཆོས་འདི་ནི་འཆིང་བ་ཐམས་ཅད་གཅོད་པར་འགྱུར་བའོ། །ཆོས་འདི་ནི་སྐྱེ་བ་དང་། རྒ་བ་དང་། ན་བ་དང་། འཆི་བ་དང་། མྱ་ངན་དང་། སྨྲེ་སྔགས་འདོན་པ་དང་། སྡུག་བསྔལ་བ་དང་། ཡིད་མི་བདེ་བ་གཅོད་པར་འགྱུར་བའོ་སྙམ་ནས་རིན་པོ་ཆེའི་སེམས་བསྐྱེད། རྙེད་དཀའ་བའི་སེམས་བསྐྱེད། སྨན་གྱི་སེམས་བསྐྱེད་དེ། ཆོས་འདི་ནི་སེམས་ཅན་ཐམས་ཅད་ཀྱི་ནད་ཉེ་བར་ཞི་བར་བྱེད་པར་འགྱུར་བས། བདག་གིས་སེམས་ཅན་ཐམས་ཅད་ཀྱི་ནད་ཉེ་བར་ཞི་བར་བྱ་བའི་ཕྱིར་ཆོས་འདི་ལྟ་བུ་བཙལ་བར་བྱའོ་ཞེས་དེ་ལྟ་བུའི་བསམ་པས་ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་ཆོས་ཚོལ་བར་བྱེད་དོ། །ཤཱ་རིའི་བུ་ཇི་ལྟར་ན་བྱང་ཆུབ་སེམས་དཔས་རིན་ཐང་མེད་པའི་སེམས་བསྐྱེད་ཅིང་བསྟི་སྟང་གིས་ཆོས་འཛིན་པར་བྱེད་ཅེ་ན། དེ་ཚིག་བཞི་པའི་ཚིགས་སུ་བཅད་པའི་ཆེད་དུ་རི་རབ་ཙམ་གྱི་རིན་པོ་ཆེའི་ཕུང་པོ་ཡོངས་སུ་བཏང་ཡང་དེ་ལ་དཀའ་བའི་འདུ་ཤེས་མི་སྐྱེད་ཀྱི། ཅི་སྟེ་གང་ཚིག་བཞི་པའི་ཚིགས་སུ་བཅད་པ་འདི་འཛིན་པ་དེ་ཉིད་ལ་ནི་དཀའ་བའི་འདུ་ཤེས་སྐྱེད་དོ། །ཆོས་ཀྱི་ཚིག་གཅིག་གི་ཆེད་དུ་སྟོང་གསུམ་གྱི་སྟོང་ཆེན་པོའི་འཇིག་རྟེན་གྱི་ཁམས་རིན་པོ་ཆེ་སྣ་བདུན་གྱིས་བཀང་བ་སྲིད་པའི་རྩེ་མོ་དང་མཉམ་པ་འདི་ཡོངས་སུ་གཏོང་ཡང་དེ་ལ་ནི་དཀའ་བའི་འདུ་ཤེས་མི་སྐྱེད་ཀྱི། ཅི་སྟེ་གང་ཆོས་ཀྱི་ཚིག་གཅིག་པོ་འདི་འཛིན་པ་དེ་ཉིད་ལ་ནི་དཀའ་བའི་འདུ་ཤེས་སྐྱེད་དོ། །ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་དེ་ལྟ་བུའི་གུས་པ་དང་། དེ་ལྟ་བུའི་རི་མོར་བྱེད་པས་ཆོས་ཡོངས་སུ་འཛིན་པར་བྱེད་དེ། ཤཱ་རིའི་བུ་འདི་ཡང་བྱང་ཆུབ་སེམས་དཔའི་ངོ་མཚར་ཡིན་ནོ། །ཤཱ་རིའི་བུ་འདི་ཇི་སྙམ་དུ་སེམས། ཅི་སྐྱེས་བུ་སྲོག་དོན་དུ་གཉེར་བས་སྲོག་གི་ཕྱིར་རིན་པོ་ཆེའི་ཕུང་པོ་འདི་ཙམ་ཡོངས་སུ་གཏོང་བ་ཡོད་དམ། གསོལ་བ། བཅོམ་ལྡན་འདས་དེ་ནི་མ་ལགས་སོ། །བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་ཆོས་འདོད་པས་ཆོས་ཀྱི་ཚིག་གཅིག་གི་དོན་གྱི་ཕྱིར་ཡང་བདག་ཉིད་ཚངས་པའི་འཇིག་རྟེན་ནས་འདོར་བའམ། ཡང་ན་སྟོང་གསུམ་གྱི་སྟོང་ཆེན་པོའི་འཇིག་རྟེན་གྱི་ཁམས་མེས་ཡོངས་སུ་གང་བ་འདི་ལ་འཇུག་ཏུ་ཟིན་ཀྱང་དཀའ་བའི་འདུ་ཤེས་སྐྱེད་པར་མི་བྱེད་ཀྱི། ཅི་སྟེ་གང་ཆོས་ཀྱི་ཚིག་འདི་འཛིན་པ་དེ་ཉིད་ལ་དཀའ་བའི་འདུ་ཤེས་སྐྱེད་པར་བྱེད་དེ། ཤཱ་རིའི་བུ་དེ་ལྟར་བྱང་ཆུབ་སེམས་དཔའ་བསྟི་སྟང་གིས་ཆོས་ཡོངས་སུ་འཛིན་པར་བྱེད་དོ། །ཤཱ་རིའི་བུ་ཇི་ལྟར་བྱང་ཆུབ་སེམས་དཔའ་བསྟི་སྟང་དུ་བྱེད་པས་ཆོས་ཀྱི་སྦྱིན་པ་སྦྱིན་པར་བྱེད་ཅེ་ན། ཤཱ་རིའི་བུ་འདི་ལ་བྱང་ཆུབ་སེམས་དཔའ་ཆོས་ཀྱི་སྦྱིན་པ་སྦྱིན་པར་བྱེད་པ་ན་ཟང་ཟིང་གི་སེམས་སྐྱེད་པར་མི་བྱེད་དེ། བདག་གི་སྦྱིན་པ་འདི་ལས་ཆོས་གོས་དང་། བསོད་སྙོམས་དང་། མལ་སྟན་དང་། ན་བའི་གསོས་སྨན་དང་། ཡོ་བྱད་རྣམས་རྙེད་པར་འགྱུར་བའམ། ཐ་ན་བདག་གི་སྦྱིན་པ་འདི་ལས་ལེགས་སོ་ཞེས་བྱ་བ་ཡང་རྙེད་པར་འགྱུར་རོ། །ཞེས་སྐྱེས་བུ་དམ་པ་དེ་དེ་ལྟ་བུའི་བསམ་པས་ཆོས་ཀྱི་སྦྱིན་པ་སྦྱིན་པར་མི་བྱེད་དོ། །འོན་ཀྱང་ཇི་ལྟར་སེམས་ཅན་འདི་དག་སྐྱེ་བ་དང་། རྒ་བ་དང་། ན་བ་དང་། འཆི་བ་དང་། མྱ་ངན་དང་། སྨྲེ་སྔགས་འདོན་པ་དང་། སྡུག་བསྔལ་བ་དང་། ཡིད་མི་བདེ་བ་དང་། འཁྲུག་པ་རྣམས་ལས་ཐར་བར་བྱ་ཞེས་དེ་ལྟར་སྦྱིན་པར་བྱེད་དེ། ཤཱ་རིའི་བུ་སྐྱེས་བུ་དམ་པ་དེ་དག་ནི་དེ་ལྟ་བུའི་བསམ་པ་དང་། དེ་ལྟ་བུའི་ཚུལ་གྱིས་ཆོས་ཀྱི་སྦྱིན་པ་སྦྱིན་པར་བྱེད་དོ། །ཤཱ་རིའི་བུ་གལ་ཏེ་ཡང་བྱང་ཆུབ་སེམས་དཔའ་ལ་ལ་ཞིག་འདི་སྐད་དུ་གལ་ཏེ་བདག་ལ་ཁྱེད་ཀྱིས་ལུས་བྱིན་པ་ལྟ་ན་བདག་ཁྱེད་ལས་ཆོས་ཀྱི་སྦྱིན་པ་འདི་ལེན་པར་འདོད་དོ་ཞེས་ཟེར་ན། ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་གང་ཞིག་དེ་ལྟ་བུའི་ཕན་པར་འདོད་པ་ཡོད་ན་འདི་སྐད་དུ་ཁྱོད་ལ་བདག་གི་ཡན་ལག་དང་ཉིང་ལག་ལས་མཆོག་དང་མཆོག་ཏུ་གྱུར་པའི་ཤ་དག་ཀྱང་ཟོ་ཤིག །འདོད་ཆགས་ཟད་པར་བྱེད་པའི་ཆོས་འདི་ཡང་ཟུང་ཤིག་ཅེས་སྨྲ་བར་བྱེད་དེ། ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་དེ་ལྟ་བུ་བྱེད་པ་ན་ཡང་དཀའ་བའི་འདུ་ཤེས་མི་སྐྱེད་ཀྱི། ཅི་སྟེ་གང་འདི་ཆོས་ཀྱི་སྦྱིན་པ་དེ་སྟེན་པ་དེ་ཉིད་ལ་ནི་དཀའ་བའི་འདུ་ཤེས་སྐྱེད་པར་བྱེད་དོ། །ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་དེ་ལྟ་བུའི་བསྟི་སྟང་གིས་ཆོས་ཀྱི་སྦྱིན་པ་སྦྱིན་པར་བྱེད། དེ་ལྟ་བུའི་བསྟི་སྟང་གིས་ཆོས་འཛིན་པར་བྱེད། དེ་ལྟ་བུའི་བསྟི་སྟང་གིས་ཆོས་ཚོལ་བར་བྱེད་དེ། ཤཱ་རིའི་བུ་དེ་ལྟ་བུའི་བསྟི་སྟང་གིས་ཆོས་ངེས་པར་ཡོངས་སུ་ཚོལ་བ་པོའི་སེམས་ཅན་ཡང་རྙེད་དཀའ་བ་ཡིན། ཤཱ་རིའི་བུ་དེ་ལྟ་བུའི་བསྟི་སྟང་གིས་ཆོས་ཡོངས་སུ་འཛིན་པ་པོའི་སེམས་ཅན་ཡང་རྙེད་དཀའ་བ་ཡིན། ཤཱ་རིའི་བུ་དེ་ལྟ་བུའི་བསྟི་སྟང་གིས་ཆོས་སྦྱིན་པ་པོའི་སེམས་ཅན་ཡང་རྙེད་དཀའ་བ་ཡིན་ཏེ། གང་རྙེད་དཀའ་བ་དེ་ནི་ངོ་མཚར་དུ་གྱུར་པ་ཡིན་ལ། དེ་དག་ནི་ལྷ་དང་བཅས་པའི་འཇིག་རྟེན་གྱིས་ཀྱང་བསྔགས་པར་བྱ་བ་ཡིན་ནོ། །ཤཱ་རིའི་བུ་ཆོས་གསུམ་པོ་འདི་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་བསྔགས་པའི་བྱ་བ་རྗེས་སུ་འཐོབ་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་འདི་ལ་འདོད་ཆགས་སྐྱེ་བར་ཡང་དག་པར་རྗེས་སུ་མཐོང་བ་ན་འདོད་ཆགས་འདི་གཞི་གང་ལས་བྱུང་བ་ཚོལ་བར་བྱེད་དེ་ཡོངས་སུ་བཙལ་བ་ན་དེའི་གཞི་མི་དམིགས་སོ། །གང་ལས་འདོད་ཆགས་དེ་སྐྱེ་བ་དང་། གང་གི་འདོད་ཆགས་དེ་སྐྱེ་བ་དང་། ཇི་ལྟར་འདོད་ཆགས་དེ་སྐྱེ་བ་དེ་ལྟར་གང་གི་སྐྱེ་བ་དེ་མི་དམིགས། གང་ལ་སྐྱེ་བ་དེ་མི་དམིགས། ཇི་ལྟར་སྐྱེ་བ་དེ་མི་དམིགས་ཏེ། འདོད་ཆགས་དེ་མི་དམིགས་པ་ནི་འདོད་ཆགས་ཀྱི་རྒྱུས་སྡིག་པའི་ལས་མི་བྱེད་ལ། དེའི་དཀར་པོའི་ཆོས་རྣམས་འཕེལ་བར་འགྱུར་ཞིང་དེ་མ་སྨད་ཅིང་བསྔགས་པར་འགྱུར་ཏེ། དེས་ན་ལྷ་དང་། ཀླུ་དང་། གནོད་སྦྱིན་དང་། མི་དང་། མི་མ་ཡིན་པ་དང་། སངས་རྒྱས་དང་ཉན་ཐོས་རྣམས་ཀྱིས་བསྔགས་པར་འགྱུར་རོ། །དེ་ཆོས་ཐམས་ཅད་ཁས་མི་ལེན་ཅིང་ཡང་དག་པར་རྗེས་སུ་མི་མཐོང་། ཆོས་ཐམས་ཅད་ནམ་མཁའ་དང་མཉམ་པར་སྟོང་པར་མཐོང་ཞིང་དེ་སངས་རྒྱས་བཅོམ་ལྡན་འདས་རྣམས་ཀྱི་བྱང་ཆུབ་ལ་ལྷག་པར་མོས་པར་འགྱུར་ཏེ། འཇིག་རྟེན་རྣམས་ཕྱིན་ཅི་ལོག་ལ་ཡང་དག་པར་ཞེན་པ་མཐོང་བ་ན་དེ་ཞེ་སྡང་མེད་པ་དང་། གནོད་སེམས་མེད་པ་བཟོད་པ་མཆོག་ལ་རབ་ཏུ་གནས་ནས་སྦྱིན་པ་དང་། ཚུལ་ཁྲིམས་དང་། བཟོད་པ་དང་། བརྩོན་འགྲུས་དང་། བསམ་གཏན་དང་། ཤེས་རབ་རྣམས་ལ་བརྩོན་འགྲུས་རྩོམ་ཞིང་སེམས་ཅན་རྣམས་ཡོངས་སུ་སྨིན་པར་བྱེད་དོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། གལ་ཏེ་གནོད་སེམས་དང་ལྡན་ན་ནི་ཁོང་ཁྲོ་བ་མང་བར་འགྱུར་རོ། །ཁོང་ཁྲོ་བ་མང་བ་ལས་ནི་གཞན་དག་ལ་ཁྲོ་བའི་ཚིག་རྗོད་པར་བྱེད་དེ། དེ་མི་དགེ་བའི་ལས་མངོན་པར་འདུ་བྱེད་དོ། །དེ་མི་དགེ་བའི་ལས་མངོན་པར་འདུ་བྱེད་པས་མི་དགེ་བའི་སྲིད་པ་མངོན་པར་སྐྱེད་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་ཡང་བཟོད་པ་དང་ལྡན་པའི་བྱང་ཆུབ་སེམས་དཔའ་ནི་བཟོད་པར་བྱེད་དོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། གང་བཟོད་པའམ། མི་བཟོད་པའི་ཆོས་རྡུལ་ཙམ་ཡང་མི་དམིགས་པ་དེས་ན་གཞན་མི་ཤེས་པའི་སེམས་ཅན་གང་ཡིན་པ་དེ་དག་ལ་སྙིང་རྗེ་སྐྱེད་པར་བྱེད་དེ། སེམས་ཅན་ཀུན་ནས་གཤེ་བར་བྱེད་པ་དང་། ཀུན་ནས་ཁྲོ་བར་བྱེད་པ་དང་། ཡོངས་སུ་སྤྱོ་བར་བྱེད་པ་དེ་དག་ཉིད་ལ་སྙིང་རྗེ་སྐྱེད་པར་བྱེད་དོ། །དེས་ན་བྱམས་པ་ཆེན་པོ་བསྒོམས་པ་ཡིན། དགེ་བའི་ཆོས་རྣམས་ཀྱང་བསྒོམས་པ་ཡིན་ཏེ། བཟོད་པ་དེ་དང་ལྡན་ན་བཟོད་པ་མཆོག་ཐོབ་པ་གང་ཡིན་པ་འདི་ནི་རྗེས་སུ་འཐུན་པའི་བཟོད་པ་ཡིན་ཏེ། དེ་སྙིང་རྗེ་ཆེན་པོ་དང་ལྡན་པས་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་ནས་སེམས་ཅན་རྣམས་ལ་ཆོས་སྟོན་པར་བྱེད། དེ་དཀའ་བ་སྤྱོད་པར་བྱེད་ཅིང་བཟོད་པ་དཀའ་བ་བྱེད་དེ། དེས་ན་བཟོད་པ་མཆོག་ཐོབ་པར་བྱེད། དེས་ན་ཐོགས་པ་མེད་པའི་ཡེ་ཤེས་མཐོང་བ་ཐོབ་པར་བྱེད། དེས་ན་སྣ་ཚོགས་པ་མེད་པའི་ཡེ་ཤེས་མཐོང་བ་ཐོབ་པར་བྱེད། དེས་ན་ཆགས་པ་མེད་པའི་ཡེ་ཤེས་མཐོང་བ་ཐོབ་པར་བྱེད་དོ། །ཤཱ་རིའི་བུ་ཆོས་བཞི་པོ་འདི་དག་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་བསྔགས་པའི་བྱ་བ་རྗེས་སུ་འཐོབ་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་གཞན་ཡང་ཆོས་བཞི་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་སྐྱེ་བ་དྲན་པར་འགྱུར་ཏེ། བཞི་གང་ཞེ་ན། ཤཱ་རིའི་བུ་འདི་ལྟར་བྱང་ཆུབ་སེམས་དཔའ་ཇི་སྐད་སྨྲ་བ་དེ་བཞིན་དུ་བྱེད་པ་ཡིན། ཇི་ལྟར་བྱེད་པ་དེ་བཞིན་དུ་སྨྲ་བ་ཡིན། ཁྲོ་བ་སྐྱེས་པ་ཡང་སྤོང་བར་བྱེད། རབ་ཏུ་སྤོང་བར་བྱེད། རྣམ་པར་སེལ་བར་བྱེད་ཅིང་ཡེ་ཤེས་ལ་རབ་ཏུ་རྟོན་པར་འགྱུར་བ་ཡིན་ཏེ། ཤཱ་རིའི་བུ་འདི་ལྟ་སྟེ། དཔེར་ན་མེ་མདག་ཆེན་པོས་གང་བའི་དོང་ནི་ཆུའི་ཕུང་པོ་ཆེན་པོས་ཉེ་བར་ཞི་བར་བྱེད་དོ། །ཤཱ་རིའི་བུ་དེ་བཞིན་དུ་བྱང་ཆུབ་སེམས་དཔའ་ཁྲོ་བ་སྐྱེས་པ་སྤོང་བར་བྱེད་པ་དང་། སེལ་བར་བྱེད་པ་དང་། ཡེ་ཤེས་ལ་རྟོན་པར་བྱེད་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་ཆོས་དང་པོ་འདི་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་སྐྱེ་བ་དྲན་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་གཞན་ཡང་འདི་ལ་བྱང་ཆུབ་སེམས་དཔའ་ཐོས་པ་ཚོལ་བ་ལ་གནས་ཤིང་དགེ་བའི་ཆོས་ཐམས་ཅད་ཚོལ་བ་ན་གཅེས་པ་མི་འདོར་བ་ཡིན། ཉམས་འོག་ཏུ་ཆུད་པའི་ཐོས་པ་མང་པོ་དང་ལྡན་པ་ཡིན། དེ་གང་དང་གང་དུ་འགྲོ་བ་ཐམས་ཅད་དུ་འདོད་པའི་བྱ་བ་རྗེས་སུ་འཐོབ་པར་འགྱུར་ལ། གཞན་དག་ཀྱང་ཐོས་པ་མང་པོ་ལ་འགོད་པར་བྱེད་ཅིང་ཆོས་དེ་དག་ལ་ཡང་ཚུལ་བཞིན་དུ་སོ་སོར་རྟོག་པར་བྱེད་དེ། འཆི་བའི་དུས་ཀྱི་ཚེ་ཡང་ཆོས་དང་ལྡན་པའི་དྲན་པས་གནས་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་འདི་ལྟ་སྟེ་དཔེར་ན་སྐྱེས་བུ་ཞིག་རྩེ་བ་དང་དགའ་བས་ཡོངས་སུ་སྤྱོད་པས་ལམ་དུ་ཞུགས་པ་ན་དེ་དུས་དེའི་ཚེ་བཀྲེས་པར་གྱུར་ལ། དེ་དུས་ཕྱིས་ཞིག་ན་སྔོན་གྱི་རྩེ་བ་དང་དགའ་བ་དེ་རྗེས་སུ་དྲན་པར་གྱུར་པ་ན་འདི་ལྟར་དེ་སྔོན་གྱི་ཡིད་ལ་བྱེད་པ་དེས་བཀྲེས་པ་ཡོད་པ་སེལ་བར་བྱེད་དོ། །ཤཱ་རིའི་བུ་དེ་བཞིན་དུ་དུས་གང་གི་ཚེ་བྱང་ཆུབ་སེམས་དཔའ་འཆི་བའི་དུས་ཀྱི་ཚོར་བས་ཉེན་པ་དེའི་ཚེ་ཆོས་དང་ལྡན་པའི་ཡིད་ཀྱིས་དང་དུ་ལེན་པར་བྱེད་དེ་དེ་ཤེས་བཞིན་དང་ལྡན་པར་འགྱུར་རོ། །དེས་ཤེས་བཞིན་དང་ལྡན་པས་མའི་མངལ་དུ་འཇུག་ཅིང་དེར་གནས་ན་ཡང་དྲན་པ་དང་ལྡན་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་ཆོས་གཉིས་པོ་འདི་དག་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་སྐྱེ་བ་དྲན་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་དག་ལེགས་པར་བསྲུང་བ་ཡིན་ཏེ། དེ་གཞན་ལ་གཤེ་བར་མི་བྱེད་དེ། གཏི་མུག་པ་ཞེའམ། རྨོངས་པ་ཞེའམ། ཁེངས་པ་ཞེའམ། ལོང་བ་ཞེའམ། འཐེང་པོ་ཞེས་གཞན་ལ་ཁྲོ་བ་སྐྱེད་པས་གནས་པ་མ་ཡིན་ནོ། །གཞན་དུ་ན་སེམས་ཅན་ཁྲོས་པ་གང་ཡིན་པ་དེ་ལ་དབུགས་འབྱིན་པར་བྱེད། བཟོད་པ་གསོལ་བར་བྱེད། རྗེས་སུ་འཐུན་པར་བྱེད། ཆོས་ཀྱིས་དགའ་བར་བྱེད། དྲན་པ་ལ་ཉེ་བར་གནས་པར་བྱེད། དེ་ལ་གསོང་པོར་སྨྲ་བར་བྱེད། ཚུར་ཤོག་ལེགས་པར་འོངས་སོ་ཞེས་སྨྲ་བར་བྱེད་དེ། མ་འཇིགས་ཤིག །མ་འཇིགས་ཤིག་ཅེས་སེམས་ཅན་དེ་དག་ལ་མི་འཇིགས་པ་སྦྱིན་པར་བྱེད། དབུགས་འབྱིན་པར་བྱེད་དེ། ཡིད་ཆེས་པར་བྱེད་ཅིང་བཟོད་པ་གསོལ་བར་བྱེད། རྗེས་སུ་འཐུན་པར་བྱེད་དོ། །སེམས་ཅན་འཁྲུལ་པའི་སྤྱོད་པ་དག་ལས་ཀྱང་ཀུན་ནས་སློང་བར་བྱེད་དེ། ཤཱ་རིའི་བུ་ཆོས་གསུམ་པོ་འདི་དག་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་སྐྱེ་བ་དྲན་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་གཞན་ཡང་བྱང་ཆུབ་སེམས་དཔའ་འདི་ལ་དགེ་བའི་ཆོས་རྣམས་ལས་ཤིན་ཏུ་འདས་ཤིང་དགེ་བའི་ཆོས་རྣམས་དང་ལྡན་པར་འགྱུར་བ་དང་། དེ་དེ་ལ་གནས་ནས་གཞན་དག་ཀྱང་དེ་ལ་རབ་ཏུ་འགོད་པར་བྱེད་པ་དང་། བསྟི་སྟང་གིས་ཆོས་ཡོངས་སུ་ཚོལ་བར་བྱེད་པ་དང་། གཞན་དག་ལ་རྒྱ་ཆེར་སྟོན་པར་བྱེད་དེ། ཤཱ་རིའི་བུ་ཆོས་བཞི་པོ་འདི་དག་དང་ལྡན་པའི་བྱང་ཆུབ་སེམས་དཔའ་སྐྱེ་བ་དྲན་པར་འགྱུར་རོ། །དེ་སྐྱེ་བ་དྲན་པས་གནས་ལྔ་རབ་ཏུ་འཐོབ་པ་ཡིན་ཏེ། ལྔ་གང་ཞེ་ན། སྔོན་ཐོས་པའི་ཆོས་གང་ཡིན་པ་དེ་དག་ལ་དེ་བཞིན་དུ་མངོན་པར་འཇུག་པ་ཉིད་དང་། དྲན་པ་མཆོག་ཏུ་འགྱུར་བ་དང་། ཤེས་རབ་རྣོ་བ་དང་། དྲན་པ་མཆོག་དང་། ཤེས་རབ་དང་ལྡན་པ་དང་། དེ་སྲོག་གི་ཕྱིར་ཡང་སྡིག་པའི་ལས་མི་བྱེད་པ་སྟེ། སྐྱེས་མ་ཐག་ཏུ་ཆོས་ལྔ་པོ་འདི་དག་འཐོབ་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་ཆོས་འདི་བཞི་ནི་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ལ་མེད་དེ། བཞི་གང་ཞེ་ན། བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ལ་གཡོ་མེད་དོ། །བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ལ་སེར་སྣ་མེད་དོ། །བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ལ་ཕྲག་དོག་དང་ཕྲ་མ་མེད་དོ། །བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ལ་བདག་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་མི་ནུས་སོ་སྙམ་དུ་ཞུམ་པའི་སེམས་མེད་དོ། །ཤཱ་རིའི་བུ་གང་ལ་ཆོས་བཞི་པོ་འདི་དག་ཡོད་པ་དེ་ནི་མཁས་པ་རྣམས་ཀྱིས་ཀྱེ་མའོ་འདི་ནི་ཚུལ་འཆོས་པའོ། །ཀྱེ་མའོ་འདི་ནི་ཁ་གསག་ཅན་ནོ། །ཀྱེ་མའོ་འདི་ནི་ཆོས་སྟོར་བའོ། །ཀྱེ་མའོ་འདི་ནི་ཀུན་ནས་ཉོན་མོངས་པའི་ཆོས་ཅན་ནོ། །ཀྱེ་མའོ་འདི་ནི་འཇིག་རྟེན་གྱི་ཟང་ཟིང་ལྷུར་ལེན་པའོ། །ཀྱེ་མའོ་འདི་ནི་གོས་དང་ཟས་ཆེད་ཆེར་བྱེད་པ་སྟེ། འཇིག་རྟེན་ན་རྐུན་པོར་རིག་པར་བྱའོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། ཤཱ་རིའི་བུ་གང་ཟག་དེ་ལྟ་བུ་ལ་ནི་དངོས་པོ་བཞི་མེད་དེ། དེའི་རྟགས་བཞི་གང་ཞེ་ན། ཟས་ལ་ཆེད་ཆེར་བྱེད་པ་ལ་གནས་པ་དང་། བདག་ལ་མཆོག་ཏུ་འདུ་ཤེས་པ་དང་། སྒྲོགས་པས་ཉེ་བར་སྟོན་པ་དང་། དེ་ལ་བྱང་ཆུབ་སེམས་དཔའི་རྩོམ་པ་གང་ཡིན་པ་འདི་ལ་བྱང་ཆུབ་སེམས་དཔའི་ཡོན་ཏན་གྱི་གནས་གཅིག་ལ་ཡང་མངོན་པར་སྦྱོར་བ་མེད་པའོ། །ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔས་བརྩམ་པར་བྱ་བའི་བྱང་ཆུབ་སེམས་དཔའི་ཡོན་ཏན་ཡང་གང་ཞེ་ན། འདི་ལྟ་སྟེ། ཚུལ་ཁྲིམས་ཡོངས་སུ་རྫོགས་པར་བྱེད་པ་ཉིད་དང་། ཚུལ་ཁྲིམས་མ་རལ་བ་ཉིད་དང་། དྲན་པ་ཉམས་པ་མེད་པ་ཉིད་དང་། ལྷུག་པར་གཏོང་བ་ཉིད་དང་། བཟོད་པའི་སྟོབས་དང་། ཁྲོ་བ་མེད་པ་ཉིད་དང་། འཚེ་བ་མེད་པ་ཉིད་དང་། ཐོས་པ་མང་བ་དང་། སྦྱིན་པ་དང་། དུལ་བ་དང་། ཡང་དག་པར་སྡོམ་པ་དང་། འདོད་པ་རྣམས་དོན་དུ་གཉེར་བ་མེད་པ་ཉིད་དང་། གནོད་སེམས་ཐམས་ཅད་ལ་དོན་དུ་གཉེར་བ་མེད་པ་ཉིད་དང་། མྱུར་དུ་མངོན་པར་འབྱུང་བ་ཉིད་དང་། ཕྲ་མ་མེད་པ་ཉིད་དང་། ཁྲོ་བ་མེད་པ་ཉིད་དང་། དགེ་བའི་ཆོས་ཡང་དག་པར་ལེན་པ་དང་། ཡེ་ཤེས་ཀྱི་སྤྱོད་པ་དང་། སེམས་ཅན་ཐམས་ཅད་ལ་བྱམས་པའི་སེམས་ཉིད་དང་། མི་འཐུན་པ་རྣམས་སྡུམ་པར་བྱེད་པ་དང་། འཁོར་བའི་འདམ་གྱི་ཟམ་པ་ཉིད་དང་། ཆོས་ཀྱི་གཏམ་བྱེད་པ་ཉིད་དང་། བཟོད་པའི་སྟོབས་སོ་སོར་རིག་པར་བྱེད་པ་དང་། ལས་ཀྱི་རྣམ་པར་སྨིན་པ་ལ་རྟོན་པ་ནི་ཤཱ་རིའི་བུ་ངེས་པར་བྱང་ཆུབ་སེམས་དཔར་འགྱུར་ཏེ། ཤཱ་རིའི་བུ་འདི་ནི་བྱང་ཆུབ་སེམས་དཔས་བརྩམས་པའི་བྱང་ཆུབ་སེམས་དཔའི་ཡོན་ཏན་རྣམས་ཡིན་ནོ། །ཤཱ་རིའི་བུ་གང་ལ་བྱང་ཆུབ་སེམས་དཔས་བརྩམས་པའི་བྱང་ཆུབ་སེམས་དཔའི་ཡོན་ཏན་གྱི་ཆོས་འདི་དག་མེད་པ་དེ་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་ཚུལ་འཆོས་པ་ངན་འགྲོའི་གནས་དང་ལྡན་པར་རིག་པར་བྱའོ། །ཤཱ་རིའི་བུ་ཆོས་བཞི་པོ་འདི་ནི་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་སྒྲུབ་པ་ཡིན་ཏེ། བཞི་གང་ཞེ་ན། ཆོས་དང་ཆོས་ཤེས་པ་དང་། སྦྱིན་པ་དང་སྦྱིན་པ་ཤེས་པ་དང་། ཞལ་ཏ་བྱེད་པའི་བྱ་བ་ཤེས་པ་དང་། ཐོས་པ་མང་པོ་བཙལ་བ་ཤེས་པའོ། །ཤཱ་རིའི་བུ་ཇི་ལྟར་ན་བྱང་ཆུབ་སེམས་དཔའི་ཆོས་གང་ཞེ་ན། འདི་ལྟ་སྟེ། དྲན་པ་ཉེ་བར་གཞག་པ་བཞི་དང་། ཡང་དག་པར་སྤོང་བ་བཞི་དང་། རྫུ་འཕྲུལ་གྱི་རྐང་པ་བཞི་དང་། དབང་པོ་ལྔ་དང་། སྟོབས་ལྔ་དང་། བྱང་ཆུབ་ཀྱི་ཡན་ལག་བདུན་དང་། འཕགས་པའི་ལམ་ཡན་ལག་བརྒྱད་དང་། ཆགས་པ་མེད་པ་དང་། ཞེ་སྡང་མེད་པ་དང་། གཏི་མུག་མེད་པ་དང་། དགེ་བ་བཅུའི་ལས་ཀྱི་ལམ་འདི་རྣམས་ནི་བྱང་ཆུབ་སེམས་དཔའི་ཆོས་ཞེས་བྱའོ། །དེ་ལ་ཆོས་ཤེས་པ་གང་ཞེ་ན། གང་དེ་རྣམས་ཚུལ་བཞིན་སྒྲུབ་པར་བྱེད་ཀྱི་ལོག་པར་མི་སྒྲུབ་པ་སྟེ། སྦྱིན་པ་དང་། དུལ་བ་དང་། ཡང་དག་པར་སྡོམ་པ་འདི་ནི་ཆོས་ཤེས་པ་ཞེས་བྱའོ། །དེ་ལ་སྦྱིན་པ་གང་ཞེ་ན། ཤཱ་རིའི་བུ་འདི་ལ་བྱང་ཆུབ་སེམས་དཔའ་ལ་གང་ཡོད་པ་དེ་ཡོངས་སུ་གཏོང་བར་བྱེད་དེ། དེ་ལག་པ་ཡོངས་སུ་གཏོང་བ་ཡིན། རྐང་པ་ཡོངས་སུ་གཏོང་བ་ཡིན། ལག་པ་དང་རྐང་པ་ཡོངས་སུ་གཏོང་བ་ཡིན། སྣ་ཡོངས་སུ་གཏོང་བ་ཡིན། མགོ་ཡོངས་སུ་གཏོང་བ་ཡིན། ཡན་ལག་དང་ཉིང་ལག་ཡོངས་སུ་གཏོང་བ་ཡིན། བུ་ཡོངས་སུ་གཏོང་བ་ཡིན། བུ་མོ་ཡོངས་སུ་གཏོང་བ་ཡིན། ཆུང་མ་ཡོངས་སུ་གཏོང་བ་ཡིན། སྲོག་ཡོངས་སུ་གཏོང་བ་ཡིན། ནོར་ཡོངས་སུ་གཏོང་བ་ཡིན། བདེ་བ་ཡོངས་སུ་གཏོང་བ་ཡིན། ལུས་ཡོངས་སུ་གཏོང་བ་ཡིན། དངོས་པོ་ཡོངས་སུ་གཏོང་བ་ཡིན། ཡུལ་ཡོངས་སུ་གཏོང་བ་ཡིན། རིན་པོ་ཆེ་ཡོངས་སུ་གཏོང་བ་ཡིན། བདོག་པ་ཐམས་ཅད་ཡོངས་སུ་གཏོང་བ་ཡིན་ཏེ། ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་དེ་ལྟར་སྦྱིན་པ་སྦྱིན་ན་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ལ་བསྙེན་པ་ཡིན་ནོ། །དེ་ལ་སྦྱིན་པ་ཤེས་པ་གང་ཞེ་ན། ཇི་ལྟར་ལྷུང་བཟེད་དུ་ཆུད་པ་དུས་སུ་སྦྱིན་པ་དང་། རུང་བ་སྦྱིན་པ་དག་སྦྱིན་པ་སྦྱིན་པར་བྱེད་དེ། ཇི་ལྟར་དགེ་སློང་རྣམས་ལ་ཆོས་གོས་དང་། བསོད་སྙོམས་དང་། གནས་མལ་དང་། ན་བའི་གསོས་སྨན་དང་། ཡོ་བྱད་རྣམས་ཡོངས་སུ་བཏང་ཞིང་མགྲོན་དུ་གཉེར་བར་བྱེད་དོ། །ཤཱ་རིའི་བུ་གང་ཡང་བྱང་ཆུབ་སེམས་དཔའ་མཁས་པས་སྨད་པའི་སྦྱིན་པ་འདི་རྣམས་ལས་རབ་ཏུ་ལྡོག་པ་ཡིན་ཏེ། འདི་ལྟ་སྟེ། དུས་སུ་མ་བྱིན་པའི་སྦྱིན་པ་དང་། རུང་བ་མ་ཡིན་པའི་སྦྱིན་པ་དང་། ཆང་གི་སྦྱིན་པ་དང་། དུག་གི་སྦྱིན་པ་དང་། གནོད་པའི་སྦྱིན་པ་དང་། འགྱོད་པའི་སྦྱིན་པ་དང་། མཚོན་ཆའི་སྦྱིན་པ་དང་། ཕྲོགས་པའི་སྦྱིན་པ་དང་། བརྙས་པའི་སྦྱིན་པ་དང་། རང་གི་ལག་གིས་མ་ཡིན་པའི་སྦྱིན་པ་དང་། རྒྱབ་ཀྱིས་ཕྱོགས་པའི་སྦྱིན་པ་དང་། འདོད་པ་ལ་སྟེན་པའི་སྦྱིན་པ་དང་། མི་འཐུན་པའི་སྦྱིན་པ་དང་། དམ་པ་མ་ཡིན་པའི་སྦྱིན་པ་དང་། སྨད་པའི་སྦྱིན་པ་དང་། དུག་དང་སྦྱར་བའི་སྦྱིན་པ་དང་། སོ་སོར་གནོད་པའི་སྦྱིན་པ་དང་། རྙེད་པའི་ཕྱིར་སྦྱིན་པ་དང་། རྟག་ཏུ་འཇིགས་པ་སྐྱེད་པའི་ཕྱིར་སྦྱིན་པ་དང་། བྱང་ཆུབ་ཡོངས་སུ་ཉམས་པར་བྱེད་པའི་སྦྱིན་པ་དང་། འཆི་བའི་ཞགས་པའི་སྦྱིན་པ་དང་། རྨོངས་པའི་སྦྱིན་པ་དང་། བསྟེན་པའི་ཕྱིར་སྦྱིན་པ་དང་། འཇིགས་པའི་སྦྱིན་པ་དག་ལས་ལྡོག་པར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་འདི་ལ་མཁས་པས་སྨད་པའི་སྦྱིན་པ་གང་ཡིན་པ་དེ་དག་ལ་ཐམས་ཅད་ཀྱི་ཐམས་ཅད་དུ་མི་སྤྱོད་ཅིང་ལྷག་པར་མི་སྤྱོད་དེ་བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ཀྱི་ཐམས་ཅད་དུ་སྤྱོད་ཅིང་ལྷག་པར་སྤྱོད་པ་གང་ཡིན་པ་དེ་ནི་བྱང་ཆུབ་སེམས་དཔའ་ཞེས་བརྗོད་པར་མི་བྱའོ། །བྱང་ཆུབ་སེམས་དཔའ་གང་ལ་བྱང་ཆུབ་སེམས་དཔའི་ཡོན་ཏན་རྣམས་དང་ལྡན་པར་གྱུར་ཀྱང་སྦྱིན་པ་ལས་ཉམས་པ་ནི་བྱང་ཆུབ་སེམས་དཔའི་ཡོན་ཏན་མེད་པ་ཡིན་ནོ། །ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་ཁྱིམ་གྱི་གནས་ནས་མངོན་པར་བྱུང་བ་གང་བྱང་ཆུབ་ཀྱི་ཡན་ལག་ལ་མངོན་པར་བརྩོན་པ་དེས་ཇི་ལྟར་སྦྱིན་པ་སྦྱིན་པར་བྱ། སྦྱིན་པ་གང་སྦྱིན་པར་བྱ། སྦྱིན་པ་ཇི་ཙམ་ཞིག་སྦྱིན་པ་སྦྱིན་པར་བྱ། ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་ཁྱིམ་གྱི་གནས་ནས་མངོན་པར་བྱུང་བ་གང་བྱང་ཆུབ་ཀྱི་ཡན་ལག་ལ་མངོན་པར་བརྩོན་པ་གང་ཡིན་པ་དེས་སོ་སོར་ཐར་པའི་སྡོམ་པས་བསྡམས་ཏེ་གནས་པར་བྱ་བ་དང་། ཆོ་ག་དང་། སྤྱོད་ཡུལ་དང་། སྤྱོད་པ་ཕུན་སུམ་ཚོགས་པར་བྱ་བ་དང་། ཁ་ན་མ་ཐོ་བ་ཕྲ་རབ་ཙམ་ལ་འཇིགས་པར་ལྟ་བ་དང་། ལུས་དང་། ངག་དང་། ཡིད་ཀྱི་ལས་ཀྱི་མཐའ་ཡོངས་སུ་དག་པ་དང་། འཚོ་བ་ཡོངས་སུ་དག་པས་བསླབ་པ་ཡང་དག་པར་བླངས་ཏེ་བསླབ་པར་བྱའོ། །དེས་ལུང་ནོད་པ་ལྷུར་བྱ་བ་དང་། ཁ་ཏོན་ལྷུར་བྱ་བ་དང་། དགོན་པ་ན་གནས་པ་ལྷུར་བྱ་བ་དང་། ཇི་ལྟར་ཐོས་པ་དང་། ཇི་ལྟར་ཀུན་ཆུབ་པར་བྱས་པའི་ཆོས་དག་གཞན་དག་ལ་རྒྱ་ཆེར་བསྟན་པར་བྱ་བ་དང་། ཆོས་ཀྱི་སྦྱིན་པ་པོ་དང་། ཆོས་ཀྱི་སྦྱིན་བདག་ཏུ་བྱའོ། །ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་ཁྱིམ་པ་གང་གིས་སངས་རྒྱས་ཀྱི་ཞིང་གང་གཱའི་ཀླུང་གི་བྱེ་མ་སྙེད་རིན་པོ་ཆེ་སྣ་བདུན་གྱིས་ཡོངས་སུ་བཀང་བ་བྱས་ཏེ། དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་རྣམས་ལ་དབུལ་བ་ཕུལ་བ་བས། ཤཱ་རིའི་བུ་རབ་ཏུ་བྱུང་བར་གྱུར་པའི་བྱང་ཆུབ་སེམས་དཔའ་གང་གིས་ཚིག་བཞི་པའི་ཚིགས་སུ་བཅད་པ་གཅིག་བསྟན་ན་འདི་ཉིད་དེ་བས་ཆེས་བསོད་ནམས་མང་དུ་སྐྱེད་དེ། ཤཱ་རིའི་བུ་དེ་བཞིན་གཤེགས་པས་ནི་རབ་ཏུ་བྱུང་བས་ཟང་ཟིང་གི་སྦྱིན་པ་བྱ་བར་མ་གནང་ངོ་། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། ཤཱ་རིའི་བུ་ང་རབ་ཏུ་བྱུང་བར་གྱུར་ནས་བསྐལ་པ་གྲངས་མེད་བསམ་གྱིས་མི་ཁྱབ་པར་ཁྱིམ་པའམ་རབ་ཏུ་བྱུང་བའི་ཆོས་གོས་དང་། བསོད་སྙོམས་ལ་མགྲོན་དུ་བོས་པར་མ་དགོངས་སོ། །གཞན་དུ་ན་ཆོས་ཁོ་ན་ལ་བསྟི་སྟང་དང་། བླ་མར་བྱས་ཏེ་ཉེ་བར་བརྟེན་ནས་གནས་པར་བྱའོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། ཤཱ་རིའི་བུ་ངའི་ཉན་ཐོས་རྣམས་ཀྱིས་ཆོས་ཀྱི་ནོར་ཅན་དུ་བྱའི་ཟང་ཟིང་གི་ནོར་ཅན་དུ་ནི་མི་བྱའོ། །ཤཱ་རིའི་བུ་གང་ཡང་ལྷུང་བཟེད་དུ་འོངས་པའི་རྙེད་པ་ཆོས་དང་འཐུན་པ་ཆོས་ཀྱིས་རྙེད་པར་འགྱུར་ན་དེས་ཚངས་པ་མཚུངས་པར་སྤྱོད་པ་རྣམས་དང་ཐུན་མོང་དུ་བཟའ་བར་བྱའོ། །གལ་ཏེ་ལ་ལ་ཞིག་འོངས་ནས་ལྷུང་བཟེད་དམ། ཆོས་གོས་སློང་ལ། གལ་ཏེ་སངས་རྒྱས་ཀྱིས་གནང་བའི་ཆོས་གོས་གསུམ་ལས་ལྷག་པར་གྱུར་ན་ཇི་ལྟར་བཏང་བ་བཞིན་དུ་སྦྱིན་པར་བྱའོ། །གལ་ཏེ་དེ་ལ་ཆོས་གོས་གསུམ་ལས་ཉུང་བར་གྱུར་ན་གང་ལ་བརྟེན་ཏེ་ཚངས་པར་སྤྱོད་པ་ལ་གནས་པ་དེ་ནི་ཡོངས་སུ་གཏང་བར་མི་བྱའོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། དེ་བཞིན་གཤེགས་པས་ཆོས་གོས་གསུམ་གཏང་བར་མི་བྱའོ། །ཞེས་གསུངས་པའི་ཕྱིར་རོ། །ཤཱ་རིའི་བུ་གལ་ཏེ་བྱང་ཆུབ་སེམས་དཔའ་ཆོས་གོས་གསུམ་ཡོངས་སུ་བཏང་སྟེ། སློང་བ་པོ་གཙོ་བོར་བྱེད་པར་གྱུར་ན་དེས་འདོད་པ་ཆུང་བ་བསྟེན་པ་མ་ཡིན་ནོ། །ཤཱ་རིའི་བུ་དེ་ལྟ་བས་ན་བྱང་ཆུབ་སེམས་དཔའ་གང་ཁྱིམ་གྱི་གནས་ནས་མངོན་པར་བྱུང་བ་དེས་ཆོས་ལ་ཀུན་ཏུ་བསྟེན་པར་བྱ་ཞིང་དེས་དེ་ལ་མངོན་པར་བརྩོན་པར་བྱའོ། །ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔས་གང་ཡང་སེ་གོལ་གཏོགས་པ་ཙམ་དུ་ཡང་སེམས་ཅན་ཐམས་ཅད་ལ་བྱམས་པའི་སེམས་བསྒོམ་པར་བྱ་སྟེ། འདི་ནི་སྦྱིན་པ་ཤེས་པ་ཞེས་བྱའོ། །ཤཱ་རིའི་བུ་ཇི་ཙམ་གྱིས་ན་བྱང་ཆུབ་སེམས་དཔའ་ཞལ་ཏ་བྱེད་པར་ཤེས་པར་བྱ་ཞེ་ན། དེས་ཆོས་དང་འཐུན་པས་ཞལ་ཏ་བྱ་ཞིང་དེ་བཞིན་གཤེགས་པའི་མཆོད་རྟེན་ལ་ཡང་ཞལ་ཏ་བྱའོ། །མཁན་པོ་དང་སློབ་དཔོན་ལ་ཡང་གུས་པ་དང་བཅས་པར་བྱའོ། །དེ་ལྟར་འདིས་བརྩོན་འགྲུས་བརྩམ་པར་བྱ་སྟེ། ཇི་ལྟར་ཇི་སྲིད་འཚོའི་མཐར་ཐུག་པའི་བར་དུ་གཉིད་དང་རྨུགས་པ་བསལ་བར་བྱ་སྟེ། ཤཱ་རིའི་བུ་འདི་ནི་བྱང་ཆུབ་སེམས་དཔའི་ཞལ་ཏ་བྱེད་པའི་བྱ་བ་ཡིན་ནོ། །ཤཱ་རིའི་བུ་དེ་ལ་ཐོས་པ་མང་པོ་ཚོལ་བའི་ཤེས་པ་གང་ཞེ་ན། ཤཱ་རིའི་བུ་གང་སངས་རྒྱས་ཀྱི་བྱང་ཆུབ་ཀྱི་ཕྱིར་བྱང་ཆུབ་ཀྱི་སེམས་བསྐྱེད་པ་ཐོབ་ནས་ཐོས་པ་མང་པོ་ལ་མངོན་པར་བརྩོན་པར་མི་བྱེད་པ་དེ་ནི་བྱང་ཆུབ་སེམས་དཔའ་བདེན་པར་སྨྲ་བ་མ་ཡིན་པར་རིག་པར་བྱའོ། །ཤཱ་རིའི་བུ་མིང་ཙམ་གྱིས་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་མི་ནུས་ཀྱི། ཤེས་རབ་དང་ལྡན་པ་ཉིད་ཀྱིས་ནི་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་ནུས་སོ། །ཤེས་རབ་དེ་ཡང་གང་ལས་ངེས་པར་འབྱུང་བ་ཡིན་ཞེ་ན། མང་དུ་ཐོས་པ་ལས་ཤེས་རབ་འབྱུང་བ་ཡིན་ཏེ། ཤཱ་རིའི་བུ་དེ་ལྟ་བས་ན་ཁྱོད་ཀྱིས་མངོན་སུམ་དུ་བྱ་ཞིང་ཁྱོད་ཀྱིས་སོ་སོར་རིག་པར་བྱའོ། །བྱང་ཆུབ་སེམས་དཔའི་ཐེག་པ་ལ་ཡང་དག་པར་ཞུགས་པའི་སེམས་ཅན་མང་དུ་ཐོས་པ་ལ་ཇི་ལྟར་ཐེ་ཙོམ་སྐྱེ་བར་འགྱུར་ཞེས་གང་བྱང་ཆུབ་སེམས་དཔའ་དེ་སེམས་ཅན་རྣམས་ལ་ཕན་འདོགས་པར་བྱེད་པའི་བྱང་ཆུབ་ལ་བདུད་སྡིག་ཅན་བར་ཆད་བྱེད་པར་འགྱུར་ཏེ། ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་བསྐྱེད་ནས་ཐོས་པ་མང་པོ་ལ་མངོན་པར་མི་བརྩོན་པ་གང་ཡིན་པ་དེ་ནི་བདུད་སྡིག་ཅན་གྱི་ཡོངས་སུ་བྱིན་གྱིས་བརླབས་པ་ཉིད་ཡིན་ནོ། །དེ་ལྟ་བས་ན་ཤཱ་རིའི་བུ་བདུད་ཀྱི་ལས་ཐམས་ཅད་ཡོངས་སུ་ཤེས་པར་འདོད་པས་བརྩོན་འགྲུས་བརྩམ་པར་བྱ། ཚིག་གཙོ་བོར་མི་བྱ། མཁས་པའི་བྱ་བ་ཤེས་པར་བྱ་ཞིང་འཇིག་རྟེན་ཐམས་ཅད་ལ་གྲགས་པའི་བྱ་བ་ཤེས་པར་བྱའོ། །ཤཱ་རིའི་བུ་དེ་ལྟ་བས་ན་འཇིག་རྟེན་ཐམས་ཅད་ལས་མངོན་པར་འཕགས་པའི་བརྩོན་འགྲུས་དང་། འཇིག་རྟེན་ཐམས་ཅད་ལས་མངོན་པར་འཕགས་པའི་ཚུལ་ཁྲིམས་ཀྱིས་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་འགྱུར་རོ། །ཤཱ་རིའི་བུ་གནས་འདི་དོན་དུ་གཉེར་བ་མང་ཡང་ནན་ཏན་གྱིས་ཡང་དག་པར་སྒྲུབ་པར་བྱེད་པ་ནི་མེད་དེ་དེ་དག་ཀྱང་དེ་བཞིན་གཤེགས་པའི་སྟོབས་བཅུ་དང་། འཇིག་རྟེན་ཐམས་ཅད་ལས་མངོན་པར་འཕགས་པའི་ཡེ་ཤེས་དང་། སྨོན་ལམ་ལས་ཡོངས་སུ་ཉམས་པར་འགྱུར་རོ། །དེ་ལྟ་བས་ན་ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔས་སེམས་ཅན་ཐམས་ཅད་གང་གིས་དགའ་བར་འགྱུར་བའི་ཚིག་དེ་སྨྲ་བར་བྱ། བྱང་ཆུབ་སེམས་དཔས་བདག་གི་དོན་དུ་སྨྲ་བར་མི་བྱ། བྱང་ཆུབ་སེམས་དཔའ་བརྩོན་འགྲུས་དམན་པར་མི་བྱའོ། །ཤཱ་རིའི་བུ་སྟོང་དུ་བགྲངས་པ་ལས་གཅིག་འགྲུབ་པའམ། འགྲུབ་པ་མེད་ཅིང་ཤཱ་རིའི་བུ་སྟོང་ཕྲག་བཅུ་དང་། སྟོང་ཕྲག་ཉི་ཤུ་དང་། སྟོང་ཕྲག་སུམ་ཅུ་དང་། སྟོང་ཕྲག་བཞི་བཅུ་དང་། སྟོང་ཕྲག་ལྔ་བཅུ་དང་། བྱེ་བ་ཕྲག་སྟོང་དུ་བགྲང་བ་ལས་གཅིག་འགྲུབ་པའམ། འགྲུབ་པ་མེད་དོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། ཤཱ་རིའི་བུ་དེ་ལྟར་གང་བྱང་ཆུབ་སེམས་དཔའི་སྤྱོད་པ་འདི་ལ་གནས་པ་དེ་ནི་མངོན་པར་འབྱུང་བར་དཀའ་བ་ཡིན་ནོ། །ཤཱ་རིའི་བུ་དེ་ལྟ་བས་ན་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་དོན་དུ་གཉེར་བས་འདི་ལྟར་སེམས་བསྐྱེད་པར་བྱ་སྟེ། བདག་གིས་སེམས་ཅན་ཐམས་ཅད་ཀྱིས་མནག་དཀའ་བ་མནག་པར་བྱ། བཟོད་དཀའ་བ་བཟོད་པར་བྱ་ཞིང་བདག་གིས་ཆོས་བཙལ་བར་བྱ་བའི་ཕྱིར། མངོན་པར་དགའ་བས་བརྩོན་པར་བྱའི་ལོངས་སྤྱོད་བཙལ་བའི་ཕྱིར་ནི་མ་ཡིན་ནོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། ཆོས་ཚོལ་བར་བྱེད་པ་ནི་ཡེ་ཤེས་ཚོལ་བར་བྱེད་པ་ཡིན་ཏེ། འདོད་པ་རྙེད་པས་རློམ་པར་མི་བྱེད། རབ་ཏུ་རློམ་པར་མི་བྱེད་དོ། །དེ་བས་ན་བྱང་ཆུབ་སེམས་དཔའི་ཐེག་པ་ལ་ཡང་དག་པར་ཞུགས་པའི་སེམས་ཅན་རྣམས་ཀྱིས་ཤེས་རབ་གཙོ་བོར་བྱའི་ལོངས་སྤྱོད་གཙོ་བོར་མི་བྱའོ། །ཤཱ་རིའི་བུ་དེ་ལྟར་ན་གང་ཆོས་ཀྱི་སྦྱིན་པ་འདི་ལ་བྱང་ཆུབ་སེམས་དཔས་མངོན་པར་བརྩོན་པར་བྱའོ། །ཤཱ་རིའི་བུ་དེ་ལ་བྱང་ཆུབ་སེམས་དཔས་མང་དུ་ཐོས་པའི་ཤེས་པ་བཙལ་བར་བྱ་ཞེ་ན། ཤཱ་རིའི་བུ་འདི་ལ་བྱང་ཆུབ་སེམས་དཔས་གང་ཐོས་པའི་ལེགས་པར་བཤད་པ་དེ་ཁྱེར་ལ་ཤོག་ཅེས་ཕྱོགས་བཞིར་སྐྱེས་བུ་བཞི་གཏང་བར་བྱ། ཁྱེད་ཅག་རྣམས་ཀྱིས་དེ་རྣམས་ན་བའི་ཉེ་གནས་དང་། ནད་གཡོག་གི་དོན་དུ་སྤྲོ་བར་བྱའོ་ཞེས་སྐྱེས་བུ་གཞན་བཞི་ཡང་གཏང་བར་བྱ་སྟེ། ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔས་ཕྱོགས་བཞིར་བྱམས་པའི་སེམས་ཀྱིས་ཁྱབ་པར་ཆོས་ཀྱི་རྗེས་སུ་འགྲོ་བའི་བློ་གཞག་པར་ཡང་བྱ། ཕྱོགས་གང་དུ་ཆོས་ཐོས་པའི་ཕྱོགས་དེར་ཡང་འགྲོ་བར་བྱའོ། །ཤཱ་རིའི་བུ་དེ་ལྟར་བྱང་ཆུབ་སེམས་དཔས་ཆོས་བཙལ་བར་བྱ་སྟེ། འདི་ནི་མང་དུ་ཐོས་པ་ཚོལ་བ་ཤེས་པ་ཡིན་ནོ། །ཤཱ་རིའི་བུ་ཆོས་བཞི་པོ་འདི་དག་ནི་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་སྒྲུབ་པར་བྱེད་པ་ཡིན་ཏེ། ཤཱ་རིའི་བུ་སྔོན་བྱུང་བ་འདས་པའི་དུས་ན་བྱང་ཆུབ་སེམས་དཔའ་མཁས་པའི་བརྩོན་འགྲུས་ཞེས་བྱ་བ་བྱུང་བར་གྱུར་ཏེ། དུས་དེའི་ཚེ་ན་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ལེགས་སྐྱེས་ཞེས་བྱ་བ་བྱུང་བར་གྱུར་པ་དེ་ཡོངས་སུ་མྱ་ངན་ལས་འདས་ནས། དེའི་བསྟན་པ་གནས་པར་གྱུར་པའི་ཚེ་བྱང་ཆུབ་སེམས་དཔའ་མཁས་པའི་བརྩོན་འགྲུས་ཞེས་བྱ་བ་དེ་མཆོད་རྟེན་ལ་ཕྱག་འཚལ་ཞིང་རྒྱུ་བ་ན་བྱང་ཆུབ་ཀྱི་ཤིང་དྲུང་ག་ལ་བ་དེ་ལོགས་སུ་སོང་སྟེ་ཕྱིན་ནས། དེས་བྱང་ཆུབ་ཀྱི་ཤིང་དེ་མཐོང་བ་ན་རབ་ཏུ་ངུ་ཞིང་བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་དེས་འདི་ལ་བཞུགས་ནས། བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་ཤིང་། བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་དེས་འདིར་བདུད་དཔུང་དང་བཅས་བཞོན་པ་དང་བཅས་པ་བཏུལ་ཏེ། སེང་གེའི་ཁྲི་འདིའི་དོན་དུ་བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་དེས་བསྐལ་པ་བརྒྱ་སྟོང་ཕྲག་དུ་མར་དཀའ་བ་སྤྱད་དེ། བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་དེས་འདི་ལ་སྐྱིལ་མོ་ཀྲུང་གཅིག་གིས་ཞག་བདུན་བཞུགས་སོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་འདི་ལ་བཞུགས་པ་ན། བཅོམ་ལྡན་འདས་དེ་མཐོང་ནས་ལྷ་བྱེ་བ་ས་ཡ་བརྒྱ་སྟོང་ཕྲག་དུ་མས་ཕྱག་འཚལ་བ་གང་ཡིན་པའི་ལྷ་དེ་དག་ནི་རྙེད་པ་ལེགས་པར་རྙེད་པ་ཡིན་ནོ། །གང་བདག་ཀྱང་དུས་ནམ་ཞིག་ན་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་འགྱུར་ཞེས་བྱང་ཆུབ་ཀྱི་ཤིང་གི་རྩ་བ་དེ་ལ་ཉིན་མཚན་བརྒྱད་དུ་རབ་ཏུ་ངུ་བར་གྱུར་ཏེ། དེའི་བར་དུ་བཟའ་བའམ། བཏུང་བ་ལ་ལོངས་མ་སྤྱད་པར་གནས་པར་གྱུར་པ་ན། གནས་གཙང་མའི་རིས་ཀྱི་ལྷ་ཉེ་བར་འོངས་ནས་དེ་ལ་འདི་སྐད་ཅེས་ལྷག་པར་གདུང་བ་མ་བྱེད་པར་སངས་རྒྱས་ཀྱི་ཡོན་ཏན་གང་ཡིན་པ་དེ་ཡོངས་སུ་ཚོལ་ཅིག །ཅེས་སྨྲས་སོ། །དེས་སྨྲས་པ། ཁྱེད་རྣམས་ཀྱིས་སངས་རྒྱས་ཀྱི་ཡོན་ཏན་ཤེས་སམ། དེ་དག་གིས་བདག་ཅག་གིས་ནི་མི་ཤེས་མོད་ཀྱི། འོན་ཀྱང་ལུས་འདི་ལ་འབྲས་ཆན་དང་ཟན་དྲོན་གྱིས་གསོས་ཏེ། བཟའ་བ་བཙལ་བ་ལ་བརྩོན་པར་གྱིས་ལ་ཁྱོད་མྱུར་དུ་འཆི་བའི་དུས་བྱེད་པར་འགྱུར་བ་མ་བྱེད་ཅིག་ཅེས་སྨྲས་སོ། །དེས་ལུས་ཉམ་ཆུང་བར་ནི་མངོན་པར་ཤེས་པ་མེད་ཀྱི་འོན་ཀྱང་གང་དུ་འགྲོ་བར་བྱ་སྟེ། འདི་ཉིད་དུ་བདག་འཆི་བའི་དུས་བྱེད་པར་བྱའོ་ཞེས་ཉིན་མཚན་བདུན་དུ་ངུ་ཞིང་འཆི་བའི་དུས་བྱའོ་ཞེས་སྨྲས་སོ། །ཤཱ་རིའི་བུ་བརྩོན་འགྲུས་དེས་བསྐལ་པ་བྱེ་བ་ཕྲག་དྲུག་ཅུ་ཕྱིར་བསྙིལ་བར་གྱུར་ཏོ། །ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་དེས་དེའི་འོག་ཏུ་བསྐལ་པ་བྱེ་བ་ཉི་ཤུ་རྩ་ལྔས་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་སོ། །དེས་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་ནས་སེམས་ཅན་བྱེ་བ་བརྒྱ་ཕྲག་སྟོང་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ལས་ཕྱིར་མི་ལྡོག་པར་ཡོངས་སུ་སྨིན་པར་བྱས་སོ། །དེའི་ཚེའི་ཚད་ནི་དཔག་ཏུ་མེད་ཅིང་འདིའི་ཉན་ཐོས་ཀྱི་དགེ་འདུན་ཡང་ཚད་མེད་ཅིང་དཔག་ཏུ་མེད་པར་གྱུར་ཏོ། །ཤཱ་རིའི་བུ་ཅིའི་ཕྱིར་མཁས་པའི་བརྩོན་འགྲུས་ཞེས་བྱ་ཞེ་ན། དེ་དུས་ཐམས་ཅད་དུ་ངེས་པར་འབྱུང་བ་ལ་བརྩོན་པ་དང་། དུས་ཐམས་ཅད་དུ་བརྩོན་འགྲུས་རྩོམ་པ་ལ་བརྩོན་པ་དང་། དུས་ཐམས་ཅད་དུ་ཐོས་པ་མང་པོ་ལ་བརྩོན་པ་སྟེ། ཤཱ་རིའི་བུ་དེ་ལྟ་བུའི་བརྩོན་འགྲུས་འདིས་བྱང་ཆུབ་སེམས་དཔའ་མཁས་པའི་བརྩོན་འགྲུས་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་སོ། །ཤཱ་རིའི་བུ་ཆོས་བཞི་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་ཤེས་རབ་དང་ལྡན་ཞིང་མྱུར་དུ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་འགྱུར་ཏེ། བཞི་གང་ཞེ་ན། ཤཱ་རིའི་བུ་བྱང་ཆུབ་སེམས་དཔའ་ཐོས་པ་མང་བ་ཡིན། ཐོས་པ་འཛིན་པ་ཡིན། ཐོས་པ་བསགས་པ་ཡིན། ཚངས་པ་མཚུངས་པར་སྤྱོད་པ་རྒན་རབས་དང་། བར་མ་དང་། གསར་བུ་ལ་གུས་པ་དང་བཅས་པ་ཡིན། དྲིས་ན་ལག་པ་གཡས་པས་ལམ་ནི་འདིའོ། །ལམ་ནི་འདིའོ་ཞེས་ཉེ་བར་སྟོན་པ་ཡིན་ཞིང་ཆོས་སྟོན་པ་ལ་བརྩོན་པ་ཡིན་ནོ། །ཤཱ་རིའི་བུ་ཆོས་བཞི་པོ་དེ་དག་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་ཤེས་རབ་དང་ལྡན་པར་འགྱུར་ཞིང་མྱུར་དུ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་འགྱུར་རོ། །དེ་ལ་ཇི་ལྟར་ཐོས་པ་བཙལ་བར་བྱ་ཞེ་ན། རང་གི་འདོད་པ་བཏང་སྟེ་མཁན་པོ་དང་། སློབ་དཔོན་གྱི་གཞུང་ལ་རབ་ཏུ་གནས་པ་དང་། གུས་པ་དང་བཅས་པར་བྱའོ། །ཆོས་བཞི་རྣམ་པར་སྤང་བར་བྱ་བ་ཡིན་ཏེ། བཞི་གང་ཞེ་ན། གཉིད་དང་། རྨུགས་པ་དང་། བྲེ་མོའི་གཏམ་དང་། ཁྲོ་བ་དང་། ཁེངས་པ་ཉིད་དེ། ཆོས་བཞི་པོ་འདི་དག་རྣམ་པར་སྤང་བར་བྱའོ། །ཆོས་བཞི་དག་ལ་གོམས་པར་བྱ་བ་ཡིན་ཏེ། བཞི་གང་ཞེ་ན། མི་ཉལ་བ་ལ་གོམས་པ་དང་། བརྩོན་འགྲུས་བརྩམས་ཤིང་འཆག་པའི་རང་བཞིན་ཅན་དུ་བྱ་བ་དང་། མཁན་པོ་དང་སློབ་དཔོན་གྱི་བསྙེན་བཀུར་ལ་གུས་པ་དང་བཅས་ཤིང་ཁེངས་པ་མེད་པ་དང་། གཞན་དག་ལ་གུས་པ་བསྐྱེད་པས་ཐོས་པ་འཛིན་པ་དང་། དེ་སེམས་ལ་བཞག་ནས་ཚིག་ཡོངས་སུ་རྫོགས་པར་བྱེད་པ་སྟེ། ཆོས་བཞི་པོ་དེ་དག་དང་ལྡན་ན་ཐོས་པ་ཚོལ་བ་ཡིན་ནོ། །ཆོས་བཞི་པོ་འདི་དག་གིས་ཀྱང་སློབ་དཔོན མཉེས་པར་བྱེད་དེ། བཞི་གང་ཞེ་ན། ངེས་པ་དང་བཤེས་པ་དང་། བཤེས་པ་བརྟན་པ་དང་། ཆོས་ལ་གནས་པ་དང་། ངེས་པའི་ཚིག་བཟང་པོ་དང་ལྡན་པ་སྟེ། ཆོས་བཞི་པོ་དེ་དག་དང་ལྡན་ན་སློབ་དཔོན་མཉེས་པར་བྱེད་དོ། །བདག་ཆོས་བཞི་དང་ལྡན་ན་དགའ་བ་འཐོབ་པར་འགྱུར་ཏེ། བཞི་གང་ཞེ་ན། བདག་གི་བཙལ་བ་དོན་ཡོད་པར་བདག་ཉིད་ཀྱིས་དགའ་བ་ཐོབ་པ་དང་། བདག་ཐོས་པའི་གཏེར་དང་ལྡན་པ་དང་། བདག་འདི་ལ་བརྟེན་ནས་ཤེས་རབ་ཀྱི་ཕུང་པོ་དང་ལྡན་པ་དང་། གཞན་དག་ལ་ཆོས་ཀྱི་སྦྱིན་པ་སྦྱིན་པ་ན་དགའ་བ་དང་རངས་པ་སྐྱེ་བར་འགྱུར་ཏེ། ཆོས་བཞི་པོ་འདི་དག་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་བདག་ཉིད་དགའ་བ་འཐོབ་པར་འགྱུར་རོ། །དེ་དེ་ལྟ་བུར་ཐོས་པ་ཚོལ་བ་ལ་གུས་པ་དང་བཅས་པ་དང་། རི་མོ་དང་ལྡན་པ་དང་། སྲི་ཞུ་དང་ལྡན་པ་དང་། བདག་ཉིད་མི་འཕྱར་བ་དང་། བདག་ཉིད་དུལ་བའི་དོན་དང་། བདག་ཉིད་ཞི་བའི་དོན་དང་། བདག་ཉིད་ཡོངས་སུ་མྱ་ངན་ལས་འདའ་བའི་དོན་དང་། གཞན་དུལ་བའི་དོན་དང་། གཞན་ཞི་བའི་དོན་དང་། གཞན་མྱ་ངན་ལས་འདའ་བའི་དོན་བྱེད་པ་ཡིན་ཏེ། བདག་གཅིག་པུའི་དོན་བྱེད་པ་ནི་མ་ཡིན་ནོ། །དེ་ལྟར་ཐོས་པ་ཚོལ་ཞིང་དེ་ལྟར་ཐོས་པ་ཡོངས་སུ་ཚོལ་བ་ན་མཐོང་བའི་ཆོས་ཀྱི་ཤེས་རབ་ཀྱང་འཕེལ་བར་བྱེད་ན་འཇིག་རྟེན་ཕ་རོལ་གྱི་ལྟ་སྨོས་ཀྱང་ཅི་དགོས་ཏེ། ཤེས་རབ་དེ་ཡང་ཐོས་པ་ནན་ཏན་དུ་བྱེད་པས་སྒྲུབ་པ་ཡིན་ནོ། །བྱང་ཆུབ་སེམས་དཔའ་ཐོས་པ་ལ་ནན་ཏན་དུ་སྒྲུབ་པར་མི་བྱེད་པ་དེས་ནི་སེར་སྐྱ་རྗེས་སུ་དྲན་པར་བྱའོ། །དེ་བས་ན་ཚངས་པ་མཚུངས་པར་སྤྱོད་པ་དག་ལ་ཡང་གུས་པ་དང་བཅས་པར་བྱ། མདུན་དུ་གནས་ཏེ་འཕྱ་བར་མི་བྱ། སྨྲ་བར་མི་བྱ། རྐང་པ་བཀྲུ་བར་མི་བྱ། མདུན་དུ་སོ་ཤིང་བཟའ་བར་མི་བྱ། མདུན་དུ་མཆིལ་མ་དོར་བར་མི་བྱ། སྒྲ་དྲག་པོས་སྨྲ་བར་མི་བྱ། མ་དྲིས་པར་སྨྲ་བར་མི་བྱ། གཏམ་གྱི་བར་ཆད་མི་བྱ། མ་སྨྲས་པར་འགྲོ་བར་མི་བྱ། སྔ་ཕྱིའི་འདུ་ཤེས་མི་བྱ། གང་ཅུང་ཟད་ཙམ་རབ་ཏུ་སྨྲ་བར་མི་བྱ། གཞན་དང་གཞན་པའི་གཏམ་མི་བྱ། དམན་པ་དང་ལྡན་པའི་གཏམ་མི་བྱ། གཞན་དྲིས་པ་ན་གཞན་གྱིས་བརྡ་སྤྲད་པར་མི་བྱ། གང་དྲིས་པ་དེ་ཉིད་བརྡ་སྤྲད་པར་བྱ། རྐུན་པོའི་གཏམ་མི་བྱ། མདོ་སྡེ་ལ་རྟོན་པར་བྱ། བདག་དང་གཞན་གྱི་སེམས་བསྲུང་བར་བྱ་ཞིང་དེ་དང་རྗེས་སུ་འཐུན་པའི་དགེ་བའི་རྩ་བ་བརྟེན་པར་བྱ། གང་སངས་རྒྱས་ཀྱི་བྱང་ཆུབ་ཡང་དག་པ་ཉིད་ལ་ཞུགས་པ་དེ་རྣམས་ལ་ཕན་གདགས་པར་བྱ། གང་མ་ཞུགས་པ་རྣམས་ལ་ཡང་གནོད་པར་མི་བྱ། དེ་ལ་རྣམ་པ་དེ་དག་གཉི་ག་ལ་ཡང་གནོད་པར་མི་བྱའོ། །གང་དག་ལ་བདག་ལ་ཁྲོས་པར་གྱུར་ནས་ཡང་དག་པ་མ་ཡིན་པ་རྗོད་པར་བྱེད་ཅིང་སྲོག་གི་བར་ཆད་བྱེད་པར་གནས་པ། ཚངས་པར་སྤྱོད་པའི་བར་ཆད་བྱེད་པ་ལ་འཇིགས་པའམ། གུས་པར་བྱ་བ་དང་། ཐ་ན་དགའ་བའི་ཚིག་གིས་ཀྱང་དེ་དང་སེམས་ཡོངས་སུ་འདྲིས་པར་བྱ། གང་ལས་ལུང་ནོད་པ་དང་། ཁ་ཏོན་འདོན་པ་དང་འབྲེལ་པ་དེ་ལ་ནི་སྟོན་པའི་འདུ་ཤེས་བསྐྱེད་ཅིང་གུས་པར་བྱའོ། །དེ་ལྟར་ཚངས་པ་མཚུངས་པར་སྤྱོད་པ་རྣམས་ལ་གུས་པར་ཡང་བྱའོ། །ཇི་ལྟར་ལམ་བསྟན་པར་བྱ་ཞེ་ན། སེམས་མཉམ་པ་ཉིད་དང་། སེམས་རབ་ཏུ་དང་བ་ཉིད་དང་། ཡང་དག་པར་འདུ་ཤེས་པ་དང་། སོམ་ཉི་མེད་པས་ལམ་ལ་ལམ་ཉེ་བར་བསྟན་པར་བྱའོ། །གལ་ཏེ་ཐེ་ཙོམ་མེད་པར་མ་གྱུར་ན་དེས་ལམ་ཤེས་པའི་སྐྱེས་བུ་དག་ལས་མཉན་ནས་ལམ་ཉེ་བར་བསྟན་པར་བྱའོ། །ཆོས་བཞི་པོ་དེ་དག་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་མྱུར་དུ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱའོ། །དེ་ལ་འདི་སྐད་ཅེས་བྱ་སྟེ། ཚུལ་བཞིན་བསྙེན་བཀུར་བྱེད་པ་དང་། །ཚུལ་བཞིན་ཆོས་ནི་ཉན་བྱེད་ཅིང་། །ཚུལ་བཞིན་ཤིན་ཏུ་དོན་སེམས་ལ། །ཆོས་རྣམས་ཀུན་གྱི་ཕ་རོལ་ཕྱིན། །ལུས་ཅན་རྣམས་ལ་ཕན་དོན་དུ། །ཆོས་དེ་ཡོངས་སུ་ཚོལ་བ་ན། །མཁན་པོ་དག་དང་ཉེ་གནས་ནི། །གུས་པ་ཀུན་གྱིས་བྱ་བ་ཡིན། །རྟག་ཏུ་བྱམས་པ་ལ་གནས་ནས། །བྱས་པ་གཟོ་དང་བྱས་པ་ཚོར། །ངོ་ཚ་ཁྲེལ་ཡོད་ཕུན་སུམ་ཚོགས། །ཕྲག་དོག་ཚུལ་འཆོས་རྣམ་པར་སྤངས། །ནད་པ་རྣམས་ནི་གསོ་བྱེད་པ། །རྙེད་དཀའ་རིན་ཆེན་འདུ་ཤེས་ཀྱིས། །དེ་དག་འདུ་ཤེས་ཉེར་གཞག་ནས། །ལེགས་པར་བཤད་པ་ཡོངས་སུ་བཙལ། །དེ་དག་ཐོས་པས་བདེར་འགྱུར་དང་། །ཤེས་རབ་རྣོན་པོའང་སྐྱེ་འགྱུར་ཞིང་། །འདོད་ཆགས་ཞེ་སྡང་གཏི་མུག་དག །ཐམས་ཅད་བྱམས་པས་ཞི་བར་འགྱུར། །དཔེར་ན་སྐྱེས་བུ་ཁྲུས་བྱས་ལ། །འགའ་ཡིས་ཕྲེང་བ་བྱིན་པ་ན། །ལག་པ་གཉིས་ཀྱིས་ཡོན་བླངས་ནས། །སྤྱི་བོ་ལ་ནི་འཇོག་པ་ལྟར། །དེ་བཞིན་ལེགས་བཤད་ཐོས་ནས་ཀྱང་། །དགའ་བའི་སེམས་ནི་སྐྱེ་འགྱུར་བ། །རྒྱལ་སྲིད་ཐོབ་པའང་དེ་ལྟ་བུའི། །དགའ་བ་རྙེད་པ་མ་ཡིན་ནོ། །རྟག་ཏུ་དེ་དག་ཆོས་སྨྲ་ལ། །མཆོག་ཏུ་བླ་མར་བྱ་བ་ཡིན། །ཆོས་ནི་རྫོགས་སངས་རྒྱལ་མཚན་ཏེ། །ཆོས་ནི་དགྲ་བཅོམ་རྒྱལ་མཚན་ཡིན། །སངས་རྒྱས་ཆོས་རྣམས་ཐོས་ནས་ཀྱང་། །བསྒྲུབ་པ་དག་ལ་རབ་གནས་ཤིང་། །ཚུལ་ཁྲིམས་ཐོས་པས་ངེས་འབྱུང་སྟེ། །དཔེར་ན་ཡར་ངོའི་ཟླ་བ་བཞིན། །མི་རྨོངས་ཞེ་སྡང་མེད་པ་ཡིས། །ལམ་མཆོག་དག་ནི་བསྟེན་པ་དང་། །བྱིས་པ་གང་ལ་གུས་བཅས་པའི། །ལྟ་བའི་གནས་ཀྱང་རྣམ་པར་སྤང་། །སེང་གེས་སྤངས་པའི་ཝ་བཞིན་དུ། །ཟིལ་གྱིས་མནན་པས་ཆོས་རྣམས་འཆད། །རླུང་གིས་སྤྲིན་གྱི་ཚོགས་བཞིན་དུ། །མུ་སྟེགས་ཅན་དེ་ཟིལ་གྱིས་མནན། །ས་ལ་སྤྲིན་གྱི་ཚོགས་བཞིན་དུ། །ཆོས་ཀྱིས་ཡོངས་སུ་ཚིམ་པར་བྱེད། །སེམས་ཅན་རྣམས་ཀྱི་སྡུག་བསྔལ་ནི། །སྨན་པའི་རྒྱལ་པོ་བཞིན་དུ་སེལ། །སེམས་ནི་རྩེ་གཅིག་མཉམ་བཞག་པས། །དེ་དག་རྣམ་རྟོག་སྤྱོད་ཡུལ་མིན། །དེ་དག་ཁྲོ་བ་ལྷུར་མི་ལེན། །འཕྱར་བ་ལ་ཡང་མི་གནས་སོ། །དུས་དང་དུས་མིན་རྣམ་ཤེས་ཤིང་། །དོན་ཤེས་པ་ཡིས་རྣམ་པར་འདྲེན། །རྟག་ཏུ་བཟོད་སྟོབས་ལ་དགའ་རྣམས། །ཤེས་པའི་ལས་རྣམས་བྱེད་པར་འགྱུར། །མཁས་དང་མཐུ་ལ་བརྟན་པ་དང་། །ཆོས་ཤེས་པ་ལ་མཁས་པ་དང་། །དཔའ་ཞིང་སེམས་སྟོབས་ཆེ་བ་སྟེ། །རླུང་གིས་ཆང་བའི་རྡུལ་བཞིན་དུ། །སྡིག་ཅན་རྣམས་ནི་ཟིལ་གྱིས་གནོན། །ཕྱོགས་རྣམས་དག་ཏུ་གསལ་གནས་པ། །ཉི་མ་འཆར་བ་ཇི་བཞིན་ནོ། །མཁས་པ་དེ་དག་རང་དོན་ལ། །ཆེད་དུ་སྨྲ་བ་བརྗོད་མི་བྱེད། །འཇིག་རྟེན་རྣམས་ལ་ཕན་པའི་ཕྱིར། །བསམ་གཏན་མངོན་ཤེས་དག་ལ་མཁས། །དེ་དག་རྣམ་པ་འདི་འདྲ་བས། །བདུད་ཀྱི་སྟེང་ན་གནས་པ་ཡིན། །སེམས་ཅན་རྣམས་ལ་བརྩེ་བའི་ཕྱིར། །འཇིག་རྟེན་མར་མེ་བྱེད་ཅིང་རྒྱུ། །ཤཱ་རིའི་བུ་གཞན་ཡང་ཆོས་བཞི་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་བཟོད་པ་སྒྲུབ་པ་ཡིན་ཏེ། བཞི་གང་ཞེ་ན། གུས་པས་ཚུལ་ཁྲིམས་སྲུང་ཞིང་སོ་སོར་ཐར་པ་ལ་ཁྱད་དུ་མི་གསོད་པ་དང་། གུས་པས་བྱམས་པ་སྒོམ་པར་བྱེད་ཅིང་གནོད་སེམས་ལ་སྦྱོར་བར་ཡང་སྤོང་བར་བྱེད་པ་དང་། ཐམས་ཅད་མཁྱེན་པ་ལ་འདུན་པ་དྲག་པོ་སྐྱེད་པར་ཡང་བྱེད་པ་དང་། དགེ་བའི་ཆོས་ཐམས་ཅད་ལ་གཅེས་པ་མི་འདོར་བ་སྟེ། ཤཱ་རིའི་བུ་ཆོས་བཞི་པོ་དེ་དག་དང་ལྡན་ན་བྱང་ཆུབ་སེམས་དཔའ་བཟོད་པ་སྒྲུབ་པ་ཡིན་ནོ། །བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་སྩལ་ཏེ། བདེ་བར་གཤེགས་པས་དེ་སྐད་གསུངས་ནས། སྟོན་པས་གཞན་ཡང་འདི་སྐད་བཀའ་སྩལ་ཏོ། །དེ་དག་སངས་རྒྱས་ཀྱིས་བསྔགས་པའི། །ཚུལ་ཁྲིམས་མ་ཉམས་མ་རལ་སྲུང་། །ཚུལ་ཁྲིམས་ཕུང་པོ་དག་པ་ལ། །དགའ་བ་ཉེ་བར་སྐྱེ་བར་འགྱུར། །ཆོས་རྣམས་མང་པོ་ཚོལ་བྱེད་ཅིང་། །ཐོས་པ་མང་ལ་ལེགས་གནས་ན། །གང་གིས་ཐེ་ཙོམ་གཅོད་བྱེད་པའི། །དེས་དེའི་ཤེས་རབ་འཕེལ་བར་འགྱུར། །སྐྱེ་དང་འཆི་བའི་རྒྱུན་དག་ལས། །འདི་དག་ཇི་ལྟར་དགྲོལ་བྱ་ཞེས། །ལྟོས་མེད་ཕན་པའི་སེམས་དག་གིས། །དམ་པའི་ཆོས་ཀྱང་སྟོན་པར་བྱེད། །གནོད་སེམས་དོན་དུ་མི་གཉེར་བས། །བྱམས་པའི་སེམས་ཀྱང་སྒོམ་པར་བྱེད། །མི་འཇིགས་པ་དང་ཕན་པའི་སེམས། །བདག་དང་སེམས་ཅན་མཉམ་པ་ཉིད། །བྱང་ཆུབ་ལ་བརྟེན་ཟིལ་གནོན་པའི། །བརྩོན་པ་དག་ཀྱང་ལེགས་པར་རྩོམ། །གང་དག་བྱང་ཆུབ་ལ་རྟེན་པའི། །དགེ་བའི་ཆོས་རྣམས་ཚོལ་བ་དང་། །བསྒོམ་པར་བྱ་བ་སྒོམ་པ་དང་། །སྤང་བར་བྱ་བའང་རྣམ་པར་སྤོང་། །ཇི་ལྟར་སྤང་བྱ་སྤངས་པ་ཡིས། །མྱུར་དུ་མྱ་ངན་འདའ་བར་འགྱུར། །དགོན་པའི་གནས་ལ་མངོན་ཕྱོགས་ཤིང་། །གནོད་པ་དག་ལ་རྒྱབ་ཀྱིས་ཕྱོགས། །ཐོས་པ་ཚོལ་ལ་བརྩོན་པ་དག །གོ་འཕང་མཆོག་ནི་རྟོགས་པར་འགྱུར། །དགེ་སློང་དགེ་སློང་མ་རྣམས་དང་། །དགེ་བསྙེན་དེ་བཞིན་དགེ་བསྙེན་མ། །ཐམས་ཅད་དང་ནི་དེ་དག་གིས། །འདྲིས་པ་རྣམ་པར་སྤང་བར་བྱ། །བྱིས་པའི་ལམ་ལ་རྟེན་པ་ཡི། །རྨོངས་པ་ཀུན་ཀྱང་སྐྱབས་མེད་དེ། །གང་ཞིག་ཆོས་ལ་རྟེན་པ་དང་། །དེ་དག་སངས་རྒྱས་མཆོག་ཏུ་གཞོལ། །མི་གང་སེམས་ཅན་དོན་དགའ་བའི། །སངས་རྒྱས་ཕ་དང་མ་ཡིན་ནོ། །གཡོ་མེད་ཆོས་ལ་འདོད་པ་དག །བྱང་ཆུབ་ལ་ནི་རབ་ཏུ་དད། །སངས་རྒྱས་དག་ཏུ་མི་ལྟ་བའི། །མི་རྣམས་མཚན་ཉིད་གསུམ་གསུངས་ཏེ། །གཡོ་དང་ཆོས་མིན་འདོད་པ་དང་། །སྙོམ་ལས་དག་ལས་རྟག་དགའ་བའོ། །འདོད་ཆགས་ཞེ་སྡང་ང་རྒྱལ་དང་། །དེ་བཞིན་སྡིག་པ་རབ་སྤངས་ནས། །བློ་ལྡན་བརྩོན་འགྲུས་ལྡན་གྱུར་པ། །བྱང་ཆུབ་ལ་ནི་རབ་གུས་འགྱུར། །བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་སྩལ་ནས། ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུ་དང་། ལྷ་དང་། མི་དང་། ལྷ་མ་ཡིན་དང་། དྲི་ཟར་བཅས་པའི་འཇིག་རྟེན་ཡི་རངས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལ་མངོན་པར་བསྟོད་དོ།། །།བྱང་ཆུབ་སེམས་དཔའི་སོ་སོར་ཐར་པ་ཆོས་བཞི་སྒྲུབ་པ་ཞེས་བྱ་བ་ཐེག་པ་ཆེན་པོའི་མདོ་རྫོགས་སོ།། །།རྒྱ་གར་གྱི་མཁན་པོ་དཱི་པཾ་ཀ་ར་ཤྲཱི་ཛྙཱ་ན་དང་། ཞུ་ཆེན་གྱི་ལོཙྪ་བ་དགེ་སློང་ཤཱཀྱ་བློ་གྲོས་དང་། དགེ་སློང་དགེ་བའི་བློ་གྲོས་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པ།།
