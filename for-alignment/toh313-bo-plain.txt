༄༅༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྱ་བྷ་དྲ་ཀ་རཱ་ཏྲཱི་ནཱ་མ་སཱུ་ཏྲ། བོད་སྐད་དུ། འཕགས་པ་མཚན་མོ་བཟང་པོ་ཞེས་བྱ་བའི་མདོ། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །འདི་སྐད་བདག་གིས་ཐོས་པ་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་རྒྱལ་པོའི་ཁབ་ན་འོད་མའི་ཚལ་བྱ་ཀ་ལན་ད་ཀ་གནས་པ་ན་བཞུགས་ཏེ། དེའི་ཚེ་དགེ་སློང་ཞིག་རྒྱལ་པོའི་ཁབ་ཀྱི་ཆུ་ཏ་པོ་ཏའི་འགྲམ་ན་འདུག་གོ། །དེ་ནས་ལྷ་ཁ་དོག་མཆོག་དང་ལྡན་པ་ཞིག་ནམ་སྲོས་ནས་དགེ་སློང་དེ་ག་ལ་བ་དེར་སོང་སྟེ་ཕྱིན་ནས་ཆུ་ཏ་པོ་ཏའི་འགྲམ་ཐམས་ཅད་སྣང་བ་རྒྱ་ཆེན་པོས་ཁྱབ་པར་སྣང་བར་བྱས་ཏེ། དགེ་སློང་དེ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། །དགེ་སློང་ཁྱོད་ཀྱིས་མཚན་མོ་བཟང་པོའི་མདོ་སྡེ་ཤེས་སམ། དེས་འདི་སྐད་ཅེས་སྨྲས་སོ། །ལྷ་བདག་གིས་ནི་མཚན་མོ་བཟང་པོའི་མདོ་སྡེ་མ་འཚལ་ན། ལྷ་ཁྱོད་ཀྱིས་མཚན་མོ་བཟང་པོའི་མདོ་སྡེ་མཁྱེན་ཏམ་མི་མཁྱེན། དེས་འདི་སྐད་ཅེས་སྨྲས་སོ། །དགེ་སློང་བདག་གིས་ཀྱང་མཚན་མོ་བཟང་པོའི་མདོ་སྡེ་མི་ཤེས་སོ། །ལྷ་འོ་ན་མཚན་མོ་བཟང་པོའི་མདོ་སྡེ་འཚལ་བ་ད་ལྟར་སུ་མཆིས། དེས་འདི་སྐད་ཅེས་སྨྲས་སོ། །རྒྱལ་པོའི་ཁབ་ཀྱི་འོད་མའི་ཚལ་བྱ་ཀ་ལན་ད་ཀ་གནས་པ་འདི་ན་བཅོམ་ལྡན་འདས་བཞུགས་ཀྱིས། ཁྱོད་བཅོམ་ལྡན་འདས་ཀྱི་ཐད་དུ་སོང་ལ་ཞུས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་དེ་ཁྱོད་ལ་ཇི་ལྟར་ལུང་བསྟན་པ་དེ་བཞིན་དུ་ཟུང་ལ་བརྩོན་པར་གྱིས་ཤིག་ཅེས་སྨྲས་ནས། ལྷ་དེ་དེ་ཉིད་དུ་མི་སྣང་བར་གྱུར་ཏོ། །དེ་ནས་དགེ་སློང་དེ་ནང་པར་ལངས་ནས་བཅོམ་ལྡན་འདས་ག་ལ་བ་དེར་སོང་སྟེ་ཕྱིན་ནས། བཅོམ་ལྡན་འདས་ཀྱི་ཞབས་ལ་མགོ་བོས་ཕྱག་འཚལ་ཏེ། ཕྱོགས་གཅིག་ཏུ་འདུག་གོ། །ཕྱོགས་གཅིག་ཏུ་འདུག་ནས་དགེ་སློང་དེས་བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཙུན་པ་མདང་དགུང་རྨོངས་ནས་བདག་གི་དེར་ལྷ་ཁ་དོག་མཆོག་དང་ལྡན་པ་ཞིག་བདག་ག་ལ་བ་དེར་མཆིས་ཏེ། ཆབ་ཏ་པོ་ཏའི་འགྲམ་ཐམས་ཅད་སྣང་བ་རྒྱ་ཆེན་པོས་ཁྱབ་པར་སྣང་བར་བགྱིས་ནས་བདག་ལ་འདི་སྐད་མཆིའོ། །དགེ་སློང་ཁྱོད་ཀྱིས་མཚན་མོ་བཟང་པོའི་མདོ་སྡེ་ཤེས་སམ། བདག་གིས་དེ་ལ་འདི་སྐད་ཅེས་བགྱིས་སོ། །ལྷ་བདག་གིས་ནི་མཚན་མོ་བཟང་པོའི་མདོ་སྡེ་མ་འཚལ་ན། ལྷ་ཁྱོད་ཀྱིས་ལྟ་མཚན་མོ་བཟང་པོའི་མདོ་སྡེ་མཁྱེན་ཏམ་མི་མཁྱེན། དེ་འདི་སྐད་ཅེས་མཆིའོ། །དགེ་སློང་བདག་གིས་ཀྱང་མཚན་མོ་བཟང་པོའི་མདོ་སྡེ་མི་ཤེས་སོ། །ལྷ་འོ་ན་མཚན་མོ་བཟང་པོའི་མདོ་སྡེ་འཚལ་བ་ད་ལྟར་སུ་མཆིས། དེ་འདི་སྐད་ཅེས་མཆིའོ། །རྒྱལ་པོའི་ཁབ་ཀྱི་འོད་མའི་ཚལ་བྱ་ཀ་ལན་ད་ཀ་གནས་པ་ན་བཅོམ་ལྡན་འདས་བཞུགས་ཀྱིས། ཁྱོད་བཅོམ་ལྡན་འདས་ཀྱི་ཐད་དུ་སོང་ལ་ཞུས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་དེ་ཁྱོད་ལ་ཇི་ལྟར་ལུང་བསྟན་པ་དེ་བཞིན་དུ་ཟུང་ལ་བརྩོན་པར་གྱིས་ཤིག་ཅེས་སྨྲས་ནས། ལྷ་དེ་དེ་ཉིད་དུ་མི་སྣང་བར་གྱུར་ཏོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་དེའི་སླད་དུ་བདག་འདིའི་དོན་ཡོངས་སུ་ཞུའོ། །བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། དགེ་སློང་ཁྱོད་ཀྱིས་ལྷ་དེ་ཤེས་སམ། དགེ་སློང་དེས་གསོལ་པ། བཙུན་པ་བདག་གིས་ལྷ་དེ་མ་འཚལ་ཏོ། །བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། ལྷའི་བུ་དེ་ནི་སུམ་ཅུ་རྩ་གསུམ་གྱི་ལྷ་རྣམས་ཀྱི་སྡེ་དཔོན་ཏེ། ཙན་དན་ཞེས་བྱའོ། །དགེ་སློང་དེས་གསོལ་པ། བཙུན་པ་བདག་མཚན་མོ་བཟང་པོའི་མདོ་སྡེ་ཐོས་པར་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། དགེ་སློང་ཁྱོད་དེའི་ཕྱིར་ལེགས་པར་རབ་ཏུ་ཉོན་ལ་ཡིད་ལ་ཟུང་ཤིག་དང་། ངས་ཁྱོད་ལ་བཤད་པར་བྱའོ། །དགེ་སློང་ཆོས་གསུམ་དང་ལྡན་ན་དགེ་སློང་མཚན་མོ་བཟང་པོའི་མདོ་སྡེ་ལ་གནས་པ་ཞེས་བྱའོ། །གསུམ་གང་ཞེ་ན། འདི་ལྟ་སྟེ། འདས་པ་ལ་རྗེས་སུ་ཕྲད་པར་མི་བྱེད་པ་དང་། མ་འོངས་པ་ལ་རེ་བ་མེད་པ་དང་། ད་ལྟར་བྱུང་བའི་ཆོས་རྣམས་ཀྱིས་མི་འཕྲོགས་པར་གནས་པའི་ཆོས་ཅན་ཏེ། ཆོས་གསུམ་པོ་དེ་དག་དང་ལྡན་ན་དགེ་སློང་མཚན་མོ་བཟང་པོའི་མདོ་སྡེ་ལ་གནས་པ་ཞེས་བྱའོ། །བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་སྩལ་ཏེ། བདེ་བར་གཤེགས་པས་དེ་སྐད་གསུངས་ནས། སྟོན་པས་གཞན་ཡང་འདི་སྐད་གསུང་ངོ་། །འདས་ལ་རྗེས་སུ་ཕྲད་མི་བྱེད། །མ་འོངས་པ་ལ་རེ་བ་མེད། །འདས་པ་གང་ཡིན་དེ་འགགས་ཏེ། །མ་འོངས་དེ་ནི་མ་ཕྱིན་པའོ། །གང་དག་ད་ལྟར་བྱུང་བའི་ཆོས། །དེ་དང་དེ་ལ་རབ་བལྟས་ནས། །རྣམ་པར་རྟོག་པས་མི་འཕྲོགས་པར། །དེ་དག་ཐམས་ཅད་ཁོང་དུ་ཆུད། །སང་ཙམ་ཤི་ཡང་སུ་ཤེས་ཀྱིས། །དེ་རིང་ཉིད་དུ་བརྟུན་ཏེ་བྱ། །འཆི་བདག་སྡེ་ཆེན་དེ་དང་ནི། །བདག་ཏུ་བཤེས་པ་མ་ཡིན་ནོ། །སེམས་ཅན་ཀུན་དང་སྲོག་ཆགས་ཀུན། །འབྱུང་པོ་ཀུན་དང་ཐམས་ཅད་ཀྱང་། །བདེ་བ་འབའ་ཞིག་གྲུབ་གྱུར་ཏེ། །ཐམས་ཅད་ཉོན་མོངས་མེད་པར་ཤོག །ཐམས་ཅད་བཟང་པོ་མཐོང་གྱུར་ཏེ། །གང་ཡང་སྡིག་པར་མ་གྱུར་ཅིག །དེ་ལྟར་གནས་ཤིང་རབ་བརྩོན་ལ། །ཉིན་མཚན་སྙོམ་ལས་མེད་པས་ན། །དེ་ཕྱིར་མཚན་མོ་བཟང་པོ་ཡི། །མདོ་སྡེ་ཐུབ་པས་རྟག་ཏུ་གསུངས། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་སེམས་ཅན་ཐམས་ཅད་རྗེས་སུ་གཟུང་བའི་ཕྱིར་མཚན་མོ་བཟང་པོའི་མདོ་འདི་བཤད་ནས་འགྲོ་ལྡིང་བའི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་གསུངས་སོ། །ཏ་ཏྱ་ཐཱ། བི་ནི་བྷ་ར་དི། བི་རི་ནི། བི་ར་ནི། བུད་དྷ་མ་ཧར་ཏཱ་ན་དྷེ། མ་ཧི་ནི་མི་ན་ནི། ནི་ནི་ནི་ནི། ཏི་ཏི་ཏི་ཏི། བི་ར་ཏི། གཽ་རི། གན་དྷ་རི། ཙན་ཌ་ལི། མ་ཏང་གི། པུ་ཀ་སི། བྲནྷ་ནི། དྲ་མི་ཌི། དྲ་མི་ཌི། ཤཱ་བྷ་དི། ས་དཱ་ལམ་བྷེ། ཧི་ནཱ་མ་ཏྱ་མ་ཨུད་ཏ་ར་ནི། མ་ཧོ་ལ་ན། ད་ལ་བྷ་ནི། ད་ལཱ་བྷ་དྲེ། མ་ཧཱ་ད་ལཱ་ནི། ཙ་ལཱ་ནི། ཨཱ་བྷེ་ཙཀྲ་བྷ་ཀི། མ་ཧཱ་ཙ་ཀྲ་བྷ་ཀི། ཤ་བ་རི། ཤ་བ་རི། མ་ཧཱ་ཤ་བ་རི། བྷུ་ཙིད་གི་ནི། བྷུ་ཙི་རིང་གི་ནི། ནི་མི་ནི་མིང་གི་ནི། ནི་མིན་དྷ་རི་བྷུ་ཏཱ་ནི་སྭཱཧཱ། ཏ་ཏྱ་ཐཱ། ཨེ་ཏུ་ཏུ་ཏུ། ན་ཏུ་མི་རྣི། ཀེ་རྣི་ཨིང་ཀུ། སུད་ཏ་རི་ཙ་ལ་ལ། ན་ར་ཏ། ན་ར་མ་ལུ་ལུ། ག་ལུ་ལུ། ཨ་བ་ཏྲ་ནོ་ནེ། ཤོ་ཀ་ནི་ར། བྷ་ར་དྷ་ས་ལེ། ཏ་ར་ཀྲེ་ཏ་རེ་དྷཱ་ཏུ། ནོད་ཏུ། ཏ་ལ་ལཱ་ལུ། བ་ཏ་བྷ་ཏི། ཡ་བྷ་ཏི་ཡེ་སྭཱ་ཧཱ། ཏ་ཏྱ་ཐཱ། ཨ་ར་ཀཱ་ཏེ། ན་ར་ཀཱ་ཏེ། བཾ་སུ་ཀ། བ་ཡཻ་ཡེ། ཀ་པོ་ཏ་ཀ་བ་ཡཻ་ཡེ་ཏ་བྷོ་ཏ་ནེ་སྭཱ་ཧཱ། དགེ་སློང་རིགས་ཀྱི་བུའམ། རིགས་ཀྱི་བུ་མོ་གང་ལ་ལ་ཞིག་གིས་སེམས་ཅན་ཐམས་ཅད་རྗེས་སུ་གཟུང་བའི་ཕྱིར་མཚན་མོ་བཟང་པོའི་མདོ་སྡེ་འདིའི་དོན་དང་། ཚིག་དང་། ཚིགས་སུ་བཅད་པ་དང་། འགྲོ་ལྡིང་བའི་སྔགས་ཀྱི་གཞི་འཆང་ངམ། འཛིན་ཏམ། ཀློག་གམ། ཡིད་ལ་བྱེད་དམ། ཀུན་ཆུབ་པར་བྱེད་དམ། གཞན་དག་ལའང་རྒྱ་ཆེར་ཡང་དག་པར་སྟོན་ན། དེའི་ལུས་ལ་མེས་མི་ཚུགས་པར་འགྱུར། དུག་གིས་མི་ཚུགས་པ་ཡིན། རྒྱལ་པོའི་ཆད་པས་མ་ཡིན། རྐུན་པོའི་ཆད་པས་མ་ཡིན་ཏེ། དུས་མ་ཡིན་པར་འཆི་བའི་དུས་བྱེད་པར་ཡང་མི་འགྱུར་ཞིང་ངེས་པར་མྱ་ངན་ལས་འདས་པར་ཉེ་བར་འགྲོ་བར་འགྱུར་རོ། །གང་དང་གང་དུ་སྐྱེས་པར་ཡང་། །མཚན་མོ་བཟང་པོའི་མདོ་བཀླགས་ན། །སྐྱེ་བ་དྲན་དང་དགྲ་བཅོམ་དང་། །དབང་ཕྱུག་ཏུ་ཡང་ངེས་པར་འགྱུར། །ལན་ཅིག་གིས་ནི་གནས་ཀུན་ཏུ། །དུས་མིན་འཆི་དང་བར་མ་དོར། །བདག་གིས་བདག་ལ་མི་སྦྱོར་བར། །དུས་མིན་འཆི་དང་ཆོས་མ་ཡིན། །སེམས་ཅན་ཐམས་ཅད་སྲུང་ཞིང་སེལ། །ནད་དང་མྱ་ངན་འཇིགས་སྐྱེ་དང་། །རྨི་ལམ་མི་ཤིས་མྱ་ངན་དུ། །བརྗོད་ན་དཔལ་དུ་བྱེད་པར་འགྱུར། །གང་གིས་བྲིས་ཏེ་སྐྱེས་པ་དང་། །བུད་མེད་ལུས་ལ་བཏགས་ན་ནི། །སྐལ་པ་བཟང་དང་གཟི་བརྗིད་དང་། །དེ་ཡི་དཔལ་ཡང་སྐྱེ་བར་འགྱུར། །ལུས་དང་ངག་དང་ཡིད་ཀྱིས་སུ། །མི་དགེ་ཅུང་ཟད་གང་བྱས་པ། །དེ་དག་ཐམས་ཅད་བཟང་པོའི་མདོས། །དུས་མ་ཡིན་པར་འཆི་ལས་སྐྱོབ། །རྒྱལ་པོ་ཆུ་དང་མེ་དང་གློག །ཆོམ་པོ་ནད་དང་དགྲ་དག་དང་། །གཡུལ་སྤྲོད་པ་དང་འཐབ་པར་ཡང་། །བཟང་པོ་དྲན་ན་སྐྱོབ་པར་བྱེད། །རིག་པ་བཟང་པོ་འདི་དང་ནི། །གསང་སྔགས་འདི་དག་རྟག་བརྗོད་ན། །མི་དག་མི་ཞུམ་དུས་ཀུན་ཏུ། །དེ་བས་འདི་ནི་ཡིད་ཀྱིས་བརྗོད། །འདས་པ་དང་། མ་བྱོན་པ་དང་། ད་ལྟར་བྱུང་བའི་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །ཕྱག་འཚལ་ནས་རིག་སྔགས་འདི་དག་བརྗོད་པར་བྱ་སྟེ། རིག་སྔགས་འདིས་བདག་འབྱོར་པར་གྱུར་ཅིག །ཏ་ཏྱ་ཐཱ། ནི་མི་ནི་མིན་དྷ་རེ། ཏེ་མི་ནི་མིན་དྷ་རེ། ཏི་མི་ཏི་མིང་ག་ལི། ཏྲཻ་ལོ་ཀ་ཨ་བ་ལོ་ཀ་ནི། ཏྲི་ཤུ་ལ་དྷ་ར་ཎ། ཨ་ཀུ་ཕི་ནི། ཀྲི་མཱི་ཀྲི་ཏི། ཀི་ལི་ཀི་ལི། ཀུད་ཏོ་ཀུད་ཏོ། ཀུད་ཏོ་ཀུད་སི། ཀུརྟི་ཀུ་པི་ཏི། བདག་དང་སེམས་ཅན་ཐམས་ཅད་ལག་ན་རྡོ་རྗེ་དང་ལྷན་ཅིག་ཏུ་འཇིགས་པ་ཐམས་ཅད་དང་། ནད་ཐམས་ཅད་དང་། དུག་ཐམས་ཅད་དང་། རིམས་ཐམས་ཅད་དང་། གདོན་ཐམས་ཅད་དང་། གདུག་པས་ཟིན་པ་ཐམས་ཅད་དང་། ཆོམ་རྐུན་ཐམས་ཅད་དང་། མི་དང་། མི་མ་ཡིན་པ་ཐམས་ཅད་དང་། འཇིགས་པ་དང་། གནོད་པ་དང་། ནད་འགོ་བ་ཐམས་ཅད་དང་། འཐབ་པ་དང་། རྙོག་པ་དང་། སྡིག་པ་དང་། རྩོད་པ་དང་། འགྱེད་པ་དང་། མཚང་འདྲུ་བ་ཐམས་ཅད་དང་། ལུས་དང་། ངག་དང་། ཡིད་ཀྱིས་ངན་པར་སྤྱད་པ་ཐམས་ཅད་དང་། ཐམས་ཅད་དུ་འཇིགས་པ་ཐམས་ཅད་ལས་སྲུངས་ཤིག །སྲུངས་ཤིག །ཏ་ཏྱ་ཐཱ། ཧི་ལི་མི་ལི། ཙི་ལི། ཨི་ལི། མི་ལི། པི་ཤ་ཙི་པཱ་རྞ་ཤ་བ་རི་ཙི་ལི་ཙི་ལི་སྭཱ་ཧཱ། བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་སྩལ་ནས། དགེ་སློང་དེ་དག་དང་། ཐམས་ཅད་དང་ལྡན་པའི་འཁོར་དེ་དག་དང་། ལྷ་དང་། མི་དང་། ལྷ་མ་ཡིན་དང་། ནམ་མཁའ་ལྡིང་དང་། དྲི་ཟ་དང་། མིའམ་ཅི་དང་། ལྟོ་འཕྱེ་ཆེན་པོ་དང་། གནོད་སྦྱིན་ཆེན་པོ་དང་། སྲིན་པོ་དང་། ཡི་དགས་དང་། ཤ་ཟ་དང་བཅས་པའི་འཇིག་རྟེན་ཡི་རངས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལ་མངོན་པར་བསྟོད་དོ། །འཕགས་པ་མཚན་མོ་བཟང་པོ་ཞེས་བྱ་བའི་མདོ་རྫོགས་སོ།། །།
