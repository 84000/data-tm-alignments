༄༅༅། །རྒྱ་གར་སྐད་དུ། ཤྲཱི་ཁ་ས་མ་ཏནྟྲ་རཱ་ཛ་ནཱ་མ། བོད་སྐད་དུ། དཔལ་ནམ་མཁའ་དང་མཉམ་པའི་རྒྱུད་ཀྱི་རྒྱལ་པོ་ཞེས་བྱ་བ། དཔལ་བདེ་བ་ཆེན་པོ་ལ་ཕྱག་འཚལ་ལོ། །འདི་སྐད་བདག་གིས་ཐོས་པ་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་སྐུ་དང་གསུང་དང་ཐུགས་རྡོ་རྗེ་བཙུན་མོའི་བྷ་ག་ལ་སྙོམས་པར་བཞུགས་སོ། །དེའི་ཚེ་འཁོར་འདུས་པ་དེ་དག་གི་ནང་ནས་མཁའ་སྤྱོད་མ། ས་སྤྱོད་མ་དང་། ས་འོག་ན་སྤྱོད་པ་མ་གསུམ་ལ་བཅོམ་ལྡན་འདས་ཀྱིས་ཞལ་འཛུམ་པ་མཛད་དེ། གསང་བའི་དོན་འདི་གསུངས་པ། །དད་ལྡན་སྙིང་རྗེར་ལྡན་དང་རྩེ་གཅིག་སེམས། །ལུས་ངག་གུས་བཅས་གཞན་ལ་ཕན་སེམས་ལྡན། །འཇིགས་མེད་བརྟུལ་ཞུགས་ལྡན་ཞིང་ཟབ་མོ་ཡི། །ལམ་ལ་སྤྲོ་བར་གྱིས་ལ་ལེགས་པར་ཉོན། །མཁའ་འགྲོ་ཁྱོད་ཀྱི་མཚན་ཉིད་ནི། །ང་ཡིས་ཁྱོད་ལ་བཤད་པར་བྱ། །མཁའ་སྤྱོད་སྟེང་སྒོའི་རླུང་ཡིན་ཏེ། །ས་སྤྱོད་བར་དུ་ཐིག་ལེའི་སྒོ། །ས་འོག་སྤྱོད་པ་ཤུ་ཀྲའི་སྒོ། །སྒོ་གསུམ་སྐུ་གསུང་ཐུགས་ཡིན་ཏེ། །དང་པོ་མཁའ་སྤྱོད་མཚན་ཉིད་ཀྱང་། །རྣམ་པ་གསུམ་དུ་ཤེས་བྱ་སྟེ། །བེམས་པོའི་རླུང་དང་རྣམ་རྟོག་རླུང་། །ཀུན་གྱི་མཚན་ཉིད་ཡེ་ཤེས་རླུང་། །མདོར་བསྟན་དེ་ཉིད་རིམ་པས་དབྱེ། །གཉིས་མེད་རླུང་ནི་མ་ནིང་སྟེ། །ཟུང་འཇུག་ཐབས་དང་ཤེས་རབ་རླུང་། །ལྡང་འཇུག་སྐུ་གསུང་ཐུགས་ཀྱི་རླུང་། །སྙིང་དང་གཞང་ཁ་མགྲིན་པ་དང་། །ལྟེ་བ་ལུས་ཀྱིས་ཁྱབ་པ་རུ། །རིགས་ལྔའི་རླུང་ནི་ཁྱབ་པར་གནས། །རྒྱུ་བ་བཞི་ནི་འབྱུང་བའི་རླུང་། །དངས་སྙིགས་འབྱེད་པ་བརྒྱད་དུ་འགྱུར། །འདི་ཉིད་རླུང་ནི་ཐུན་བརྒྱད་དེ། །ཐུན་རེ་ལ་ནི་འཕོ་བ་གཉིས། །རླུང་གི་འཕོ་བ་བཅུ་དྲུག་ཡིན། །འཕོ་བ་གཅིག་ལ་ཆུ་ཚོད་གཉིས། །ཆུ་ཚོད་རེ་ལ་དབྱུག་གུ་གཉིས། །རླུང་གི་དབྱུག་གུ་དྲུག་ཅུ་བཞི། །དེ་བཞིན་ཆུ་ཚོད་སུམ་ཅུ་གཉིས། །ཉི་ཁྲི་ཆིག་སྟོང་དྲུག་བརྒྱ་སོགས། །སེམས་དང་སེམས་མིན་འདྲེས་པའི་རླུང་། །རགས་པ་དང་ནི་ཕྲ་བ་ལ། །མཁས་པས་སེམས་ཀྱི་བཞོན་པར་གཟུང་། །རྣམ་རྟོག་གཟུང་འཛིན་སེམས་ཀྱི་རྟ། །གཟུང་འཛིན་གཉིས་འགགས་དབུ་མར་གནས། །དེ་ཉིད་ཡེ་ཤེས་རྡོ་རྗེའི་རླུང་། །དབྱེ་བ་ཐབས་དང་ཤེས་རབ་གཉིས། །འདི་རྣམས་མཁའ་སྤྱོད་རང་བཞིན་ཏེ། །དེ་དག་མན་ངག་མཁས་པས་བརྟག །ཡོན་ཏན་སྐྱོན་དང་རྟགས་རྣམས་ཀྱིས། །རླུང་གི་མཚན་ཉིད་ཤེས་པར་བྱ། །དེ་དག་ཀུན་གྱི་རྟེན་ཉིད་ཀྱང་། །གཞོལ་མ་དང་ནི་གཡོན་མ་དང་། །འདི་ནི་གཞུ་ཡི་རྣམ་པར་གནས། །བེམས་པོ་རླུང་གི་ལམ་ཡིན་ཏེ། །དབང་པོ་མངོན་སུམ་ལམ་ཡིན་པས། །སྣ་ཡི་རི་རབ་གླིང་བཞི་ལ། །ཉི་ཟླ་བྱང་ནུབ་ལྷོར་རྒྱུག་གིས། །ཐབས་དང་ཤེས་རབ་སྐོར་སོགས་ལ། །འབད་པས་སེམས་ནི་གཟུང་བར་བྱ། །འདི་དག་བེམས་པོས་ཡང་དག་གནས། །མི་ཤེས་པ་ཡི་རབ་རིབ་ཀྱིས། །རིན་ཆེན་སེམས་ནི་དྲི་ཅན་བྱས། །འདི་ནི་རྣམ་རྟོག་རླུང་ཡིན་ཏེ། །འདི་ཉིད་ཀྱིས་ནི་ལམ་གྱུར་བ། །རོ་མ་དང་ནི་རྐྱང་མའོ། །སྦྲུལ་འཁྱིལ་ལྟ་བུའི་རྣམ་པ་སྟེ། །རྣལ་འབྱོར་གྱི་ནི་དམིགས་པའི་ཡུལ། །གནས་དེར་རྣམ་རྟོག་རླུང་འགག་གོ། །རང་རིག་གདོད་ནས་མ་སྐྱེས་པ། །འདི་ཉིད་ཡེ་ཤེས་རྡོ་རྗེའི་རླུང་། །ཀུན་འདར་མ་ཡི་རྩ་ལ་བརྟེན། །ཙཎྜཱ་ལཱི་ནི་ཤིང་ལྟ་བུ། །རང་རིག་ཡེ་ཤེས་སྤྱོད་ཡུལ་ཏེ། །གནས་དེར་འབད་དེ་རང་རིག་གཟུང་། །མཁའ་སྤྱོད་རླུང་གི་རང་བཞིན་ཕྱིར། །འདི་དག་རྡོ་རྗེའི་ལམ་ཡིན་ནོ། །དེ་ནས་ས་སྤྱོད་བཤད་བྱ་སྟེ། །རང་རིག་ཡེ་ཤེས་གང་འདོད་པས། །ཙཎྜཱ་ལཱི་ཡི་བྱེ་བྲག་བརྟག །དང་པོར་རང་བཞིན་ཙཎྜཱ་ལཱི། །གཉིས་པ་གཏུམ་མོ་ཙཎྜཱ་ལཱི། །གསུམ་པ་ཁ་སྦྱར་ཙཎྜཱ་ལཱི། །དེ་བཞིན་གཅེར་བུའི་ཙཎྜཱ་ལཱི། །དེ་ཉིད་ལ་ཡང་དབྱེ་བ་སྟེ། །གཅེར་བུ་གཏུམ་མོ་ཤིན་ཏུ་གཏུམ། །ཡང་དག་གཏུམ་མོ་གཞན་སྦྱར་དང་། །བྱེ་བྲག་ཀུན་ལ་སྦྱར་བར་བྱ། །གཅེར་བུ་ཤིན་ཏུ་གཅེར་བུ་དང་། །ཤིན་ཏུ་གཅེར་བུ་གཅེར་བུ་སོགས། །འདི་དག་ཀུན་ལ་སྦྱར་བར་བྱ། །དབྱེ་ན་རྣམ་པ་བཅུ་གཉིས་ཏེ། །ཡན་ལག་བཅུ་ཚན་བཞི་དང་བརྒྱད། །ས་སྤྱོད་མ་ཡི་རང་བཞིན་ཏེ། །སྟོང་ཕྲག་བདུན་ཅུ་རྩ་གཉིས་སོགས། །གཞན་སྐྱེས་ཀྱི་ནི་ཁྲོ་མོ་ཙམ། །ཕྲ་བ་དེ་ལས་བཟློག་པའོ། །བརྒྱ་དང་ཉི་ཤུ་སྐྲ་ཡི་རྩེ། །སུམ་ཅུ་རྩ་གཉིས་སྐྲ་ཡི་རྩེ། །ལྔ་ཡི་ཆ་སྟེ་གཙོ་བོ་གསུམ། །གསུམ་འདུས་ཡུངས་ཀར་གྱི་ནི་ཚད། །དབུས་དང་གཡས་གཡོན་ཕྲ་སྦོམ་དང་། །གཉིས་འགྱུར་བྱེ་བྲག་ཤེས་པར་བྱ། །དངས་མ་དང་ནི་སྙིགས་མའི་རྩར། །རང་བཞིན་དུ་ནི་གནས་པ་ལ། །ཡི་གེ་ལྟ་བུའི་རྩ་མདུད་ཀྱང་། །དང་པོའི་སྒྲ་རུ་གང་ཤེས་པ། །ས་སྤྱོད་མ་ཡི་རང་བཞིན་ནོ། །གནས་ལ་སོགས་པ་གཞན་ནས་ཤེས། །འདི་ཉིད་ནོར་བུ་རིན་པོ་ཆེ། །འདི་ལ་བརྟེན་ནས་ཐམས་ཅད་འབྱུང་། །རིག་འཆོལ་གནས་འཆོལ་ཚད་འཆོལ་དང་། །དབྱིབས་དང་ཁ་དོག་གྲངས་ལ་སོགས། །འཆོལ་པ་ལས་ནི་དངོས་གྲུབ་མིན། །དེ་ཉིད་བྲལ་བའི་རང་བཞིན་ནི། །ས་སྤྱོད་མ་ཡི་དངོས་གྲུབ་སྟེ། །ཐམས་ཅད་ཀུན་གྱི་རྟེན་ཡིན་ནོ། །ས་འོག་སྤྱོད་མའི་རང་བཞིན་ཏེ། །དེ་ནས་ཡང་དག་རབ་བཤད་པ། །བྱང་ཆུབ་སེམས་ནི་རྣམ་པ་གསུམ། །དང་པོ་ཆོས་ཉིད་སེམས་ཡིན་ཏེ། །འདི་ནི་ཤེས་བརྗོད་ཡུལ་ལས་འདས། །གཉིས་པ་རང་རིག་བྱང་ཆུབ་སེམས། །དཔེར་ན་རྐྱེན་གྱིས་ཤེལ་སྒོང་བཞིན། །གསུམ་པ་བེམས་པོའི་བྱང་ཆུབ་སེམས། །སིཧླ་དང་ནི་ག་པུར་ཏེ། །དོན་དམ་ཀུན་རྫོབ་གཉིས་སུ་གསུངས། །ལས་ཀྱི་རླུང་དང་ལས་ཀྱི་མེས། །གཡོ་བ་ཀུན་རྫོབ་བྱང་ཆུབ་སེམས། །ཡེ་ཤེས་རླུང་དང་ཡེ་ཤེས་མེས། །གཡོ་བ་དོན་དམ་སེམས་ཡིན་ཏེ། །དངས་སྙིགས་ལས་ནི་རིམ་པ་ཡིན། །དབྱིབས་དང་ཁ་དོག་ལས་ཀྱིས་ནི། །སེམས་ཀྱི་ཁྱད་པར་ཤེས་པར་བྱ། །ལས་ཀྱི་རླུང་གིས་མི་གཡོ་ཞིང་། །ལས་ཀྱི་མེ་ཡིས་མི་འཛག་པར། །ཁ་བ་ལྟ་བུའི་མཚན་ཉིད་ལས། །མི་ཤིགས་ཧྃ་གི་དབྱིབས་འདྲ་སྟེ། །ནོར་བ་ཀུནྡའི་མདོག་ལྟ་བུ། །དཀར་ལ་འཚེར་དང་གཟི་མདངས་སྣུམ། །བདེ་བ་ཆེན་པོའི་སྐུར་འབྱུང་བ། །འདི་ནི་དོན་དམ་བྱང་ཆུབ་སེམས། །དམར་ལ་དངས་པའི་རང་བཞིན་ལ། །ཁྱད་པར་བཞི་ནི་ཐུན་མོང་སྟེ། །འདི་ནི་ཤེས་རབ་བྱང་ཆུབ་སེམས། །དེ་ལས་བཟློག་པའི་དམན་པ་གཉིས། །འོག་སྒོར་གནས་ཤིང་སྲིད་པ་སྐྱེ། །དམན་པའི་རྐྱེན་གྱིས་གཡོ་བར་བྱེད། །དེ་ཕྱིར་དོན་དམ་ཀུན་རྫོབ་ཀྱིས། །སེམས་ཀྱི་མཚན་ཉིད་ཤེས་པར་བྱ། །རྒྱུ་འཆོལ་བ་ལས་དངོས་གྲུབ་མེད། །རྐྱེན་འཆོལ་བ་ལས་དེ་བཞིན་ཏེ། །བླ་མའི་ཞལ་ནས་ཤེས་པར་བྱ། །དོན་དམ་སེམས་ཀྱི་སྤྱོད་ཡུལ་མིན། །ཀུན་རྫོབ་ཚོར་བ་བདེ་བའི་ཡུལ། །ས་འོག་སྤྱོད་པའི་རང་བཞིན་ཏེ། །རྡོ་རྗེ་གསུམ་བསྒོམ་མི་ཕྱེད་པ། །དེ་ཉིད་རྡོ་རྗེ་སེམས་དཔའ་སྟེ། །ཡི་གེ་དང་ནི་ཡན་ལག་དང་། །བསྡུས་པའི་དོན་དང་སྤྱི་ཡི་དོན། །སྦས་པའི་དོན་དང་མཐར་ཐུག་དོན། །སྣ་ཚོགས་སེམས་ལས་འབྱུང་བར་འགྱུར། །ནོར་བུ་དང་ནི་སིལ་སྙན་བཞིན། །རྟོག་མེད་རྟེན་ལས་སྣ་ཚོགས་འབྱུང་། །ཨཱ་ལི་ཀཱ་ལི་མ་ནིང་དང་། །དྲུག་ཏུ་སྦྱོར་བ་བརྟག་པར་བྱ། །སྒྲ་མཐའི་ཡི་གེ་བསམ་མི་ཁྱབ། །ཡན་ལག་རྣམ་པ་གསུམ་ཡིན་ཏེ། །རྒྱུ་འབྲས་བརྡ་དང་བརྡ་ཅན་དང་། །བརྗོད་བྱ་དང་ནི་རྗོད་བྱེད་དོ། །དེ་ལས་བཟློག་པ་བསྡུས་པར་བཤད། །ཐེག་པ་གསུམ་དང་དེ་ལས་གཞན། །ཐུན་མོང་སྤྱི་རུ་ངེས་པ་སྟེ། །བྱེ་བྲག་རང་ལས་ཤེས་པར་བྱ། །སྦས་པ་དབང་དང་སྔགས་སུ་དབྱེ། །སྔགས་ལ་བཞི་སྟེ་དབང་ལ་གཉིས། །ཆོས་ནི་སྤྱི་དང་བྱེ་བྲག་གོ། །དེ་དག་ཀུན་གྱི་མཚན་ཉིད་ཀྱང་། །བླ་མའི་ཞལ་ནས་ཤེས་པར་བྱ། །སྣོད་དང་སྣོད་མིན་བྱེ་བྲག་ཀྱང་། །བླ་མས་ལེགས་པར་བརྟག་བྱ་སྟེ། །དེ་ཉིད་ཡང་དག་རྟོགས་གྱུར་ན། །སྟོན་པ་མེད་ཅིང་བསྟན་པ་མེད། །བསམ་གཏན་ལ་སོགས་འཇིག་པ་མེད། །སངས་རྒྱས་གཟུགས་ནི་སྣ་ཚོགས་འབྱུང་། །བྱ་བ་བཅུ་གཉིས་ལ་སོགས་པ། །ཆོས་ཀྱི་འཁོར་ལོ་སྣ་ཚོགས་བསྐོར། །དམན་པ་རྣམས་ལ་བྱ་བའི་རྒྱུད། །བྱ་མེད་རྣལ་འབྱོར་དེ་ལྟག་ལ། །དེ་རྗེས་རྣལ་འབྱོར་ཉིད་གསུངས་ཏེ། །རྣལ་འབྱོར་ནང་པ་དེ་ལྟག་ལ། །ཕྱི་ནས་རྣལ་འབྱོར་མ་རྒྱུད་ནི། །ཤེས་རབ་སྙིང་པོ་བརྩེར་གྱུར་པའི། །ཐབས་དང་ཤེས་རབ་བསྲུབ་ཤིང་ལ། །གཉིས་མེད་ཐུགས་ཀྱིས་བསྲུབས་པ་ལས། །བདུད་རྩིའི་རོ་ནི་ཉི་ཤུ་བཞི། །ནོར་བུ་རིན་ཆེན་རྒྱུད་ཅེས་བརྗོད། །འཇིག་རྟེན་འཇིག་རྟེན་འདས་པ་ཡི། །དངོས་གྲུབ་ཐམས་ཅད་འདི་ལས་འབྱུང་། །ཇི་ལྟར་མ་དཀྲོགས་འོ་མ་ལས། །འདོད་པ་གང་ཡང་འབྱུང་བ་མིན། །དེ་བཞིན་ལོ་གྲངས་མང་པོ་ལས། །དངོས་གྲུབ་མྱུར་དུ་འབྱུང་བ་མིན། །དེ་ཕྱིར་རིན་ཆེན་ཉི་ཤུ་བཞི། །བདུད་རྩི་རོ་འདྲ་སེང་གེའི་གཟུགས། །གནས་ནི་ཉི་ཤུ་བཞི་ཉིད་ཀྱང་། །སྐུ་གསུང་ཐུགས་ཀྱིས་བྱིན་བརླབས་ཏེ། །མཁའ་འགྲོ་ཉི་ཤུ་བཞི་ལ་གཏད། །ཕྱག་རྒྱ་བཞི་ཡིས་རྒྱས་བཏབ་བོ། །དམ་དུ་སྦ་ཞིང་དམ་དུ་སྤེལ། །རྒྱ་ལས་འདས་ན་དམ་ཚིག་ཉམས། །སྔགས་དང་དབྱིབས་དང་ཆོས་བདག་ནི། །སྐུ་གསུང་ཐུགས་ཀྱི་རང་བཞིན་ཏེ། །དེ་ཡི་ངོ་བོར་རྒྱུད་ཀུན་བརྟག །སྔགས་ལ་གཉིས་ཏེ་ཆོས་ལ་གཉིས། །དབྱིབས་ཀྱང་དེ་བཞིན་ཤེས་པར་བྱ། །རེ་རེའི་བྱེ་བྲག་དབྱེ་བ་ཡང་། །བླ་མའི་ཞལ་ནས་ཤེས་པར་བྱ། །ཕྱི་ནང་སྦས་པ་མཐར་ཐུག་བཞི། །རྒྱུད་ཀུན་ཕྱག་རྒྱ་བཞི་ཡིན་ཏེ། །དེ་ཉིད་དབྱེ་བ་མཁས་པས་བརྟག །དེ་ཉིད་རྟོགས་ན་རྒྱུད་དང་ལྡན། །དེ་ལས་བཟློག་པ་འདས་པའོ། །རང་བཞིན་གྱིས་ནི་ཙཎྜཱ་ལཱི། །གནས་ནི་ཉི་ཤུ་རྩ་བཞི་སྟེ། །རྩ་གསུམ་གྱིས་ནི་བྱིན་གྱིས་བརླབས། །གནས་ནི་ཉི་ཤུ་རྩ་བཞི་རུ། །མཁའ་འགྲོ་ཉི་ཤུ་རྩ་བཞི་ལ། །རྒྱུད་ནི་ཉི་ཤུ་རྩ་བཞི་གཏད། །པུ་རུ་རབ་གཏུམ་གསང་རྡོ་རྗེ། །ཛ་རུ་གཏུམ་མིག་གསང་བ་གཅོད། །ཨོ་རུ་འོད་ལྡན་བསམ་མི་ཁྱབ། །ཨཱ་རུ་སྣ་རྩེ་ནམ་མཁའ་མཉམ། །གོ་རུ་བསྟན་བློ་ནམ་མཁའ་ཆེ། །ར་རུ་མིའུ་ཐུང་སྐུ་གསུང་ཐུགས། །དེ་རུ་ལང་ཀ་རིན་ཆེན་ཕྲེང་། །མ་རུ་ཤིང་གྲིབ་དམ་ཚིག་ཆེ། །ཀ་རུ་ཨེ་རཱུ་སྟོབས་པོ་ཆེ། །ཨོ་རུ་འཇིགས་བྱེད་ཡེ་ཤེས་གསང་། །ཏྲི་རུ་རླུང་ཤུགས་ཡེ་ཤེས་ཕྲེང་། །ཀོ་རུ་ཆང་འཐུང་ཡེ་ཤེས་འབར། །ཀུ་རུ་ཤྱཱ་མ་ཟླ་བའི་ཕྲེང་། །ལ་རུ་ཤིན་བཟང་རིན་ཆེན་འབར། །ཀ་རུ་རྟ་རྣ་ཉི་མའི་འཁོར། །ཧེ་རུ་ཁ་ཁ་ཡེ་ཤེས་རྒྱལ། །བྲེ་རུ་འཁོར་ཤུགས་མཁའ་འགྲོ་གསང་། །གྲྀ་རུ་ཁཎྜ་གསང་མེ་འབར། །སཽ་རུ་ཤོཎྜི་གསང་བདུད་རྩི། །སུ་རུ་འཁོར་ལོ་དུར་ཁྲོད་རྒྱན། །ན་རུ་ཤིན་དཔའ་རྡོ་རྗེ་རྒྱལ། །སི་རུ་སྟོབས་ཆེན་ཡེ་ཤེས་རྒྱལ། །མ་རུ་འཁོར་སྒྱུར་ཆགས་པའི་རྒྱལ། །ཀུ་རུ་བརྩོན་འགྲུས་མཁའ་འགྲོ་སྡོམ། །དེ་དག་གཏད་དོ་ལེགས་པར་སྦོས། །ནང་དུ་ཙཎྚཱ་ལཱི་དག་ནི། །ཉི་ཤུ་བཞི་ནི་གཏད་པ་སྟེ། །དང་པོ་གསང་བ་གཉིས་པ་ཡང་། །རང་བཞིན་ཡིན་ཏེ་དེ་ནས་དབང་། །རྐྱེན་ནི་ཐ་མས་གསུངས་པ་སྟེ། །མཁའ་འགྲོ་ཁྱེད་ཀྱི་རང་བཞིན་ཏེ། །དེ་ནས་དངོས་གྲུབ་གང་འདོད་སྦྱིན། །དངོས་གྲུབ་དམ་པ་གང་འདོད་པས། །དབང་དང་དམ་ཚིག་ལྡན་བྱས་ཏེ། །རྟ་དང་གླང་པོ་ལ་སོགས་དང་། །བུ་དང་ཆུང་མ་ནོར་གྱི་དབྱིག །བསྙེན་བཀུར་དག་གིས་བསྙེན་བྱས་ཏེ། །བདུད་རྩི་རོ་ཡི་ར་ལི་ལས། །བླ་མའི་ཞལ་ནས་དངོས་གྲུབ་བླང་། །ཡིད་ལ་ཟུང་ལ་དྲན་པར་གྱིས། །དེ་ནས་འཁོར་འདུས་པ་དེ་དག་ཐམས་ཅད་རྗེས་སུ་ཡི་རང་བར་གྱུར་ཏོ། །སྐུ་གསུང་ཐུགས་ཀྱི་རང་བཞིན་དཔལ་ནམ་མཁའ་དང་མཉམ་པའི་རྒྱུད་ཀྱི་རྒྱལ་པོ་ཞེས་བྱ་བ་རྫོགས་སོ།། །།པཎྜི་ཏ་ག་ཡ་དྷ་ར་དང་། དགེ་སློང་ཤཱཀྱ་ཡེ་ཤེས་ཀྱིས་ཞུས་ཤིང་བསྒྱུར་བའོ།། །།
