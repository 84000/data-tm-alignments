༄༅༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྱ་ཨ་གྲ་པྲ་དཱི་པ་དྷཱ་ར་ཎཱི་བིདྱ་རཱ་ཛ། བོད་སྐད་དུ། འཕགས་པ་རིག་སྔགས་ཀྱི་རྒྱལ་པོ་སྒྲོན་མ་མཆོག་གི་གཟུངས། སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །འདི་སྐད་བདག་གིས་ཐོས་པའི་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་མཉན་ཡོད་ན་རྒྱལ་བུ་རྒྱལ་བྱེད་ཀྱི་ཚལ་མགོན་མེད་ཟས་སྦྱིན་གྱི་ཀུན་དགའ་ར་བ་ན་དགེ་སློང་སྟོང་ཉིས་བརྒྱ་ལྔ་བཅུའི་དགེ་སློང་གི་དགེ་འདུན་ཆེན་པོ་དང་། ལྷ་དང་མི་འབུམ་ཕྲག་དུ་མ་དང་། བྱང་ཆུབ་སེམས་དཔའ་འབུམ་ཕྲག་དུ་མ་དང་། ཐབས་ཅིག་ཏུ་བཞུགས་ཏེ། ཡོངས་སུ་བསྐོར་ཅིང་མདུན་གྱིས་བལྟས་ནས་ཆོས་སྟོན་ཏོ། །སངས་རྒྱས་ཀྱི་ཞིང་འདི་ནས་སངས་རྒྱས་ཀྱི་ཞིང་བྱེ་བ་བརྒྱ་ཕྲག་སྟོང་འདས་པ་ན་འཇིག་རྟེན་གྱི་ཁམས་མེ་ཏོག་མཐའ་ཡས་ཞེས་བྱ་བ་དེ་ན་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་སྒྲོན་མ་མཆོག་ཅེས་བྱ་བ་ད་ལྟར་ཡང་བཞུགས་ཏེ་འཚོ་ཞིང་གཞེས་ལ་ཆོས་ཀྱང་སྟོན་པ་དེས་བྱང་ཆུབ་སེམས་དཔའ་འོད་ཆེན་པོ་དང་། འོད་དཔག་མེད་ཅེས་བྱ་བ་བྱང་ཆུབ་སེམས་དཔའ་གཉིས་འཇིག་རྟེན་གྱི་ཁམས་མི་མཇེད་འདིར་བཏང་ངོ་། །དེ་ནས་བྱང་ཆུབ་སེམས་དཔའ་དེ་གཉིས་རྒྱལ་བུ་རྒྱལ་བྱེད་ཀྱི་ཚལ་མགོན་མེད་ཟས་སྦྱིན་གྱི་ཀུན་དགའ་ར་བ་ག་ལ་བ་དང་། བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ཤཱཀྱ་ཐུབ་པ་ག་ལ་བ་དེར་སོང་སྟེ་ཕྱིན་ནས། བཅོམ་ལྡན་འདས་ཀྱི་ཞབས་ལ་མགོ་བོས་ཕྱག་འཚལ་ཏེ། བཅོམ་ལྡན་འདས་ལ་ལན་གསུམ་བསྐོར་བ་བྱས་ནས་ཕྱོགས་གཅིག་ཏུ་འདུག་གོ། །ཕྱོགས་གཅིག་ཏུ་འདུག་ནས་བྱང་ཆུབ་སེམས་དཔའ་དེ་གཉིས་ཀྱིས་བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་སངས་རྒྱས་ཀྱི་ཞིང་བྱེ་བ་ཕྲག་འབུམ་འདས་པ་ན་འཇིག་རྟེན་གྱི་ཁམས་མེ་ཏོག་མཐའ་ཡས་ཞེས་བགྱི་བ་དེ་ན་བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་སྒྲོན་མ་མཆོག་ཅེས་བགྱི་བ་ད་ལྟར་བཞུགས་ཏེ། འཚོ་ཞིང་གཞེས་ལ་ཆོས་ཀྱང་སྟོན་ཏོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་སྒྲོན་མའི་མཆོག་དེས་བདག་ཅག་གཉིས་འཇིག་རྟེན་གྱི་ཁམས་མི་མཇེད་འདིར་བཏང་སྟེ། བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་སྒྲོན་མའི་མཆོག་དེས། བཅོམ་ལྡན་འདས་ལ་གནོད་པ་ཆུང་ངམ། ཉམ་ང་བ་ཉུང་ངམ། བསྐྱོད་པ་ཡང་ངམ། འཚོའམ། སྟོབས་དང་། བདེ་བ་དང་། ཁ་ན་མ་ཐོ་བ་མ་མཆིས་པ་ལ་རེག་པར་གནས་སམ་ཞེས་སྙུན་གསོལ་ལོ། །འདི་སྐད་ཅེས་ཀྱང་གསུང་ངོ་། །འདི་ན་མིའམ། མི་མ་ལགས་པའམ། འབྱུང་པོའམ། ཤ་ཟའམ། གནོད་སྦྱིན་ནམ། སྲིན་པོའམ། གྲུལ་བུམ་མམ། གྲུལ་བུམ་གྱི་སྡེ་དཔོན་ནམ། རྒྱལ་པོའམ། བློན་པོ་ཆེན་པོའམ། ཆོམ་རྐུན་ནམ། སྦྲུལ་ལམ། སྲིན་བུའམ། སྲིན་བུ་ལུ་ཏའམ། སྦྲང་བུ་ལྕགས་ཕྱེའམ། ཉེ་བའི་ཉོན་མོངས་པ་གང་ཡང་རུང་བདག་འཚེ་བར་བགྱིད་པ་འགའ་ཡང་མ་མཆིས་སམ་ཞེས་གསུང་ངོ་། །སེམས་ཅན་རྣམས་ཀྱི་དོན་དང་། སྨན་པ་དང་། བདེ་བ་དང་། མདངས་དང་ལྡན་པར་བགྱི་བ་དང་། གཟི་བརྗིད་དང་ལྡན་པར་བགྱི་བ་དང་། གྲགས་པ་དང་ལྡན་པར་བགྱི་བ་དང་། ཡོན་ཏན་དང་ལྡན་པར་བགྱི་བ་དང་། སྟོབས་དང་ལྡན་པར་བགྱི་བ་དང་། མདོག་དང་ལྡན་པར་བགྱི་བ་དང་། བསྲུང་བ་དང་། སྦ་བ་དང་། བསྐྱབ་པའི་སླད་དུ་བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་སྒྲོན་མའི་མཆོག་དེས་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ཀྱང་བསྐུར་ཏེ། འདི་ལྟ་སྟེ། ཛྭཱ་ལེ་ཛྭཱ་ལེ། ཛྭཱ་ལེ་ག་ཏེ། ཧུཀྐུ། མུཀྐུ། སཾ་མ་ཏེ། མ་ཧཱ་སཾ་མ་ཏེ། མ་ཧཱ་ཛྭཱ་ལེ། ཛྭཱ་བ་ལ་མེ། ས་ཧེ་ས་ཧེ། མ་ཧཱ་ཛྭཱ་ལེ། ཨུཀྐེ། མུཀྐེ། ཤ་མེ། སཾ་མ་ཧེ། མ་ཧཱ་སཾ་མ་ཧེ། མ་ཧཱ་ཤ་མེ་སྭཱ་ཧཱ། དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་པ། ཀུན་དགའ་བོ་ཁྱོད་ཀྱིས་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ཟུང་ཤིག །ལོང་ཤིག །ཆོང་ཤིག །ལྷོགས་ཤིག །ཀུན་ཆུབ་པར་གྱིས་ཤིག །གཞན་དག་ལ་ཡང་རྒྱ་ཆེར་རབ་ཏུ་སྟོན་ཅིག །ཀུན་དགའ་བོ་སངས་རྒྱས་འབྱུང་བ་རྙེད་པར་དཀའོ། །གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་རྙེད་པར་དཀའོ། །ཀུན་དགའ་བོ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་དག་འཛིན་པ་ཡང་ཤིན་ཏུ་རྙེད་པར་དཀའོ། །ཀུན་དགའ་བོ་རིགས་ཀྱི་བུའམ། རིགས་ཀྱི་བུ་མོ་གང་ལ་ལ་ནི་གཟུངས་ཀྱི་གསང་སྔགས་གཞི་འདི་དག་འཛིན་པ་དང་། ལེན་པ་དང་། འཆང་བ་དང་། ཀློག་པ་དང་། ཀུན་ཆུབ་པར་བྱེད་པ་དང་། གཞན་དག་ལ་ཡང་རྒྱ་ཆེར་རབ་ཏུ་སྟོན་པ་དེ་ནི་ཚེ་རབས་བདུན་དུ་ཚེ་རབས་དྲན་པར་འགྱུར་རོ། །ཚེ་འདི་ཉིད་ལ་ཡང་སྔོན་གྱི་ལས་མ་གཏོགས་པར་གཞན་གྱི་གནོད་པ་ཐམས་ཅད་ལས་ཡང་དག་པར་འདའ་བ་དང་། འགྲོ་བ་ཐམས་ཅད་དང་རྗེས་སུ་མཐུན་པར་འགྱུར་རོ། །ཀུན་དགའ་བོ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ནི་སངས་རྒྱས་བྱེ་བ་ཕྲག་བདུན་ཅུ་རྩ་བདུན་གྱིས་གསུངས་ཏེ། གང་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ལས་འདའ་བར་བྱེད་པ་དེ་ནི་དེ་དག་ཐམས་ཅད་ལ་འཁུས་པ་ཡིན་ནོ། །དེ་ནས་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་མ་ཕམ་པ་སྟན་ལས་ལངས་ཏེ། གོས་ཕྲག་པ་གཅིག་ཏུ་གཟར་ནས་པུས་མོ་གཡས་པའི་ལྷ་ང་ས་ལ་བཙུགས་ཏེ། བཅོམ་ལྡན་འདས་ག་ལ་བ་དེ་ལོགས་སུ་ཐལ་མོ་སྦྱར་བ་བཏུད་ནས། བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་བདག་གིས་ཀྱང་སེམས་ཅན་རྣམས་ཀྱི་དོན་དང་། སྨན་པ་དང་། བདེ་བ་དང་། མདངས་དང་ལྡན་པར་བགྱི་བ་དང་། གཟི་བརྗིད་དང་ལྡན་པར་བགྱི་བ་དང་། གྲགས་པ་དང་ལྡན་པར་བགྱི་བ་དང་། ཡོན་ཏན་དང་ལྡན་པར་བགྱི་བ་དང་། སྟོབས་དང་ལྡན་པར་བགྱི་བ་དང་། མདོག་དང་ལྡན་པར་བགྱི་བ་དང་། བསྲུང་བ་དང་། སྦ་བ་དང་། བསྐྱབ་པའི་སླད་དུ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་རྣམས་བརྗོད་པར་བགྱིའོ། །འདི་ལྟ་སྟེ། ཨཏྟེ། བཏྟེ། ནཏྟེ། ཀུ་ནཏྟེ། པུཀྐ་སེ། ཀུ་པུཀྐ་སེ། ཀ་སེ། ཨ་ན་ཀ་སེ། ནཱ་མ་ཀ་སེ། ཨ་ནུ་ཏ། པྲ་ནུ་ཏ། ཁ་ཛེ། མ་ཧཱ་ཁ་ཛེ། སད་པ་ཕེ། ཏ་ཌ་ཕེ། ཏུ་ཌ་ཕེ། ཧི་ལི་མི་ལི་ཏི་ལེ་སྭཱ་ཧཱ། དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་པ། ཀུན་དགའ་བོ་ཁྱོད་ཀྱིས་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ཟུང་ཤིག །ལོང་ཤིག །ཆོང་ཤིག །ལྷོགས་ཤིག །ཀུན་ཆུབ་པར་གྱིས་ཤིག །གཞན་དག་ལ་ཡང་རྒྱ་ཆེར་ཡང་དག་པར་རབ་ཏུ་སྟོན་ཅིག །ཀུན་དགའ་བོ་སངས་རྒྱས་འབྱུང་བ་རྙེད་པར་དཀའོ། །གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་རྙེད་པར་དཀའོ། །ཀུན་དགའ་བོ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་འཛིན་པ་ཡང་ཤིན་ཏུ་རྙེད་པར་དཀའོ། །ཀུན་དགའ་བོ་རིགས་ཀྱི་བུའམ། རིགས་ཀྱི་བུ་མོ་གང་ལ་ལ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་འཛིན་པ་དང་། ལེན་པ་དང་། འཆང་བ་དང་། ཀློག་པ་དང་། ཀུན་ཆུབ་པར་བྱེད་པ་དང་། གཞན་དག་ལ་ཡང་རྒྱ་ཆེར་རབ་ཏུ་སྟོན་པ་དེ་ནི་ཚེ་རབས་བཅུ་གཉིས་སུ་ཚེ་རབས་དྲན་པར་འགྱུར་རོ། །ཚེ་འདི་ཉིད་ལ་ཡང་སྔོན་གྱི་ལས་མ་གཏོགས་པར་གཞན་གྱི་གནོད་པ་ཐམས་ཅད་ལས་ཡང་དག་པར་འདའ་བ་དང་། འགྲོ་བ་ཐམས་ཅད་དང་རྗེས་སུ་མཐུན་པར་འགྱུར་རོ། །ཀུན་དགའ་བོ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ནི་སངས་རྒྱས་བྱེ་བ་ཕྲག་བརྒྱད་ཅུ་རྩ་གཉིས་ཀྱིས་གསུངས་ཏེ། །གང་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ལས་འདའ་བར་བྱེད་པ་ནི་དེ་དག་ཐམས་ཅད་ལ་འཁུས་པ་ཡིན་ནོ། །དེ་ནས་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་སྟན་ལས་ལངས་ཏེ། གོས་ཕྲག་པ་གཅིག་ཏུ་གཟར་ནས་པུས་མོ་གཡས་པའི་ལྷ་ང་ས་ལ་བཙུགས་ཏེ། བཅོམ་ལྡན་འདས་ག་ལ་བ་དེ་ལོགས་སུ་ཐལ་མོ་སྦྱར་བ་བཏུད་ནས་བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་བདག་གིས་ཀྱང་སེམས་ཅན་རྣམས་ཀྱི་དོན་དང་། སྨན་པ་དང་། བདེ་བ་དང་། མདངས་དང་ལྡན་པར་བགྱི་བ་དང་། གཟི་བརྗིད་དང་ལྡན་པར་བགྱི་བ་དང་། གྲགས་པ་དང་ལྡན་པར་བགྱི་བ་དང་། ཡོན་ཏན་དང་ལྡན་པར་བགྱི་བ་དང་། སྟོབས་དང་ལྡན་པར་བགྱི་བ་དང་། མདོག་དང་ལྡན་པར་བགྱི་བ་དང་། བསྲུང་བ་དང་། སྦ་བ་དང་། བསྐྱབ་པའི་སླད་དུ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་རྣམས་བརྗོད་པར་བགྱིའོ། །དེ་འདི་ལྟ་སྟེ། ཨཀྑེ། བཀྑེ། ན་ཛི་ཏེ། ནཱ་གཱ་ས་རེ། ན་ཧི་ལི། ཛ་ཧི་ལི། ཛ་ཧི་ལི་བ་ཏེ། ཙཎྜཱི་ལ་བ་ཏེ་སྭཱ་ཧཱ། དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་པ། ཀུན་དགའ་བོ་ཁྱོད་ཀྱིས་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ཟུང་ཤིག །ལོང་ཤིག །ཆོང་ཤིག །ལྷོགས་ཤིག །ཀུན་ཆུབ་པར་གྱིས་ཤིག །གཞན་དག་ལ་ཡང་རྒྱ་ཆེར་རབ་ཏུ་སྟོན་ཅིག །ཀུན་དགའ་བོ་སངས་རྒྱས་འབྱུང་བ་རྙེད་པར་དཀའོ། །གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་རྙེད་པར་དཀའོ། །ཀུན་དགའ་བོ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་འཛིན་པ་ཡང་ཤིན་ཏུ་རྙེད་པར་དཀའོ། །ཀུན་དགའ་བོ་རིགས་ཀྱི་བུའམ། རིགས་ཀྱི་བུ་མོ་གང་ལ་ལས་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་འཛིན་པ་དང་། ལེན་པ་དང་། འཆང་བ་དང་། ཀློག་པ་དང་། ཀུན་ཆུབ་པར་བྱེད་པ་དང་། གཞན་དག་ལ་ཡང་རྒྱ་ཆེར་རབ་ཏུ་སྟོན་པ་དེ་ནི་ཚེ་རབས་བཅུ་གསུམ་དུ་ཚེ་རབས་དྲན་པར་འགྱུར་རོ། །ཚེ་འདི་ཉིད་ལ་ཡང་སྔོན་གྱི་ལས་མ་གཏོགས་པ་གཞན་གྱི་གནོད་པ་ཐམས་ཅད་ལས་ཡང་དག་པར་འདའ་བ་དང་། འགྲོ་བ་ཐམས་ཅད་དང་རྗེས་སུ་མཐུན་པར་འགྱུར་རོ། །ཀུན་དགའ་བོ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ནི་སངས་རྒྱས་བྱེ་བ་ཕྲག་བརྒྱད་ཅུ་རྩ་གཉིས་ཀྱིས་གསུངས་ཏེ། གང་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ལས་འདའ་བར་བྱེད་པ་དེ་ནི་དེ་དག་ཐམས་ཅད་ལ་འཁུས་པ་ཡིན་ནོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་དགེ་སློང་རྣམས་ལ་བཀའ་སྩལ་པ། དགེ་སློང་དག་ངས་ཀྱང་སེམས་ཅན་རྣམས་ཀྱི་དོན་དང་། ཕན་པ་དང་། བདེ་བ་དང་། མདངས་དང་ལྡན་པར་བྱ་བ་དང་། གཟི་བརྗིད་དང་ལྡན་པར་བྱ་བ་དང་། གྲགས་པ་དང་ལྡན་པར་བྱ་བ་དང་། ཡོན་ཏན་དང་ལྡན་པར་བྱ་བ་དང་། སྟོབས་དང་ལྡན་པར་བྱ་བ་དང་། མདོག་དང་ལྡན་པར་བྱ་བ་དང་། བསྲུང་བ་དང་། སྦ་བ་དང་། བསྐྱབ་པའི་ཕྱིར་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་རྣམས་བཤད་དོ། །འདི་ལྟ་སྟེ། ཨཊྚེ། བཊྚེ། ན་ཊྚེ། ཀུ་ན་ཊྚེ། ཊ་ཀེ། ཨེ་ཌ་ཀེ། ཏ་ཏ་བེ། ཏུ་ཌ་བེ། ར་ཊ་ཀེ། ལོ་ལེ། ཧུ་ལེ། ཧི་ལི། མི་ལེ། ཏི་ལེ། ཤཱི་ལེ། ཤཱི་ལ། ཨ་ཌྷེ། བ་ཌྷེ། ན་ཌྷེ། ཀུ་ན་ཌྷེ། ཨ་ཌབ་ཏཱི། པ་ན་བ་ཏཱི། ཀ་སེ་ཀ་སེ་ཀ་སེ། ཕ་སེ་ཕ་སེ་ཕ་སེ། མ་ཧཱ་ཕ་སེ་ཕ་སེ་སྭཱ་ཧཱ། དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་པ། ཀུན་དགའ་བོ་ཁྱོད་ཀྱིས་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ཟུང་ཤིག །ལོང་ཤིག །ཆོང་ཤིག །ལྷོགས་ཤིག །ཀུན་ཆུབ་པར་གྱིས་ཤིག །གཞན་དག་ལ་ཡང་རྒྱ་ཆེར་རབ་ཏུ་སྟོན་ཅིག །ཀུན་དགའ་བོ་སངས་རྒྱས་འབྱུང་བ་རྙེད་པར་དཀའོ། །གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་རྙེད་པར་དཀའོ། །ཀུན་དགའ་བོ་གཟུངས་ཀྱི་གསང་སྔགས་འདི་དག་སྟོན་པ་ཡང་ཤིན་ཏུ་རྙེད་པར་དཀའོ། །ཀུན་དགའ་བོ་རིགས་ཀྱི་བུའམ། རིགས་ཀྱི་བུ་མོ་གང་ལ་ལ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་འཛིན་པ་དང་། ལེན་པ་དང་། འཆང་བ་དང་། ཀློག་པ་དང་། ཀུན་ཆུབ་པར་བྱེད་པ་དང་། གཞན་དག་ལ་ཡང་རྒྱ་ཆེར་རབ་ཏུ་སྟོན་པ་དེ་ནི་ཚེ་རབས་བཅུ་བཞིར་ཚེ་རབས་དྲན་པར་འགྱུར་རོ། །ཚེ་འདི་ཉིད་ལ་ཡང་སྔོན་གྱི་ལས་མ་གཏོགས་པར་གཞན་གྱི་གནོད་པ་ཐམས་ཅད་ལས་འདའ་བ་དང་། འགྲོ་བ་ཐམས་ཅད་དང་རྗེས་སུ་མཐུན་པར་འགྱུར་རོ། །ཀུན་དགའ་བོ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ནི་སངས་རྒྱས་བྱེ་བ་ཕྲག་བརྒྱད་ཅུ་རྩ་བཞིས་གསུངས་ཏེ། གང་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ལས་འདའ་བར་བྱེད་པ་དེ་ནི་དེ་དག་ཐམས་ཅད་ལ་འཁུས་པ་ཡིན་ནོ། །རིགས་ཀྱི་བུའམ། རིགས་ཀྱི་བུ་མོ་གང་ལ་ལ་དག་གིས་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ཤིང་སྐམ་པོ་གཤེར་བག་མེད་པ་ལ་བསྲུང་བ་བྱས་ཏེ། ཡོངས་སུ་བསྐྱབ་པ་དང་། ཡོངས་སུ་གཟུང་བ་དང་། བཅིང་བ་དང་། ཞི་བ་དང་། བདེ་ལེགས་སུ་འགྱུར་བ་བྱས་ན་ཤིང་དེ་ལ་ལྕུ་གུ་དང་། ལོ་མ་དག་ཀྱང་སྐྱེ་བར་འགྱུར་ཏེ། མེ་ཏོག་དང་། འབྲས་བུ་རྣམས་ཀྱང་མངོན་པར་འགྲུབ་པར་འགྱུར་ན། མིར་གྱུར་པ་ལྟ་ཅི་སྨོས། ནམཿསརྦ་བུདྡྷེ་བྷྱཿ། སིད་དྷ་ནྟུ་དྷཱ་ར་ཎཱི། མནྟྲ་པཱ་ད་ནི་སྭཱ་ཧཱ། དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ཡང་དགེ་སློང་རྣམས་ལ་བཀའ་སྩལ་པ། དགེ་སློང་དག་སེམས་ཅན་རྣམས་ཀྱི་དོན་དང་། ཕན་པ་དང་། བདེ་བ་དང་། མདངས་དང་ལྡན་པར་བྱ་བ་དང་། གཟི་བརྗིད་དང་ལྡན་པར་བྱ་བ་དང་། གྲགས་པ་དང་ལྡན་པར་བྱ་བ་དང་། ཡོན་ཏན་དང་ལྡན་པར་བྱ་བ་དང་། སྟོབས་དང་ལྡན་པར་བྱ་བ་དང་། མདོག་དང་ལྡན་པར་བྱ་བ་དང་། བསྲུང་བ་དང་། སྦ་བ་དང་། བསྐྱབ་པའི་ཕྱིར་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་གཞན་དག་ཀྱང་བཤད་པར་བྱའོ། །དེ་འདི་ལྟ་སྟེ། ཨཏྟེ། མཏྟེ། པ་ཀེ་པཱ་ར་ཀེ། པེ་ཀེ་ཌཱ། དྲ་མ་ཏི། ཏ་ར་མ་ཏི། ཊུ་ཌ་མ་ཊི། ཊུ་ཌ་མ་ཊི། ཊུ་ཌ་ཊུ་རེ། སུ་རེ། ཧི་ལེ། མུ་ལེ། ཏི་ལེ་སྭཱ་ཧཱ། དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་པ། ཀུན་དགའ་བོ་ཁྱོད་ཀྱིས་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ཟུང་ཤིག །ལོང་ཤིག །ཆོང་ཤིག །ཀློགས་ཤིག །ཀུན་ཆུབ་པར་གྱིས་ཤིག །གཞན་དག་ལ་ཡང་རྒྱ་ཆེར་རབ་ཏུ་སྟོན་ཅིག །ཀུན་དགའ་བོ་སངས་རྒྱས་འབྱུང་བ་རྙེད་པར་དཀའོ། །གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་རྙེད་པར་དཀའོ། །ཀུན་དགའ་བོ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་འཛིན་པ་ཡང་ཤིན་ཏུ་རྙེད་པར་དཀའོ། །ཀུན་དགའ་བོ་རིགས་ཀྱི་བུའམ། རིགས་ཀྱི་བུ་མོ་གང་ལ་ལ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་འཛིན་པ་དང་། ལེན་པ་དང་། འཆང་བ་དང་། ཀློག་པ་དང་། ཀུན་ཆུབ་པར་བྱེད་པ་དང་། གཞན་དག་ལ་ཡང་རྒྱ་ཆེར་རབ་ཏུ་སྟོན་པ་དེ་ནི་ཚེ་རབས་ཉི་ཤུ་རྩ་གཅིག་ཏུ་ཚེ་རབས་དྲན་པར་འགྱུར་རོ། །ཚེ་འདི་ཉིད་ལ་ཡང་སྔོན་གྱི་ལས་མ་གཏོགས་པར་གཞན་གྱི་གནོད་པ་ཐམས་ཅད་ལས་འདའ་བ་དང་། འགྲོ་བ་ཐམས་ཅད་དང་རྗེས་སུ་མཐུན་པར་འགྱུར་རོ། །གང་ལ་ལ་ཞིག་ལ་བསྲུང་བ་བྱས་ན་དེ་ལ་ཡང་འབྱུང་པོ་ཐམས་ཅད་ཀྱིས་མི་ཚུགས་སོ། །ལྷ་དང་། ཀླུ་དང་། གནོད་སྦྱིན་དང་། དྲི་ཟ་དང་། ལྷ་མ་ཡིན་དང་། ནམ་མཁའ་ལྡིང་དང་། མིའམ་ཅི་དང་། ལྟོ་འཕྱེ་ཆེན་པོ་དང་། མི་དང་། མི་མ་ཡིན་པ་ཐམས་ཅད་ཀྱིས་མི་ཚུགས་སོ། །སྦྲུལ་དང་། སྲིན་བུ་དང་། སྲིན་བུ་ལུ་ཏ་དང་། སྦྲང་བུ་ལྕགས་ཕྱེ་ཐམས་ཅད་ཀྱིས་མི་ཚུགས་སོ། །རིམས་རླུང་དང་། མཁྲིས་པ་དང་། བད་ཀན་དང་། འདུས་པ་དང་། དུས་ཚིགས་འགྱུར་བ་ལས་བྱུང་བ་ཐམས་ཅད་ཀྱིས་མི་ཚུགས་སོ། །ཞག་གཅིག་པའམ། ཞག་གཉིས་པའམ། ཞག་གསུམ་པའམ། ཞག་བཞི་པའམ། ཞག་བདུན་པའམ། ཟླ་བ་ཕྱེད་པའམ། ཟླ་བ་གཅིག་པའམ། ལོ་གཅིག་པའམ། ནད་རྒྱུན་རིངས་སམ། སྐད་ཅིག་པ་རྣམས་ཀྱིས་མི་ཚུགས་སོ། །འབྲས་སམ། ཕོལ་མིག་གམ། རྐང་ཤུའམ། ལྷོག་པའམ། མེ་དཔལ་ལམ། ཤུ་བའམ། ཁྱི་རྔོའམ། གཡན་པའམ། མཛེ་ཐམས་ཅད་དམ། ལྷོག་པ་ཐམས་ཅད་དམ། གཤེད་བྱེད་ཐམས་ཅད་དམ། སྒྱུར་བའི་ལས་ཐམས་ཅད་དམ། བྱད་ཐམས་ཅད་དང་། རོ་ལངས་ཐམས་ཅད་ཀྱིས་ཀྱང་མི་ཚུགས་སོ། །ཀུན་དགའ་བོ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ནི་སངས་རྒྱས་བྱེ་བ་ཕྲག་དགུ་བཅུ་རྩ་དགུས་གསུངས་ཏེ། གང་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ལས་འདའ་བར་བྱེད་པ་དེ་ནི་དེ་དག་ཐམས་ཅད་ལ་འཁུས་པ་ཡིན་ནོ། །ཀུན་དགའ་བོ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ནི་རྒྱལ་པོའི་ཕོ་བྲང་ན་འདུག་ན་ཡང་ཡིད་ལ་བྱའོ། །ཆོམ་རྐུན་གྱི་ནང་ན་འདུག་ན་ཡང་ཡིད་ལ་བྱའོ། །སྲིན་པོའི་ནང་ན་འདུག་ན་ཡང་ཡིད་ལ་བྱའོ། །གཅན་གཟན་ཁྲོ་བོའི་ནང་ན་འདུག་ན་ཡང་ཡིད་ལ་བྱའོ། །འབྲོག་དགོན་པའི་ནང་ན་འདུག་ན་ཡང་ཡིད་ལ་བྱའོ། །ཀླུང་དང་བགྲོད་དཀའ་བ་ན་འདུག་ན་ཡང་ཡིད་ལ་བྱའོ། །མེའི་ནང་ན་འདུག་ན་ཡང་ཡིད་ལ་བྱའོ། །ཆུའི་ནང་ན་འདུག་ན་ཡང་ཡིད་ལ་བྱའོ། །དེ་ཡིད་ལ་བྱས་པ་དེ་རྣམས་ལས་འཇིགས་པ་དེ་དག་ཐམས་ཅད་མི་འབྱུང་ངོ་། །དེས་དུག་ཟོས་པ་ཐམས་ཅད་ཀྱང་ཁ་ཟས་ཉིད་དུ་འགྱུར་རོ། །ཀུན་དགའ་བོ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་འཛིན་པ་ནི་འབྱུང་པོ་ཐམས་ཅད་ཀྱིས་མི་ཚུགས་སོ། །ཡི་དགས་ཐམས་ཅད་དང་། སྲུལ་པོ་ཐམས་ཅད་དང་། གཤེད་བྱེད་ཐམས་ཅད་དང་། སྒྱུར་བའི་ལས་ཐམས་ཅད་དང་། བྱད་ཐམས་ཅད་དང་། མི་དང་མི་མ་ཡིན་པ་ཐམས་ཅད་ཀྱིས་མི་ཚུགས་ཏེ། ཀུན་དགའ་བོ་མདོར་ན་དེ་ལ་གཞན་གྱི་གནོད་པ་ཐམས་ཅད་ཀྱིས་མི་ཚུགས་སོ། །ཀུན་དགའ་བོ་གསད་པའི་རིགས་པ་ནི་ཆད་པས་བྲལ་བར་འགྱུར་རོ། །ཆད་པས་གཅད་པའི་རིགས་པ་ནི་བརྡེགས་པས། བརྡེགས་པའི་རིགས་པ་ནི་བརྒྱད་པས། བརྒྱད་པའི་རིགས་པ་ནི་བསྡིགས་པས། བསྡིགས་པའི་རིགས་པ་ནི་མིག་ཚུགས་སུ་བལྟས་པས་ཐར་པར་འགྱུར་རོ། །ཀུན་དགའ་བོ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་གིས་གང་ལ་བསྲུང་བ་བྱས་པའམ། ཡོངས་སུ་བསྐྱབ་པ་བྱས་པའམ། ཡོངས་སུ་གཟུང་བས་བཅིངས་པའམ། ཞི་བ་དང་། བདེ་ལེགས་སུ་འགྱུར་བ་བྱས་པ་གཞན་དུ་འགྱུར་བ་དེ་ནི་གནས་མེད་དེ། ལྷ་དང་བཅས། བདུད་དང་བཅས། ཚངས་པ་དང་བཅས་པའི་འཇིག་རྟེན་དང་། དགེ་སྦྱོང་དང་། བྲམ་ཟེར་བཅས་པའི་སྐྱེ་དགུ་དང་། ལྷ་དང་། མི་དང་། ལྷ་མ་ཡིན་གྱི་ནང་ན་ཡང་ངས་མ་མཐོང་ངོ་། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་ནི་གནོད་པ་དང་། ནད་འགོ་བ་དང་། འཁྲུག་པ་ཐམས་ཅད་ལས་ངེས་པར་འབྱུང་བར་བྱིན་གྱིས་བརླབས་པའི་ཕྱིར་རོ། །དེ་ནས་ཆོས་ཀྱི་རྣམ་གྲངས་འདི་རྫོགས་པའི་རྗེས་ལ་རྒྱལ་པོ་ཆེན་པོ་བཞིས་བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་བདག་ཅག་གིས་ཀྱང་ཆོས་ཀྱི་རྣམ་གྲངས་འདི་རྒྱ་ཆེར་རྒྱས་པར་བགྱི་བ་དང་། ཆོས་སྨྲ་བ་དེ་དག་རྗེས་སུ་ཡོངས་སུ་གཟུང་བ་དང་། དམ་པའི་ཆོས་ཅི་ནས་ཀྱང་ཡུན་རིང་དུ་གནས་པ་དང་། བདུད་ཐམས་ཅད་ཚར་གཅོད་པའི་སླད་དུ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་རྣམས་རྗེས་སུ་སྩལ་བར་འཚལ་ན། བཅོམ་ལྡན་འདས་ཀྱིས་དེ་ལ་ལེགས་སོ་ཞེས་བགྱི་བ་གནང་བར་མཛད་དུ་གསོལ། དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་རྒྱལ་པོ་ཆེན་པོ་བཞི་ལ་ཐུགས་བརྩེ་བའི་ཕྱིར་ཅང་མི་གསུང་བར་གྱུར་པས་གནང་སྟེ། རྒྱལ་པོ་ཆེན་པོ་བཞི་པོ་དེ་དག་གིས་ཀྱང་བཅོམ་ལྡན་འདས་ཀྱིས་ཅང་མི་གསུང་བར་གྱུར་པས་གནང་བར་རིག་ནས་ཆེད་དུ་བརྗོད་པ་ཆེད་དུ་བརྗོད་དོ། །དེ་ནི་འདི་ལྟ་སྟེ། ད་ནཊྜེ། དེ་ནཊྜེ། མ་ཧཱ་ནཊྜེ། ཛམྦྷ་ནེ། སྟམྦྷ་ནེ། ཊ་ཊེ། ཊ་ཊ་ནེ། མ་ཧཱ་ཊ་ཊ་ནེ། ཀུཊྚེ། ཀུཊྚ་ནེ། མ་ཧཱ་ཀུཊྚ་ནེ། དྷརྨ་ལེ། ཙརྨ་ལེ་སྭཱ་ཧཱ། བཅོམ་ལྡན་འདས་འདི་དག་ནི་གཟུངས་འདིའི་འཁོར་གྱི་ཚིག་ལྟ་བུར་གདའ་བ་ལགས་ཏེ། རིགས་ཀྱི་བུའམ། རིགས་ཀྱི་བུ་མོ་གང་ལ་ལ་གཟུངས་ཀྱི་གསང་སྔགས་ཀྱི་གཞི་འདི་དག་འཛིན་པ་དང་། བཟླས་ཏེ་བཏོན་ནས་འཆང་བ་འདི་ནི་སྔོན་གྱི་ལས་མ་གཏོགས་པར་གཞན་གྱི་གནོད་པ་ཐམས་ཅད་ལས་འདའ་བ་དང་། འགྲོ་བ་ཐམས་ཅད་དང་རྗེས་སུ་མཐུན་པར་འགྱུར་རོ། །དེ་ལ་ནི་དམ་པའི་ཆོས་ཐམས་ཅད་ཡོངས་སུ་གཟུང་བ་ལ་ཡང་བར་ཆད་འབྱུང་བར་མི་འགྱུར་ལགས་སོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་རྒྱལ་པོ་ཆེན་པོ་བཞི་པོ་དག་ལ་གྲོགས་པོ་དག་ཁྱོད་བསྟན་པ་ཡུན་རིང་དུ་གནས་པར་བྱ་བའི་ཕྱིར་ཞུགས་པ་ལེགས་སོ་ལེགས་སོ་ཞེས་བྱ་བ་བྱིན་ནོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་པ། ཀུན་དགའ་བོ་ཁྱོད་ཀྱིས་དེ་དག་རྫུ་འཕྲུལ་ཆེ་བར་འགྱུར་བ་དང་། ཕན་ཡོན་ཆེ་བ་དང་། བྱིན་ཆེ་བར་འགྱུར་བའི་ཕྱིར་གཟུངས་ཀྱི་ཆོས་ཀྱི་རྣམ་གྲངས་འདི་ཟུང་ཤིག །ལོང་ཤིག །ཆོང་ཤིག །ལྷོགས་ཤིག །ཀུན་ཆུབ་པར་གྱིས་ཤིག །གཞན་དག་ལ་ཡང་རྒྱ་ཆེར་རབ་ཏུ་སྟོན་ཅིག་གསོལ་པ། བཅོམ་ལྡན་འདས་རྒྱས་པར་བགྱི་བའི་སླད་དུ་བདག་གིས་གཟུང་ལགས་སོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་དོན་འདི་ཉིད་རྒྱས་པར་བསྟན་པའི་ཕྱིར་དེའི་ཚེ་ཚིགས་སུ་བཅད་པ་འདི་དག་བཀའ་སྩལ་ཏོ། །གཟུངས་སྔགས་ཀྱི་ནི་མདོ་སྡེ་འདི། །སངས་རྒྱས་ཀུན་གྱིས་གསུངས་པ་སྟེ། །སེམས་ཅན་རྣམས་ལ་ཕན་པའི་ཕྱིར། །དེ་ནི་ངས་ཀྱང་བཤད་པ་ཡིན། །གང་ཞིག་གཟུངས་སྔགས་མདོ་སྡེ་འདི། །འཛིན་ཏམ་ཡང་ན་ཀློག་བྱེད་པ། །ཚེ་རབས་ཉི་ཤུ་རྩ་གཅིག་ཏུ། །དེ་ནི་ཚེ་རབས་དྲན་པར་འགྱུར། །གཟུངས་སྔགས་འདི་རྗོད་བྱེད་དེ་ལ། །མི་མིན་གང་ཡང་རུང་བ་དང་། །གནོད་སྦྱིན་གྲུལ་བུམ་སྲིན་པོ་དག །ནམས་ཀྱང་བར་ཆད་མི་བྱེད་དོ། །གཟུངས་སྔགས་འདི་རྗོད་བྱེད་དེ་ལ། །འདོད་ཁམས་དབང་ཕྱུག་གང་ཡིན་པ། །སྡིག་ཅན་བདུད་དེ་ནམ་དུ་ཡང་། །བར་ཆད་བྱེད་པར་མི་འགྱུར་རོ། །གཟུངས་སྔགས་འདི་འཛིན་བྱེད་དེ་ཡི། །ཕ་མ་དང་ནི་ཕུ་ནུ་དང་། །སྲིང་མོ་མཛའ་བཤེས་གཉེན་དག་ཀྱང་། །ལོག་པར་ལྟུང་བར་མི་འགྱུར་རོ། །དེ་ཕྱིར་གཟུངས་སྔགས་མདོ་སྡེ་འདི། །འཇིགས་མེད་གང་གིས་ཐོས་གྱུར་ནས། །གཟུངས་སྔགས་འདི་རྗོད་བྱེད་པ་དེས། །ནམས་ཀྱང་འཇིགས་པར་མི་བྱའོ། །གཟུངས་སྔགས་འདི་འཛིན་བྱེད་དེ་ཡི། །བསྐལ་པ་བྱེ་བ་སྟོང་རྣམས་སུ། །སྒྲིབ་པའི་ལས་ནི་གང་བྱས་པ། །ཞག་བདུན་གྱིས་ནི་བྱང་བར་འགྱུར། །བྱང་ཆུབ་སེམས་དཔའི་བསོད་ནམས་གང་། །བསྐལ་པ་བྱེ་བར་བསྒྲུབས་པ་ཀུན། །གཟུངས་སྔགས་འདི་འཛིན་བྱེད་པ་དེས། །ཞག་བདུན་གྱིས་ནི་ཟིལ་གྱིས་ནོན། །ཕྱོགས་བཞིའི་འཇིག་རྟེན་སྐྱོང་བ་བོ། །རྒྱལ་པོ་ཆེན་པོ་བཞི་པོ་དག །ལུས་ངན་དང་ནི་ཡུལ་འཁོར་སྲུང་། །མིག་མི་བཟང་དང་འཕགས་སྐྱེས་པོ། །ཁ་དོག་ལྡན་ཞིང་གྲགས་དང་ལྡན། །འཁོར་བཅས་བློན་བཅས་དེ་དག་ཀྱང་། །གཟུངས་སྔགས་འདི་ནི་རྗོད་བྱེད་པ། །དེ་ལ་རྟག་ཏུ་སྲུང་བར་བྱེད། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་སྩལ་ནས། ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་དང་། ཉན་ཐོས་ཆེན་པོ་དེ་དག་དང་། འཕགས་པ་བྱམས་པ་དང་། འཇམ་དཔལ་ལ་སོགས་པ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་དང་། རྒྱལ་པོ་ཆེན་པོ་བཞི་དང་། ལྷ་དང་། མི་དང་། ལྷ་མ་ཡིན་དང་། དྲི་ཟར་བཅས་པའི་འཇིག་རྟེན་ཡི་རངས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལ་མངོན་པར་བསྟོད་དོ།། །།འཕགས་པ་རིག་སྔགས་ཀྱི་རྒྱལ་པོ་སྒྲོན་མ་མཆོག་གི་གཟུངས་རྫོགས་སོ།།
