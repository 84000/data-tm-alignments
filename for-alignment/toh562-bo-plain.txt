༄༅༅། །རྒྱ་གར་སྐད་དུ། མ་ཧཱ་ཤཱི་ཏ་བ་ནཱི་སཱུ་ཏྲ། བོད་སྐད་དུ། བསིལ་བའི་ཚལ་ཆེན་པོའི་མདོ། དཀོན་མཆོག་གསུམ་ལ་ཕྱག་འཚལ་ལོ། །འཇིག་རྟེན་མཁས་པ་རྫོགས་སངས་རྒྱས། །གང་གིས་ཐོག་མར་རིག་སྔགས་རྣམས། །འཛམ་བུའི་གླིང་དུ་གསུངས་པ་ཡི། །སངས་རྒྱས་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །ད་ལྟར་བྱུང་དང་འདས་པ་དང་། །མ་བྱོན་སངས་རྒྱས་གང་ལགས་པ། །ཐམས་ཅད་ལ་ཡང་བདག་ཕྱག་འཚལ། །དེ་དག་ཀུན་ལ་སྐྱབས་སུ་མཆི། །འཁོར་ལོ་བསྐོར་བ་བླ་མེད་པ། །མར་མེ་མཛད་ལ་ཕྱག་འཚལ་ལོ། །ཆོས་ཀྱི་རྒྱལ་པོ་འོད་མཛད་པ། །ཐམས་ཅད་གནོན་ལ་ཕྱག་འཚལ་ལོ། །ཐམས་ཅད་མཁྱེན་པ་གྲགས་ལྡན་པ། །པདྨའི་བླ་ལ་ཕྱག་འཚལ་ལོ། །སྤྱན་དང་ལྡན་པ་སྐྱོབ་མཛད་པ། །རབ་འཐོར་གཤེགས་ལ་ཕྱག་འཚལ་ལོ། །དེད་དཔོན་བླ་ན་མེད་པ་པོ། །གྲགས་པའི་བླ་ལ་ཕྱག་འཚལ་ལོ། །ཉི་ཟླ་ལྟ་བུར་འོད་མཛད་པ། །བདེ་མཛད་དེ་ལ་ཕྱག་འཚལ་ལོ། །ཆོས་ཀྱི་མེ་ལོང་སྟོན་པ་པོ། །དོན་གཟིགས་ལ་ཡང་ཕྱག་འཚལ་ལོ། །དུལ་བའི་ཚུལ་ཁྲིམས་འཇིགས་མེད་པ། །རྒྱལ་ཞེས་བྱ་ལ་ཕྱག་འཚལ་ལོ། །མཚན་མཆོག་སུམ་ཅུ་རྩ་གཉིས་ལྡན། །སྐར་རྒྱལ་ལ་ཡང་ཕྱག་འཚལ་ལོ། །རྫོགས་པའི་སངས་རྒྱས་དཔལ་ལྡན་པ། །རྣམ་པར་གཟིགས་ལ་ཕྱག་འཚལ་ལོ། །འོད་ཅན་གཟི་བརྗིད་ལྡན་པ་པོ། །གཙུག་ཏོར་ཅན་ལ་ཕྱག་འཚལ་ལོ། །ལེགས་པར་མཆོད་པ་གྲགས་ལྡན་པ། །ཐམས་ཅད་སྐྱོབ་ལ་ཕྱག་འཚལ་ལོ། །བདུད་དཔུང་རབ་ཏུ་འཇོམས་པ་པོ། །ལོག་དད་སེལ་ལ་ཕྱག་འཚལ་ལོ། །བྲམ་ཟེ་དཔལ་དང་ལྡན་པ་པོ། །གསེར་ཐུབ་དེ་ལ་ཕྱག་འཚལ་ལོ། །སེམས་ཅན་ཀུན་ལ་ཐུགས་བརྩེ་བ། །འོད་སྲུང་དེ་ལ་ཕྱག་འཚལ་ལོ། །ཤཱཀྱ་སེང་གེ་འོད་མཛད་པ། །གསེར་མདོག་ཅན་ལ་ཕྱག་འཚལ་ལོ། །སེར་སྣ་མེད་པ་ཐུགས་རྗེ་ཅན། །བྱམས་པ་ལ་ཡང་ཕྱག་འཚལ་ལོ། །སངས་རྒྱས་ཀུན་ལའང་ཕྱག་འཚལ་ཏེ། །དེ་དག་ལ་ནི་བདག་སྐྱབས་མཆི། །གང་དག་གིས་ནི་ཆོས་བསྟན་པ། །སངས་རྒྱས་ཀུན་ལ་ཕྱག་འཚལ་ལོ། །གང་ལ་སངས་རྒྱས་གུས་མཛད་པའི། །ཆོས་ལ་བདག་ནི་ཕྱག་འཚལ་ལོ། །གང་ལ་ཕུལ་ན་དོན་ཆེ་བའི། །དགེ་འདུན་ལ་ཡང་ཕྱག་འཚལ་ལོ། །ཉན་ཐོས་ཤེས་རབ་མཆོག་གྱུར་པ། །ཤཱ་རིའི་བུ་ལ་ཕྱག་འཚལ་ལོ། །རྫུ་འཕྲུལ་ཅན་གྱི་མཆོག་གྱུར་པ། །མཽད་གལ་བུ་ལ་ཕྱག་འཚལ་ལོ། །ཆོས་ཀྱི་གཏམ་ལ་འཇིགས་མེད་པ། །ཀཱ་ཏྱཱའི་བུ་ལ་ཕྱག་འཚལ་ལོ། །ཉན་ཐོས་སྦྱངས་པའི་མཆོག་གྱུར་པ། །འོད་སྲུང་ལ་ཡང་ཕྱག་འཚལ་ལོ། །སྐྱོབ་པ་ཡི་ནི་དབང་གྱུར་པ། །ཀཽཎྜི་ནྱ་ལ་ཕྱག་འཚལ་ལོ། །མང་དུ་ཐོས་པ་རིག་སྔགས་འཆང་། །ཀུན་དགའ་བོ་ལ་ཕྱག་འཚལ་ལོ། །ལུས་ངན་དང་ནི་ཡུལ་འཁོར་སྲུང་། །མིག་མི་བཟང་དང་འཕགས་སྐྱེས་པོ། །ཕྱོགས་བཞི་ཁོ་ར་ཁོར་ཡུག་གི །རྒྱལ་ཆེན་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །གནོད་སྦྱིན་རྣམས་ཀྱི་སྡེ་དཔོན་མཆོག །ཉི་ཤུ་བརྒྱད་ལ་ཕྱག་འཚལ་ལོ། །ཕ་མ་མཁན་པོ་སློབ་དཔོན་དང་། །ལྷ་རྣམས་ལ་ནི་ཕྱག་འཚལ་ལོ། །དེ་དག་ལ་ནི་ཕྱག་འཚལ་ནས། །རིག་སྔགས་འདི་ནི་གྲུབ་གྱུར་ཅིག །གང་གི་དོན་དུ་བྱ་བ་ཡི། །དོན་དེ་བདག་ལ་གྲུབ་གྱུར་ཅིག །སྡང་བར་སེམས་ཤིང་གནོད་བྱེད་པའི། །མི་མ་ཡིན་རྣམས་གུད་དུ་དེང་། །གནོད་པར་འདོད་པ་གླགས་ལྟ་རྣམས། །སྟོན་པའི་བསྟན་པ་མཉན་པར་གྱིས། །བྱམས་པར་སེམས་ཤིང་ཕན་འདོད་པའི། །མི་མ་ཡིན་པ་གང་ཡིན་རྣམས། །ཐམས་ཅད་འདུས་ཤིང་མཐུན་གྱུར་ནས། །འཇིག་རྟེན་སྐྱོང་བ་རྣམས་ཀྱི་ཚིག །སྟོན་པས་རྗེས་སུ་ཡི་རང་བ། །དགའ་བའི་སེམས་ཀྱིས་བདག་ལ་ཉོན། །བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་ནི་རྒྱལ་པོ་ཆེན་པོ་བཞིའི་བསྲུང་བ་ཆེན་པོ་སྟེ། འཁོར་བཞི་པོ་རྣམས་ལ་ཡོངས་སུ་ཁྱབ་པའོ། །གནོད་སྦྱིན་དང་། སྲིན་པོ་དང་། དྲི་ཟ་དང་། ཀླུ་དང་། ནམ་མཁའ་ལྡིང་དང་། གསང་བ་པ་དང་། འབྱུང་པོ་དང་། གྲུལ་བུམ་དང་། ཡི་དགས་དང་། སྲུལ་པོ་དང་། ཤ་ཟ་དང་། ལྷ་མ་ཡིན་དང་། རླུང་ལྷ་དང་། སྐེམ་བྱེད་དང་། འཚེ་རྒྱུ་དང་། སྨྱོ་བྱེད་དང་། མི་འམ་ཅི་དང་། གྱེན་དུ་རྒྱུ་དང་། རྒྱུ་བ་དང་། འདུན་པ་དང་། ངན་བྱེད་དང་། གཟི་བྱིན་འཕྲོག་པ་དང་། བརྗེད་བྱེད་དང་། རིམས་དྲག་པོ་དང་། རིམས་ཞག་གཅིག་པ་དང་། ཞག་གཉིས་པ་དང་། ཞག་གསུམ་པ་དང་། ཞག་བཞི་པ་དང་། མི་དང་མི་མ་ཡིན་པ། སྡང་བར་སེམས་པ་དང་། གླགས་ལྟ་བ་དང་། མི་ཕན་པར་འདོད་པ་དང་། གནོད་པར་བྱེད་པ། སངས་རྒྱས་བཅོམ་ལྡན་འདས་ཀྱི་བསྟན་པ་ལ་མི་དགའ་བ་དང་། གནོད་པར་འདོད་པ་དང་། ཕན་པར་མི་འདོད་པ་དང་། བདེ་བར་མི་འདོད་པ་དང་། གྲུབ་པ་དང་། བདེ་བར་མི་འདོད་པ་དང་། འཁོར་བཞི་པོ་རྣམས་ལ་མི་དགའ་བ་དང་། གནོད་པར་འདོད་པ་དང་། ཕན་པར་མི་འདོད་པ་དང་། བདེ་བར་མི་འདོད་པ་དང་། གྲུབ་པ་དང་། བདེ་བར་མི་འདོད་པ་དང་། མིང་འདི་ཞེས་བྱ་བ་ལ་མི་དགའ་བ་དང་། གནོད་པར་འདོད་པ་དང་། ཕན་པར་མི་འདོད་པ་དང་། བདེ་བར་མི་འདོད་པ་དང་། གྲུབ་པ་དང་། བདེ་བར་མི་འདོད་པ་གང་ཡིན་པ་དེ་དག་ནི་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་ཐོས་ནས་དེངས་ཤིག །སྐྲག་པར་གྱུར་ཅིག །དངངས་པར་གྱུར་ཅིག །ཀུན་ཏུ་དངངས་པར་གྱུར་ཅིག །འདི་ན་མ་འཁོད་ཅིག །ངན་སེམས་དང་གནོད་སེམས་ཀྱི་མགོ་བོ་ཚལ་པ་བདུན་དུ་འགས་ཏ་རེ། གནོད་སྦྱིན་དང་། སྲིན་པོ་དང་། དྲི་ཟ་དང་། ཀླུ་དང་། ནམ་མཁའ་ལྡིང་དང་། གསང་བ་པ་དང་། འབྱུང་པོ་དང་། གྲུལ་བུམ་དང་། ཡི་དགས་དང་། སྲུལ་པོ་དང་། ཤ་ཟ་དང་། ལྷ་མ་ཡིན་དང་། རླུང་ལྷ་དང་། སྐེམ་བྱེད་དང་། འཚེ་རྒྱུ་དང་། སྨྱོ་བྱེད་དང་། མི་འམ་ཅི་དང་། གྱེན་དུ་རྒྱུ་དང་། རྒྱུ་བ་དང་། འདུན་པ་དང་། ངན་བྱེད་དང་། གཟི་བྱིན་འཕྲོག་པ་དང་། བརྗེད་བྱེད་དང་། རིམས་དྲག་པོ་དང་། རིམས་ཞག་གཅིག་པ་དང་། ཞག་གཉིས་པ་དང་། ཞག་གསུམ་པ་དང་། ཞག་བཞི་པ་དང་། མི་དང་མི་མ་ཡིན་པ་སྡང་བར་སེམས་པ་མ་ཡིན་པ་རྣམས་དང་། གླགས་མི་ལྟ་བ་དང་། ཕན་པར་འདོད་པ་དང་། གནོད་པར་མི་འདོད་པ་དག་སངས་རྒྱས་བཅོམ་ལྡན་འདས་ཀྱི་བསྟན་པ་ལ་དགའ་བ་དང་། དོན་དུ་འགྱུར་བར་འདོད་པ་དང་། ཕན་པར་འདོད་པ་དང་། བདེ་བར་འདོད་པ་དང་། གྲུབ་པ་དང་། བདེ་བར་འདོད་པ་དང་། མིང་འདི་ཞེས་བྱ་བ་ལ་དགའ་བ་དང་། དོན་དུ་འགྱུར་བར་འདོད་པ་དང་། ཕན་པར་འདོད་པ་དང་། བདེ་བར་འདོད་པ་དང་། གྲུབ་པ་དང་། བདེ་བར་འདོད་པ་གང་ཡིན་པ་དེ་དག་ནི་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་ཐོས་ནས་འདིར་འཁོད་ཅིག །མ་སྐྲག་ཅིག །མ་དངངས་པར་གྱུར་ཅིག །ཀུན་ཏུ་དངངས་པར་མ་གྱུར་ཅིག །མི་འཇིགས་པ་ཁོ་ནར་གྱུར་ཅིག །མིང་འདི་ཞེས་བྱ་བའི་དོན་དང་། ཕན་པ་དང་། བདེ་བ་དང་། བདེ་བར་གནས་པར་བྱ་བའི་ཕྱིར། བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་བདག་གིས་བསྟན་ཅིང་བཤད་པར་བྱའོ། །དེ་ལ་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་བདག་ལ་བསྲུང་བ་ནི་འདི་ཡིན་ཏེ། སྱཱད་ཡ་ཐེ་དན། ཁ་ཊེ་ཁ་ཊེ། ཁཏྟྱ་སི། པ་ལ་ཀ་བ་ཊེ། རོ་ག་བྷ་དྲི་ག་ཎེ། ཧི་ལི་ཧི་ལི། དུ་མ་ཏེ། གྲྀཏྟ་ཏི། ཨ་ཛ་ཊི། ཀ་ཐ་རི། མ་སཱ་ར་ཀལྤེ། ས་མནྟེ་ན། ཙ་ཏུདྡི་ཤི །ཡོ་ཛ་ན་ཤ་ཏ། ཨཱཏྨ་རཀྵ། ཨ་ན་ཏི་ཀྲ་མ་ཎི། སརྦ་བི་ཧེ་ཋ་ཀེ་བྷྱཿན་མོ་བྷ་ག་བ་ཏེ། བུད་དྷ་སྱ། སིད་དྷྱནྟུ་མནྟྲ་པ་དཱ། ད་ར་དུ་བི་དྱཱ། བྲཧྨ་ཎོ། མ་ན་དུ་སྭཱ་ཧཱ། འདི་སྐད་བདག་གིས་ཐོས་པ་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་རྒྱལ་པོའི་ཁབ་ན། དུར་ཁྲོད་ཆེན་པོ་བསིལ་བའི་ཚལ་འཇིགས་སུ་རུང་བ་སྐྱི་བུང་ཞེས་བྱེད་པ་ན། དགེ་སློང་སྟོང་ཉིས་བརྒྱ་ལྔ་བཅུའི་དགེ་སློང་གི་དགེ་འདུན་ཆེན་པོ་དང་ཐབས་ཅིག་ཏུ་བཞུགས་སོ། །དེ་ནས་རྒྱལ་པོ་ཆེན་པོ་བཞི་ལུས་ངན་དང་། ཡུལ་འཁོར་སྲུང་དང་། མིག་མི་བཟང་དང་། འཕགས་སྐྱེས་པོ་བུ་དང་བཅས། བློན་པོ་དང་བཅས། འཁོར་དང་བཅས། མངག་གཞུག་པ་དང་བཅས། ཕོ་ཉ་དང་བཅས། གཡོག་འཁོར་དང་བཅས་པ་ཁ་དོག་འཕགས་པ་དག་ནམ་ཕྱེད་ན་ལྷགས་ནས་གྲོས་མཐུན་པར་དུར་ཁྲོད་ཆེན་པོ་བསིལ་བའི་ཚལ་དུ་དོང་སྟེ། རང་བཞིན་གྱི་ཁ་དོག་དང་མཐུས་འོད་ཆེན་པོས་སྣང་བར་བྱས་ནས། བཅོམ་ལྡན་འདས་ག་ལ་བ་དེར་སོང་སྟེ་ཕྱིན་ནས། བཅོམ་ལྡན་འདས་ཀྱི་ཞབས་གཉིས་ལ་མགོ་བོས་ཕྱག་འཚལ་ནས་ལན་གསུམ་བསྐོར་བ་བྱས་ནས་ཐལ་མོ་སྦྱར་ཏེ། བཅོམ་ལྡན་འདས་འབའ་ཞིག་ལ་ཕྱག་འཚལ་ཞིང་ཕྱོགས་གཅིག་ཏུ་འཁོད་དོ། །ཕྱོགས་གཅིག་ཏུ་འཁོད་ནས་རྒྱལ་པོ་ཆེན་པོ་བཞིས་བཅོམ་ལྡན་འདས་ལ་ཚིགས་སུ་བཅད་པས་བསྟོད་པ། འཇིག་རྟེན་མཁས་པ་རྫོགས་སངས་རྒྱས། །དཔའ་ཆེན་ཁྱོད་ལ་ཕྱག་འཚལ་ལོ། །གཽ་ཏམ་ཁྱོད་ཀྱིས་གང་མཁྱེན་པ། །དེ་ནི་ལྷ་རྣམས་མ་འཚལ་ཏོ། །ལན་གཉིས་ལན་གསུམ་དུ་ཡང་རྒྱལ་པོ་ཆེན་པོ་བཞིས་བཅོམ་ལྡན་འདས་ལ་ཚིགས་སུ་བཅད་ནས་བསྟོད་པ། འཇིག་རྟེན་མཁས་པ་རྫོགས་སངས་རྒྱས། །དཔའ་ཆེན་ཁྱོད་ལ་ཕྱག་འཚལ་ལོ། །གཽ་ཏམ་ཁྱོད་ཀྱིས་གང་མཁྱེན་པ། །དེ་ནི་ལྷ་རྣམས་མ་འཚལ་ཏོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ལ་རྒྱལ་པོ་ཆེན་པོ་བཞིས་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་ཡུལ་དང་འཕྲོད་ལགས་སམ། འཚོ་བ་དང་ལྡན་ལགས་སམ། བཙུན་པ་བཅོམ་ལྡན་འདས་སྐུ་བདེ་ལགས་སམ། སྙུན་མི་མངའ་ལགས་སམ། ཉམ་ང་བ་མ་མཆིས་སམ། བཅོམ་ལྡན་འདས་ཀྱི་སྐུ་ལ་གནོད་པ་མི་མངའ་ལགས་སམ། བཅོམ་ལྡན་འདས་ལ་གནོད་སྦྱིན་དང་། སྲིན་པོ་དང་། དྲི་ཟ་དང་། ཀླུ་དང་། ནམ་མཁའ་ལྡིང་དང་། གསང་བ་པ་དང་། འབྱུང་པོ་དང་། གྲུལ་བུམ་དང་། ཡི་དགས་དང་། སྲུལ་པོ་དང་། ཤ་ཟ་དང་། ལྷ་མ་ཡིན་དང་། རླུང་ལྷ་དང་། སྐེམ་བྱེད་དང་། འཚེ་རྒྱུ་དང་། སྨྱོ་བྱེད་དང་། མི་འམ་ཅི་དང་། གྱེན་དུ་རྒྱུ་དང་། རྒྱུ་བ་དང་། འདུན་པ་དང་། ངན་བགྱིད་དང་། གཟི་བྱིན་འཕྲོག་པ་དང་། བརྗེད་བྱེད་དང་། རིམས་དྲག་པོ་དང་། རིམས་ཞག་གཅིག་པ་དང་། ཞག་གཉིས་པ་དང་། ཞག་གསུམ་པ་དང་། ཞག་བཞི་པ་དང་། མི་དང་། མི་མ་ཡིན་པ་སྡང་བར་སེམས་པ་དང་། གླགས་ལྟ་བ་དང་། ཕན་པར་མི་འདོད་པ་དང་། གནོད་པར་བགྱིད་པ་དག་ལས་གནོད་པར་བགྱིད་པར་སེམས་པ་ལྟ་མ་མཆིས་ལགས་སམ། དེ་སྐད་ཅེས་གསོལ་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་རྒྱལ་པོ་ཆེན་པོ་བཞི་ལ་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། །ཀྱེ་རྒྱལ་པོ་ཆེན་པོ་དག་ང་ལ་ནི་ཐམས་ཅད་ཡོད་དེ། ཡུལ་དང་ཡང་འཕྲོད། འཚོ་བ་དང་ཡང་ལྡན། ལུས་ཀྱང་བདེ། ནད་ཀྱང་མེད། ཉམ་ང་བ་ཡང་མེད་དོ། །ཀྱེ་རྒྱལ་པོ་ཆེན་པོ་དག་ངའི་ལུས་ལ་ནི་གནོད་པ་ཡང་མེད་དོ། །ཀྱེ་རྒྱལ་པོ་ཆེན་པོ་དག་ངས་ནི་འཇིག་རྟེན་ལྷ་དང་བཅས་པ། བདུད་དང་བཅས་པ། ཚངས་པ་དང་བཅས་པ། དགེ་སྦྱོང་དང་། བྲམ་ཟེར་བཅས་པའི་སྐྱེ་དགུ་དང་། ལྷ་དང་མིར་བཅས་པའི་ནང་ན་གང་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ལ་གནོད་པ་བྱེད་པར་སེམས་པ་ཡང་དག་པར་མ་མཐོང་ངོ་། །ཀྱེ་རྒྱལ་པོ་ཆེན་པོ་དག་དེ་ལྟ་མོད་ཀྱི། ཁྱོད་ཀྱི་འཁོར་རྣམས་ངའི་འཁོར་ལ་གནོད་པ་བྱེད་པར་སེམས་སོ། །དེ་སྐད་ཅེས་བཀའ་སྩལ་པ་དང་། བཅོམ་ལྡན་འདས་ལ་རྒྱལ་པོ་ཆེན་པོ་བཞིས་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་བདག་ཅག་གསོལ་བ་མཆིས་ཏེ། དེའི་སླད་དུ་བཅོམ་ལྡན་འདས་ལ་བལྟ་བ་དང་ཕྱག་བགྱི་བ་དང་། བསྙེན་བཀུར་བགྱི་བ་དང་། བཅོམ་ལྡན་འདས་ཉིད་ལ་ཞུ་བའི་སླད་དུ་བདག་ཅག་བཅོམ་ལྡན་འདས་ཀྱི་སྤྱན་སྔར་མཆིས་ལགས་སོ། །དེ་ཅིའི་སླད་དུ་ཞེ་ན། བཙུན་པ་བཅོམ་ལྡན་འདས་དགོན་པ་དང་། བས་མཐའི་གནས་མལ་འདི་དག་ན་ཤིན་ཏུ་སྡང་བའི་གནོད་སྦྱིན་དང་། སྲིན་པོ་དང་། དྲི་ཟ་དང་། ཀླུ་དང་། ནམ་མཁའ་ལྡིང་དང་། གསང་བ་པ་དང་། འབྱུང་པོ་དང་། གྲུལ་བུམ་དང་། ཡི་དགས་དང་། སྲུལ་པོ་དང་། ཤ་ཟ་དང་། ལྷ་མ་ཡིན་དང་། རླུང་ལྷ་དང་། སྐེམ་བྱེད་དང་། འཚེ་རྒྱུ་དང་། སྨྱོ་བྱེད་དང་། མི་འམ་ཅི་དང་། གྱེན་དུ་རྒྱུ་དང་། རྒྱུ་བ་དང་། འདུན་པ་དང་། ངན་བྱེད་དང་། གཟི་བྱིན་འཕྲོག་པ་དང་། བརྗེད་བྱེད་དང་། རིམས་དྲག་པོ་དང་། རིམས་ཞག་གཅིག་པ་དང་། ཞག་གཉིས་པ་དང་། ཞག་གསུམ་པ་དང་། ཞག་བཞི་པ་དང་། མི་དང་མི་མ་ཡིན་པ་སྡང་བར་སེམས་པ་དང་། གླགས་ལྟ་བ་དང་། ཕན་པར་མི་སེམས་པ་དང་། གནོད་པར་བགྱིད་པ་དག་མཆིས་སོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་ཀྱི་ཉན་ཐོས་དེ་དག་ཀྱང་དེ་ན་གནས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱི་ཉན་ཐོས་དེ་དག་ནི་སྲོད་དང་ཐོ་རངས་མི་ཉལ་བར་སྦྱོར་བ་ལ་བརྩོན་པར་གནས་ལགས་སོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་གནོད་སྦྱིན་ནམ་སྲིན་པོ་ནས་གནོད་པར་བགྱིད་པའི་བར་དག་བཅོམ་ལྡན་འདས་ཀྱི་བསྟན་པ་འདི་ལ་དད་པའི་སེམས་དང་ལྡན་པ་ནི་ཉུང་ཟད་ཅིག་མཆིས་པར་བས་སོ། །གང་དག་བཅོམ་ལྡན་འདས་ཀྱི་བསྟན་པ་འདི་ལ་མ་དད་པའི་སེམས་དང་ལྡན་པ་ནི་ཆེས་མང་ལགས་ཏེ། དེ་དག་ནི་བཙུན་པ་བཅོམ་ལྡན་འདས་ཀྱི་ཉན་ཐོས་སྲོད་དང་ཐོ་རངས་ལ་མི་ཉལ་ཞིང་སྦྱོར་བ་ལ་བརྩོན་པར་གནས་པ་རྣམས་ལ་གནོད་པར་བགྱིད་པར་སེམས་ལགས་ཏེ། དེ་དག་དགག་པའི་སླད་དུ་གནོད་སྦྱིན་མ་རུངས་པ་མ་དད་པ་རྣམས་དད་པར་བགྱི་བ་དང་། དད་པ་རྣམས་སླར་ཞིང་དད་པར་བགྱི་བ་དང་། འཁོར་བཞི་པོ་རྣམས་བདེ་ལེགས་དང་། བདེ་བར་གནས་པར་བགྱི་བ་དང་། མ་ལུས་པར་ཤིན་ཏུ་མཉམ་པར་གཞོག་པར་འགྱུར་བའི་སླད་དུ། བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་བཤད་པར་འཚལ་ལོ། །མི་མ་ཡིན་པ་སྡང་སེམས་ཅན། །རྒྱལ་བའི་བསྟན་ལ་མ་དད་པ། །དེ་དག་བདག་གིས་བཟློག་པར་བྱ། །རྒྱལ་ཆེན་རྣམས་ནི་བདག་ལ་ཉོན། །མདོ་ཆེན་འདི་ནི་ཐོས་ནས་ཀྱང་། །མི་རྣམས་ལ་ནི་གནོད་མ་བྱེད། །གནོད་སྦྱིན་སྲིན་པོ་ཀླུ་རྣམས་དང་། །ནམ་མཁའ་ལྡིང་དང་གསང་བ་པ། །ཕན་པར་མི་འདོད་གང་ཡང་རུང་། །དེ་དག་ཐམས་ཅད་བཟློག་པར་བྱ། །ལྷ་དང་ལྷ་མིན་རླུང་ལྷ་དང་། །ལྷ་མོའི་ཚོགས་ཀྱིས་གང་བ་ཡི། །གཞལ་མེད་ཁང་མཛེས་དེ་དག་ནི། །མེ་དང་འདྲ་བར་འོད་གསལ་ཞིང་། །ཉི་མ་ཟླ་བའི་ཚུལ་འདྲ་བ། །རིན་ཆེན་ལས་བྱས་ཤིན་ཏུ་སྣང་། །བསོད་ནམས་བྱས་པ་དག་གིས་གང་། །རབ་སྣང་གདོན་རྣམས་རྣམ་པར་སྤངས། །མི་ལ་གནོད་བྱེད་གང་ཡིན་པའི། །འབྱུང་པོ་གྲུལ་བུམ་སྲིན་པོ་དང་། །ཡི་དགས་སྲུལ་པོ་དེ་དག་གིས། །རྨད་བྱུང་དེ་དག་མི་མཐོང་སྟེ། །གཟི་བྱིན་ལྡན་པའི་ལྷ་རྣམས་ཀྱི། །གྲོང་ཁྱེར་དུ་ཡང་དེ་མི་འགྲོ། །འབྱུང་པོ་ཆེན་པོ་འདུ་བ་ན། །དེ་ཡང་དེ་དག་འདུར་མི་དབང་། །སྟན་ནམ་ཡང་ན་ཆུ་ཡང་རུང་། །ཟས་དང་སྐོམ་ཡང་མི་དབང་ངོ་། །ལུས་ངན་གྱི་ནི་ཕོ་བྲང་དུ། །ནམ་ཡང་དེ་དག་འགྲོ་མི་འགྱུར། །རྒྱལ་པོ་ཆེན་པོ་རྣམས་ཀྱི་ཚིག །འབྱུང་པོ་གང་ཞིག་མི་ཉན་པ། །དེ་ཡི་ཁ་ནས་ཁྲག་དྲོན་དག །མྱུར་དུ་རབ་ཏུ་འབབ་པར་འགྱུར། །དེ་ཡི་བརླ་ཡི་གཡས་རོལ་ནས། །ཕོལ་མིག་དག་ཀྱང་འབྱུང་འགྱུར་ཏེ། །དེ་ནི་ནད་ཀྱིས་འཆི་འགྱུར་ཞིང་། །གནོད་སྦྱིན་གཟུགས་ཀྱང་སྟོན་པར་འགྱུར། །འཇིག་རྟེན་སྐྱོང་གིས་བཤད་པ་ཡི། །མདོ་ལས་གང་ཞིག་འགལ་བྱེད་པ། །འཇིགས་པ་ཐམས་ཅད་དེ་ལ་འོང་། །མགོ་བོ་ཚལ་པ་བདུན་དུ་འགས། །རྒྱལ་པོ་ཆེན་པོ་རྣམས་ཀྱི་ཚིག །གནོད་སྦྱིན་གང་དག་མི་ཉན་པ། །འབྱུང་པོ་སྲིན་པོ་གང་ཡོད་པ། །ངན་ཕྲེང་ལྷག་པའི་ངན་ཕྲེང་རྣམས། །ལྕགས་ལས་བྱས་པའི་འཁོར་ལོ་ཡི། །མཚོན་ཆ་སྤུ་གྲིའི་སོ་འདྲ་བས། །གནོད་སྦྱིན་དེ་དག་བརླག་པར་བྱ། །རྒྱལ་པོ་ཆེན་པོ་དཔལ་དང་ལྡན། །གོ་ཆ་སྲ་ཞིང་མཚོན་རྣོ་ལ། །སྟན་རྣམས་ལ་ནི་འཁོད་པ་དག །ཤར་ཕྱོགས་ན་ནི་ཡུལ་འཁོར་སྲུང་། །ལྷོ་ཕྱོགས་ན་ནི་འཕགས་སྐྱེས་པོ། །ནུབ་ཕྱོགས་ན་ནི་མིག་མི་བཟང་། །བྱང་ཕྱོགས་ན་ནི་ལུས་ངན་པོ། །དེ་དག་དེ་ན་འཁོད་པ་ཡི། །ཉེ་འཁོར་གནོད་སྦྱིན་འདི་རྣམས་གནས། །སེང་གེའི་གཟུགས་དང་སྟག་གི་གཟུགས། །མི་སྡུག་པ་དག་འགྲོ་ཞིང་རྒྱུ། །གླང་ཆེན་གཟུགས་དང་རྟ་ཡི་གཟུགས། །ཁ་ཅིག་མ་ཧེའི་གཟུགས་ཅན་ལ། །དྲེད་མོ་དང་ནི་ཝ་ཡི་གཟུགས། །ལུག་དང་སྤྱང་ཀི་ཕག་རྒོད་རྣམས། །རྣ་བ་འཕྱང་དང་རྣ་བོ་ཆེ། །རྣ་བ་གཅིག་པ་ཐ་དད་པ། །སྒྲ་ཆེར་འབྱིན་ཅིང་འཇིགས་བྱེད་པ། །ས་གཞི་རབ་ཏུ་གཡོ་བར་བྱེད། །བོང་བུ་དང་ནི་རྔ་མོ་ཁྱི། །ཆ་ལུགས་གཞན་ཡང་ཉེ་འཁོར་འཁོད། །རྒྱལ་པོ་ཆེན་པོ་སྟོབས་པོ་ཆེ། །དེ་དག་ཕན་ཚུན་འདུ་བ་ན། །ལྷ་ཡི་ཤིང་རྟ་བཅས་པ་ཡིས། །ནམ་མཁའ་ལས་ནི་འགྲོ་བར་བྱེད། །གནོད་སྦྱིན་དག་གི་བདག་པོ་ཡང་། །མི་རྣམས་ལ་ནི་བདེ་ལེགས་བྱེད། །སྱཱད་ཡ་ཐེ་དན། བ་ཀ་ཧུ་མུ་ལེ། ཧ་ཤི་ཎ་ཤ་ཤི་ཎ། བ་ན་མུ་ཧ་ལེ་ས་མུ་ཧཱ་ལེ། ཨུ་དུ་ཧཱ་ལེ། ས་མ་ཧཱ་ལེ། པྲ་ཤ་མ་མཱི། རཱཀྵ་ས། ཨ་མ་ནུ་ཥྱཱ། བཱ་རེ་མི། ཡཀྵ་ནི། བཱ་རཱ་མི་པྲེ་ཏ་མ་བྷི་ཐཻ། དུ་མ་ནུ་ཥྱཱ། མ་ས་པ་ཏ་ཧ་དྷ་མུར་ནན། སུ་ཏི་ཥྱ་ཏི། རྒྱལ་པོ་ཆེན་པོ་རྣམ་ཐོས་བུ། །མི་ལ་ཞོན་པ་སྡེར་བཅས་པས། །འདི་སྨྲས་ནས་ནི་ཚུབ་མ་ལངས། །དེ་བཞིན་རི་ཡང་གཡོས་པར་གྱུར། །རྒྱལ་པོ་ཆེན་པོ་རྣམ་ཐོས་བུའི། །རིན་ཆེན་ཤིང་རྟ་བཅས་པ་ན། །ལྷ་དང་མི་ཡི་སྒྲ་སྐད་གྲག །རྒྱལ་པོ་ཆེན་པོ་འཆི་བ་ན། །དངོས་པོ་རྣམས་ཀྱང་འབར་བར་འགྱུར། །སྲིན་པོ་འདོད་དགུར་བསྒྱུར་བའི་གཟུགས། །དེ་ཡི་མདུན་ན་རྒྱུག་པར་བགྱིད། །མི་མ་ཡིན་པའི་གཟུགས་རྣམས་ཀྱིས། །འཇིགས་ཤིང་སྐྲག་པ་བསྐྱེད་པར་བགྱིད། །ཆེ་ཞིང་གཏུམ་ལ་མི་བཟད་པ། །དྲག་ཅིང་མཐོང་ན་མི་སྡུག་ལ། །སྤུ་ཡང་རིང་ཞིང་སེན་མོ་རིང་། །ལག་ན་རལ་གྲི་ཐོ་བ་ཐོགས། །སོ་ནི་ཟངས་ཅན་འཇིགས་པར་བགྱིད། །སྣ་ཆེ་རྔ་མོ་སྟག་གི་འདྲ། །དྲག་ཅིང་ལག་པ་ཁྲག་དམར་བ། །ཡན་ལག་མ་ཚང་ཕྱེད་ཚལ་ལུས། །མིག་སེར་ལུས་ཀྱང་ཤིན་ཏུ་སེར། །དྲི་མ་ཅན་ལ་ལུས་སྐེམ་པ། །སྐྲ་བརྫེས་ལྕགས་ཀྱི་མཆུ་ཅན་ལ། །མཚོན་ཆ་ཚབས་ཆེར་འདེབས་པར་བགྱིད། །རྣ་བ་གཞུ་རྣ་གསུས་པོ་ཆེ། །སྦྲུལ་གྱི་གཟུགས་ཅན་བེའུ་གཟུགས། །ཕག་གི་གཟུགས་ཅན་ལྟོ་བོ་ཆེ། །མགོ་བོ་ཆེ་ལ་ལག་རིང་བ། །རྣ་འཕྱང་བ་ལ་ནུ་མ་འཕྱང་། །དེ་འདྲའི་གནོད་སྦྱིན་མང་པོ་ཡི། །གཡོག་འཁོར་དག་གིས་ཡོངས་སུ་བསྐོར། །གནོད་སྦྱིན་བྱེ་བ་ཕྲག་སྟོང་རྣམས། །དེ་དག་མ་ནི་ལིམ་ལོག་ཅན། །གཡོག་འཁོར་དེ་དག་ལྷན་ཅིག་ཏུ། །བདག་ནི་ཕྱག་འཚལ་སླད་དུ་མཆི། །གཽ་ཏམ་དཔའ་བོ་ཁྱོད་ལ་ནི། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །དྲག་ཅིང་རྒོད་ལ་གཏུམ་པ་དང་། །མགོ་བོ་ཆེ་ལ་སྐྲ་རིང་བ། །མི་རྣམས་དང་ནི་རི་དགས་དང་། །བྱ་རྣམས་གཟི་བྱིན་འཕྲོག་པར་བགྱིད། །མེ་འབར་ཟ་དང་མང་དུ་ཟ། །མིག་གཅིག་པ་དང་མིག་སེར་དང་། །སྣ་དང་རྣ་བ་བཅད་པ་དང་། །ཆད་པ་མང་པོས་བཅད་པ་དང་། །མགོ་མ་མཆིས་དང་མགོ་བཅད་དང་། །ལུས་ཀྱང་སྒུར་ལ་གཉེར་མ་གདོང་། །མིག་ལ་མེ་འབར་རྐང་མེད་པ། །མི་རྣམས་འཇིགས་པར་བགྱིད་པ་སྟེ། །འཕེལ་ཀ་ཆེ་ལ་རྒོད་བག་ཅན། །དེ་བཞིན་འགྲོ་ཞིང་རྒྱུ་བར་བགྱིད། །དེ་འདྲའི་གནོད་སྦྱིན་མང་པོ་ཡི། །གཡོག་འཁོར་དག་གིས་ཡོངས་སུ་བསྐོར། །གནོད་སྦྱིན་བྱེ་བ་ཕྲག་སྟོང་རྣམས། །དེ་དག་མ་ནི་རྐུན་མ་ལགས། །གཡོག་འཁོར་དེ་དག་ལྷན་ཅིག་ཏུ། །བདག་ནི་ཕྱག་འཚལ་སླད་དུ་མཆིས། །གཽ་ཏམ་དཔའ་བོ་ཁྱོད་ལ་ནི། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །བདག་ཅག་གི་ནི་གནས་རྣམས་ན། །ཤིང་རྟ་ཞེས་བགྱི་སྲིན་མོ་མཆིས། །དེ་ལ་བུ་ཕོ་སྟོབས་ལྡན་པ། །སྟོང་ཕྲག་བཅུ་ཡི་གྲངས་སྙེད་མཆིས། །དེ་དག་བདག་གི་གཡོག་ལགས་ཏེ། །དེ་དག་ཕྱག་འཚལ་སླད་དུ་མཆིས། །གཽ་ཏམ་དཔའ་བོ་ཁྱོད་ལ་ནི། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །བདག་ཅག་གི་ནི་གནས་རྣམས་ན། །མགྲིན་རིངས་ཞེས་བགྱི་སྲིན་མོ་མཆིས། །སྲིན་མོ་ཤིན་ཏུ་གནག་པ་དང་། །སྲིན་མོ་ནག་མོ་ཞེས་བགྱི་དང་། །སྲིན་མོ་རབ་གནག་རྣམས་མཆིས་ཏེ། །དེ་དག་གཽ་ཏམ་མཐུན་པར་བལྟ། །ས་ཡི་གནོད་སྦྱིན་བདག་ཉིད་ཆེ། །དེ་དག་ས་སྟེངས་གནས་པ་སྟེ། །གཡོག་འཁོར་དེ་དག་རྣམས་ཀྱིས་ནི། །ཡོངས་སུ་བསྐོར་ཏེ་འདིར་མཆིས་ལགས། །དེ་དག་ཀུན་དང་ལྷན་ཅིག་ཏུ། །བདག་ནི་ཕྱག་འཚལ་སླད་དུ་མཆིས། །གཽ་ཏམ་དཔའ་བོ་ཁྱོད་ལ་ནི། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །བདག་གི་མདུན་ན་རྒྱུག་པ་ཡི། །ཡི་དགས་གནོད་སྦྱིན་སྲིན་པོ་རྣམས། །ལུས་ཆེན་ཤུགས་ཀྱང་དྲག་པོ་སྟེ། །སོ་ཡང་ཆེ་ལ་ལྟོ་བོ་ཆེ། །རྣ་འཕྱང་འཇིགས་པ་ཆེན་པོ་སྟེ། །སློ་མ་རྣ་ལ་རབ་ཏུ་འཕྱང་། །ཡན་ལག་དྲུག་པ་སོ་བརྟན་པ། །གུར་གུམ་མདོག་ཅན་ཐོ་བ་འདྲ། །སོ་ཡང་རྩུབ་ཅིང་རྣོ་བ་དང་། །མདུང་ཐུང་གསལ་ཞིང་འཇིགས་པར་བགྱིད། །གླང་ཆེན་མགོ་དང་བུམ་སྣ་ཅན། །མགོ་བོ་སྟག་དང་སེང་གེ་འདྲ། །ལྕགས་ཀྱི་སོ་ཅན་ལྕགས་ཀྱི་སྐྲ། །གདོན་རྣམས་རྣམ་པར་སྐྲག་བགྱིད་པ། །ལག་ན་ལྕགས་ཀྱི་གཏུན་བུ་ཐོགས། །ལག་ན་ལྕགས་ཀྱི་ཐོ་བ་ཐོགས། །སོ་རྣོན་ལག་པ་རིང་བ་སྟེ། །འཇིགས་པའི་གཟུགས་ཅན་ལྕགས་ཀྱི་མཆུ། །མགོ་བོ་ཆེ་ལ་སྐྲ་གཡེངས་པ། །གནག་ཅིང་མིག་སེར་སྣ་ཞོམ་པ། །བུམ་པའི་མགྲིན་དང་མིག་དམར་དང་། །མིག་གཅིག་པ་དང་ལྟོ་བོ་ཆེ། །མཆུ་འཕྱངས་པ་དང་ལག་གཅིག་པ། །རྐང་གཅིག་པ་དང་རྐང་གཉིས་པ། །ཁ་མེད་ཁ་གཉིས་དྲག་པོ་སྟེ། །མི་རྣམས་རབ་ཏུ་འཇིགས་པར་བགྱིད། །ལག་ན་ཁར་བ་དབྱུག་པ་ཐོགས། །མདུང་རྩེ་གསུམ་པ་སྐྲག་ཅིང་འཇིགས། །སྐྲག་ཅིང་འཇིགས་པར་བྱེད་པ་སྟེ། །གདོན་རྣམས་རྣམ་པར་སྐྲག་བགྱིད་པ། །སྐྲག་པར་བགྱིད་ཅིང་སྨྲ་བར་བགྱིད། །སེམས་ཀྱང་འཁྲུལ་བར་བགྱིད་པ་སྟེ། །སྨྱོ་བར་བགྱིད་ཅིང་རྨོངས་པར་བགྱིད། །དེ་ནས་གཟི་བྱིན་འཕྲོག་པར་བགྱིད། །དེ་དག་འཇིག་རྟེན་འཇིགས་པ་སྟེ། །སྟོབས་པོ་ཆེ་ལ་བསྟེན་པར་དཀའ། །སྟོང་ཕྲག་དྲུག་ཅུ་རྩ་བཞི་པོ། །རྒྱལ་པོ་ཆེན་པོའི་མདུན་ན་རྒྱུག །དེ་འདྲའི་གནོད་སྦྱིན་མང་པོ་ཡི། །གཡོག་འཁོར་དག་གིས་ཡོངས་སུ་བསྐོར། །གནོད་སྦྱིན་བྱེ་བ་ཕྲག་སྟོང་རྣམས། །དེ་དག་ཕྱག་འཚལ་སླད་དུ་མཆིས། །གཽ་ཏམ་དཔའ་བོ་ཁྱོད་ལ་ནི། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །ས་ག་ལྔས་རྩེན་ཟླ་བ་དང་། །སེང་གེ་སྡེ་དང་རབ་གནོན་དང་། །མ་ལ་སྦྱིན་བརྩོན་དྲག་པོ་དང་། །གང་པོ་དང་ནི་ནོར་འཛིན་དང་། །ཙནྡན་དང་ནི་འདོད་མཆོག་དང་། །སྣ་ཚོགས་སྡེ་དང་གཏོར་མ་ལེན། །ཡིད་བཞིན་རབ་འབྱོར་བཞིན་བཟང་དང་། །འདམ་བུ་ཅན་དང་ཤིན་ཏུ་རྒྱལ། །འདོད་པ་ཐམས་ཅད་འབྱོར་པ་དང་། །དབང་པོ་རྒྱལ་བ་ཁྱུ་མཆོག་དང་། །ནོར་བུ་བཟང་སྲས་རྫོགས་བཟང་དང་། །འདམ་བུ་ས་མཆོག་འདམ་བུ་དགའ། །ཁབ་ཀྱི་སྤུ་ཅན་རིང་བཟང་དང་། །ལག་མཆོག་སྐྱེ་དགུའི་བདག་པོ་དང་། །འདམ་བུའི་རྒྱུད་དང་མདུང་རིང་དང་། །རིག་པ་སྟོབས་ཅན་ནོར་བུ་དང་། །འཇོག་པོ་ནག་པོ་ཆེན་པོ་དང་། །ལ་བ་ཅན་དང་དྲེའུ་དང་གཉིས། །ཀླུ་ཡི་རྒྱལ་པོ་ས་སྲུང་བུ། །ཡན་ལག་ཆེན་པོ་རྫུ་འཕྲུལ་ཆེ། །ནམ་མཁའ་ལྡིང་ནི་ཤུགས་ཆེན་ལྡན། །འདབ་ཆགས་ཀུན་གྱི་མཆོག་གྱུར་པ། །བྱ་ཡི་གཟུགས་ཀྱིས་མཆི་བགྱིད་ཅིང་། །རྒྱ་མཚོ་ཀུན་ཏུ་འཁྲུག་པར་བགྱིད། །ལྷ་དང་ལྷ་མིན་དྲི་ཟ་རྣམས། །འཇིགས་པར་བགྱིད་དང་དགེ་བའང་རུང་། །མ་དད་པ་དང་དད་པ་རྣམས། །དེ་དག་ཐམས་ཅད་འདིར་མཆིས་ཏེ། །གཽ་ཏམ་དཔའ་བོ་ཁྱོད་ལ་ནི། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །འཕྲོག་མ་ཞེས་བགྱི་གནོད་སྦྱིན་མོ། །འཇིགས་པར་བགྱིད་པ་སྟོབས་ཆེན་མོ། །དེ་ཡི་བུ་ནི་ལྔ་བརྒྱ་རྣམས། །འཇིགས་བགྱིད་གནོད་སྦྱིན་མཐུ་རྩལ་ཅན། །བདག་ཉིད་ཀྱི་ནི་གཡོག་ལགས་ཏེ། །དེ་དག་ཕྱག་འཚལ་སླད་དུ་མཆིས། །གཽ་ཏམ་དཔའ་བོ་ཁྱོད་ལ་ནི། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །མ་ག་དྷཱ་ན་ཐམས་ཅད་མཐོང་། །ཀ་ཤི་ཀ་ནི་པདྨ་འཆང་། །ཡངས་པ་ཅན་ན་རྒྱལ་མཆོག་གནས། །གྱད་ཡུལ་ན་ནི་གྲགས་འཛིན་གནས། །རྒྱལ་པོའི་ཁབ་ན་ཀུ་བེ་ར། །སེར་སྐྱའི་གནས་ན་དྲག་པོ་སྟེ། །ཏ་མ་ལ་ན་འབྱོར་པ་གནས། །མི་ཏི་ལ་ན་ཀུན་ཏུ་རྒྱལ། །མ་མོ་ཆེན་མོ་འདི་བརྒྱད་ནི། །གནོད་སྦྱིན་རྣམས་ཀྱི་སྡེ་དཔོན་མཆོག །རྫུ་འཕྲུལ་ཆེན་པོ་སྐལ་བ་ཆེ། །སྟོབས་ཆེན་དེ་དག་ཐམས་ཅད་ནི། །ཡིད་དགེ་དད་པའི་སེམས་དང་ལྡན། །རྫོགས་སངས་རྒྱས་ལ་སྐྱབས་མཆིས་པས། །དེ་དག་དང་ནི་འདུས་ནས་ཀྱང་། །རྒྱལ་པོ་ཆེན་པོ་སྟོབས་ཆེན་གྱིས། །ཡང་དག་རྫོགས་སངས་རྒྱས་བསྟན་ལ། །མི་རྣམས་ལ་ནི་སྙིང་བརྩེའི་སླད། །གནོད་སྦྱིན་འབྱུང་པོ་ཡི་དགས་དང་། །སྡང་བའི་གདོན་རྣམས་བཟློག་པ་དང་། །མི་མ་ཡིན་རྣམས་བཟློག་པ་ཡི། །བསྲུང་བ་འདི་ནི་ལེགས་པར་བགྱིས། །དེ་ནས་རྒྱལ་པོ་རྣམ་ཐོས་བུ། །ཐལ་སྦྱར་འདུག་སྟེ་སྨྲས་པ་ནི། །ཁོན་དང་འཇིགས་པ་ཀུན་འདས་པ། །རྣམ་གྲོལ་ཟག་པ་མི་མངའ་བ། །གཽ་ཏམ་དཔའ་བོ་ཁྱོད་ལ་ནི། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །ཁྱོད་ལ་ལྷ་དང་གྲུབ་པ་དང་། །བྲམ་ཟེ་ཐམས་ཅད་ཕྱག་འཚལ་ཏེ། །གནོད་སྦྱིན་བྱེ་བ་དྲུག་ཁྲི་དང་། །བཞི་སྟོང་དག་ཀྱང་ལྷགས་ནས་ནི། །ཁོ་ར་ཁོར་ཡུག་ཡོངས་བསྐོར་ནས། །ཐལ་མོ་སྦྱར་ཏེ་འཁོད་གྱུར་ལགས། །བདག་ནི་རྒྱལ་པོ་རྣམ་ཐོས་བུ། །བྱང་ཕྱོགས་སྐྱོང་བར་བགྱིད་པ་སྟེ། །འཇིག་རྟེན་མགོན་པོའི་ཉེ་འཁོར་དུ། །གནོད་སྦྱིན་རྒྱུ་སྐར་སྨྱོ་བྱེད་དང་། །བརྗེད་བྱེད་དང་ནི་མི་འམ་ཅི། །སྐེམ་བྱེད་དང་ནི་རིམས་དྲག་དང་། །དེ་བཞིན་རིམས་ནད་ཞག་བཞི་པ། །དེ་དག་བདག་གིས་བཟློག་པར་བགྱི། །སྱཱད་ཡ་ཐེ་དན། ཨུ་ཧ་ཧཾ། ཨུ་ཧ་ཧཾ། མ་ཧཱ་ཧཾ། ཨུ་ཧ་མ་ཧཱ་ཧཾ། ས་ལ་མུ་ཛི། ཨི་ལཱི་མི་ལི། ཨི་ལཱི་མི་ལི། ཀི་ལཱི་མི་ལི། ཙི་རི་ཊི། ཨུ་བྷི་ཨུ་ཏུ་བྷི། པི་ཏི་ལི་ཡེ། མ་ཊ་མ་ལི། མ་ནནྡ་ཡེ་སྭཱ་ཧཱ། རྒྱལ་པོ་ཆེན་པོ་རྣམ་ཐོས་ཀྱི་བུ་ལུས་ངན་ལྕང་ལོ་ཅན་གྱི་བདག་པོ་ཕྱོགས་སྐྱོང་བར་གྱུར་ཅིག །གནོད་སྦྱིན་དང་། རྒྱུ་སྐར་དང་། སྨྱོ་བྱེད་དང་། བརྗེད་བྱེད་དང་། མི་འམ་ཅི་དང་། གྱེན་དུ་རྒྱུ་དང་། རྒྱུ་བ་དང་། འདུན་པ་དང་། ངན་བྱེད་དང་། གཟི་བྱིན་འཕྲོག་པ་རྣམས་བཟློག་པར་གྱུར་ཅིག །ལྷའི་ནང་ན་ལྷའི་རྒྱལ་པོ་ནོར་ལྷའི་བུ་ཞེས་བགྱི་བ་མཆིས་ཏེ། དེ་ལ་བདག་ཕྱག་འཚལ་ལོ། །དེ་ཡང་བཅོམ་ལྡན་འདས་ཉིད་ལ་ཕྱག་འཚལ་ཏེ། རྫོགས་པའི་སངས་རྒྱས་དཔའ་བོ་སྟེ། །ཀུན་མཁྱེན་ཀུན་གཟིགས་ཁྱོད་ལ་འདུད། །ཡང་དག་དཔའ་བོ་ཁྱོད་ལ་འདུད། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །དེ་ནས་རྒྱལ་པོ་ཡུལ་འཁོར་སྲུང་། །ཐལ་སྦྱར་འདུག་སྟེ་སྨྲས་པ་ནི། །ཆོས་ཀྱི་རྒྱལ་པོ་འོད་མཛད་པ། །སངས་རྒྱས་དཔལ་ལ་བདག་ཕྱག་འཚལ། །གཽ་ཏམ་དཔའ་བོ་ཁྱོད་ལ་ནི། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །གཽ་ཏམ་ཁྱོད་ཀྱིས་གང་མཁྱེན་པ། །དེ་ནི་ལྷ་རྣམས་མ་འཚལ་ཏོ། །སྲིན་པོ་མཐོང་ན་རབ་འཇིགས་པ། །སྟོང་ཕྲག་དྲུག་ཅུ་རྩ་བཞི་རྣམས། །ཁོ་ར་ཁོར་ཡུག་ཡོངས་བསྐོར་ནས། །ཐལ་མོ་སྦྱར་ཏེ་འཁོད་གྱུར་ལགས། །བདག་ནི་རྒྱལ་པོ་ཡུལ་འཁོར་སྲུང་། །ཤར་ཕྱོགས་སྐྱོང་བར་བགྱིད་པ་སྟེ། །འཇིག་རྟེན་མགོན་པོའི་ཉེ་འཁོར་དུ། །སྲིན་པོ་རྣམས་དང་ཤ་ཟ་དང་། །དྲི་ཟ་དང་ནི་རླུང་ལྷ་རྣམས། །དེ་དག་བདག་གིས་བཟློག་པར་བགྱི། །སྱཱད་ཡ་ཐེ་དན། ཨི་ཏི་མི་ཊཱི། ཁུ་ར་ཊེ། ཕུ་ར་ཊེ། མ་ཏི་བི་མ་ཊི། ཨུ་མ་ཎི་ཨཀྐེ། མཀྐེ། ནཀྐེ། ཨ་ཊུ་མི། བ་ཊུ་མི། མ་བྷེ། ཨེ་རི་ཊི། པི་རི་ཊི་སྭཱ་ཧཱ། རྒྱལ་པོ་ཆེན་པོ་ཡུལ་འཁོར་སྲུང་ཕྱོགས་སྐྱོང་བར་གྱུར་ཅིག །སྲིན་པོ་དང་། ཡི་དགས་དང་། དྲི་ཟ་དང་། རླུང་ལྷ་དང་། མི་འམ་ཅི་དང་། གཟི་བྱིན་འཕྲོག་པ་རྣམས་བཟློག་པར་གྱུར་ཅིག །ལྷའི་ནང་ན་ལྷའི་རྒྱལ་པོ་ནོར་ལྷའི་བུ་ཞེས་བགྱི་བ་མཆིས་ཏེ། དེ་ལ་བདག་ཕྱག་འཚལ་ལོ། །དེ་ཡང་བཅོམ་ལྡན་འདས་ཉིད་ལ་ཕྱག་འཚལ་ཏེ། རྫོགས་པའི་སངས་རྒྱས་དཔའ་བོ་སྟེ། །ཀུན་མཁྱེན་ཀུན་གཟིགས་ཁྱོད་ལ་འདུད། །ཡང་དག་དཔའ་བོ་ཁྱོད་ལ་འདུད། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །དེ་ནས་རྒྱལ་པོ་འཕགས་སྐྱེས་པོ། །ཐལ་སྦྱར་འདུག་སྟེ་སྨྲས་པ་ནི། །འཇིག་རྟེན་མགོན་པོ་སྐྱབས་མཛད་པ། །འཇིག་རྟེན་ཀུན་ལ་ཕན་པ་མཛད། །གཽ་ཏམ་དཔའ་བོ་ཁྱོད་ལ་ནི། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །གྲུལ་བུམ་ཡི་དགས་སྲུལ་པོ་རྣམས། །སྟོང་ཕྲག་དྲུག་ཅུ་རྩ་བཞི་རྣམས། །ཁོ་ར་ཁོར་ཡུག་ཡོངས་བསྐོར་ནས། །ཐལ་མོ་སྦྱར་ཏེ་འཁོད་གྱུར་ལགས། །བདག་ནི་རྒྱལ་པོ་འཕགས་སྐྱེས་པོ། །ལྷོ་ཕྱོགས་སྐྱོང་བར་བགྱིད་པ་སྟེ། །འཇིག་རྟེན་མགོན་པོའི་ཉེ་འཁོར་དུ། །འབྱུང་པོ་རྒྱུ་བ་གཟི་བྱིན་འཕྲོག །གྲུལ་བུམ་ཡི་དགས་སྲུལ་པོ་རྣམས། །དེ་དག་བདག་གིས་བཟློག་པར་བགྱི། །སྱཱད་ཡ་ཐེ་དན། ཨ་ལེ། ཨི་ལེ་ལེ། ཀི་ལེ་ཀི་ལེ་ལེ། ཀུ་པ་ཀུ་པ་ས། སི་ལེ་སི་ལེ་ལེ། སི་ལི་ལི་ལི། ལི་ལི་ལི་ལི་ལི། ཧི་ཧི་སི་ལེ། མ་ཏི་སུ་མ་ཏི། སུ་སུ་མ་ཏི། སུ་སུ་སུ་སུ། སུ་སུ་མ་ཏི། ཧི་ལི་ཥ་ཧི་ལི་ཥ་སྭཱ་ཧཱ། རྒྱལ་པོ་ཆེན་པོ་འཕགས་སྐྱེས་པོ་ཕྱོགས་སྐྱོང་བར་གྱུར་ཅིག །གྲུལ་བུམ་དང་། ཡི་དགས་དང་། སྲུལ་པོ་དང་། འབྱུང་པོ་དང་། རྒྱུ་བ་དང་། གཟི་བྱིན་འཕྲོག་པ་རྣམས་བཟློག་པར་གྱུར་ཅིག །ལྷའི་ནང་ན་ལྷའི་རྒྱལ་པོ་ནོར་ལྷའི་བུ་ཞེས་བགྱི་བ་མཆིས་ཏེ། དེ་ལ་བདག་ཕྱག་འཚལ་ལོ། །དེ་ཡང་བཅོམ་ལྡན་འདས་ཉིད་ལ་ཕྱག་འཚལ་ཏེ། རྫོགས་པའི་སངས་རྒྱས་དཔའ་བོ་སྟེ། །ཀུན་མཁྱེན་ཀུན་གཟིགས་ཁྱོད་ལ་འདུད། །ཡང་དག་དཔའ་བོ་ཁྱོད་ལ་འདུད། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །དེ་ནས་རྒྱལ་པོ་མིག་མི་བཟང་། །ཐལ་སྦྱར་འདུག་སྟེ་སྨྲས་པ་ནི། །མུན་པར་གྱུར་པའི་འཇིག་རྟེན་ལ། །བཅོམ་ལྡན་ལམ་རྣམས་སྟོན་མཛད་པ། །གཽ་ཏམ་དཔའ་བོ་ཁྱོད་ལ་ནི། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །འཇིག་རྟེན་དག་ན་ཉི་ཟླ་འདྲ། །རིན་ཆེན་སྙིང་པོ་སྟོབས་པོ་ཆེ། །ཀླུ་དང་མཁའ་ལྡིང་གསང་བ་པ། །སྟོང་ཕྲག་དྲུག་ཅུ་རྩ་བཞི་རྣམས། །ཁོ་ར་ཁོར་ཡུག་ཡོངས་བསྐོར་ནས། །ཐལ་མོ་སྦྱར་ཏེ་འཁོད་གྱུར་ལགས། །བདག་ནི་རྒྱལ་པོ་མིག་མི་བཟང་། །ནུབ་ཕྱོགས་སྐྱོང་བར་བགྱིད་པ་སྟེ། །འཇིག་རྟེན་མགོན་པོའི་ཉེ་འཁོར་དུ། །ཀླུ་དང་མཁའ་ལྡིང་གསང་བ་པ། །ལྷ་མིན་དབང་པོ་སྟོབས་པོ་ཆེ། །དེ་དག་བདག་གིས་བཟློག་པར་བགྱི། །སྱཱད་ཡ་ཐེ་དན། དགེ་དགེ། སུ་བ་ཏི་ཡེ། པ་ཊེ། ཨཊ། ཀ་མཊྚེ། བི་མ་བི་ད་མ། བི་ད་ད་མ། བྷི་ད་ད་མ་ནི། ཨ་བྷི་ན་མེ། གཙྪ་བྷུ་ཏེ། ད་ལ་ཕི་ཡེ། བ་ར་སུརྟྲ་ཙི་ཙི་ལེ། ཙི་ལི་ལི་ལི་ལི་ལི་ལི་སྭཱ་ཧཱ། རྒྱལ་པོ་ཆེན་པོ་མིག་མི་བཟང་ཕྱོགས་སྐྱོང་བར་གྱུར་ཅིག །ཀླུ་དང་། ནམ་མཁའ་ལྡིང་དང་། གསང་བ་པ་དང་། ལྷ་མ་ཡིན་དང་། གཟི་བྱིན་འཕྲོག་པ་རྣམས་བཟློག་པར་གྱུར་ཅིག །ལྷའི་ནང་ན་ལྷའི་རྒྱལ་པོ་ནོར་ལྷའི་བུ་ཞེས་བགྱི་བ་མཆིས་ཏེ། དེ་ལ་བདག་ཕྱག་འཚལ་ལོ། །དེ་ཡང་བཅོམ་ལྡན་འདས་ཉིད་ལ་ཕྱག་འཚལ་ཏེ། རྫོགས་པའི་སངས་རྒྱས་དཔའ་བོ་སྟེ། །ཀུན་མཁྱེན་ཀུན་གཟིགས་ཁྱོད་ལ་འདུད། །ཡང་དག་དཔའ་བོ་ཁྱོད་ལ་འདུད། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །དེ་ནས་རྒྱལ་པོ་ཆེན་པོ་བཞི་བླ་གོས་ཕྲག་པ་གཅིག་ཏུ་གཟར་ནས་པུས་མོ་གཡས་པའི་ལྷ་ང་ས་ལ་བཙུགས་ཏེ་ཐལ་མོ་སྦྱར་ནས། བཅོམ་ལྡན་འདས་འབའ་ཞིག་ལ་ཕྱག་འཚལ་ཞིང་གྲོས་མཐུན་པ་དང་། ཚིག་མཐུན་པ་དང་། སྐད་མཐུན་པར་བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་ནི་རྒྱལ་པོ་ཆེན་པོ་བཞིའི་བསྲུང་བ་སྟེ་འཁོར་བཞི་པོ་རྣམས་ལ་ཡོངས་སུ་ཁྱབ་པ་ལགས་སོ། །གནོད་སྦྱིན་དང་། སྲིན་པོ་ནས་གནོད་པ་བགྱིད་པའི་བར་དུ་བཟློག་པར་བགྱི་བ་དང་། འཁོར་བཞི་པོ་རྣམས་ལ་སྲུང་བ་དང་། སྐྱོབ་པ་དང་། སྦེད་པ་དང་། བདེ་ལེགས་སུ་འགྱུར་བ་དང་། བདེ་བར་གནས་པར་བགྱིད་པ་ལགས་སོ། །དེ་སྐད་ཅེས་གསོལ་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་རྒྱལ་པོ་ཆེན་པོ་བཞི་ལ་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། །ཀྱེ་རྒྱལ་པོ་ཆེན་པོ་དག་རིག་སྔགས་ཀྱི་རྒྱལ་པོ་ཆེན་པོ་བཟུང་ཟིན་ནོ། །བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་ངས་ཡོངས་སུ་བཟུང་ཟིན་ནོ། །ཀྱེ་རྒྱལ་པོ་ཆེན་པོ་དག་གཞན་ཡང་ངས་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་ལ་རྗེས་སུ་ཡི་རང་བའི་རིག་སྔགས་ཀྱི་རྒྱལ་མོ་ཆེན་མོ་བརྗོད་ཀྱིས། དེ་ལེགས་པར་རབ་ཏུ་ཉོན་ལ་ཡིད་ལ་ཟུང་ཤིག་དང་ངས་བཤད་དོ། །བཅོམ་ལྡན་འདས་དེ་ལྟར་འཚལ་ལོ། །ཞེས་རྒྱལ་པོ་ཆེན་པོ་བཞིས་བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ་ཉན་ཏོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ཕྱག་རིན་པོ་ཆེ་གཡས་པས་ན་བཟའ་ཆོས་གོས་ཕྱེ་ནས་འབྱུང་པོའི་ཚོགས་དེ་དག་ལ་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ། །ངས་འདི་བསྟན་པར་བྱའོ། །རྒྱལ་པོ་ཆེན་པོ་རྣམ་ཐོས་ཀྱི་བུ་ལ་དབྱེ་བར་བྱའོ། །ཐམས་ཅད་ལ་གོ་བར་བྱའོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ཀྱིས་རིག་སྔགས་འདི་བཀའ་སྩལ་ཏོ། །སྱཱད་ཡ་ཐེ་དན། ཧི་ལཱི་ཧི་ལཱི། བྷི་སི་ནི། བཀྐེ །ཨ་ཧ་ར་ཡེ། ཏ་མ་ཏི། ཏ་མ་ཏི། ཏི་ལ་བཀྐེ། བ་ལཱི་ཏ་ཀེ། ད་རི་ཏི་ན། ད་དྷི་ན་ད། ཕུ་ཊྱི་ཕུ་ཕུ་ཊྱི། གུ་ཊྱི་གུ་གུ་ཊྱི། ཧི་ཙ་ཀཱནྟི། ནཱ་མཱ། ཀཱན་ཏཱི་སྭཱ་ཧཱ། བཅོམ་ལྡན་འདས་ཀྱིས་རིག་སྔགས་ཀྱི་རྒྱལ་མོ་ཆེན་མོ་འདི་བཀའ་སྩལ་པ་ན་ས་འདི་ཀུན་ཏུ་གཡོས། འབྱུང་པོ་ཐམས་ཅད་རབ་ཏུ་འདར་ནས་ཀྱེ་མ་ཀྱེ་ཧུད་ཀྱེ་མ་སྡུག་བསྔལ་ལོ་ཞེས་སྒྲ་ཆེན་པོ་བྱུང་ངོ་། །སྲིན་པོ་རྣམས་ཀྱིས་ཀྱང་འདི་སྐད་ཅེས་སྨྲས་ཏེ། འདི་ལྟར་ཡང་དག་རྫོགས་སངས་རྒྱས། །སེམས་ཅན་ཀུན་ལ་ཐུགས་བརྩེ་ནས། །འབྱུང་པོ་ཀུན་འཇིགས་ཉེན་པ་ཡི། །རིག་སྔགས་འདི་འདྲ་སྲུང་བར་མཛད། །དེ་ནས་ཀུན་མཁྱེན་སྟོན་པ་པོ། །སེམས་ཅན་ཀུན་ལ་ཐུགས་བརྩེ་བས། །གནོད་སྦྱིན་སྲིན་པོ་དྲི་ཟ་དང་། །ཀླུ་དང་མཁའ་ལྡིང་གསང་བ་པ། །འབྱུང་པོ་རྒྱུ་བ་གཟི་བྱིན་འཕྲོག །གྲུལ་བུམ་ཡི་དགས་སྲུལ་པོ་དང་། །ལྷ་མིན་གཤེད་བྱེད་རོ་ལངས་དང་། །བརྗེད་བྱེད་དང་ནི་མི་འམ་ཅི། །སྐེམ་བྱེད་དང་ནི་རིམས་དྲག་དང་། །དེ་བཞིན་བྲམ་ཟེ་སྲིན་པོ་དང་། །སྨྱོ་བྱེད་རིམས་ནད་ཞག་གསུམ་པ། །དེ་བཞིན་ཞག་ནི་བཞི་པ་རྣམས། །སྐྲག་ཅིང་རབ་ཏུ་འདར་འགྱུར་ཏེ། །ཕྱོགས་བཅུར་རྒྱུག་པ་གཟིགས་ནས་ཀྱང་། །དེ་དག་ལ་ནི་ཐུགས་བརྩེའི་ཕྱིར། །རིག་སྔགས་འདི་ནི་བརྗོད་པར་མཛད། །སྱཱད་ཡ་ཐེ་དན། ཧི་རི་མ་ཁ། ཏི་རི་ཁ་ཁ། ཀ་ཊ་མ་ཊཱ། སིད་དྷ་མ་ཧཱ། ཧ་ཏ་རི། ཨུ་ཊ་ཏ་ནི། ད་ཊ་ད་ཊ། ད་ཏི་ལི། ད་མཏྟེ། ད་མཏྟེ། དྷུ་དྷུ་རི། ད་ད་རི། ད་ར་མི་ར། ཀི་རི་ཀཱི་ཡེ། ཀ་ཏ་བ་རེ་ཎི། སུ་སུ་མཱ་ལེ་སྭཱ་ཧཱ། གསང་སྔགས་ཆེན་པོ་རིག་སྔགས་ཆེན་པོ་འདི་ལས་འགལ་བར་མ་བྱེད་ཅིག །བཅོམ་ལྡན་འདས་ཀྱི་གསུང་གི་ཆོས་ལ་ཡང་དག་པར་སོ་སོར་རྟོགས་ཤིག །ཁྱེད་ཀུན་ལུས་ཞིག་སྟེ་ཤི་བའི་འོག་ཏུ་ངན་སོང་ངན་འགྲོ་ལོག་པར་ལྟུང་བ་སེམས་ཅན་དམྱལ་བ་རྣམས་སུ་སྐྱེ་ཞིང་འཁོར་བར་འགྲོ་བ་རེ། དེ་ནས་རྒྱལ་པོ་ཆེན་པོ་བཞིས་བཅོམ་ལྡན་འདས་ཀྱི་རིག་སྔགས་ཀྱི་རྒྱལ་མོ་ཆེན་མོ་སྔོན་མ་ཐོས་པ་འདི་ཐོས་ནས་སྐྲག་ཅིང་དངངས་ཏེ་མི་དགའ་ལ་སྐྱི་བུང་ཞེས་བྱེད་ནས་ཐལ་མོ་སྦྱར་ཏེ། བཅོམ་ལྡན་འདས་ལ་ཕྱག་འཚལ་ཞིང་གྲོས་མཐུན་པ་དང་། ཚིག་མཐུན་པ་དང་། སྐད་མཐུན་པར་བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །སྡང་བའི་འབྱུང་པོ་ཟད་བགྱིད་པ། །རིག་སྔགས་འདི་ནི་ལེགས་པར་གསུངས། །མི་ཡི་འཇིག་རྟེན་བདེ་བགྱིད་པའི། །རིག་སྔགས་བདག་ཅག་བརྗོད་པར་འཚལ། །དེ་ལ་རིག་སྔགས་འདི་ལགས་ཏེ། །འཇིག་རྟེན་མགོན་པོ་བདག་ལ་གསོན། །སྱཱད་ཡ་ཐེ་དན། ཨུ་ཧུ་ཧ་ལ། ཀུ་ལི་ཛི་བ་ཨ་བྷུ་ཡ། མ་སུ་ར་བེ་ལ། ཨད་དྱཱ་བད་དྱཱ། ཨད་དྱཱ་བ་ཏེ། ནད་ད་བད་དྱེ། མ་རུ་དྷྱ། མ་རུ་དྷྱ། བ་དེ་མ་རུ་ད་དྱ་ཕ་ལ་སྭཱ་ཧཱ། བཅོམ་ལྡན་འདས་འབྱུང་པོ་ཐམས་ཅད་བཟློག་པ་འདི་གཟུང་བར་གསོལ། སྱཱད་ཡ་ཐེ་དན། ཨི་ལི་ཨི་ལི། མི་ལི་མི་ལི། ཀི་ལི་ཀི་ལི། བ་ན་ར། ནི་ལ་པ་ཡ། བྷུ་བྷུ་བྷུ་ར། ཐི་རཱུ་ཕུ་ཕུ་ཕུ་ར། ག་ཤ་ལཱི། རཽ་ཁ་ད་ཡ་མ་ལ། ཁུ་ལི་ཁུ་ལི་བ་ཡ་སྭཱ་ཧཱ། རིག་སྔགས་ཀྱི་རྒྱལ་མོ་འདི་ནི་ལོ་གཅིག་གི་བར་དུ་མཚམས་གཅོད་པ་དང་། འབྱུང་པོ་ཐམས་ཅད་བཟློག་པ་དང་། དོན་ཐམས་ཅད་གྲུབ་པ་ལགས་སོ། །དེ་ནས་གནོད་སྦྱིན་དང་། སྲིན་པོ་དང་། དྲི་ཟ་ནས་གནོད་པར་བགྱིད་པའི་བར་རྣམས་ཀྱིས། བཅོམ་ལྡན་འདས་ཀྱི་རིག་སྔགས་ཀྱི་རྒྱལ་མོ་ཆེན་མོ་སྔོན་མ་ཐོས་པ་འདི་ཡང་ཐོས། རྒྱལ་པོ་ཆེན་པོ་བཞིའི་བསྲུང་བ་སྔོན་མ་ཐོས་པ་འདི་ཡང་ཐོས་ནས་སྐྲག་ཅིང་དངངས་ལ་མི་དགའ་སྟེ། སྐྱི་བུང་ཞེས་བྱེད་ནས་ས་ལ་འཛུལ་ཀུན་ཏུ་འཛུལ། ཀུན་ཏུ་རབ་ཏུ་འཛུལ་བར་བྱེད་དོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ས་གཞི་རྡོ་རྗེར་སྤྲུལ་ནས་དེ་དག་ཕྱོགས་བཞིར་རྒྱུག་གོ། །དེ་ནས་རྒྱལ་པོ་ཆེན་པོ་བཞིས་ཕྱོགས་བཞིར་མེའི་ཕུང་པོ་ཆེན་པོ་སྤྲུལ་ཏོ། །དེ་ནས་ནམ་མཁའ་ལ་རྒྱུག་གོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ནམ་མཁའ་གསེར་དུ་སྤྲུལ་ཏོ། །དེ་ནས་འཇིག་རྟེན་སྐྱོང་བ་རྒྱལ་པོ་ཆེན་པོ་རྣམ་ཐོས་ཀྱི་བུ་ལྷ་དང་དྲི་ཟས་བསྐོར་ཅིང་མདུན་དུ་བདར་ཏེ། སྟེང་གི་བར་སྣང་ལ་འཕགས་ནས་གནོད་སྦྱིན་དང་སྲིན་པོ་ནས་གནོད་པར་བྱེད་པའི་བར་རྣམས་གཞོམ་པ་དང་། སེམས་ཅན་ཐམས་ཅད་ལ་སྙིང་བརྩེ་བ་དང་། འབྱུང་པོ་ཐམས་ཅད་བཟློག་པ་དང་། འཁོར་བཞི་པོ་རྣམས་བསྲུང་བ་དང་། བསྐྱབ་པ་དང་། སྦ་བ་དང་། བདེ་ལེགས་སུ་འགྱུར་བ་དང་། བདེ་བར་གནས་པར་བྱ་བའི་ཕྱིར་རིག་སྔགས་འདི་བརྗོད་དོ། །སྱཱད་ཡ་ཐེ་དན། གི་མི་གི་མི་གི་མི་གི་མི། གི་ནི་གི་ནི་གི་ནི་གི་ནི། ན་ཁི་མི་ན་ཁི། ནི་ཥ་ནི་ཥ་ནི་ཥ། ཨུ་བ་ཨ། དི་ཤྭ་དི་ཤྭ། བི་དི་ཤ། སརྦ་དི་ཤྭ། ཨ་དྷི་བ་ཏི། གི་ར། བི་ས། ཛ་ནི་བྷུ་མི་པ་ཏི། ཨི་ལི་ལཱི། ཨི་ལི་ལཱི། ཀི་ལི་ལཱི། ཀི་ལི་ཀི་ལི་ལཱི། ཀི་ལི་ཀི་ལི་ལཱི། ཀི་ལི་ཀི་ལི་ལཱི། ཀི་ལི་ཀི་ལི་ལཱི། ཀི་ལི་ཀི་ལི་ལི། ཀི་ལི་ལི་ལི་ལི་ལི་ལི། ཀི་ལི་ལི་ལི་ལི་ལི་ལི་ལི་ལི། ཨད་དྱི། ནད་དྱི། ཀུ་ནད་དྱི། མ་ཧཱ་ཀུ་ནད་དྱི། ཀུ་ལད་དྱི་ཁུ་ལད་དྱི། ཁུ་ཁུ་ལད་དྱི་མ་ཧཱ་ཁུ་ལད་དྱི། ཧུ་ལུ་ཧུ་ལུ། ལུ། ཧུ་ལུ་ལུ་ལུ། ཧུ་ལུ་ལུ་ལུ་ལུ། ཧུ་ལུ་ལུ་ལུ་ལུ། ལུ། ཧུ་ལུ་ལུ་ལུ་ལུ་ལུ་ལུ་ལུ་ལུ་ལུ། ལུ། ཧ་ལ་ཧ་ལ། ཧི་ལི་ཧི་ལི། ལ་ཧ་ཧ་ལུ། ཧུ་ལེ། ཨི་ཧ། ཧུ་ལེ་ཧུ་ལེ། ཨུ་ལད་ཏྱི། ཧི་ལེ། ཨུ་ཧ་ཧི་ལེ། ཧི་ལ་ཧི་ལེ། ཧུ་ལུ། ཧི། ཧུ་ལུ་ཧུ་ལུ་ཧུ་ལུ། ཧུ་ལུ་ཧུ་ལེ། ཧ་ཧ་ཧ་ཧ་ཧ་ཧ། སྭཱ་ཧཱ། རྒྱལ་པོ་ཆེན་པོ་རྣམ་ཐོས་ཀྱི་བུ་ཕྱོགས་སྐྱོང་བར་གྱུར་ཅིག །ལྷ་རྣམས་ཀྱི་ནང་ན་ལྷའི་རྒྱལ་པོ་ནོར་ལྷའི་བུ་ཞེས་བགྱི་བ་མཆིས་ཏེ། དེ་ལ་བདག་ཕྱག་འཚལ་ལོ། །དེ་ཡང་བཅོམ་ལྡན་འདས་ཉིད་ལ་ཕྱག་འཚལ་ཏེ། རྫོགས་པའི་སངས་རྒྱས་དཔའ་བོ་སྟེ། །ཀུན་མཁྱེན་ཀུན་གཟིགས་ཁྱོད་ལ་འདུད། །རྫོགས་པའི་སངས་རྒྱས་རྐང་གཉིས་མཆོག །ཤེས་རབ་ཆེན་པོ་ཁྱོད་ལ་འདུད། །གཽ་ཏམ་ཁྱོད་ཀྱིས་གང་མཁྱེན་པ། །དེ་ནི་ལྷ་རྣམས་མ་འཚལ་ཏོ། །ཡང་དག་དཔའ་བོ་ཁྱོད་ལ་འདུད། །ཆོས་བཞིན་མཐུན་པར་བལྟ་ཞིང་མཆིས། །གནོད་སྦྱིན་དང་། སྲིན་པོ་དང་། དྲི་ཟ་ནས་གནོད་པར་བགྱིད་པའི་བར་རྣམས་མིང་འདི་ཞེས་བྱ་བའི་ལུས་ལ་བབས་པ་ལས་གུད་དུ་དེང་ཤིག །གལ་ཏེ་མིང་འདི་ཞེས་བྱ་བའི་ལུས་ལས་མྱུར་དུ་གུད་དུ་མ་དོང་ན། དེ་དག་ལ་མྱུར་དུ་འཇིགས་པ་ཆེན་པོ་དང་། ནད་དྲག་པོས་འདེབས་པར་འགྱུར་ལ། དེའི་རྐྱེན་གྱིས་མགོ་བོ་དང་། སྙིང་ཚལ་པ་བརྒྱར་འགས་པར་འགྱུར་ཏེ། དེའི་མོད་ལ་སྡུག་བསྔལ་དུ་མ་འབྱུང་བ་དང་བརླག་པ་དང་མཐའ་མེད་པར་འགྱུར་རོ། །འབྲུམ་བུའི་གདོན་གང་ཡིན་པ་དང་། སྐྱ་རེངས་འཆར་ཀའི་གདོན་གང་ཡིན་པ་དང་། གྲིབ་གནོན་གྱི་གདོན་གང་ཡིན་པ་དེ་དག་ཀྱང་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་ཐོས་ནས་མིང་འདི་ཞེས་བྱ་བའི་ལུས་ལས་མྱུར་དུ་གུད་དུ་དེང་ཤིག །གལ་ཏེ་མྱུར་དུ་གུད་དུ་མ་དོང་ན་དེ་དག་ལ་མྱུར་དུ་འཇིགས་པ་ཆེན་པོ་དང་། ནད་དྲག་པོས་འདེབས་པར་འགྱུར་ལ། དེའི་རྐྱེན་གྱིས་མགོ་བོ་དང་། སྙིང་ཚལ་པ་བརྒྱར་འགས་པར་འགྱུར་ཏེ། དེའི་མོད་ལ་སྡུག་བསྔལ་དུ་མ་འབྱུང་བ་དང་། བརླག་པ་དང་མཐའ་མེད་པར་འགྱུར་རོ། །གཟི་བྱིན་དང་ནི་ཁྲག་ཟ་བ། །ཉིན་མཚན་རྒྱུ་བ་གང་ཡིན་པ། །དེ་དག་གིས་ཀྱང་ལེགས་གསུངས་པའི། །མདོ་བཤད་ཐོས་ནས་གཏང་བར་གྱིས། །ཤ་ཟ་ལུས་འཕགས་སྐེམ་བྱེད་དང་། །འབྱུང་པོ་འཇིགས་བྱེད་སྲུལ་པོ་རྣམས། །དེ་དག་གིས་ཀྱང་ལེགས་གསུངས་པའི། །མདོ་འདི་བཤད་པ་ཐོས་ནས་ཐོང་། །ལྷ་དང་ལྷ་མིན་རླུང་ལྷ་དང་། །དྲི་ཟ་དང་ནི་གདོན་གྱི་བླ། །འབྱུང་པོ་དང་ནི་ཤ་ཟ་དང་། །དེ་བཞིན་གནོད་སྦྱིན་སྲིན་པོ་རྣམས། །ཀུན་ཏུ་འཁྲུགས་ཤིང་ཀུན་ཏུ་གཡོས། །སྐྲག་ཅིང་ཀུན་ཏུ་རབ་ཏུ་འགུལ། །གདོན་སྡང་གནོད་པ་བྱེད་པ་རྣམས། །ལྟུང་ཞིང་རབ་ཏུ་ཕམ་པར་འགྱུར། །ས་གཞི་དག་ཀྱང་རབ་ཏུ་གཡོས། །སྒྲ་བོ་ཆེ་དག་སྒྲོགས་པར་བྱེད། །ཡི་དགས་དང་ནི་གྲུལ་བུམ་རྣམས། །སྡང་བའི་གདོན་ནི་ཟློག་བྱེད་པའི། །མདོ་འདི་དག་ནི་ཐོས་གྱུར་ནས། །འཇིགས་པ་དྲག་པོས་འདེབས་པར་འགྱུར། །སངས་རྒྱས་ཆོས་དང་དགེ་འདུན་ལ། །དད་པའི་གནོད་སྦྱིན་གང་ཡིན་པ། །དེ་དག་མགུ་ཞིང་ཡིད་དགའ་ནས། །ལེགས་པར་རྗེས་སུ་ཡི་རང་ངོ་། །དེ་ནས་རྒྱལ་པོ་ཆེན་པོ་བཞི་བར་སྣང་ལས་བབས་ནས་བཅོམ་ལྡན་འདས་ག་ལ་བ་དེར་སོང་སྟེ་ལྷགས་ནས། བཅོམ་ལྡན་འདས་ཀྱི་ཞབས་ལ་མགོ་བོས་ཕྱག་འཚལ་ཏེ་ཐལ་མོ་སྦྱར་ནས་བཅོམ་ལྡན་འདས་ལ་ཕྱག་འཚལ་ཞིང་། གྲོས་མཐུན་པ་དང་། ཚིག་མཐུན་པར་བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་རྒྱལ་པོ་ཆེན་པོ་བཞིས་ཞབས་ལ་ཕྱག་བགྱིས་པ་དང་། འཁོར་བཞི་པོ་རྣམས་སྲུང་བ་དང་། ཡོངས་སུ་སྐྱོབ་པ་དང་། གནོད་སྦྱིན་དང་། སྲིན་པོ་ནས་གནོད་པར་བགྱིད་པའི་བར་རྣམས་འཇོམས་པ་ལགས་སོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་དགེ་སློང་དང་། དགེ་སློང་མ་དང་། དགེ་བསྙེན་དང་། དགེ་བསྙེན་མ་གང་ལ་བསྲུང་བ་དང་། དེ་དག་གིས་ཀྱང་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་གཟུང་བར་བགྱི། བཅང་བར་བགྱི། བརྗོད་པར་བགྱི། ཀུན་ཆུབ་པར་བགྱིའོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་དགེ་སློང་ངམ། དགེ་སློང་མའམ། དགེ་བསྙེན་ནམ། དགེ་བསྙེན་མ་གང་ལ་ལ་བདག་བསྲུང་བ་དང་། བདག་སྦ་བ་དང་། གཞན་བསྲུང་བ་དང་། གཞན་སྦ་བའི་སླད་དུ་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་འཛིན་པ་དང་། འཆང་བ་དང་། བརྗོད་པ་དང་། ཀུན་ཆུབ་པར་བགྱིད་པ་དེ་ལ་ནི་གནོད་སྦྱིན་དང་། སྲིན་པོ་ནས་གནོད་པར་བགྱིད་པའི་བར་རྣམས་ཉེ་འཁོར་དུ་ཡང་གནས་པར་མི་འགྱུར་ན། གནོད་པར་བགྱིད་པ་ལྟ་ག་ལ་སེམས་ལགས། དེ་གང་དུ་མཚན་གཅིག་ཙམ་གནས་པ་དེར་ཡང་གནོད་སྦྱིན་དང་། སྲིན་པོ་ནས་གནོད་པ་བགྱིད་པའི་བར་རྣམས་ཉེ་འཁོར་དུ་ཡང་གནས་པར་མི་འགྱུར་ན། དེ་གང་དུ་རྟག་ཏུ་གནས་པ་དེར་ལྟ་སྨོས་ཀྱང་ཅི་འཚལ་ལགས། བཙུན་པ་བཅོམ་ལྡན་འདས་གང་ལ་མི་མ་ལགས་པའི་གནོད་པས་ཐེབས་པ་དེའི་མདུན་དུ་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་བརྗོད་ན། དེ་མྱུར་བ་ཁོ་ནར་བདེ་ལེགས་དང་། ནད་མ་མཆིས་པ་དང་། བདེ་བར་འགྱུར་ཞིང་ཐར་བར་འགྱུར་ལ། མི་མ་ལགས་པ་རྣམས་དེ་ལ་ལོག་པ་མི་བགྱིད་ཅིང་མྱུར་དུ་གུད་དུ་མཆི་བར་འགྱུར་ལགས་ཏེ། བཙུན་པ་བཅོམ་ལྡན་འདས་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་དེ་ལྟར་དོན་ཆེ་བ་ལགས་སོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་དགེ་སློང་ངམ། དགེ་སློང་མའམ། དགེ་བསྙེན་ནམ། དགེ་བསྙེན་མ་གང་ལ་ལ་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་འཛིན་པ་དང་། འཆང་བ་དང་། བརྗོད་པ་དང་། ཀུན་ཆུབ་པར་བགྱིད་པ་དང་། སྦྱོར་བར་བགྱིད་པ་དེ་ལ་ནི་དད་པ་ཅན་གྱི་གནོད་སྦྱིན་དགེ་བསྙེན་རྣམས་རྒྱལ་པོ་ཆེན་པོ་བཞིའི་གནས་སུ་མཆིས་ནས་མིང་སྨོས་པར་བགྱིད་དེ། ཀྱེ་རྒྱལ་བོ་ཆེན་པོ་དག་དགེ་སློང་ངམ། དགེ་སློང་མའམ། དགེ་བསྙེན་ནམ། དགེ་བསྙེན་མ་ཆེ་གེ་མོ་ཞིག་བདག་བསྲུང་བ་དང་། བདག་སྦ་བ་དང་། གཞན་བསྲུང་བ་དང་། གཞན་སྦ་བའི་ཕྱིར་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་འཛིན་པར་བྱེད། འཆང་བར་བྱེད། རྗོད་པར་བྱེད། ཀུན་ཆུབ་པར་བྱེད། སྦྱོར་བར་བྱེད་དོ་ཞེས་སྒྲོགས་པར་བགྱིད་ལགས་སོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་དད་པ་ཅན་གྱི་གནོད་སྦྱིན་དགེ་བསྙེན་རྣམས་རྒྱལ་པོའི་ཕོ་བྲང་ལྕང་ལོ་ཅན་དུ་མཆིས་ནས་རྒྱལ་པོ་ཆེན་པོ་རྣམ་ཐོས་ཀྱི་བུའི་མདུན་དུ་གནོད་སྦྱིན་རྣམས་འདུས་ཤིང་འབྱུང་པོ་ཐམས་ཅད་ཚོགས་པའི་ནང་དུ་ཡང་བསྒྲགས་ནས། དེར་དེ་དག་མངོན་པར་དགའ་བར་འགྱུར་ཏེ། བཙུན་པ་བཅོམ་ལྡན་འདས་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་ནི་དེ་ལྟར་དོན་ཆེ་བ་ལགས་སོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་དགེ་སློང་ངམ། དགེ་སློང་མའམ། དགེ་བསྙེན་ནམ། དགེ་བསྙེན་མ་གང་ལ་ལ་བདག་བསྲུང་བ་དང་། བདག་སྦ་བ་དང་། གཞན་བསྲུང་བ་དང་། གཞན་སྦ་བའི་སླད་དུ་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་འཛིན་པ་དང་། འཆང་བ་དང་། བརྗོད་པ་དང་ཀུན་ཆུབ་པར་བགྱིད་པ་དང་། སྦྱོར་བར་བགྱིད་པ་དེ་ནི་ཚངས་པར་སྤྱད་པར་བགྱིའོ། །ཆང་མི་བཏུང་བ་དང་། དྲན་པ་དང་། ཤེས་བཞིན་ཅན་དུ་མཆིས་པ་དང་། ཟས་རྣམ་པ་ལྔ་ལ་འདི་ལྟ་སྟེ། བུ་རམ་དང་། སྦྲང་རྩི་དང་། ཏིལ་དང་། ཤ་དང་། ཉ་ཤ་རྣམས་ཡོངས་སུ་སྤང་བར་བགྱིའོ། །དེ་ཅིའི་སླད་དུ་ཞེ་ན། བཙུན་པ་བཅོམ་ལྡན་འདས་གནོད་སྦྱིན་དང་། སྲིན་པོ་ནས་གནོད་པར་བགྱིད་པའི་བར་རྣམས་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་འཛིན་པ་དང་། འཆང་བ་དང་། བརྗོད་པ་དང་ཀུན་ཆུབ་པར་བགྱིད་པ་དང་། སྦྱོར་བ་དེ་ལ་དེའི་སྐབས་ཁོ་ནར་བརླག་པ་དང་། མཐའ་མ་མཆིས་པར་བགྱིད་པ་དག་མཆིས་ཏེ། དེ་དག་གཞོམ་པའི་སླད་དུ་ཡི་དམ་བགྱི་འཚལ་ལགས་སོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་དགེ་སློང་ངམ། དགེ་སློང་མའམ། དགེ་བསྙེན་ནམ། དགེ་བསྙེན་མ་གང་ལ་ལ་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་འཛིན་པ་དང་། འཆང་བ་དང་། བརྗོད་པ་དང་། ཀུན་ཆུབ་པར་བགྱིད་པ་དང་། སྦྱོར་བར་བགྱིད་པ་དེ་ལ་ནི་དད་པ་ཅན་གྱི་གནོད་སྦྱིན་དགེ་བསྙེན་རྣམས་གོས་དང་། ཟས་དང་། མལ་ཆ་དང་། སྟན་དང་། བྲོ་གསོས་སྨན་དང་། ཡོ་བྱད་རྣམས་ཀྱིས་སྤྲོ་བ་བསྐྱེད་པར་བགྱིད་དེ། བཙུན་པ་བཅོམ་ལྡན་འདས་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་ནི་དེ་ལྟར་དོན་ཆེ་བ་ལགས་སོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་གཞན་མུ་སྟེགས་ཅན་བཅོམ་ལྡན་འདས་ཀྱི་གསུང་རབ་ལ་མ་དད་པ་དང་། མངོན་པར་མི་དགའ་བ་དང་། གནོད་པར་འཚལ་བ་དང་། སྨན་པར་མི་འཚལ་བ་དང་། གྲུབ་པ་དང་། བདེ་བར་མི་འཚལ་བ་གང་ལ་ལ་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་འཛིན་པ་དང་། འཆང་བ་དང་། བརྗོད་པ་དང་། ཀུན་ཆུབ་པར་བགྱིད་པ་དང་། སྦྱོར་བར་བགྱིད་པ་དེ་ལ་ནི་དོན་དེ་འགྲུབ་པར་མི་བགྱིད་དེ། མི་མ་ལགས་པ་རྣམས་ཀྱིས་བར་མ་དོར་འགུམས་པ་དང་། སྡུག་བསྔལ་དུ་མ་འབྱུང་བ་དང་། བརླག་པ་དང་། མཐའ་མ་མཆིས་པར་བགྱིད་ལགས་སོ། །དེ་ཅིའི་སླད་དུ་ཞེ་ན། དེ་ནི་དེ་དག་ལ་གནོད་པར་འཚལ་བ་དང་། སྨན་པར་མི་འཚལ་བ་དང་། གྲུབ་པ་དང་། བདེ་བར་མི་འཚལ་བ་ལགས་པའི་སླད་དུ་སྟེ། དེ་དག་ལ་ནི་ལོག་པར་ལྟ་བ་གང་ལགས་པ་དེ་བརླག་པར་བགྱིད་ལགས་སོ། །བཙུན་པ་བཅོམ་ལྡན་འདས་འཇིག་རྟེན་ན་སེམས་ཅན་རྣམས་བསྲུང་བ་དང་། བསྐྱབ་པ་དང་། སྦ་བའི་སླད་དུ་རྒྱལ་པོ་ཆེན་པོ་བཞིས་བསྲུང་བ་བགྱིས་པ་གང་ཅི་ཡང་རུང་བའི་བསྲུང་བ་དེ་དག་གི་ནང་ན་ནི་འདི་མཆོག་དང་། གཙོ་བོ་དང་། རབ་མཆོག་ལགས་སོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ལ་རྒྱལ་པོ་ཆེན་པོ་བཞིས་ཚིགས་སུ་བཅད་པ་འདི་དག་གསོལ་ཏོ། །མི་མཆོག་གང་ཞིག་མདོ་འདི་དག །ཡོངས་སུ་རྫོགས་པར་བསྲུང་བ་དང་། །བདེ་ལེགས་བགྱིད་པ་དེ་ལ་ནི། །སྲིན་པོ་རྣམས་ཀྱང་འཚེ་མི་བགྱིད། །སུ་ཡང་རུང་སྟེ་མདོ་འདི་དག །མི་མཆོག་གིས་ནི་གསུངས་པ་བཞིན། །ཡོངས་སུ་རྫོགས་ཤིང་མ་ཆད་པར། །ཤིན་ཏུ་ཚང་བར་འཆང་བགྱིད་ན། །དེ་ལ་གནོད་སྦྱིན་ཡི་དགས་དང་། །འབྱུང་པོ་གྲུལ་བུམ་སྲིན་པོ་དང་། །སྨྱོ་བྱེད་རྒྱུ་བ་ཀླུ་རྣམས་དང་། །དེ་ལྟ་བུ་དག་འཚེ་མི་བགྱིད། །དེ་ལ་མི་མ་ལགས་དང་གདོན། །མྱུར་དུ་གུད་དུ་མཆི་བར་འགྱུར། །དེ་ལ་ཅིས་ཀྱང་མི་ཚུགས་ཏེ། །དེ་ལ་འཚེ་བར་ཡོང་མི་ནུས། །རྒྱལ་པོ་ཆེན་པོ་བཞི་པོ་ཡིས། །བཅོམ་ལྡན་ཕྱག་འཚལ་ཞེས་བརྗོད་དེ། །སངས་རྒྱས་ཞབས་ལ་ཕྱག་འཚལ་ནས། །དེ་ཉིད་དུ་ནི་མི་སྣང་གྱུར། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་དེའི་མཚན་མོ་འདས་ཏེ་ནམ་ལངས་ནས་དགེ་སློང་གི་དགེ་འདུན་གྱི་གུང་ལ་བཞུགས་ཏེ་དགེ་སློང་རྣམས་ལ་བཀའ་སྩལ་པ། དགེ་སློང་དག་མདང་སུམ་རྒྱལ་པོ་ཆེན་པོ་བཞི་བུ་དང་བཅས། བློན་པོ་དང་བཅས། འཁོར་དང་བཅས། མངག་གཞུག་པ་དང་བཅས། ཕོ་ཉ་དང་བཅས། གཡོག་འཁོར་དང་བཅས་པ་ཁ་དོག་འཕགས་པ་དག་ནམ་སྲོས་ནས་ལྷགས་ནས་གྲོས་མཐུན་པར་དུར་ཁྲོད་ཆེན་པོ་བསིལ་བའི་ཚལ་དུ་འོངས་ཏེ། རང་གི་འོད་ཀྱིས་སྣང་བར་བྱས་ནས་ལུས་ངན་དང་། ཡུལ་འཁོར་སྲུང་དང་། མིག་མི་བཟང་དང་། འཕགས་སྐྱེས་པོ་རྣམས་ང་ག་ལ་བ་དེར་འོངས་ཏེ་ལྷགས་ནས། ངའི་རྐང་པ་གཉིས་ལ་མགོ་བོས་ཕྱག་འཚལ་ཏེ་ཕྱོགས་གཅིག་ཏུ་འཁོད་དོ། །ཕྱོགས་གཅིག་ཏུ་འཁོད་ནས། རྒྱལ་པོ་ཆེན་པོ་བཞིས་ངའི་ཐད་དུ་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་བརྗོད་ནས་དོང་ངོ་། །ཁྱོད་ཀྱིས་དེ་ལེགས་པར་རབ་ཏུ་ཉོན་ལ་ཡིད་ལ་ཟུང་ཤིག་དང་ངས་བཤད་དོ། །བཙུན་པ་དེ་ལྟར་འཚལ་ལོ་ཞེས་གསོལ་ཏེ། དགེ་སློང་རྣམས་ཀྱིས་བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ་མཉན་ནས། བཅོམ་ལྡན་འདས་ཀྱིས་དགེ་སློང་དང་། དགེ་སློང་མ་དང་། དགེ་བསྙེན་དང་། དགེ་བསྙེན་མ་རྣམས་ལ་བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་འདི་རྒྱས་པར་བཤད་དོ། །བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་སྩལ་ནས། དགེ་སློང་དང་། དགེ་སློང་མ་དང་། དགེ་བསྙེན་དང་། དགེ་བསྙེན་མ་དེ་དག་ཡི་རངས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལ་མངོན་པར་བསྟོད་དོ།། །།འདིའི་ཆོ་ག་ནི་ཟས་དཀར་གསུམ་ཟ་བ་དང་། ལེགས་པར་ཁྲུས་བྱས་པ་དང་། སྨྱུང་བ་བྱས་པ་དང་། ཚུལ་ཁྲིམས་དང་། ལྡན་པ་དང་། བསམ་པ་རྒྱ་ཆེན་པོ་དང་ལྡན་པས་རྒྱལ་པོ་ཆེན་པོ་བཞིའི་གཟུགས་ལྡོང་རོས་སམ་བཙག་གིས་བྲིས་ལ། སྤོས་ཐམས་ཅད་དང་ལྡན་པའི་དཀྱིལ་འཁོར་གྲུ་བཞི་པ་བྱས་ནས་སངས་རྒྱས་ཀྱི་སྤྱན་སྔར་དུས་གསུམ་དུ་བརྗོད་པར་བྱའོ། །བསིལ་བའི་ཚལ་གྱི་མདོ་ཆེན་པོ་རྫོགས་སོ།། །།པཎྜི་ཏ་ཤཱི་ལེནྡྲ་བོ་དྷི་དང་། ཛྙཱ་ན་སིདྡྷི་དང་། ཤཱཀྱ་པྲ་བྷ་དང་། ལོ་ཙཱ་བ་བནྡེ་ཡེ་ཤེས་སྡེས་བསྒྱུར་ཅིང་ཞུས་ཏེ་སྐད་གསར་ཆད་ཀྱིས་ཀྱང་བཅོས་ཏེ་གཏན་ལ་ཕབ་པ།། །།ཕྱིས་འགོས་གཞོན་ནུ་དཔལ་གྱིས་ཆག་ལོ་ཙཱ་བའི་ཕྱག་དཔེ་ལས་ཞུས་སོ།།
