༄༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྱཱ་བ་ལོ་ཀི་ཏེ་ཤྭ་ར་སྱ་ནཱ་མ་ཨཥྚ་ཤ་ཏ་ཀཾ། བོད་སྐད་དུ། འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་གི་མཚན་བརྒྱ་རྩ་བརྒྱད་པ། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །འདི་སྐད་བདག་གིས་ཐོས་པ་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་གི་གནས། ཕོ་བྲང་རི་པོ་ཊ་ལའི་རྩེ་མོ་ན། མེ་ཏོག་དྲི་ཞིམ་པོ་སྣ་ཚོགས་ཀྱིས་བཀྲམ་པ། འཛམ་བུ་ན་དའི་གསེར་གྱི་ཁ་དོག་དང་འདྲ་བ། རིན་པོ་ཆེ་སྣ་ཚོགས་ཀྱིས་ཀུན་ཏུ་འབར་བའི་ས་ཕྱོགས་ན་བཞུགས་ཏེ། དེ་ན་ལྷ་དང་། ཀླུ་དང་། གནོད་སྦྱིན་དང་། སྲིན་པོ་དང་། དྲི་ཟ་དང་། ལྷ་མ་ཡིན་དང་། ནམ་མཁའ་ལྡིང་དང་། མིའམ་ཅི་དང་། ལྟོ་འཕྱེ་ཆེན་པོ་དང་། མི་དང་མི་མ་ཡིན་པ་བྱེ་བ་ཁྲག་ཁྲིག་བརྒྱ་སྟོང་དུ་མས་བཅོམ་ལྡན་འདས་ལ་བསྟི་སྟང་དུ་བྱས། བླ་མར་བྱས། རིམ་གྲོར་བྱས། མཆོད་ཅིང་བཀུར་སྟིར་བྱས་ཏེ། ཡོངས་སུ་བསྐོར་ནས་མདུན་དུ་བྱས་པ་ལ་ཆོས་སྟོན་ཏེ། ཐོག་མར་དགེ་བ། བར་དུ་དགེ་བ། ཐ་མར་དགེ་བ། དོན་བཟང་པོ། ཚིག་འབྲུ་བཟང་པོ། ཡོངས་སུ་བྱང་བ། ཡོངས་སུ་དག་པ། མ་འདྲེས་པ་ཡོངས་སུ་རྫོགས་པ། ཡོངས་སུ་གསལ་བ། ཚངས་པར་སྤྱོད་པ་འབའ་ཞིག་ཡང་དག་པར་རབ་ཏུ་སྟོན་ཏོ། །དེ་ནས་ཚངས་པ་ལ་སོགས་པས་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་ལ་བསྟོད་པ། ཀྱེ་ཀྱེ་སངས་རྒྱས་བཅོམ་ལྡན་འདས། བྱ་བ་བྱས་ཤིང་བྱེད་པ་བྱས། །ཁུར་བོར་བདག་དོན་རྗེས་སུ་བརྙེས། །སྲིད་པར་སྦྱོར་བ་ཡོངས་སུ་ཟད། །ཡང་དག་བཀའ་ལ་རྣམ་གྲོལ་སེམས། །ཤེས་རབ་ཤིན་ཏུ་རྣམ་པར་གྲོལ། །ཀླུ་ཆེན་པོ་ནི་ཅང་ཤེས་པོ། །སེམས་ལ་དབང་སྒྱུར་ཐམས་ཅད་ཀྱི། །མཆོག་ཏུ་ཕ་རོལ་ཕྱིན་པ་ཐོབ། །ཡེ་ཤེས་ཚོགས་ནི་ཡོངས་སུ་རྫོགས། །སྲིད་པའི་དགོན་པ་རབ་ཏུ་རྒལ། །གཞན་ལ་ཕན་པ་མཛད་ལ་བརྩོན། །ཐུགས་རྗེ་དང་ནི་འབྲེལ་པའི་ཐུགས། །སྐྱེ་དགུ་ཀུན་ལ་རབ་ཏུ་བྱམས། །དགེ་བ་རབ་བརྩོན་བྱམས་ལས་བྱུང་། །མཐའ་ཡས་སེམས་ཅན་སྒྲོལ་ལ་མཁས། །བདེ་བར་གཤེགས་པ་ཉིད་ལས་སྐྱེས། །བདེ་བར་གཤེགས་པ་ཉིད་ཀྱི་བཀའ། །ཁམས་གསུམ་གྱི་ནི་གཉེན་གཅིག་པོ། །འདོད་ཆགས་ཞེ་སྡང་གཏི་མུག་མེད། །དྲི་མ་གསུམ་ནི་རབ་ཏུ་སྤངས། །རིག་པ་གསུམ་གྱི་ཕ་རོལ་ཕྱིན། །མངོན་པར་ཤེས་པ་དྲུག་ཀྱང་བརྙེས། །ནྱ་གྲོ་དྷ་ནི་ལྟ་བུར་ཟླུམ། །སྐྱེས་བུ་ཆེན་པོའི་མཚན་རྣམས་ནི། །སུམ་ཅུ་རྩ་གཉིས་དག་དང་ལྡན། །དཔེ་བྱད་བཟང་པོ་བརྒྱད་ཅུས་ཀྱང་། །སྐུ་ནི་རྣམ་པ་ཀུན་ཏུ་བརྒྱན། །སྐུ་མདོག་གསེར་གྱི་ཁ་དོག་འདྲ། །སྐུ་ཚད་ཆེ་ཞིང་དཀར་ཤམ་པོ། །ནཱ་ག་གེ་སར་གཞོན་ནུ་ལྟར། །ཅུང་ཟད་དམར་བ་འདྲ་བའི་མདོག །དབུ་ཡི་ཐོར་ཚུགས་དག་དང་ལྡན། །ཐོར་ཚུགས་མང་པོས་ཉེ་བར་གཡོགས། །དབུ་ལ་འོད་དཔག་མེད་པ་བཞུགས། །འོད་ཟེར་དག་ནི་རྣམ་པར་འཕྲོ། །འདོམ་གང་འོད་ནི་རབ་ཏུ་འབར། །གསེར་གྱི་རི་འདྲ་གྲགས་པས་ཁྱབ། །གཟི་བརྗིད་དག་ཀྱང་རབ་ཏུ་ཆེ། །ཤར་གྱི་རི་ལ་འཆར་བ་ཡི། །ཉི་མའི་གཙུག་ཏོར་དག་དང་ལྡན། །ནོར་བུ་རིན་ཆེན་འབར་བ་ཡི། །སེར་ལ་ཁས་ནི་སྐུ་ཕྱེད་བརྒྱན། །ས་དག་ལ་ནི་རབ་ཏུ་མཁས། །ཕ་རོལ་ཕྱིན་པ་བཅུ་རྣམས་སྤྱོད། །ཚུལ་ཁྲིམས་མ་ཉམས་རལ་བ་མེད། །ཐོང་ཀ་སེང་གེ་འཕྱོང་བ་འདྲ། །སྐུ་ནི་འཇམ་ཞིང་ལྡེམ་བག་ཅན། །གཟིགས་ཤིང་གཤེགས་པས་ཁྱུ་མཆོག་འདྲ། །གཡས་སུ་འཁྱིལ་བའི་དུང་ཡང་འདྲ། །དཔྲལ་བ་ཟླ་བ་ཁམ་པ་འདྲ། །དཔྲལ་བའི་དབྱིབས་ཀྱི་རྒྱ་ཡང་ཆེ། །སྨིན་མཚམས་བར་ཡང་ཡོང་མ་ཆད། །སྤྱན་ནི་ནོར་བུ་རིན་ཆེན་འབར། །ཤངས་ཀྱི་དབྱིབས་ནི་ཅུང་ཟད་མཐོ། །ཚེམས་ཀྱི་སྒྲོས་ཀྱིས་རྣམ་པར་གནོན། །ལྗགས་ཀྱིས་སྟོང་གསུམ་ཁེབས་མཛད་ཅིང་། །སྙན་གྱི་ཁུང་ཡང་རབ་ཏུ་ཟབ། །མགུར་ཡང་བུམ་པའི་མགྲིན་པ་འདྲ། །དཔུང་བ་དག་ཀྱང་རབ་ཏུ་ཟླུམ། །ཕྱག་སོར་རྣམས་ཀྱི་ཚིགས་ཀྱང་རིང་། །སེན་མོ་འཇམ་ཞིང་ཅུང་ཟད་དམར། །ཕྱག་སོར་རྣམས་ཀྱི་བར་ཡང་འབྲེལ། །ཕྱག་དང་ཞབས་ཀྱི་མཐིལ་རྣམས་ལ། །འཁོར་ལོ་དག་གིས་རབ་ཏུ་བརྒྱན། །སྟོན་ཀའི་པདྨ་འདྲ་བར་མཛེས། །སྐུ་མདོག་འཇམ་ཞིང་ཤ་ཡང་རྒྱས། །གསུང་ནི་ཚངས་པའི་དབྱངས་ལྟར་ཟབ། །ཡིད་དུ་འོང་ཞིང་དགའ་བར་འགྱུར། །བལྟ་ན་སྡུག་ཅིང་མདངས་དགའ་བ། །པདྨ་ལས་བྱུང་པདྨའི་མདོག །པདྨ་དག་ལས་ཀུན་དུ་སྐྱེས། །པདྨ་དག་གི་གདན་ལ་བཞུགས། །ཕྱག་ན་པདྨ་བཟང་པོ་བསྣམས། །རིལ་བ་སྤྱི་བླུགས་ཕྱག་ན་བསྣམས། །ཀྲྀཥྞ་ས་རིའི་པགས་པ་བསྣམས། །དབྱུག་པ་དང་ནི་བགྲང་ཕྲེང་འཛིན། །བཀྲ་ཤིས་དག་གིས་གཙང་མར་འགྱུར། །སྔ་ནས་གསུང་ཞིང་གསོང་པོར་སྨྲ། །བདུད་རྩིའི་ཆར་ནི་རབ་ཏུ་འབེབས། །ཡིད་བཞིན་ནོར་བུ་རིན་ཆེན་ལྟར། །སེམས་ཅན་ཐམས་ཅད་དགའ་བར་བྱེད། །བཟང་པོར་སྣང་བའི་ལྗོན་ཤིང་ལྟར། །སེམས་ཅན་ཀུན་གྱི་གསོས་སུ་གྱུར། །སངས་རྒྱས་ཀུན་གྱི་སྤྲུལ་པ་ལགས། །བདེ་བར་གཤེགས་པའི་ཆ་ལུགས་མངའ། །བདེ་བར་གཤེགས་པའི་རིང་བསྲེལ་འཛིན། །བ་སྤུ་ཉག་མ་རེ་རེ་ཡང་། །སེམས་ཅན་ཀུན་གྱི་གཙོ་བོར་འགྱུར། །བསོད་ནམས་མཛད་ཅིང་དགེ་བ་མཛད། །བཤོལ་བ་མེད་ཅིང་ངེས་པ་སྟེ། །རབ་ཏུ་གྲགས་པའི་བརྩོན་འགྲུས་ཅན། །འཁོར་བ་དག་ལས་ཀུན་ཏུ་རྒལ། །དམ་ཆོས་རྒྱལ་ཚབ་མངའ་གསོལ་བ། །ཡིད་བཞིན་འཁོར་ལོ་བདེ་བར་སྐོར། །དོན་ཡོད་ཞགས་པས་ཉོན་མོངས་འདྲེན། །འཕགས་པ་རྟ་མགྲིན་བདུད་བཞི་འཇོམས། །ཀྲོ་དྷ་མགྲིན་སྔོན་གདུག་པ་སེལ། །ཆུ་སྲིན་འཛིན་ཁྲིའི་འཇིགས་ལས་སྒྲོལ། །ལྷ་མོ་སྒྲོལ་མ་རྗེས་སུ་འབྲང་། །ཁྲོ་གཉེར་ཅན་ནི་བཀའ་བཞིན་བྱེད། །རྒྱལ་བ་ཅན་ནི་ཚུལ་དང་ལྡན། །མཐུ་ཆེན་ལྡན་ཞིང་དྲན་དང་ལྡན། །ཡོན་ཏན་ལྡན་ཞིང་བྱམས་དང་ལྡན། །ཤེས་རབ་ལྡན་ཞིང་དཔལ་དང་ལྡན། །ཚུལ་ཁྲིམས་ལྡན་ཞིང་དོན་དང་ལྡན། །དོན་རྣམས་ཀུན་ནི་རིག་པ་པོ། །ཐེ་ཚོམ་ཐམས་ཅད་གཅོད་པ་པོ། །ཆོས་རྣམས་ཐམས་ཅད་སྨྲ་བ་པོ། །འཇིག་རྟེན་ཀུན་གྱི་སྟོན་པ་པོ། །ཞལ་བཟངས་ཟླ་བ་ཉ་བ་འདྲ། །སྐུ་སྨད་རིན་ཆེན་ཀུན་གྱིས་བརྒྱན། །གསེར་གྱི་མཆོད་སྡོང་འདྲ་བར་གནས། །འོད་ཟེར་ཉི་མ་སྟོང་བས་ལྷག །ཚངས་ལ་སོགས་པས་ཕྱག་བགྱིས་པའི། །འཇིག་རྟེན་མགོན་ལ་ཕྱག་འཚལ་ལོ། །གང་ལ་ལ་ཞིག་འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་ལ་མཚན་བརྒྱ་རྩ་བརྒྱད་ཀྱིས་བསྟོད་པ་བྱས་པ་དེའི་མཚམས་མེད་པ་ལྔ་བྱས་པའི་ལས་ཀྱི་སྒྲིབ་པ་ཡོངས་སུ་བྱང་བར་འགྱུར་རོ། །དཀྱིལ་འཁོར་ཐམས་ཅད་དུ་འཇུག་པར་འགྱུར་རོ། །སྔགས་ཐམས་ཅད་ཀྱང་འགྲུབ་པར་འགྱུར་རོ། །བསྐལ་པ་སྟོང་གི་བར་དུ་ཡང་ངན་འགྲོར་མི་སྐྱེའོ། །མནར་མེད་པར་མི་ལྟུང་ངོ་། །སུ་ཞིག་ནང་པར་ལངས་ནས། ཀློག་གམ་ཀློག་ཏུ་འཇུག་པ་དང་། ཁ་ཏོན་དུ་བྱེད་པ་དེའི་ལུས་ལ་མཛེ་དང་ཤུ་བ་དང་གློ་ན་བ་དང་དབུགས་མི་བདེ་བ་ལ་སོགས་པ་ནད་ཐམས་ཅད་དང་བྲལ་བར་འགྱུར་རོ། །སྐྱེས་སོ་ཅོག་ཏུ་སྐྱེ་བ་དྲན་པར་འགྱུར་རོ། །ལྷའི་སྲས་པོ་དང་འདྲའོ། །འཆི་བའི་དུས་ཀྱི་ཚེ་ན་ཡང་འཇིག་རྟེན་གྱི་ཁམས་བདེ་བ་ཅན་དུ་སྐྱེ་བར་འགྱུར་རོ། །གར་སྐྱེས་གར་གནས་ཀྱང་འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་དང་མི་འབྲལ་བར་འགྱུར་རོ། །རྟག་ཏུ་བཟླས་བརྗོད་བྱས་ན་ཡིད་གཞུངས་པར་འགྱུར་རོ། །སེམས་དཔའ་བར་འགྱུར་རོ། །དབྱངས་སྙན་པར་འགྱུར་རོ། །སྐལ་བ་བཟང་པོར་འགྱུར་རོ། །གཙུག་ལག་ཐམས་ཅད་ལ་མཁས་པར་འགྱུར་རོ། །ཚིག་བཙུན་པར་འགྱུར་རོ། །བསྟོད་པ་འདིས་བསྟོད་ན་སངས་རྒྱས་བཅོམ་ལྡན་འདས་གང་གཱའི་ཀླུང་དྲུག་ཅུ་རྩ་གཉིས་སྙེད་ཀྱི་བྱེ་མ་སྙེད་དག་ལ་མཆོད་པ་བྱས་པར་འགྱུར་ཏེ། རྣམ་པར་སྨིན་པ་འདྲ་བས་ན་བྱེ་བྲག་མེད་དོ། །རྒྱལ་བ་ཐམས་ཅད་ཀྱི་རིང་བསྲེལ་རིན་པོ་ཆེ་ཟ་མ་ཏོག་ཅེས་བྱ་བ། འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་གི་མཚན་བརྒྱ་རྩ་བརྒྱད་པ་རྫོགས་སོ།། །།
