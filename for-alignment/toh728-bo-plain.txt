༄༅། །རྒྱ་གར་སྐད་དུ། ཏཱ་ར་དེ་བཱི་ནཱ་མ་ཨཥྚ་ཤ་ཏ་ཀ་ན། བོད་སྐད་དུ། ལྷ་མོ་སྒྲོལ་མའི་མཚན་བརྒྱ་རྩ་བརྒྱད་པ་ཞེས་བྱ་བ། ལྷ་མོ་ཆེན་མོ་སྒྲོལ་མ་རྣམ་པར་རྒྱལ་བའི་ནོར་བུ་རྣམ་པ་སྣ་ཚོགས་ཀྱི་དོ་ཤལ་དང་། ན་བཟའ་དང་། ཤམ་ཐབས་ཀྱི་གཤལ་ཀས་འཇིག་རྟེན་གྱི་ཁམས་ཆེན་པོའི་བླ་རེ་བྱས་པ་ལ་ཕྱག་འཚལ་ལོ། །དེ་ནས་ལྷ་མོ་སྒྲོལ་མས་དེའི་ཚེ་སྔོན་མ་མཐོང་མ་ཐོས་པའི་གཟུངས་ཀྱི་མཆོག་བརྗོད་པ། ཨོཾ་ཏྲཻ་ལོ་ཀྱ་བི་ཛ་ཡ། ཨ་ཙིཏྟ་ཛཱ་ཡ། ཨ་ནི་ར་ཙི་ཛ་ཏཱ་ཡ། སུ་ཛཱ་ཡ། བི་ཛཱ་ཡ། མ་ཧཱ་ཛཱ་ཡ། བི་ཛཱ་ཡ། ཛ་ཡ་ཛ་ཡ། ཧེ་ཧེ། སྨ་ར་སྨ་ར། བ་ར་བ་ར། བར་ད། མ་ཧཱ་དེ་བི། བི་ལ་ས། བི་ལ་ས། བི་ལ་ས། བ་ཛ་ཨ་མོ་ཏ་ཀ་རི་བི་ཛ་ཡ་དེ་བི། ཧཱི་ཧཱི། ཀི་ནི་ཀི་ནི་པི་ལམ་བ་སེ། སྨ་ར་སྨ་ར། མ་ཧཱ་པྲ་ཛྙཱ་ཨ་ནི་བ་ཏ། བཛྲ་མ་མ་ཧཱ་ཀཱ་རུ་ཎི་ཀ །བྷ་གི་ནི་ཧྲཱུཾ་ཧྲཱུཾ། སྥུཊ་སྥུཊ། སྥོ་ཊ་སྥོ་ཊ། ཨཱ་བེ་ཤ་ཡ། ཨཱ་བེ་ཤ་ཡ། དྷུ་ན་དྷུ་ན། བི་དྷུ་ན་བི་དྷུ་ན། ཀཾ་པ་ཀཾ་པ། ཀཾ་པ་ཡ། སུ་ར་བྷི་གནྡྷ། ཨ་ཏི་སི་ཏ། མུ་ཁེ་ཧ་ལ་ད་བ་ལ། སཾ་ག་རཱ་ཛ་ན། བི་མ་བི་གི །བི་ནཱ་ཤ་ཡ། བི་ནཱ་ཤ་ཡ། ཧེ་བྷ་ག་བ་ཏི། སནྟྲ་ས་ཡ། བི་གྷ་ནཱ་ན། པ་རི་བ་ལཱ་ཡ། ཨ་སྨཱ་ཀཾ་ཨ་སྨཱ་ཀཾ། མཱ་ད་ནི་མཱ་ད་ནི། སཾ་བོ་དྷ་ཡ། སཾ་མོ་ཧ་ཡ། ཧ་ར་ཧ་ར། ཧི་རི་ཧི་རི། ཧུ་རུ་ཧུ་རུ། བི་ཏ་ཀ་ར། བ་ར་ཛ། ཧཾ་ཧཾ། ཏྲོ་ཏཱ་ཡ། ཧཾ་ཧཾ། སནྟྲ་ཏཱ་ཡ། སནྟྲོ་ཡ། མནྟ་མནྟ། མཪྡ་མཪྡ། ཙ་ར་ཙ་ར། སཾ་ཙ་ར་སཾ་ཙ་ར། བེ་ག་བེ་ག་བེ་ག་བ་ཏི། ན་མོ་ན་མོ་ན་མཿསྭཱ་ཧཱ། དེ་ནས་ལྷ་དང་ལྷ་མིན་ཚོགས། །སྲིན་པོ་གནོད་སྦྱིན་མིའམ་ཅི། །ལྟོ་འཕྱེ་ཆེན་པོའི་ཚོགས་རྣམས་དང་། །འབྱུང་པོ་དྲི་ཟ་ལོག་འདྲེན་རྣམས། །འཇིག་རྟེན་གསུམ་དུ་གྲགས་པའི་མཆོག །གསང་སྔགས་ཆེན་པོ་འདི་ཐོས་ནས། །སྐྲག་ཅིང་མགོ་བོ་སྨད་གྱུར་ཏེ། །ལག་པ་ཆེན་པོ་ཡོངས་བསྐྱོད་ནས། །གང་གིས་འདི་ནས་ས་གསུམ་འདི། །མཐའ་དག་ཁྱབ་པར་བྱས་པའི་ཕྱིར། །འདོད་པའི་ལྷ་ཡི་ངག་བཟློག་པ། །སྒྲོལ་མ་སྤྱན་ཡངས་ཕྱག་འཚལ་ལོ། །སྤྱན་རས་གཟིགས་ཀྱི་རིགས་སྨོས་པ། །ཆེ་བ་ཉིད་ཀྱི་གྲགས་པ་འོད། །སྤྱན་ཡངས་ཁྱོད་ནི་སེམས་ཅན་རྣམས། །ཕན་བཞེད་སླད་དུ་འབྱུང་བ་ལགས། །དེ་ནས་དབང་ཕྱུག་ལ་སོགས་པ། །ལྷ་དང་ལྷ་མིན་འདུས་ཀུན་དང་། །རིག་སྔགས་འཆང་གི་བདག་པོ་དང་། །དེ་བཞིན་རིག་སྔགས་འཆང་བ་རྣམས། །བླ་གོས་ཕྲག་པ་གཅིག་གཟར་ཏེ། །བསྐོར་བ་མཆོག་ནི་བྱས་ནས་ཀྱང་། །པུས་མོའི་ལྷ་ང་ས་སྟེང་དུ། །བཙུགས་ནས་ཚིག་ནི་འདི་སྐད་སྨྲས། །སྒྲོལ་མ་འོད་འབར་དེ་ལ་ནི། །ཕྱག་འཚལ་ཕྱག་འཚལ་ཕྱག་འཚལ་ལོ། །མིང་བརྒྱ་རྩ་བརྒྱད་དམ་པ་ཡི། །དཔེ་མེད་མིང་ནི་བརྗོད་པར་བགྱི། །བཀྲ་ཤིས་རྣམ་རྒྱལ་དཔལ་བཟང་བ། །སྐྱེ་དགུའི་དབང་ཕྱུག་མངོན་ཤེས་མྱུར། །དབང་ཕྱུག་རབ་དགའ་འཇིགས་ལ་དྲག །རྒོད་བག་མེད་ཅིང་དྲེགས་པ་མེད། །པདྨའི་སྤྱན་དང་པདྨའི་ཞལ། །པདྨའི་ལྟེ་བ་རབ་གཙང་སྦྱིན། །སྤྱན་བཟང་སྤྱན་ནི་ཡངས་པ་སྟེ། །སྔོ་བ་ཨུཏྤལ་སྔོན་པོའི་འོད། །སྐྱོབ་པ་སྐྱབས་འོས་བདེ་བ་སྦྱིན། །དེ་བཞིན་སྟོབས་སྦྱིན་གཟུགས་སྦྱིན་པ། །མགོན་པོ་མགོན་མཛད་དྲག་མོ་སྟེ། །དཀོན་མཆོག་གསུམ་པོ་རྒྱལ་མཛད་མ། །སྒྲོས་དམར་སེན་མོ་དམར་བ་སྟེ། །སྤྱན་རྩ་དམར་པོས་གཟིགས་པར་མཛད། །དགའ་མཛད་དགའ་སྟོབས་པདྨ་ལགས། །པདྨའི་མདོག་ཅན་པད་མདོག་དགྱེས། །འཇིག་རྟེན་གསུམ་འདུལ་ཞི་བ་སྟེ། །འཇིགས་པའི་ཆ་ལུགས་སྟོབས་མོ་ཆེ། །འདུལ་དུ་འཆུག་མ་འདུལ་མཛད་མ། །བསྟོད་ལྡན་མི་མཆོག་དམ་པ་ལགས། །རབ་གསལ་འདུལ་བ་རྡོ་རྗེ་ཅན། །རྣམ་པར་རྒྱལ་བ་འོད་གསལ་བ། །བྱང་ཆུབ་རྡུལ་མེད་དྲི་མ་མེད། །ཟླ་བ་ལྷ་མཆོག་རྣམས་ཀྱི་རྗེད། །སྲེད་བྱ་དང་ནི་སྲེད་བྱའི་བར། །རྒྱུ་ཞིང་གཤེགས་ལ་འོད་ཟེར་མཛེས། །ཆུ་ལྷ་གཡུལ་འགྱེད་འགྱིང་བག་ཅན། །སྐྱེས་མཆོག་རང་བཞིན་ཡིད་འཕྲོས་རྒྱས། །གཙུག་ཕུད་གསུམ་པོ་ལྟོ་ཚིལ་གསུམ། །སྤྱན་གསུམ་དྲི་མ་གསུམ་སེལ་བ། །སྲིད་པ་སེལ་མཛད་ཁོང་དགེ་བ། །ཚིག་ལྡན་མར་ཡང་དེ་བཞིན་གྲགས། །རྒྱལ་བ་སྦྱིན་པ་བསོད་ནམས་སྦྱིན། །སྡིག་པ་འཇོམས་མ་འཇིགས་པ་མེད། །ནོར་སྦྱིན་མདོག་སྦྱིན་ཞི་བ་མ། །གཟུགས་བཟང་ཕྱལ་བཟང་ནུ་མ་བཟང་། །སྒྱུ་འཕྲུལ་སྒྱུ་འཕྲུལ་ལྡན་ཞིང་འདུལ། །དུལ་བ་བསྲུང་པ་རྡུལ་སེལ་བ། །ཕྲེང་བ་ཐོགས་མ་རྒྱགས་མཛད་མ། །རྒྱགས་པ་སྦྱིན་མ་དྲེགས་སྤོང་མ། །དྲག་མོ་ལགས་ཏེ་གཟི་བསྐྱེད་མ། །དཀར་མོ་ཞི་མཛད་བདེ་བ་མོ། །སྐྱེ་དགུའི་བདག་མོ་ཚངས་པར་ལྡན། །ཚངས་སྦྱིན་དབང་ཕྱུག་དམ་པ་ལགས། །ཟླ་བའི་འོད་ཅན་ཟླ་བའི་ཞལ། །ཟླ་བའི་གཟི་ནི་འཕྲོག་པར་མཛད། །ཉི་མའི་གཟི་འཕྲོག་དགྱེས་པ་མོ། །ཡིད་འོང་བཟང་ཞིང་ཞི་བ་མོ། །བདེ་བ་བདེ་ལྡན་མུ་དང་སྟེགས། །བཀྲ་ཤིས་བགེགས་རྣམས་སེལ་མཛད་མ། །ནོར་སྤེལ་བསོད་ནམས་རྣམ་པར་རྒྱལ། །འཛམ་བུའི་ཆུ་བོ་གསེར་དངུལ་འོད། །འཇིགས་པ་སེལ་བ་འཇིགས་མེད་སྦྱིན། །སྒྲོལ་མ་ལགས་ཏེ་ཟླ་བའི་ཞལ། །སེམས་ཅན་ཀུན་ལ་མེད་མི་འགྱུར། །མཆོད་གནས་ཕྱག་གནས་དགའ་མཛད་མ། །ཆ་ལུགས་སྔོ་བ་ཆ་ལུགས་བཟང་། །རྒྱན་དམར་རྒྱན་དུ་ཐོགས་པ་སྟེ། །ན་བཟའ་གསེར་མནབས་གང་ག་མ། །དཀར་མོ་འབར་བ་རྣམ་པར་གཟིགས། །གིའུ་ཝང་གི་ཐིག་ལེ་སྤོས། །འཁོར་ལྡན་རྡོ་རྗེ་ཆུ་ལྷ་ཉིད། །དཀྱིལ་འཁོར་མཆོག་མངའ་འཇིགས་མཛད་མ། །ཚེམས་དཀར་བཞད་ཅིང་ཚེམས་དཀར་འཛུམ། །དཔུང་རྒྱན་སྙན་ཆ་གདུབ་འཁོར་ཐོགས། །དོ་ཤལ་འབར་བའི་ཆུན་པོ་ཅན། །འོག་པག་བཟང་ལྡན་ཞལ་བཟང་མ། །གྲོགས་ནི་ནིམ་པའི་ལོ་མ་ཐོགས། །ནོར་བུའི་ཅོད་པན་ནོར་བུ་འཆང་། །ནོར་བུའི་རྒྱན་རྣམས་རྒྱན་དུ་ཐོགས། །ནོར་བུའི་འོད་ཟེར་སྣང་བ་ཅན། །ནོར་བུའི་དཀྱིལ་འཁོར་རྒྱན་མངའ་བ། །འབར་བའི་མདོག་མངའ་རྡོ་རྗེ་ཅན། །སྐྱེས་པ་བུད་མེད་སད་པར་མཛད། །ཞི་བ་དབྱིག་ཐོགས་མཆེ་བ་ཅན། །ཚངས་པའི་ཤུགས་ཅན་རླུང་དང་ལྡན། །བུད་མེད་ལགས་ཏེ་མཁའ་ལས་གཤེགས། །ལྷ་མོ་མཆོག་གི་འོད་འཕྲོག་མ། །འོད་དང་ལྡན་པ་འདོད་པས་འདུལ། །འདོད་པའི་ཁྲོ་བོ་རབ་འཇོམས་མ། །དྲག་པོ་ཐོར་ཚུགས་གཙུག་པུད་ཅན། །ཚངས་པ་ཚངས་པའི་གསུང་ལྡན་རྒྱལ། །བིང་ཀའི་སྐད་འདྲ་ཕེག་བཟང་མ། །འཇིག་རྟེན་དབང་ཕྱུག་རིགས་མཆོག་སྐྱེས། །དབྱིག་པའི་ཤིང་ལོ་རྡོ་རྗེ་ཐོགས། །དྲག་མོ་རིན་ཆེན་འོད་ཀྱིས་བརྒྱན། །གཤེགས་ན་སྙན་ཆ་དྲི་མེད་གཡོ། །གཤིན་རྗེ་བ་ཡི་ཡིད་ལ་འཇུག །རིག་སྔགས་འཆང་མ་གཟུགས་འཆང་མ། །སྟོབས་ལྔ་སྟོབས་དང་ལྡན་པར་མཛད། །སྟོབས་དང་སྟོབས་མེད་མངའ་བ་སྟེ། །ཤེས་རབ་མཚོན་གྱིས་གཅོད་པར་མཛད། །སྒྱུ་རྩལ་ལགས་ཏེ་སྒྱུ་རྩལ་ལྡན། །དེ་བཞིན་བརྗོད་དང་སྨྲ་བ་ལགས། །སྟེང་ནས་ཁྱམས་དང་ཁྱམས་བཟང་ཚུལ། །ཙམ་པ་ཀ་ཡི་ཆུན་པོས་གསལ། །སེར་མོ་ཙན་པ་ལེགས་པའི་རྟེན། །དྲག་མོ་ཕྱག་ན་ཆུན་པོ་ཐོགས། །སྐལ་བ་བཟང་མོ་གཤལ་ཀ་ཅན། །དེ་བཞིན་ཤིན་ཏུ་ཤས་ཆེར་དགྱེས། །རིགས་མཚན་ཤིན་ཏུ་སྟོབས་དང་ལྡན། །རྡོ་རྗེ་ཁྲོ་བོ་དགའ་མཛད་མ། །བདེ་ལྡན་བློ་ལྡན་རྡོ་རྗེའི་ཐབས། །བློ་མཛད་ཤེས་རབ་སྐྱེད་པར་མཛད། །ལྷ་མོ་ལམ་མེ་ཅོད་པན་ཅན། །ལྷ་རྣམས་ཀུན་ལས་ཚུལ་ལྷག་པ། །སྲེད་མེད་བུ་མོ་སྡིག་པ་སེལ། །བཀྲ་ཤིས་ངོ་ཚ་ཤེས་དང་དགྱེས། །རབ་ཏུ་དགའ་མ་འགྲོགས་ན་དགའ། །སྤྱོད་ཚུལ་དགའ་ལ་དགའ་དང་ལྡན། །དགའ་མ་དགའ་བར་མཛད་པ་སྟེ། །ཆོས་རྣམས་ཀུན་གྱི་ལེགས་པའི་རྟེན། །སངས་རྒྱས་སྤྲུལ་པ་སྦྲུལ་མཛད་མ། །ང་རྒྱལ་སེལ་བར་མཛད་པ་སྟེ། །ལས་དང་ཉོན་མོངས་རབ་འཇོམས་ཤིང་། །འདོད་པའི་སྡེ་རྣམས་རབ་འཇོམས་མ། །ཤེས་པ་བཅུ་མངའ་སངས་རྒྱས་པ། །ཕ་རོལ་ཕྱིན་པ་བཅུ་ལ་གནས། །ས་བཅུ་དག་ལ་མངའ་བརྙེས་པ། །སྟོབས་རྣམས་བཅུ་བོས་ཆེར་གསལ་མ། །རྡོ་རྗེ་ཅོད་པན་རྡོ་རྗེ་འཆང་། །རྡོ་རྗེ་ཕྲ་མོ་ལྷ་མིན་སེལ། །དུས་ཀྱི་རྡོ་རྗེ་རབ་ཏུ་རྒྱགས། །རྡོ་རྗེ་ལྕགས་ཀྱུ་བདེ་དང་དཔལ། །རྡོ་རྗེ་ཞགས་པ་ཞགས་བ་བཟང་། །སྒེག་མོ་རྡོ་རྗེ་ཕྱག་ན་བསྣམས། །ཡིད་འོང་ཡིད་ལ་ཚིམ་པར་མཛད། །ཕ་རོལ་བགེགས་རྣམས་སེལ་མཛད་མ། །ཚིམ་མཛད་ཤས་ཆེར་བདེ་བ་སྟེ། །རོལ་ཅིང་གར་ཐབས་སྟོན་པར་མཛད། །ཆོས་ཅན་ཧ་ལ་ཧ་ལའི་འོད། །ཆོས་སྦྱིན་རྣམ་པར་གཟིགས་པ་ལགས། །ཁྱོད་ནི་རྡོ་རྗེ་འཕྲོ་བ་དང་། །བདུད་རྩི་རྡོ་རྗེར་བརྗོད་པ་སྟེ། །རྡོ་རྗེ་འབར་བ་སྟོབས་མོ་ཆེ། །ལྷ་མོ་ཆེན་མོ་ཁྱོད་ཕྱག་འཚལ། །བདག་ཅག་རྣམས་ཀྱིས་བསྟོད་པ་ཡིས། །མཚན་བརྒྱ་རྩ་བརྒྱད་མཆོག་འདི་བརྗོད། །ཀྱེ་ཀྱེ་བཅོམ་ལྡན་མཆོག་ཉིད་ནི། །སྨྲ་བ་རྣམས་ཀྱི་མཆོག་ལགས་གསུང་། །དེ་ནས་ཟླ་བའི་འོད་འདྲ་ཞིང་། །འབར་ལ་འཕྲོ་བར་ལྡན་པ་ཡི། །ལྷ་མོ་སྒྲོལ་མ་བརྗིད་བག་ཅན། །འཛུམ་བཞིན་དུ་ནི་རབ་སྨྲས་པ། །དབང་ཕྱུག་ཆེན་པོ་ལ་སོགས་ཀུན། །ང་ཡི་ཚིག་ནི་མཉན་པར་གྱིས། །ཐམས་ཅད་རྟག་ཏུ་མཐུན་པར་ནི། །སྟོབས་བཅུ་དག་ལ་བརྩོན་པར་གྱིས། །དེ་བཞིན་གཤེགས་པའི་ཚུལ་ཆེན་ཡང་། །དུས་ཀུན་ཏུ་ནི་བསྲུང་བར་གྱིས། །ཞི་བའི་ལམ་ལ་དམིགས་པས་ནི། །འཁོར་བའི་རྒྱ་མཚོ་མི་བཟད་ཅིང་། །འཇིགས་པ་ཆེན་པོའི་གཞིར་གྱུར་པ། །རྒལ་དཀའ་ཁྱེད་ཀྱིས་རྒལ་བར་གྱིས། །མྱ་ངན་འདས་པའི་གྲོང་ཁྱེར་ལ། །དམིགས་ནས་བརྩོན་ཏེ་འཇུག་པར་གྱིས། །དེ་ནས་ཉི་འོད་འབར་བ་ཡི། །ཕྲེང་བ་ལྟ་བུ་མཛེས་པའི་ཞལ། །དེ་བཞིན་གཤེགས་པའི་འབྲུག་གི་སྒྲ། །འདྲ་བའི་ཚིག་ནི་བཀའ་སྩལ་པ། །ཀྱེ་མ་རིག་སྔགས་སྒྲོལ་མའི་མཆོག །མུན་པ་སེལ་བར་བྱེད་པ་དང་། །མཐུ་སྟོབས་རྒྱ་ཆེན་སྐྱེད་བྱེད་པ། །སྒྲིབ་པ་ཐམས་ཅད་གཅོད་བྱེད་པ། །ཚོགས་རྣམས་སྐྱེད་པའི་གཟི་བརྗིད་ཀྱིས། །དག་པར་བྱས་པ་མ་ཡིན་པས། །ཤས་ཆེར་དགའ་བར་བྱེད་པ་ཡི། །གཟུངས་འདི་ཐོས་པར་མི་འགྱུར་རོ། །སྔོན་ཆད་སངས་རྒྱས་བརྩོན་བྱས་པའི། །བརྟན་པོ་གཟི་བརྗིད་དག་རྣམས་ལ། །དམ་ཚིག་ཆེན་པོ་རིག་སྔགས་ཀྱི། །དམ་ཚིག་འདི་ནི་འབྱུང་བར་འགྱུར། །ཉོན་མོངས་མཐའ་དག་རབ་འཇོམས་ཤིང་། །སྡིག་པ་ཐམས་ཅད་སེལ་བྱེད་པ། །སེམས་ཅན་ཐམས་ཅད་རབ་འབྱེད་པའི། །གཟུངས་སྔགས་རྡོ་རྗེ་མཆོག་འཛིན་པ། །སུ་ཞིག་མཉམ་པར་བཞག་པ་ཡིས། །ཉན་ཏམ་འཆང་ངམ་ཀློག་བྱེད་ན། །རང་བྱུང་གིས་ནི་གསུངས་པ་བཞིན། །དེ་ཡི་དོན་ཀུན་གྲུབ་པར་འགྱུར། །ཉོན་མོངས་ཀུན་ལས་རྣམ་གྲོལ་ཞིང་། །སེམས་ཅན་ཀུན་ལས་ལྷག་པའི་མཆོག །འཇིག་རྟེན་དག་ཏུ་ཆོས་ཤེས་ལ། །འཛུད་པའི་དེ་བཞིན་གཤེགས་དང་མཚུངས། །སུ་ཞིག་མིང་བརྒྱ་རྩ་བརྒྱད་འདི། །འདོན་ཞིང་སྒྲོལ་མ་ཟླ་བའི་ཞལ། །དེ་བཞིན་སྟོད་བྱེད་དེ་དག་ནི། །སངས་རྒྱས་ཉིད་ཀྱང་ཐོབ་པར་འགྱུར། །དེ་ཡི་དོན་ཀྱང་གྲུབ་འགྱུར་ཞིང་། །རྒྱལ་པོ་རྣམས་དང་སྟོབས་ཆེན་མི། །གཞན་ཡང་སེམས་ཅན་བགྲང་བ་གང་། །ཐམས་ཅད་དེ་ཡི་དབང་དུ་འགྱུར། །ལྷ་མོ་སྒྲོལ་མ་ཞལ་བཟང་བའི། །མིང་བརྒྱ་རྩ་བརྒྱད་གྲགས་པ་འདི། །འཇིག་རྟེན་དག་ན་རྒྱགས་པ་དང་། །ཉོན་མོངས་མི་རིགས་ཀུན་སེལ་བའོ། །དེ་ནས་བདུད་རིས་གྱེན་དུ་གཡོས་པའི་རི་བོ་འཁྲུག་བྱེད་པ། །དེ་བཞིན་གཤེགས་པའི་རི་བོ་མཆོག་དང་འདྲ་བའི་སྐུ་མངའ་བ། །འོད་གསལ་ལྷ་དང་མི་རྣམས་ལེགས་པར་སད་པར་མཛད་པ་པོ། །རྒྱལ་སྲས་ཐུ་བོ་རྣམས་ཀྱིས་ཁྱབ་པར་སྟོབས་ཀྱིས་བརྟེན་པ་ཡིས། །གསུང་རབ་རྟག་ཏུ་ཚོགས་ཆེན་བརྡ་སྤྲོད་ལ། །དོན་ཆེན་གྲགས་ཤིང་ཕན་ཕྱིར་ཞུགས་པ་གང་། །རབ་དེས་ཆོས་དེ་དང་ནི་སྡུད་བྱེད་དང་། །དེ་བཞིན་ཁྱད་པར་སེམས་པའི་རིམས་སེལ་བས། །བཟོད་མེད་ཉོན་མོངས་སེལ་བ་དང་། །ཞི་བའི་ཡིད་ནི་སྐྱེད་པའི་ཚིག །རྒྱལ་བའི་མཆོག་གིས་བཀའ་སྩལ་ནས། །གཞེས་པར་གྱུར་ཏེ་མི་གསུང་བཞུགས། །དེ་ནས་སྒྲོལ་མཆོག་ལྷ་མོས་གཙོར་གྱུར་དགའ་བར་བྱེད་ཅིང་ཡིད་འཕྲོག་ལ། །དགའ་བ་རྒྱ་ཆེན་སྐྱེད་ཅིང་ཚེ་དང་བདེ་སྦྱིན་བསྟན་པའི་ཚིག་ཐོས་པས། །ལྷ་ཚོགས་ལྷ་མིན་ཚོགས་བཅས་མཐའ་དག་དགའ་བས་རྒྱས་ཤིང་རྨད་བྱུང་ནས། །རབ་ཏུ་དགའ་ཞིང་སྤྱི་བོས་ཐལ་མོ་སྦྱར་ཏེ་སྨྲ་བ་ཞི་བར་གྱུར། །སྒྲོལ་མའི་མཆོག་ཅེས་བྱ་བའི་གཟུངས་ཏེ་ལྷ་མོ་སྒྲོལ་མའི་མཚན་བརྒྱ་རྩ་བརྒྱད་པ་རྫོགས་སོ།། །།
